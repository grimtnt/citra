274 SharedPtr<CodeSet> ElfReader::LoadInto(u32 vaddr) {
275  LOG_DEBUG(Loader, "String section: {}", header->e_shstrndx);
276 
277  // Should we relocate?
278  relocate = (header->e_type != ET_EXEC);
279 
280  if (relocate) {
281  LOG_DEBUG(Loader, "Relocatable module");
282  entryPoint += vaddr;
283  } else {
284  LOG_DEBUG(Loader, "Prerelocated executable");
285  }
286  LOG_DEBUG(Loader, "{} segments:", header->e_phnum);
287 
288  // First pass : Get the bits into RAM
289  u32 base_addr = relocate ? vaddr : 0;
290 
291  u32 total_image_size{};
292  for (unsigned int i{}; i < header->e_phnum; ++i) {
293  Elf32_Phdr* p = &segments[i];
294  if (p->p_type == PT_LOAD) {
295  total_image_size += (p->p_memsz + 0xFFF) & ~0xFFF;
296  }
297  }
298 
299  std::vector<u8> program_image(total_image_size);
300  std::size_t current_image_position{};
301 
302  SharedPtr<CodeSet> codeset{CodeSet::Create("", 0)};
303 
304  for (unsigned int i{}; i < header->e_phnum; ++i) {
305  Elf32_Phdr* p{&segments[i]};
306  LOG_DEBUG(Loader, "Type: {} Vaddr: {:08X} Filesz: {:08X} Memsz: {:08X} ", p->p_type,
307  p->p_vaddr, p->p_filesz, p->p_memsz);
308 
309  if (p->p_type == PT_LOAD) {
310  CodeSet::Segment* codeset_segment{};
311  u32 permission_flags{p->p_flags & (PF_R | PF_W | PF_X)};
312  if (permission_flags == (PF_R | PF_X)) {
313  codeset_segment = &codeset->CodeSegment();
314  } else if (permission_flags == (PF_R)) {
315  codeset_segment = &codeset->RODataSegment();
316  } else if (permission_flags == (PF_R | PF_W)) {
317  codeset_segment = &codeset->DataSegment();
318  } else {
319  LOG_ERROR(Loader, "Unexpected ELF PT_LOAD segment id {} with flags {:X}", i,
320  p->p_flags);
321  continue;
322  }
323 
324  if (codeset_segment->size != 0) {
325  LOG_ERROR(Loader,
326  "ELF has more than one segment of the same type. Skipping extra "
327  "segment (id {})",
328  i);
329  continue;
330  }
331 
332  u32 segment_addr{base_addr + p->p_vaddr};
333  u32 aligned_size{(p->p_memsz + 0xFFF) & ~0xFFF};
334 
335  codeset_segment->offset = current_image_position;
336  codeset_segment->addr = segment_addr;
337  codeset_segment->size = aligned_size;
338 
339  memcpy(&program_image[current_image_position], GetSegmentPtr(i), p->p_filesz);
340  current_image_position += aligned_size;
341  }
342  }
343 
344  codeset->entrypoint = base_addr + header->e_entry;
345  codeset->memory = std::make_shared<std::vector<u8>>(std::move(program_image));
346 
347  LOG_DEBUG(Loader, "Done loading.");
348 
349  return codeset;
350 }

379 ResultStatus AppLoader_ELF::Load(Kernel::SharedPtr<Kernel::Process>& process) {
380  if (is_loaded)
381  return ResultStatus::ErrorAlreadyLoaded;
382 
383  if (!file.IsOpen())
384  return ResultStatus::Error;
385 
386  // Reset read pointer in case this file has been read before.
387  file.Seek(0, SEEK_SET);
388 
389  std::size_t size{file.GetSize()};
390  std::unique_ptr<u8[]> buffer{new u8[size]};
391  if (file.ReadBytes(&buffer[0], size) != size)
392  return ResultStatus::Error;
393 
394  ElfReader elf_reader{&buffer[0]};
395  SharedPtr<CodeSet> codeset{elf_reader.LoadInto(Memory::PROCESS_IMAGE_VADDR)};
396  codeset->name = filename;
397 
398  process = Kernel::Process::Create(std::move(codeset));
399  process->svc_access_mask.set();
400  process->address_mappings = default_address_mappings;
401 
402  // Attach the default resource limit (APPLICATION) to the process
403  process->resource_limit =
404  Kernel::ResourceLimit::GetForCategory(Kernel::ResourceLimitCategory::APPLICATION);
405 
406  process->Run(48, Kernel::DEFAULT_STACK_SIZE);
407 
408  is_loaded = true;
409  return ResultStatus::Success;
410 }
