12 ResultCode ErrEula::ReceiveParameter(const Service::APT::MessageParameter& parameter) {
13  if (parameter.signal != Service::APT::SignalType::Request) {
14  LOG_ERROR(Applet_ErrEula, "unsupported signal {}", static_cast<u32>(parameter.signal));
15  UNIMPLEMENTED();
16  // TODO(Subv): Find the right error code
17  return ResultCode(-1);
18  }
19 
20  // The LibAppJustStarted message contains a buffer with the size of the framebuffer shared
21  // memory.
22  // Create the SharedMemory that will hold the framebuffer data
23  Service::APT::CaptureBufferInfo capture_info{};
24  ASSERT(sizeof(capture_info) == parameter.buffer.size());
25 
26  memcpy(&capture_info, parameter.buffer.data(), sizeof(capture_info));
27 
28  // TODO: allocated memory never released
29  using Kernel::MemoryPermission;
30  // Allocate a heap block of the required size for this applet.
31  heap_memory = std::make_shared<std::vector<u8>>(capture_info.size);
32  // Create a SharedMemory that directly points to this heap block.
33  framebuffer_memory = Kernel::SharedMemory::CreateForApplet(
34  heap_memory, 0, capture_info.size, MemoryPermission::ReadWrite, MemoryPermission::ReadWrite,
35  "ErrEula Memory");
36 
37  // Send the response message with the newly created SharedMemory
38  Service::APT::MessageParameter result{};
39  result.signal = Service::APT::SignalType::Response;
40  result.buffer.clear();
41  result.destination_id = Service::APT::AppletId::Application;
42  result.sender_id = id;
43  result.object = framebuffer_memory;
44 
45  SendParameter(result);
46  return RESULT_SUCCESS;
47 }

57 void ErrEula::Update() {
58  if (auto cb{Core::System::GetInstance().GetQtCallbacks().erreula}) {
59  cb(config);
60  Finalize();
61  return;
62  }
63 
64  switch (config.error_type) {
65  case ErrEulaErrorType::ErrorCode: {
66  LOG_INFO(Applet_ErrEula, "Error Code: {:#010X}", config.error_code);
67  break;
68  }
69  case ErrEulaErrorType::LocalizedErrorText:
70  case ErrEulaErrorType::ErrorText: {
71  LOG_INFO(Applet_ErrEula, "Error Code: {:#010X}", config.error_code);
72  LOG_INFO(Applet_ErrEula, "Error Text: {}", Common::UTF16ToUTF8(config.error_text));
73  break;
74  }
75  case ErrEulaErrorType::Agree:
76  case ErrEulaErrorType::Eula:
77  case ErrEulaErrorType::EulaDrawOnly:
78  case ErrEulaErrorType::EulaFirstBoot: {
79  LOG_INFO(Applet_ErrEula, "EULA accepted");
80  Service::CFG::GetCurrentModule()->AcceptEULA();
81  break;
82  }
83  }
84 
85  config.return_code = ErrEulaResult::Success;
86  Finalize();
87 }

96  message.sender_id = id;
