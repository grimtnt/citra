135 void ScheduleEvent(s64 cycles_into_future, const EventType* event_type, u64 userdata) {
136  ASSERT(event_type != nullptr);
137  s64 timeout{static_cast<s64>(GetTicks()) + cycles_into_future};
138 
139  // If this event needs to be scheduled before the next advance(), force one early
140  if (!is_global_timer_sane)

145 }

151 void UnscheduleEvent(const EventType* event_type, u64 userdata) {
152  auto itr{std::remove_if(event_queue.begin(), event_queue.end(), [&](const Event& e) {
153  return e.type == event_type && e.userdata == userdata;
154  })};
155 
156  // Removing random items breaks the invariant so we have to re-establish it.
157  if (itr != event_queue.end()) {
158  event_queue.erase(itr, event_queue.end());
159  std::make_heap(event_queue.begin(), event_queue.end(), std::greater<>());
160  }
161 }

179 void ForceExceptionCheck(s64 cycles) {
180  cycles = std::max<s64>(0, cycles);
181  if (downcount > cycles) {
182  slice_length -= downcount - cycles;

187 void MoveEvents() {
188  for (Event ev; ts_queue.Pop(ev);) {
189  ev.fifo_order = event_fifo_id++;
190  event_queue.emplace_back(std::move(ev));
191  std::push_heap(event_queue.begin(), event_queue.end(), std::greater<>());
192  }
193 }

195 void Advance() {
196  MoveEvents();
197 
198  s64 cycles_executed{slice_length - downcount};
199  global_timer += cycles_executed;
200  slice_length = MAX_SLICE_LENGTH;
201 
202  is_global_timer_sane = true;
203 
204  while (!event_queue.empty() && event_queue.front().time <= global_timer) {
205  Event evt = std::move(event_queue.front());
206  std::pop_heap(event_queue.begin(), event_queue.end(), std::greater<>());
207  event_queue.pop_back();
208  evt.type->callback(evt.userdata, global_timer - evt.time);
209  }
210 
211  is_global_timer_sane = false;
212 
213  // Still events left (scheduled in the future)
214  if (!event_queue.empty()) {
215  slice_length = static_cast<int>(
216  std::min<s64>(event_queue.front().time - global_timer, MAX_SLICE_LENGTH));
217  }
218 
219  downcount = slice_length;
220 }
