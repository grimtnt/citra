84  std::tuple<float, float> GetAnalog(int axis_x, int axis_y) const {
85  if (!Core::System::GetInstance().IsShellOpen())
86  return std::make_tuple(0.0f, 0.0f);
87  float x{GetAxis(axis_x)};
88  float y{GetAxis(axis_y)};
89  y = -y; // 3DS uses an y-axis inverse from SDL
90 
91  // Make sure the coordinates are in the unit circle,
92  // otherwise normalize it.
93  float r{x * x + y * y};
94  if (r > 1.0f) {
95  r = std::sqrt(r);
96  x /= r;
97  y /= r;
98  }
99 
100  return std::make_tuple(x, y);
101  }

154 static std::shared_ptr<SDLJoystick> GetSDLJoystickByGUID(const std::string& guid, int port) {
155  std::lock_guard<std::mutex> lock{joystick_map_mutex};
156  const auto it{joystick_map.find(guid)};
157  if (it != joystick_map.end()) {
158  while (it->second.size() <= port) {
159  auto joystick{std::make_shared<SDLJoystick>(guid, it->second.size(), nullptr,
160  [](SDL_Joystick*) {})};
161  it->second.emplace_back(std::move(joystick));
162  }
163  return it->second[port];
164  }
165  auto joystick{std::make_shared<SDLJoystick>(guid, 0, nullptr, [](SDL_Joystick*) {})};
166  return joystick_map[guid].emplace_back(std::move(joystick));
167 }

173 static std::shared_ptr<SDLJoystick> GetSDLJoystickBySDLID(SDL_JoystickID sdl_id) {
174  std::lock_guard<std::mutex> lock{joystick_map_mutex};
175  auto sdl_joystick{SDL_JoystickFromInstanceID(sdl_id)};
176  const std::string guid{GetGUID(sdl_joystick)};
177  auto map_it{joystick_map.find(guid)};
178  if (map_it != joystick_map.end()) {
179  auto vec_it{std::find_if(map_it->second.begin(), map_it->second.end(),
180  [&sdl_joystick](const std::shared_ptr<SDLJoystick>& joystick) {
181  return sdl_joystick == joystick->GetSDLJoystick();
182  })};
183  if (vec_it != map_it->second.end()) {
184  // This is the common case: There is already an existing SDL_Joystick mapped to a
185  // SDLJoystick. return the SDLJoystick
186  return *vec_it;
187  }
188  // Search for a SDLJoystick without a mapped SDL_Joystick...
189  auto nullptr_it{std::find_if(map_it->second.begin(), map_it->second.end(),
190  [](const std::shared_ptr<SDLJoystick>& joystick) {
191  return !joystick->GetSDLJoystick();
192  })};
193  if (nullptr_it != map_it->second.end()) {
194  // ... and map it
195  (*nullptr_it)->SetSDLJoystick(sdl_joystick);
196  return *nullptr_it;
197  }
198  // There is no SDLJoystick without a mapped SDL_Joystick
199  // Create a new SDLJoystick
200  auto joystick{std::make_shared<SDLJoystick>(guid, map_it->second.size(), sdl_joystick)};
201  return map_it->second.emplace_back(std::move(joystick));
202  }
203  auto joystick{std::make_shared<SDLJoystick>(guid, 0, sdl_joystick)};
204  return joystick_map[guid].emplace_back(std::move(joystick));
205 }

207 void InitJoystick(int joystick_index) {
208  std::lock_guard<std::mutex> lock{joystick_map_mutex};
209  SDL_Joystick* sdl_joystick{SDL_JoystickOpen(joystick_index)};
210  if (!sdl_joystick) {
211  LOG_ERROR(Input, "failed to open joystick {}", joystick_index);
212  return;
213  }
214  std::string guid{GetGUID(sdl_joystick)};
215  if (joystick_map.find(guid) == joystick_map.end()) {
216  auto joystick{std::make_shared<SDLJoystick>(guid, 0, sdl_joystick)};
217  joystick_map[guid].emplace_back(std::move(joystick));
218  return;
219  }
220  auto& joystick_guid_list{joystick_map[guid]};
221  const auto it{std::find_if(
222  joystick_guid_list.begin(), joystick_guid_list.end(),
223  [](const std::shared_ptr<SDLJoystick>& joystick) { return !joystick->GetSDLJoystick(); })};
224  if (it != joystick_guid_list.end()) {
225  (*it)->SetSDLJoystick(sdl_joystick);
226  return;
227  }
228  auto joystick{std::make_shared<SDLJoystick>(guid, joystick_guid_list.size(), sdl_joystick)};
229  joystick_guid_list.emplace_back(std::move(joystick));
230 }

245 void HandleGameControllerEvent(const SDL_Event& event) {
246  switch (event.type) {
247  case SDL_JOYBUTTONUP: {
248  auto joystick{GetSDLJoystickBySDLID(event.jbutton.which)};
249  if (joystick) {
250  joystick->SetButton(event.jbutton.button, false);
251  }
252  break;
253  }
254  case SDL_JOYBUTTONDOWN: {
255  auto joystick{GetSDLJoystickBySDLID(event.jbutton.which)};
256  if (joystick) {
257  joystick->SetButton(event.jbutton.button, true);
258  }
259  break;
260  }
261  case SDL_JOYHATMOTION: {
262  auto joystick{GetSDLJoystickBySDLID(event.jhat.which)};
263  if (joystick) {
264  joystick->SetHat(event.jhat.hat, event.jhat.value);
265  }
266  break;
267  }
268  case SDL_JOYAXISMOTION: {
269  auto joystick{GetSDLJoystickBySDLID(event.jaxis.which)};
270  if (joystick) {
271  joystick->SetAxis(event.jaxis.axis, event.jaxis.value);
272  }
273  break;
274  }
275  case SDL_JOYDEVICEREMOVED:
276  LOG_DEBUG(Input, "Controller removed with Instance_ID {}", event.jdevice.which);
277  CloseJoystick(SDL_JoystickFromInstanceID(event.jdevice.which));
278  break;
279  case SDL_JOYDEVICEADDED:
280  LOG_DEBUG(Input, "Controller connected with device index {}", event.jdevice.which);
281  InitJoystick(event.jdevice.which);
282  break;
283  }
284 }

291 void PollLoop() {
292  if (SDL_Init(SDL_INIT_JOYSTICK) < 0) {
293  LOG_CRITICAL(Input, "SDL_Init(SDL_INIT_JOYSTICK) failed with: {}", SDL_GetError());
294  return;
295  }
296  if (SDL_SetHint(SDL_HINT_JOYSTICK_ALLOW_BACKGROUND_EVENTS, "1") == SDL_FALSE) {
297  LOG_ERROR(Input, "Failed to set Hint for background events", SDL_GetError());
298  }
299 
300  SDL_Event event;
301  while (initialized) {
302  // Wait for 10 ms or until an event happens
303  if (SDL_WaitEventTimeout(&event, 10)) {
304  // Don't handle the event if we are configuring
305  if (polling) {
306  event_queue.Push(event);
307  } else {
308  HandleGameControllerEvent(event);
309  }
310  }
311  }
312  CloseSDLJoysticks();
313  SDL_QuitSubSystem(SDL_INIT_JOYSTICK);
314 }

366 class SDLAnalog final : public Input::AnalogDevice {
367 public:
368  SDLAnalog(std::shared_ptr<SDLJoystick> joystick_, int axis_x_, int axis_y_, float deadzone_)
369  : joystick{std::move(joystick_)}, axis_x{axis_x_}, axis_y{axis_y_}, deadzone{deadzone_} {}
370 
371  std::tuple<float, float> GetStatus() const override {
372  float x;
373  float y;
374  std::tie(x, y) = joystick->GetAnalog(axis_x, axis_y);
375  const float r{std::sqrt((x * x) + (y * y))};
376  if (r > deadzone) {
377  return std::make_tuple(x / r * (r - deadzone) / (1 - deadzone),
378  y / r * (r - deadzone) / (1 - deadzone));
379  }
380  return std::make_tuple<float, float>(0.0f, 0.0f);
381  }
382 
383 private:
384  std::shared_ptr<SDLJoystick> joystick;
385  const int axis_x;
386  const int axis_y;
387  const float deadzone;
388 };

409  std::unique_ptr<Input::ButtonDevice> Create(const Common::ParamPackage& params) override {
410  const std::string guid{params.Get("guid", "0")};
411  const int port{params.Get("port", 0)};
412 
413  auto joystick{GetSDLJoystickByGUID(guid, port)};
414 
415  if (params.Has("hat")) {
416  const int hat{params.Get("hat", 0)};
417  const std::string direction_name{params.Get("direction", "")};
418  Uint8 direction;
419  if (direction_name == "up") {
420  direction = SDL_HAT_UP;
421  } else if (direction_name == "down") {
422  direction = SDL_HAT_DOWN;
423  } else if (direction_name == "left") {
424  direction = SDL_HAT_LEFT;
425  } else if (direction_name == "right") {
426  direction = SDL_HAT_RIGHT;
427  } else {
428  direction = 0;
429  }
430  // This is necessary so accessing GetHat with hat won't crash
431  joystick->SetHat(hat, SDL_HAT_CENTERED);
432  return std::make_unique<SDLDirectionButton>(joystick, hat, direction);
433  }
434 
435  if (params.Has("axis")) {
436  const int axis{params.Get("axis", 0)};
437  const float threshold{params.Get("threshold", 0.5f)};
438  const std::string direction_name{params.Get("direction", "")};
439  bool trigger_if_greater{};
440  if (direction_name == "+") {
441  trigger_if_greater = true;
442  } else if (direction_name == "-") {
443  trigger_if_greater = false;
444  } else {
445  trigger_if_greater = true;
446  LOG_ERROR(Input, "Unknown direction {}", direction_name);
447  }
448  // This is necessary so accessing GetAxis with axis won't crash
449  joystick->SetAxis(axis, 0);
450  return std::make_unique<SDLAxisButton>(joystick, axis, threshold, trigger_if_greater);
451  }
452 
453  const int button{params.Get("button", 0)};
454  // This is necessary so accessing GetButton with button won't crash
455  joystick->SetButton(button, false);
456  return std::make_unique<SDLButton>(joystick, button);
457  }

594 class SDLAnalogPoller final : public SDLPoller {
595 public:
596  void Start() override {
597  SDLPoller::Start();
598 
599  // Reset stored axes
600  analog_xaxis = -1;
601  analog_yaxis = -1;
602  analog_axes_joystick = -1;
603  }
604 
605  Common::ParamPackage GetNextInput() override {
606  SDL_Event event;
607  while (event_queue.Pop(event)) {
608  if (event.type != SDL_JOYAXISMOTION || std::abs(event.jaxis.value / 32767.0) < 0.5) {
609  continue;
610  }
611  // An analog device needs two axes, so we need to store the axis for later and wait for
612  // a second SDL event. The axes also must be from the same joystick.
613  int axis{event.jaxis.axis};
614  if (analog_xaxis == -1) {
615  analog_xaxis = axis;
616  analog_axes_joystick = event.jaxis.which;
617  } else if (analog_yaxis == -1 && analog_xaxis != axis &&
618  analog_axes_joystick == event.jaxis.which) {
619  analog_yaxis = axis;
620  }
621  }
622  Common::ParamPackage params{};
623  if (analog_xaxis != -1 && analog_yaxis != -1) {
624  auto joystick = GetSDLJoystickBySDLID(event.jaxis.which);
625  params.Set("engine", "sdl");
626  params.Set("port", joystick->GetPort());
627  params.Set("guid", joystick->GetGUID());
628  params.Set("axis_x", analog_xaxis);
629  params.Set("axis_y", analog_yaxis);
630  analog_xaxis = -1;
631  analog_yaxis = -1;
632  analog_axes_joystick = -1;
633  return params;
634  }
635  return params;
636  }

639  int analog_xaxis{-1};
640  int analog_yaxis{-1};
