174 std::vector<u16> Rgb2Yuv(const QImage& source, int width, int height) {
175  auto buffer{std::vector<u16>(width * height)};
176  auto dest{buffer.begin()};
177  bool write{false};
178  int py{}, pu{}, pv{};
179  for (int y{}; y < height; ++y) {
180  for (int x{}; x < width; ++x) {
181  QRgb rgb{source.pixel(x, y)};
182  int r{qRed(rgb)};
183  int g{qGreen(rgb)};
184  int b{qBlue(rgb)};
185 
186  // The following transformation is a reverse of the one in Y2R using ITU_Rec601
187  int y{YuvTable::Y(r, g, b)};
188  int u{YuvTable::U(r, g, b)};
189  int v{YuvTable::V(r, g, b)};
190 
191  if (write) {
192  pu = (pu + u) / 2;
193  pv = (pv + v) / 2;
194  *(dest++) =
195  static_cast<u16>(std::clamp(py, 0, 0xFF) | (std::clamp(pu, 0, 0xFF) << 8));
196  *(dest++) =
197  static_cast<u16>(std::clamp(y, 0, 0xFF) | (std::clamp(pv, 0, 0xFF) << 8));
198  } else {
199  py = y;
200  pu = u;
201  pv = v;
202  }
203  write = !write;
204  }
205  }
206  return buffer;
207 }

209 std::vector<u16> ProcessImage(const QImage& image, int width, int height, bool output_rgb = false,
210  bool flip_horizontal = false, bool flip_vertical = false) {
211  std::vector<u16> buffer(width * height);
212  if (image.isNull()) {
213  return buffer;
214  }
215  QImage scaled{
216  image.scaled(width, height, Qt::KeepAspectRatioByExpanding, Qt::SmoothTransformation)};
217  QImage transformed{
218  scaled.copy((scaled.width() - width) / 2, (scaled.height() - height) / 2, width, height)
219  .mirrored(flip_horizontal, flip_vertical)};
220  if (output_rgb) {
221  QImage converted{transformed.convertToFormat(QImage::Format_RGB16)};
222  std::memcpy(buffer.data(), converted.bits(), width * height * sizeof(u16));
223  } else {
224  return CameraUtil::Rgb2Yuv(transformed, width, height);
225  }
226  return buffer;
227 }
