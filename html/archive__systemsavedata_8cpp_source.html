22 std::string GetSystemSaveDataPath(const std::string& mount_point, const Path& path) {
23  const std::vector<u8> vec_data{path.AsBinary()};
24  u32 save_low{};
25  u32 save_high{};
26  std::memcpy(&save_low, &vec_data[4], sizeof(u32));
27  std::memcpy(&save_high, &vec_data[0], sizeof(u32));
28  return fmt::format("{}{:08X}/{:08X}/", mount_point, save_low, save_high);

35 Path ConstructSystemSaveDataBinaryPath(u32 high, u32 low) {
36  std::vector<u8> binary_path{};
37  binary_path.reserve(8);
38 
39  // Append each word byte by byte
40 
41  // First is the high word
42  for (unsigned i{}; i < 4; ++i)
43  binary_path.push_back((high >> (8 * i)) & 0xFF);
44 
45  // Next is the low word
46  for (unsigned i{}; i < 4; ++i)
47  binary_path.push_back((low >> (8 * i)) & 0xFF);
48 
49  return {binary_path};
50 }
