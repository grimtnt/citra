52  void StartReceive() {
53  socket.async_receive_from(
54  boost::asio::buffer(receive_buffer), receive_endpoint,
55  [this](const boost::system::error_code& error, std::size_t bytes_transferred) {
56  HandleReceive(error, bytes_transferred);

60 private:
61  void HandleReceive(const boost::system::error_code& error, std::size_t bytes_transferred) {
62  if (auto type{Response::Validate(receive_buffer.data(), bytes_transferred)}) {
63  switch (*type) {
64  case Type::Version: {
65  Response::Version version{};
66  std::memcpy(&version, &receive_buffer[sizeof(Header)], sizeof(Response::Version));
67  callback.version(std::move(version));
68  break;
69  }
70  case Type::PortInfo: {
71  Response::PortInfo port_info{};
72  std::memcpy(&port_info, &receive_buffer[sizeof(Header)],
73  sizeof(Response::PortInfo));
74  callback.port_info(std::move(port_info));
75  break;
76  }
77  case Type::PadData: {
78  Response::PadData pad_data{};
79  std::memcpy(&pad_data, &receive_buffer[sizeof(Header)], sizeof(Response::PadData));
80  callback.pad_data(std::move(pad_data));
81  break;
82  }
83  }
84  }
85  StartReceive();
86  }

88  void HandleSend(const boost::system::error_code& error) {
89  // Send a request for getting port info for the pad
90  Request::PortInfo port_info{1, {pad_index, 0, 0, 0}};
91  auto port_message{Request::Create(port_info, client_id)};
92  std::memcpy(&send_buffer1, &port_message, PORT_INFO_SIZE);
93  std::size_t len{socket.send_to(boost::asio::buffer(send_buffer1), send_endpoint)};
94 
95  // Send a request for getting pad data for the pad
96  Request::PadData pad_data{Request::PadData::Flags::Id, pad_index, EMPTY_MAC_ADDRESS};
97  auto pad_message{Request::Create(pad_data, client_id)};
98  std::memcpy(send_buffer2.data(), &pad_message, PAD_DATA_SIZE);
99  std::size_t len2{socket.send_to(boost::asio::buffer(send_buffer2), send_endpoint)};
100  StartSend(timer.expiry());
101  }

152 void Client::OnPadData(Response::PadData data) {
153  LOG_TRACE(Input, "PadData packet received");
154  if (data.packet_counter <= packet_sequence) {
155  LOG_WARNING(
156  Input,
157  "PadData packet dropped because its stale info. Current count: {} Packet count: {}",
158  packet_sequence, data.packet_counter);
159  return;
160  }
161  packet_sequence = data.packet_counter;
162  // Due to differences between the 3ds and cemuhookudp motion directions, we need to invert
163  // accel.x and accel.z and also invert pitch and yaw. See
164  // https://github.com/citra-emu/citra/pull/4049 for more details on gyro/accel
165  Math::Vec3f accel{Math::MakeVec<float>(-data.accel.x, data.accel.y, -data.accel.z)};
166  Math::Vec3f gyro{Math::MakeVec<float>(-data.gyro.pitch, -data.gyro.yaw, data.gyro.roll)};
167  {
168  std::lock_guard<std::mutex> guard(status->update_mutex);
169 
170  status->motion_status = {accel, gyro};
171 
172  // TODO: add a setting for "click" touch. Click touch refers to a device that differentiates
173  // between a simple "tap" and a hard press that causes the touch screen to click.
174  bool is_active{data.touch_1.is_active != 0};
175 
176  float x{};
177  float y{};
178 
179  if (is_active && status->touch_calibration) {
180  u16 min_x{status->touch_calibration->min_x};
181  u16 max_x{status->touch_calibration->max_x};
182  u16 min_y{status->touch_calibration->min_y};
183  u16 max_y{status->touch_calibration->max_y};
184 
185  x = (std::clamp(static_cast<u16>(data.touch_1.x), min_x, max_x) - min_x) /
186  static_cast<float>(max_x - min_x);
187  y = (std::clamp(static_cast<u16>(data.touch_1.y), min_y, max_y) - min_y) /
188  static_cast<float>(max_y - min_y);
189  }
190 
191  status->touch_status = {x, y, is_active};
192  }
193 }

195 void Client::StartCommunication(const std::string& host, u16 port, u8 pad_index, u32 client_id) {
196  SocketCallback callback{[this](Response::Version version) { OnVersion(version); },
197  [this](Response::PortInfo info) { OnPortInfo(info); },
198  [this](Response::PadData data) { OnPadData(data); }};
199  LOG_INFO(Input, "Starting communication with UDP input server on {}:{}", host, port);
200  socket = std::make_unique<Socket>(host, port, pad_index, client_id, callback);
201  thread = std::thread{SocketLoop, this->socket.get()};
202 }

224 CalibrationConfigurationJob::CalibrationConfigurationJob(
225  const std::string& host, u16 port, u8 pad_index, u32 client_id,
226  std::function<void(Status)> status_callback,
227  std::function<void(u16, u16, u16, u16)> data_callback) {
228 
229  std::thread([=] {
230  constexpr u16 CALIBRATION_THRESHOLD{100};
231 
232  u16 min_x{UINT16_MAX}, min_y{UINT16_MAX};
233  u16 max_x{}, max_y{};
234 
235  Status current_status{Status::Initialized};
236  SocketCallback callback{[](Response::Version version) {}, [](Response::PortInfo info) {},
237  [&](Response::PadData data) {
238  if (current_status == Status::Initialized) {
239  // Receiving data means the communication is ready now
240  current_status = Status::Ready;
241  status_callback(current_status);
242  }
243  if (!data.touch_1.is_active)
244  return;
245  LOG_DEBUG(Input, "Current touch: {} {}", data.touch_1.x,
246  data.touch_1.y);
247  min_x = std::min(min_x, static_cast<u16>(data.touch_1.x));
248  min_y = std::min(min_y, static_cast<u16>(data.touch_1.y));
249  if (current_status == Status::Ready) {
250  // First touch - min data (min_x/min_y)
251  current_status = Status::Stage1Completed;
252  status_callback(current_status);
253  }
254  if (data.touch_1.x - min_x > CALIBRATION_THRESHOLD &&
255  data.touch_1.y - min_y > CALIBRATION_THRESHOLD) {
256  // Set the current position as max value and finishes
257  // configuration
258  max_x = data.touch_1.x;
259  max_y = data.touch_1.y;
260  current_status = Status::Completed;
261  data_callback(min_x, min_y, max_x, max_y);
262  status_callback(current_status);
263 
264  complete_event.Set();
265  }
266  }};
267  Socket socket{host, port, pad_index, client_id, callback};
268  std::thread worker_thread{SocketLoop, &socket};
269  complete_event.Wait();
270  socket.Stop();
271  worker_thread.join();
272  })
273  .detach();
274 }
