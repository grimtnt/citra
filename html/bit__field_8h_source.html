38 /*
39  * Abstract bitfield class
40  *
41  * Allows endianness-independent access to individual bitfields within some raw
42  * integer value. The assembly generated by this class is identical to the
43  * usage of raw bitfields, so it's a perfectly fine replacement.
44  *
45  * For BitField<X,Y,Z>, X is the distance of the bitfield to the LSB of the
46  * raw value, Y is the length in bits of the bitfield. Z is an integer type
47  * which determines the sign of the bitfield. Z must have the same size as the
48  * raw integer.
49  *
50  *
51  * General usage:
52  *
53  * Create a new union with the raw integer value as a member.
54  * Then for each bitfield you want to expose, add a BitField member
55  * in the union. The template parameters are the bit offset and the number
56  * of desired bits.
57  *
58  * Changes in the bitfield members will then get reflected in the raw integer
59  * value and vice-versa.
60  *
61  *
62  * Sample usage:
63  *
64  * union SomeRegister
65  * {
66  * u32 hex;
67  *
68  * BitField<0,7,u32> first_seven_bits; // unsigned
69  * BitField<7,8,u32> next_eight_bits; // unsigned
70  * BitField<3,15,s32> some_signed_fields; // signed
71  * };
72  *
73  * This is equivalent to the little-endian specific code:
74  *
75  * union SomeRegister
76  * {
77  * u32 hex;
78  *
79  * struct
80  * {
81  * u32 first_seven_bits : 7;
82  * u32 next_eight_bits : 8;
83  * };
84  * struct
85  * {
86  * u32 : 3; // padding
87  * s32 some_signed_fields : 15;
88  * };
89  * };
90  *
91  *
92  * Caveats:
93  *
94  * 1)
95  * BitField provides automatic casting from and to the storage type where
96  * appropriate. However, when using non-typesafe functions like printf, an
97  * explicit cast must be performed on the BitField object to make sure it gets
98  * passed correctly, e.g.:
99  * printf("Value: %d", (s32)some_register.some_signed_fields);
100  *
101  * 2)
102  * Not really a caveat, but potentially irritating: This class is used in some
103  * packed structures that do not guarantee proper alignment. Therefore we have
104  * to use #pragma pack here not to pack the members of the class, but instead
105  * to break GCC's assumption that the members of the class are aligned on
106  * sizeof(StorageType).
107  * TODO(neobrain): Confirm that this is a proper fix and not just masking
108  * symptoms.
109  */
110 #pragma pack(1)
111 template <std::size_t Position, std::size_t Bits, typename T>
112 struct BitField {
113 private:
114  // We hide the copy assigment operator here, because the default copy
115  // assignment would copy the full storage value, rather than just the bits
116  // relevant to this particular bit field.
117  // We don't delete it because we want BitField to be trivially copyable.
118  constexpr BitField& operator=(const BitField&) = default;
119 
120  // StorageType is T for non-enum types and the underlying type of T if
121  // T is an enumeration. Note that T is wrapped within an enable_if in the
122  // former case to workaround compile errors which arise when using
123  // std::underlying_type<T>::type directly.
124  using StorageType = typename std::conditional_t<std::is_enum<T>::value, std::underlying_type<T>,
125  std::enable_if<true, T>>::type;
126 
127  // Unsigned version of StorageType
128  using StorageTypeU = std::make_unsigned_t<StorageType>;
129 
130 public:
132  static constexpr std::size_t position{Position};
133  static constexpr std::size_t bits{Bits};
134  static constexpr StorageType mask = (((StorageTypeU)~0) >> (8 * sizeof(T) - bits)) << position;
135 
141  static constexpr FORCE_INLINE StorageType FormatValue(const T& value) {
142  return ((StorageType)value << position) & mask;
143  }
144 
150  static constexpr FORCE_INLINE T ExtractValue(const StorageType& storage) {
151  if (std::numeric_limits<T>::is_signed) {
152  std::size_t shift = 8 * sizeof(T) - bits;
153  return (T)((storage << (shift - position)) >> shift);
154  } else {
155  return (T)((storage & mask) >> position);
156  }
157  }
158 
159  // This constructor and assignment operator might be considered ambiguous:
160  // Would they initialize the storage or just the bitfield?
161  // Hence, delete them. Use the Assign method to set bitfield values!
162  BitField(T val) = delete;
163  BitField& operator=(T val) = delete;
164 
165  // Force default constructor to be created
166  // so that we can use this within unions
167  constexpr BitField() = default;
168 
169  constexpr FORCE_INLINE operator T() const {
170  return Value();
171  }
172 
173  constexpr FORCE_INLINE void Assign(const T& value) {
174  storage = (storage & ~mask) | FormatValue(value);
175  }
176 
177  constexpr T Value() const {
178  return ExtractValue(storage);
179  }
180 
181  // TODO: we may want to change this to explicit operator bool() if it's bug-free in VS2015
182  constexpr FORCE_INLINE bool ToBool() const {
183  return Value() != 0;
184  }
185 
186 private:
187  StorageType storage;
188 
189  static_assert(bits + position <= 8 * sizeof(T), "Bitfield out of range");
190 
191  // And, you know, just in case people specify something stupid like bits=position=0x80000000
192  static_assert(position < 8 * sizeof(T), "Invalid position");
193  static_assert(bits <= 8 * sizeof(T), "Invalid number of bits");
194  static_assert(bits > 0, "Invalid number of bits");
195  static_assert(std::is_trivially_copyable_v<T>, "T must be trivially copyable in a BitField");
196 };
