40 static void MapPages(PageTable& page_table, u32 base, u32 size, u8* memory, PageType type) {
41  LOG_DEBUG(HW_Memory, "Mapping {} onto {:08X}-{:08X}", (void*)memory, base * PAGE_SIZE,
42  (base + size) * PAGE_SIZE);
43 
44  RasterizerFlushVirtualRegion(base << PAGE_BITS, size * PAGE_SIZE,
45  FlushMode::FlushAndInvalidate);
46 
47  u32 end{base + size};
48  while (base != end) {
49  ASSERT_MSG(base < PAGE_TABLE_NUM_ENTRIES, "out of range mapping at {:08X}", base);
50 
51  page_table.attributes[base] = type;
52  page_table.pointers[base] = memory;
53 
54  base += 1;
55  if (memory != nullptr)
56  memory += PAGE_SIZE;
57  }
58 }

84 static u8* GetPointerFromVMA(const Kernel::Process& process, VAddr vaddr) {
85  u8* direct_pointer{};
86 
87  auto& vm_manager{process.vm_manager};
88 
89  auto it{vm_manager.FindVMA(vaddr)};
90  ASSERT(it != vm_manager.vma_map.end());
91 
92  auto& vma = it->second;
93  switch (vma.type) {
94  case Kernel::VMAType::AllocatedMemoryBlock:
95  direct_pointer = vma.backing_block->data() + vma.offset;
96  break;
97  case Kernel::VMAType::BackingMemory:
98  direct_pointer = vma.backing_memory;
99  break;
100  case Kernel::VMAType::Free:
101  return nullptr;
102  default:
103  UNREACHABLE();
104  }
105 
106  return direct_pointer + (vaddr - vma.base);
107 }

138 template <typename T>
139 T Read(const VAddr vaddr) {
140  const u8* page_pointer{current_page_table->pointers[vaddr >> PAGE_BITS]};
141  if (page_pointer) {
142  // NOTE: Avoid adding any extra logic to this fast-path block
143  T value;
144  std::memcpy(&value, &page_pointer[vaddr & PAGE_MASK], sizeof(T));
145  return value;
146  }
147 
148  // The memory access might do an MMIO or cached access, so we have to lock the HLE kernel state
149  std::lock_guard<std::recursive_mutex> lock{HLE::g_hle_lock};
150 
151  PageType type{current_page_table->attributes[vaddr >> PAGE_BITS]};
152  switch (type) {
153  case PageType::Unmapped:
154  LOG_ERROR(HW_Memory, "unmapped Read{} @ 0x{:08X}", sizeof(T) * 8, vaddr);
155  return 0;
156  case PageType::Memory:
157  ASSERT_MSG(false, "Mapped memory page without a pointer @ {:08X}", vaddr);
158  break;
159  case PageType::RasterizerCachedMemory: {
160  RasterizerFlushVirtualRegion(vaddr, sizeof(T), FlushMode::Flush);
161 
162  T value;
163  std::memcpy(&value, GetPointerFromVMA(vaddr), sizeof(T));
164  return value;
165  }
166  case PageType::Special:
167  return ReadMMIO<T>(GetMMIOHandler(vaddr), vaddr);
168  default:
169  UNREACHABLE();
170  }
171 }

176 template <typename T>
177 void Write(const VAddr vaddr, const T data) {
178  u8* page_pointer{current_page_table->pointers[vaddr >> PAGE_BITS]};
179  if (page_pointer) {
180  // NOTE: Avoid adding any extra logic to this fast-path block
181  std::memcpy(&page_pointer[vaddr & PAGE_MASK], &data, sizeof(T));
182  return;
183  }
184 
185  // The memory access might do an MMIO or cached access, so we have to lock the HLE kernel state
186  std::lock_guard<std::recursive_mutex> lock{HLE::g_hle_lock};
187 
188  PageType type = current_page_table->attributes[vaddr >> PAGE_BITS];
189  switch (type) {
190  case PageType::Unmapped:
191  LOG_ERROR(HW_Memory, "unmapped Write{} 0x{:08X} @ 0x{:08X}", sizeof(data) * 8, (u32)data,
192  vaddr);
193  return;
194  case PageType::Memory:
195  ASSERT_MSG(false, "Mapped memory page without a pointer @ {:08X}", vaddr);
196  break;
197  case PageType::RasterizerCachedMemory: {
198  RasterizerFlushVirtualRegion(vaddr, sizeof(T), FlushMode::Invalidate);
199  std::memcpy(GetPointerFromVMA(vaddr), &data, sizeof(T));
200  break;
201  }
202  case PageType::Special:
203  WriteMMIO<T>(GetMMIOHandler(vaddr), vaddr, data);
204  break;
205  default:
206  UNREACHABLE();
207  }

267 u8* GetPhysicalPointer(PAddr address) {
268  struct MemoryArea {
269  PAddr paddr_base;
270  u32 size;
271  };
272 
273  static constexpr MemoryArea memory_areas[] = {
274  {VRAM_PADDR, VRAM_SIZE},
275  {IO_AREA_PADDR, IO_AREA_SIZE},
276  {DSP_RAM_PADDR, DSP_RAM_SIZE},
277  {FCRAM_PADDR, FCRAM_N3DS_SIZE},
278  {N3DS_EXTRA_RAM_PADDR, N3DS_EXTRA_RAM_SIZE},
279  };
280 
281  const auto area{
282  std::find_if(std::begin(memory_areas), std::end(memory_areas), [&](const auto& area) {
283  return address >= area.paddr_base && address < area.paddr_base + area.size;
284  })};
285 
286  if (area == std::end(memory_areas)) {
287  LOG_ERROR(HW_Memory, "unknown GetPhysicalPointer @ 0x{:08X}", address);
288  return nullptr;
289  }
290 
291  if (area->paddr_base == IO_AREA_PADDR) {
292  LOG_ERROR(HW_Memory, "MMIO mappings are not supported yet. phys_addr=0x{:08X}", address);
293  return nullptr;
294  }
295 
296  u32 offset_into_region{address - area->paddr_base};
297 
298  u8* target_pointer{};
299  switch (area->paddr_base) {
300  case VRAM_PADDR:
301  target_pointer = vram.data() + offset_into_region;
302  break;
303  case DSP_RAM_PADDR:
304  target_pointer = Core::DSP().GetDspMemory().data() + offset_into_region;
305  break;
306  case FCRAM_PADDR:
307  for (const auto& region : Kernel::memory_regions) {
308  if (offset_into_region >= region.base &&
309  offset_into_region < region.base + region.size) {
310  target_pointer =
311  region.linear_heap_memory->data() + offset_into_region - region.base;
312  break;
313  }
314  }
315  break;
316  case N3DS_EXTRA_RAM_PADDR:
317  target_pointer = n3ds_extra_ram.data() + offset_into_region;
318  break;
319  default:
320  UNREACHABLE();
321  }
322 
323  return target_pointer;
324 }

326 void RasterizerMarkRegionCached(PAddr start, u32 size, bool cached) {
327  if (start == 0) {
328  return;
329  }
330 
331  u32 num_pages{((start + size - 1) >> PAGE_BITS) - (start >> PAGE_BITS) + 1};
332  PAddr paddr{start};
333 
334  for (unsigned i{}; i < num_pages; ++i, paddr += PAGE_SIZE) {
335  boost::optional<VAddr> maybe_vaddr = PhysicalToVirtualAddress(paddr);
336  // While the physical <-> virtual mapping is 1:1 for the regions supported by the cache,
337  // some games (like Pokemon Super Mystery Dungeon) will try to use textures that go beyond
338  // the end address of VRAM, causing the Virtual->Physical translation to fail when flushing
339  // parts of the texture.
340  if (!maybe_vaddr) {
341  LOG_ERROR(HW_Memory,
342  "Trying to flush a cached region to an invalid physical address {:08X}",
343  paddr);
344  continue;
345  }
346  VAddr vaddr{*maybe_vaddr};
347 
348  PageType& page_type{current_page_table->attributes[vaddr >> PAGE_BITS]};
349 
350  if (cached) {
351  // Switch page type to cached if now cached
352  switch (page_type) {
353  case PageType::Unmapped:
354  // It is not necessary for a process to have this region mapped into its address
355  // space, for example, a system module need not have a VRAM mapping.
356  break;
357  case PageType::Memory:
358  page_type = PageType::RasterizerCachedMemory;
359  current_page_table->pointers[vaddr >> PAGE_BITS] = nullptr;
360  break;
361  default:
362  UNREACHABLE();
363  }
364  } else {
365  // Switch page type to uncached if now uncached
366  switch (page_type) {
367  case PageType::Unmapped:
368  // It is not necessary for a process to have this region mapped into its address
369  // space, for example, a system module need not have a VRAM mapping.
370  break;
371  case PageType::RasterizerCachedMemory: {
372  u8* pointer = GetPointerFromVMA(vaddr & ~PAGE_MASK);
373  if (pointer == nullptr) {
374  // It's possible that this function has been called while updating the pagetable
375  // after unmapping a VMA. In that case the underlying VMA will no longer exist,
376  // and we should just leave the pagetable entry blank.
377  page_type = PageType::Unmapped;
378  } else {
379  page_type = PageType::Memory;
380  current_page_table->pointers[vaddr >> PAGE_BITS] = pointer;
381  }
382  break;
383  }
384  default:
385  break;
386  }
387  }
388  }
389 }

417 void RasterizerFlushVirtualRegion(VAddr start, u32 size, FlushMode mode) {
418  // Since pages are unmapped on shutdown after video core is shutdown, the renderer may be
419  // null here
420  if (VideoCore::g_renderer == nullptr) {
421  return;
422  }
423 
424  VAddr end{start + size};
425 
426  auto CheckRegion{[&](VAddr region_start, VAddr region_end) {
427  if (start >= region_end || end <= region_start) {
428  // No overlap with region
429  return;
430  }
431 
432  VAddr overlap_start{std::max(start, region_start)};
433  VAddr overlap_end{std::min(end, region_end)};
434 
435  PAddr physical_start{TryVirtualToPhysicalAddress(overlap_start).value()};
436  u32 overlap_size{overlap_end - overlap_start};
437 
438  auto* rasterizer{VideoCore::g_renderer->GetRasterizer()};
439  switch (mode) {
440  case FlushMode::Flush:
441  rasterizer->FlushRegion(physical_start, overlap_size);
442  break;
443  case FlushMode::Invalidate:
444  rasterizer->InvalidateRegion(physical_start, overlap_size);
445  break;
446  case FlushMode::FlushAndInvalidate:
447  rasterizer->FlushAndInvalidateRegion(physical_start, overlap_size);
448  break;
449  }
450  }};
451 
452  CheckRegion(LINEAR_HEAP_VADDR, LINEAR_HEAP_VADDR_END);
453  CheckRegion(NEW_LINEAR_HEAP_VADDR, NEW_LINEAR_HEAP_VADDR_END);
454  CheckRegion(VRAM_VADDR, VRAM_VADDR_END);
455 }

473 void ReadBlock(const Kernel::Process& process, const VAddr src_addr, void* dest_buffer,
474  const std::size_t size) {
475  auto& page_table{process.vm_manager.page_table};
476 
477  std::size_t remaining_size{size};
478  std::size_t page_index{src_addr >> PAGE_BITS};
479  std::size_t page_offset{src_addr & PAGE_MASK};
480 
481  while (remaining_size > 0) {
482  const std::size_t copy_amount{std::min(PAGE_SIZE - page_offset, remaining_size)};
483  const VAddr current_vaddr{static_cast<VAddr>((page_index << PAGE_BITS) + page_offset)};
484 
485  switch (page_table.attributes[page_index]) {
486  case PageType::Unmapped: {
487  LOG_ERROR(HW_Memory,
488  "unmapped ReadBlock @ 0x{:08X} (start address = 0x{:08X}, size = {})",
489  current_vaddr, src_addr, size);
490  std::memset(dest_buffer, 0, copy_amount);
491  break;
492  }
493  case PageType::Memory: {
494  DEBUG_ASSERT(page_table.pointers[page_index]);
495 
496  const u8* src_ptr{page_table.pointers[page_index] + page_offset};
497  std::memcpy(dest_buffer, src_ptr, copy_amount);
498  break;
499  }
500  case PageType::Special: {
501  MMIORegionPointer handler{GetMMIOHandler(page_table, current_vaddr)};
502  DEBUG_ASSERT(handler);
503  handler->ReadBlock(current_vaddr, dest_buffer, copy_amount);
504  break;
505  }
506  case PageType::RasterizerCachedMemory: {
507  RasterizerFlushVirtualRegion(current_vaddr, static_cast<u32>(copy_amount),
508  FlushMode::Flush);
509  std::memcpy(dest_buffer, GetPointerFromVMA(process, current_vaddr), copy_amount);
510  break;
511  }
512  default:
513  UNREACHABLE();
514  }
515 
516  page_index++;
517  page_offset = 0;
518  dest_buffer = static_cast<u8*>(dest_buffer) + copy_amount;
519  remaining_size -= copy_amount;
520  }
521 }

595 void ZeroBlock(const Kernel::Process& process, const VAddr dest_addr, const std::size_t size) {
596  auto& page_table{process.vm_manager.page_table};
597  std::size_t remaining_size{size};
598  std::size_t page_index{dest_addr >> PAGE_BITS};
599  std::size_t page_offset{dest_addr & PAGE_MASK};
600 
601  static const std::array<u8, PAGE_SIZE> zeros = {};
602 
603  while (remaining_size > 0) {
604  const std::size_t copy_amount{std::min(PAGE_SIZE - page_offset, remaining_size)};
605  const VAddr current_vaddr{static_cast<VAddr>((page_index << PAGE_BITS) + page_offset)};
606 
607  switch (page_table.attributes[page_index]) {
608  case PageType::Unmapped: {
609  LOG_ERROR(HW_Memory,
610  "unmapped ZeroBlock @ 0x{:08X} (start address = 0x{:08X}, size = {})",
611  current_vaddr, dest_addr, size);
612  break;
613  }
614  case PageType::Memory: {
615  DEBUG_ASSERT(page_table.pointers[page_index]);
616 
617  u8* dest_ptr{page_table.pointers[page_index] + page_offset};
618  std::memset(dest_ptr, 0, copy_amount);
619  break;
620  }
621  case PageType::Special: {
622  MMIORegionPointer handler{GetMMIOHandler(page_table, current_vaddr)};
623  DEBUG_ASSERT(handler);
624  handler->WriteBlock(current_vaddr, zeros.data(), copy_amount);
625  break;
626  }
627  case PageType::RasterizerCachedMemory: {
628  RasterizerFlushVirtualRegion(current_vaddr, static_cast<u32>(copy_amount),
629  FlushMode::Invalidate);
630  std::memset(GetPointerFromVMA(process, current_vaddr), 0, copy_amount);
631  break;

647 void CopyBlock(const Kernel::Process& process, VAddr dest_addr, VAddr src_addr,
648  const std::size_t size) {
649  auto& page_table{process.vm_manager.page_table};
650  std::size_t remaining_size{size};
651  std::size_t page_index{src_addr >> PAGE_BITS};
652  std::size_t page_offset{src_addr & PAGE_MASK};
653 
654  while (remaining_size > 0) {
655  const std::size_t copy_amount{std::min(PAGE_SIZE - page_offset, remaining_size)};
656  const VAddr current_vaddr{static_cast<VAddr>((page_index << PAGE_BITS) + page_offset)};
657 
658  switch (page_table.attributes[page_index]) {
659  case PageType::Unmapped: {
660  LOG_ERROR(HW_Memory,
661  "unmapped CopyBlock @ 0x{:08X} (start address = 0x{:08X}, size = {})",
662  current_vaddr, src_addr, size);
663  ZeroBlock(process, dest_addr, copy_amount);
664  break;
665  }
666  case PageType::Memory: {
667  DEBUG_ASSERT(page_table.pointers[page_index]);
668  const u8* src_ptr{page_table.pointers[page_index] + page_offset};
669  WriteBlock(process, dest_addr, src_ptr, copy_amount);
670  break;
671  }
672  case PageType::Special: {
673  MMIORegionPointer handler{GetMMIOHandler(page_table, current_vaddr)};
674  DEBUG_ASSERT(handler);
675  std::vector<u8> buffer(copy_amount);
676  handler->ReadBlock(current_vaddr, buffer.data(), buffer.size());
677  WriteBlock(process, dest_addr, buffer.data(), buffer.size());
678  break;
679  }
680  case PageType::RasterizerCachedMemory: {
681  RasterizerFlushVirtualRegion(current_vaddr, static_cast<u32>(copy_amount),
682  FlushMode::Flush);
683  WriteBlock(process, dest_addr, GetPointerFromVMA(process, current_vaddr), copy_amount);
684  break;
685  }
686  default:
687  UNREACHABLE();
688  }
689 
690  page_index++;
691  page_offset = 0;
692  dest_addr += static_cast<VAddr>(copy_amount);
693  src_addr += static_cast<VAddr>(copy_amount);
694  remaining_size -= copy_amount;
695  }
696 }

742 boost::optional<PAddr> TryVirtualToPhysicalAddress(const VAddr addr) {
743  if (addr == 0) {
744  return 0;
745  } else if (addr >= VRAM_VADDR && addr < VRAM_VADDR_END) {
746  return addr - VRAM_VADDR + VRAM_PADDR;
747  } else if (addr >= LINEAR_HEAP_VADDR && addr < LINEAR_HEAP_VADDR_END) {
748  return addr - LINEAR_HEAP_VADDR + FCRAM_PADDR;
749  } else if (addr >= NEW_LINEAR_HEAP_VADDR && addr < NEW_LINEAR_HEAP_VADDR_END) {
750  return addr - NEW_LINEAR_HEAP_VADDR + FCRAM_PADDR;
751  } else if (addr >= DSP_RAM_VADDR && addr < DSP_RAM_VADDR_END) {
752  return addr - DSP_RAM_VADDR + DSP_RAM_PADDR;
753  } else if (addr >= IO_AREA_VADDR && addr < IO_AREA_VADDR_END) {
754  return addr - IO_AREA_VADDR + IO_AREA_PADDR;
755  } else if (addr >= N3DS_EXTRA_RAM_VADDR && addr < N3DS_EXTRA_RAM_VADDR_END) {
756  return addr - N3DS_EXTRA_RAM_VADDR + N3DS_EXTRA_RAM_PADDR;
757  }
758 
759  return boost::none;
760 }
