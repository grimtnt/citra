40 bool ServerSession::ShouldWait(Thread* thread) const {
41  // Closed sessions should never wait, an error will be returned from svcReplyAndReceive.
42  if (parent->client == nullptr)
43  return false;
44  // Wait if we have no pending requests, or if we're currently handling a request.
45  return pending_requesting_threads.empty() || currently_handling != nullptr;
46 }

48 void ServerSession::Acquire(Thread* thread) {
49  ASSERT_MSG(!ShouldWait(thread), "object unavailable!");
50 
51  // If the client endpoint was closed, don't do anything. This ServerSession is now useless and
52  // will linger until its last handle is closed by the running application.
53  if (parent->client == nullptr)
54  return;
55 
56  // We are now handling a request, pop it from the stack.
57  ASSERT(!pending_requesting_threads.empty());
58  currently_handling = pending_requesting_threads.back();
59  pending_requesting_threads.pop_back();
60 }

62 ResultCode ServerSession::HandleSyncRequest(SharedPtr<Thread> thread) {
63  // The ServerSession received a sync request, this means that there's new data available
64  // from its ClientSession, so wake up any threads that may be waiting on a svcReplyAndReceive or
65  // similar.
66 
67  // If this ServerSession has an associated HLE handler, forward the request to it.
68  if (hle_handler != nullptr) {
69  hle_handler->HandleSyncRequest(SharedPtr<ServerSession>(this));
70  }
71 
72  if (thread->status == THREADSTATUS_RUNNING) {
73  // Put the thread to sleep until the server replies, it will be awoken in
74  // svcReplyAndReceive for LLE servers.
75  thread->status = THREADSTATUS_WAIT_IPC;
76 
77  if (hle_handler != nullptr) {
78  // For HLE services, we put the request threads to sleep for a short duration to
79  // simulate IPC overhead, but only if the HLE handler didn't put the thread to sleep for
80  // other reasons like an async callback. The IPC overhead is needed to prevent
81  // starvation when a thread only does sync requests to HLE services while a
82  // lower-priority thread is waiting to run.
83 
84  // This delay was approximated in a homebrew application by measuring the average time
85  // it takes for svcSendSyncRequest to return when performing the SetLcdForceBlack IPC
86  // request to the GSP:GPU service in a n3DS with firmware 11.6. The measured values have
87  // a high variance and vary between models.
88  thread->WakeAfterDelay(39000);
89  } else {
90  // Add the thread to the list of threads that have issued a sync request with this
91  // server.
92  pending_requesting_threads.push_back(std::move(thread));
93  }
94  }
95 
96  // If this ServerSession does not have an HLE implementation, just wake up the threads waiting
97  // on it.
98  WakeupAllWaitingThreads();
99  return RESULT_SUCCESS;
100 }
