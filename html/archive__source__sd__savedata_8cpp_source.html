43 ResultVal<std::unique_ptr<ArchiveBackend>> ArchiveSource_SDSaveData::Open(u64 program_id) {
44  std::string concrete_mount_point{GetSaveDataPath(mount_point, program_id)};
45  if (!FileUtil::Exists(concrete_mount_point)) {
46  // When a SaveData archive is created for the first time, it is not yet formatted and the
47  // save file/directory structure expected by the game has not yet been initialized.
48  // Returning the NotFormatted error code will signal the game to provision the SaveData
49  // archive with the files and folders that it expects.
50  return ERR_NOT_FORMATTED;
51  }
52 
53  auto archive{std::make_unique<SaveDataArchive>(std::move(concrete_mount_point))};
54  return MakeResult<std::unique_ptr<ArchiveBackend>>(std::move(archive));
55 }

57 ResultCode ArchiveSource_SDSaveData::Format(u64 program_id,
58  const FileSys::ArchiveFormatInfo& format_info) {
59  std::string concrete_mount_point{GetSaveDataPath(mount_point, program_id)};
60  FileUtil::DeleteDirRecursively(concrete_mount_point);
61  FileUtil::CreateFullPath(concrete_mount_point);
62 
63  // Write the format metadata
64  std::string metadata_path{GetSaveDataMetadataPath(mount_point, program_id)};
65  FileUtil::IOFile file{metadata_path, "wb"};
66 
67  if (file.IsOpen()) {
68  file.WriteBytes(&format_info, sizeof(format_info));
69  return RESULT_SUCCESS;
70  }
71  return RESULT_SUCCESS;
72 }
