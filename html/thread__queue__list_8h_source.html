40  T get_first() {
41  Queue* cur = first;
42  while (cur != nullptr) {
43  if (!cur->data.empty()) {
44  return cur->data.front();

52  T pop_first() {
53  Queue* cur = first;
54  while (cur != nullptr) {
55  if (!cur->data.empty()) {
56  auto tmp = std::move(cur->data.front());
57  cur->data.pop_front();
58  return tmp;
59  }
60  cur = cur->next_nonempty;
61  }

128  struct Queue {

140  void link(Priority priority) {
141  Queue* cur = &queues[priority];
142 
143  for (int i = priority - 1; i >= 0; --i) {
144  if (queues[i].next_nonempty != UnlinkedTag()) {
145  cur->next_nonempty = queues[i].next_nonempty;
146  queues[i].next_nonempty = cur;
147  return;
148  }
149  }
150 
151  cur->next_nonempty = first;
152  first = cur;
153  }
