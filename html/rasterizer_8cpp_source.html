237 Rasterizer::VertexArrayInfo Rasterizer::AnalyzeVertexArray(bool is_indexed) {
238  const auto& regs{Pica::g_state.regs};
239  const auto& vertex_attributes{regs.pipeline.vertex_attributes};
240 
241  u32 vertex_min{};
242  u32 vertex_max{};
243  if (is_indexed) {
244  const auto& index_info{regs.pipeline.index_array};
245  PAddr address{vertex_attributes.GetPhysicalBaseAddress() + index_info.offset};
246  const u8* index_address_8{Memory::GetPhysicalPointer(address)};
247  const u16* index_address_16{reinterpret_cast<const u16*>(index_address_8)};
248  bool index_u16{index_info.format != 0};
249 
250  vertex_min = 0xFFFF;
251  vertex_max = 0;
252  std::size_t size{regs.pipeline.num_vertices * (index_u16 ? 2 : 1)};
253  res_cache.FlushRegion(address, size, nullptr);
254  for (u32 index{}; index < regs.pipeline.num_vertices; ++index) {
255  u32 vertex{
256  static_cast<u32>(index_u16 ? index_address_16[index] : index_address_8[index])};
257  vertex_min = std::min(vertex_min, vertex);
258  vertex_max = std::max(vertex_max, vertex);
259  }
260  } else {
261  vertex_min = regs.pipeline.vertex_offset;
262  vertex_max = regs.pipeline.vertex_offset + regs.pipeline.num_vertices - 1;
263  }
264 
265  u32 vertex_num{vertex_max - vertex_min + 1};
266  u32 vs_input_size{};
267  for (auto& loader : vertex_attributes.attribute_loaders) {
268  if (loader.component_count != 0) {
269  vs_input_size += loader.byte_count * vertex_num;
270  }
271  }
272 
273  return {vertex_min, vertex_max, vs_input_size};
274 }

276 void Rasterizer::SetupVertexArray(u8* array_ptr, GLintptr buffer_offset, GLuint vs_input_index_min,
277  GLuint vs_input_index_max) {
278  const auto& regs{Pica::g_state.regs};
279  const auto& vertex_attributes{regs.pipeline.vertex_attributes};
280  PAddr base_address{vertex_attributes.GetPhysicalBaseAddress()};
281 
282  state.draw.vertex_array = hw_vao.handle;
283  state.draw.vertex_buffer = vertex_buffer.GetHandle();
284  state.Apply();
285 
286  std::array<bool, 16> enable_attributes{};
287 
288  for (const auto& loader : vertex_attributes.attribute_loaders) {
289  if (loader.component_count == 0 || loader.byte_count == 0) {
290  continue;
291  }
292 
293  u32 offset{};
294  for (u32 comp{}; comp < loader.component_count && comp < 12; ++comp) {
295  u32 attribute_index = loader.GetComponent(comp);
296  if (attribute_index < 12) {
297  if (vertex_attributes.GetNumElements(attribute_index) != 0) {
298  offset = Common::AlignUp(
299  offset, vertex_attributes.GetElementSizeInBytes(attribute_index));
300 
301  u32 input_reg{
302  static_cast<u32>(regs.vs.GetRegisterForAttribute(attribute_index))};
303  GLint size{
304  static_cast<GLint>(vertex_attributes.GetNumElements(attribute_index))};
305  GLenum type{vs_attrib_types[static_cast<u32>(
306  vertex_attributes.GetFormat(attribute_index))]};
307  GLsizei stride{static_cast<GLsizei>(loader.byte_count)};
308  glVertexAttribPointer(input_reg, size, type, GL_FALSE, stride,
309  reinterpret_cast<GLvoid*>(buffer_offset + offset));
310  enable_attributes[input_reg] = true;
311 
312  offset += vertex_attributes.GetStride(attribute_index);
313  }
314  } else {
315  // Attribute ids 12, 13, 14 and 15 signify 4, 8, 12 and 16-byte paddings,
316  // respectively
317  offset = Common::AlignUp(offset, 4);
318  offset += (attribute_index - 11) * 4;
319  }
320  }
321 
322  PAddr data_addr{base_address + loader.data_offset +
323  (vs_input_index_min * loader.byte_count)};
324 
325  u32 vertex_num{vs_input_index_max - vs_input_index_min + 1};
326  u32 data_size{loader.byte_count * vertex_num};
327 
328  res_cache.FlushRegion(data_addr, data_size, nullptr);
329  std::memcpy(array_ptr, Memory::GetPhysicalPointer(data_addr), data_size);
330 
331  array_ptr += data_size;
332  buffer_offset += data_size;
333  }
334 
335  for (std::size_t i{}; i < enable_attributes.size(); ++i) {
336  if (enable_attributes[i] != hw_vao_enabled_attributes[i]) {
337  if (enable_attributes[i]) {
338  glEnableVertexAttribArray(i);
339  } else {
340  glDisableVertexAttribArray(i);
341  }
342  hw_vao_enabled_attributes[i] = enable_attributes[i];
343  }
344 
345  if (vertex_attributes.IsDefaultAttribute(i)) {
346  u32 reg{regs.vs.GetRegisterForAttribute(i)};
347  if (!enable_attributes[reg]) {
348  const auto& attr{Pica::g_state.input_default_attributes.attr[i]};
349  glVertexAttrib4f(reg, attr.x.ToFloat32(), attr.y.ToFloat32(), attr.z.ToFloat32(),
350  attr.w.ToFloat32());
351  }
352  }
353  }
354 }

393 static GLenum GetCurrentPrimitiveMode(bool use_gs) {
394  const auto& regs{Pica::g_state.regs};
395  if (use_gs) {
396  switch ((regs.gs.max_input_attribute_index + 1) /
397  (regs.pipeline.vs_outmap_total_minus_1_a + 1)) {
398  case 1:

400  case 2:
401  return GL_LINES;

405  return GL_TRIANGLES;

481 bool Rasterizer::Draw(bool accelerate, bool is_indexed) {
482  const auto& regs{Pica::g_state.regs};
483 
484  bool shadow_rendering{regs.framebuffer.output_merger.fragment_operation_mode ==
485  Pica::FramebufferRegs::FragmentOperationMode::Shadow};
486 
487  const bool has_stencil{regs.framebuffer.framebuffer.depth_format ==
488  Pica::FramebufferRegs::DepthFormat::D24S8};
489 
490  const bool write_color_fb{shadow_rendering || state.color_mask.red_enabled == GL_TRUE ||
491  state.color_mask.green_enabled == GL_TRUE ||
492  state.color_mask.blue_enabled == GL_TRUE ||
493  state.color_mask.alpha_enabled == GL_TRUE};
494 
495  const bool write_depth_fb{
496  (state.depth.test_enabled && state.depth.write_mask == GL_TRUE) ||
497  (has_stencil && state.stencil.test_enabled && state.stencil.write_mask != 0)};
498 
499  const bool using_color_fb{regs.framebuffer.framebuffer.GetColorBufferPhysicalAddress() != 0 &&
500  write_color_fb};
501  const bool using_depth_fb{
502  !shadow_rendering && regs.framebuffer.framebuffer.GetDepthBufferPhysicalAddress() != 0 &&
503  (write_depth_fb || regs.framebuffer.output_merger.depth_test_enable != 0 ||
504  (has_stencil && state.stencil.test_enabled))};
505 
506  MathUtil::Rectangle<s32> viewport_rect_unscaled{
507  // These registers hold half-width and half-height, so must be multiplied by 2
508  regs.rasterizer.viewport_corner.x, // left
509  regs.rasterizer.viewport_corner.y + // top
510  static_cast<s32>(Pica::float24::FromRaw(regs.rasterizer.viewport_size_y).ToFloat32() *
511  2),
512  regs.rasterizer.viewport_corner.x + // right
513  static_cast<s32>(Pica::float24::FromRaw(regs.rasterizer.viewport_size_x).ToFloat32() *
514  2),
515  regs.rasterizer.viewport_corner.y // bottom
516  };
517 
518  Surface color_surface;
519  Surface depth_surface;
520  MathUtil::Rectangle<u32> surfaces_rect;
521  std::tie(color_surface, depth_surface, surfaces_rect) =
522  res_cache.GetFramebufferSurfaces(using_color_fb, using_depth_fb, viewport_rect_unscaled);
523 
524  const u16 res_scale{static_cast<u16>(
525  color_surface != nullptr ? color_surface->res_scale
526  : (depth_surface == nullptr ? 1u : depth_surface->res_scale))};
527 
528  MathUtil::Rectangle<u32> draw_rect{
529  static_cast<u32>(std::clamp<s32>(static_cast<s32>(surfaces_rect.left) +
530  viewport_rect_unscaled.left * res_scale,
531  surfaces_rect.left, surfaces_rect.right)), // Left
532  static_cast<u32>(std::clamp<s32>(static_cast<s32>(surfaces_rect.bottom) +
533  viewport_rect_unscaled.top * res_scale,
534  surfaces_rect.bottom, surfaces_rect.top)), // Top
535  static_cast<u32>(std::clamp<s32>(static_cast<s32>(surfaces_rect.left) +
536  viewport_rect_unscaled.right * res_scale,
537  surfaces_rect.left, surfaces_rect.right)), // Right
538  static_cast<u32>(std::clamp<s32>(static_cast<s32>(surfaces_rect.bottom) +
539  viewport_rect_unscaled.bottom * res_scale,
540  surfaces_rect.bottom, surfaces_rect.top))}; // Bottom
541 
542  // Bind the framebuffer surfaces
543  state.draw.draw_framebuffer = framebuffer.handle;
544  state.Apply();
545 
546  if (shadow_rendering) {
547  if (!allow_shadow || color_surface == nullptr) {
548  return true;
549  }
550  glFramebufferParameteri(GL_DRAW_FRAMEBUFFER, GL_FRAMEBUFFER_DEFAULT_WIDTH,
551  color_surface->width * color_surface->res_scale);
552  glFramebufferParameteri(GL_DRAW_FRAMEBUFFER, GL_FRAMEBUFFER_DEFAULT_HEIGHT,
553  color_surface->height * color_surface->res_scale);
554  glFramebufferTexture2D(GL_DRAW_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, 0, 0);
555  glFramebufferTexture2D(GL_DRAW_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, GL_TEXTURE_2D, 0,
556  0);
557  state.image_shadow_buffer = color_surface->texture.handle;
558  } else {
559  glFramebufferTexture2D(GL_DRAW_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D,
560  color_surface != nullptr ? color_surface->texture.handle : 0, 0);

623  }

659  CheckBarrier(state.image_shadow_texture_px = surface->texture.handle);
660  } else {
661  state.image_shadow_texture_px = 0;
662  }
663

665  regs.texturing.GetCubePhysicalAddress(CubeFace::NegativeX);
666  surface = res_cache.GetTextureSurface(info);
667  if (surface != nullptr) {
668  CheckBarrier(state.image_shadow_texture_nx = surface->texture.handle);
669  } else {
670  state.image_shadow_texture_nx = 0;
671  }
672

674  regs.texturing.GetCubePhysicalAddress(CubeFace::PositiveY);
675  surface = res_cache.GetTextureSurface(info);
676  if (surface != nullptr) {
677  CheckBarrier(state.image_shadow_texture_py = surface->texture.handle);
678  } else {
679  state.image_shadow_texture_py = 0;
680  }
681

683  regs.texturing.GetCubePhysicalAddress(CubeFace::NegativeY);
684  surface = res_cache.GetTextureSurface(info);
685  if (surface != nullptr) {
686  CheckBarrier(state.image_shadow_texture_ny = surface->texture.handle);
687  } else {
688  state.image_shadow_texture_ny = 0;
689  }
690

692  regs.texturing.GetCubePhysicalAddress(CubeFace::PositiveZ);
693  surface = res_cache.GetTextureSurface(info);
694  if (surface != nullptr) {
695  CheckBarrier(state.image_shadow_texture_pz = surface->texture.handle);
696  } else {

758 
759  // Viewport can have negative offsets or larger
760  // dimensions than our framebuffer sub-rect.
761  // Enable scissor test to prevent drawing
762  // outside of the framebuffer region
763  state.scissor.enabled = true;
764  state.scissor.x = draw_rect.left;
765  state.scissor.y = draw_rect.bottom;
766  state.scissor.width = draw_rect.GetWidth();
767  state.scissor.height = draw_rect.GetHeight();
768  state.Apply();

771  bool succeeded{true};
772  if (accelerate) {
773  succeeded = AccelerateDrawBatchInternal(is_indexed, use_gs);
774  } else {
775  state.draw.vertex_array = sw_vao.handle;
776  state.draw.vertex_buffer = vertex_buffer.GetHandle();
777  shader_program_manager->UseTrivialVertexShader();
778  shader_program_manager->UseTrivialGeometryShader();

783  for (std::size_t base_vertex{}; base_vertex < vertex_batch.size();

790  vertex_buffer.Map(vertex_size, sizeof(HardwareVertex));

794  }
795  }

798

800  for (unsigned texture_index{}; texture_index < pica_textures.size(); ++texture_index) {
801  state.texture_units[texture_index].texture_2d = 0;
802  }
803  state.texture_cube_unit.texture_cube = 0;
804  if (allow_shadow) {
805  state.image_shadow_texture_px = 0;
806  state.image_shadow_texture_nx = 0;
807  state.image_shadow_texture_py = 0;
808  state.image_shadow_texture_ny = 0;
809  state.image_shadow_texture_pz = 0;
810  state.image_shadow_texture_nz = 0;
811  state.image_shadow_buffer = 0;
812  }
813  state.Apply();
814 
815  if (shadow_rendering) {
816  glMemoryBarrier(GL_TEXTURE_FETCH_BARRIER_BIT | GL_SHADER_IMAGE_ACCESS_BARRIER_BIT |
817  GL_TEXTURE_UPDATE_BARRIER_BIT | GL_FRAMEBUFFER_BARRIER_BIT);
818  }
819 
820  if (need_texture_barrier && GLAD_GL_ARB_texture_barrier) {
821  glTextureBarrier();
822  }
823 
824  // Mark framebuffer surfaces as dirty
825  MathUtil::Rectangle<u32> draw_rect_unscaled{
826  draw_rect.left / res_scale, draw_rect.top / res_scale, draw_rect.right / res_scale,
827  draw_rect.bottom / res_scale};
828 
829  if (color_surface != nullptr && write_color_fb) {
830  auto interval{color_surface->GetSubRectInterval(draw_rect_unscaled)};
831  res_cache.InvalidateRegion(boost::icl::first(interval), boost::icl::length(interval),
832  color_surface);
833  }
834  if (depth_surface != nullptr && write_depth_fb) {
835  auto interval{depth_surface->GetSubRectInterval(draw_rect_unscaled)};
836  res_cache.InvalidateRegion(boost::icl::first(interval), boost::icl::length(interval),
837  depth_surface);
838  }
839 
840  return succeeded;
841 }

1452  src_surface->is_tiled ? output_gap / 8 : output_gap);
