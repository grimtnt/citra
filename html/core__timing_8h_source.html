26 // The timing we get from the assembly is 268,111,855.956 Hz
27 // It is possible that this number isn't just an integer because the compiler could have
28 // optimized the multiplication by a multiply-by-constant division.
29 // Rounding to the nearest integer should be fine
30 constexpr u64 BASE_CLOCK_RATE_ARM11{268111856};

54 inline s64 usToCycles(s64 us) {
55  if (us / 1000000 > MAX_VALUE_TO_MULTIPLY) {
56  LOG_ERROR(Core_Timing, "Integer overflow, use max value");
57  return std::numeric_limits<s64>::max();
58  }
59  if (us > MAX_VALUE_TO_MULTIPLY) {
60  LOG_DEBUG(Core_Timing, "Time very big, do rounding");
61  return BASE_CLOCK_RATE_ARM11 * (us / 1000000);
62  }
63  return (BASE_CLOCK_RATE_ARM11 * us) / 1000000;
64 }
