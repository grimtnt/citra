140 Module::Module() {
141  // Open the SharedExtSaveData archive 0xF000000B and create the gamecoin.dat file if it doesn't
142  // exist
143  FileSys::Path archive_path{ptm_shared_extdata_id};
144  auto archive_result{
145  Service::FS::OpenArchive(Service::FS::ArchiveIdCode::SharedExtSaveData, archive_path)};
146  // If the archive didn't exist, create the files inside
147  if (archive_result.Code() == FileSys::ERR_NOT_FORMATTED) {
148  // Format the archive to create the directories
149  Service::FS::FormatArchive(Service::FS::ArchiveIdCode::SharedExtSaveData,
150  FileSys::ArchiveFormatInfo(), archive_path);
151  // Open it again to get a valid archive now that the folder exists
152  archive_result =
153  Service::FS::OpenArchive(Service::FS::ArchiveIdCode::SharedExtSaveData, archive_path);
154  ASSERT_MSG(archive_result.Succeeded(), "Could not open the PTM SharedExtSaveData archive!");
155 
156  FileSys::Path gamecoin_path{"/gamecoin.dat"};
157  Service::FS::CreateFileInArchive(*archive_result, gamecoin_path, sizeof(GameCoin));
158  FileSys::Mode open_mode = {};
159  open_mode.write_flag.Assign(1);
160  // Open the file and write the default gamecoin information
161  auto gamecoin_result{
162  Service::FS::OpenFileFromArchive(*archive_result, gamecoin_path, open_mode)};
163  if (gamecoin_result.Succeeded()) {
164  auto gamecoin{std::move(gamecoin_result).Unwrap()};
165  gamecoin->backend->Write(0, sizeof(GameCoin), true,
166  reinterpret_cast<const u8*>(&default_game_coin));
167  }
168  }
169 }

171 void SetPlayCoins(u16 play_coins) {
172  FileSys::Path archive_path{ptm_shared_extdata_id};
173  auto archive_result{
174  Service::FS::OpenArchive(Service::FS::ArchiveIdCode::SharedExtSaveData, archive_path)};
175 
176  ASSERT_MSG(archive_result.Succeeded(), "Could not open the PTM SharedExtSaveData archive!");
177 
178  FileSys::Path gamecoin_path("/gamecoin.dat");
179  FileSys::Mode open_mode = {};
180  open_mode.read_flag.Assign(1);
181  open_mode.write_flag.Assign(1);
182  // Open the file and write the gamecoin information
183  auto gamecoin_result{
184  Service::FS::OpenFileFromArchive(*archive_result, gamecoin_path, open_mode)};
185  if (gamecoin_result.Succeeded()) {
186  auto gamecoin{std::move(gamecoin_result).Unwrap()};
187  GameCoin game_coin{};
188  gamecoin->backend->Read(0, sizeof(GameCoin), reinterpret_cast<u8*>(&game_coin));
189  game_coin.total_coins = play_coins;
190  gamecoin->backend->Write(0, sizeof(GameCoin), true,
191  reinterpret_cast<const u8*>(&game_coin));
192  }
193 }
