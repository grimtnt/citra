16 StereoBuffer16 DecodeADPCM(const u8* const data, const std::size_t sample_count,
17  const std::array<s16, 16>& adpcm_coeff, ADPCMState& state) {
18  // GC-ADPCM with scale factor and variable coefficients.
19  // Frames are 8 bytes long containing 14 samples each.
20  // Samples are 4 bits (one nibble) long.
21 
22  constexpr std::size_t FRAME_LEN{8};
23  constexpr std::size_t SAMPLES_PER_FRAME{14};
24  constexpr std::array<int, 16> SIGNED_NIBBLES = {
25  {0, 1, 2, 3, 4, 5, 6, 7, -8, -7, -6, -5, -4, -3, -2, -1}};
26 
27  const std::size_t ret_size{sample_count % 2 == 0 ? sample_count
28  : sample_count + 1}; // Ensure multiple of two.
29  StereoBuffer16 ret{ret_size};
30 
31  int yn1{state.yn1}, yn2{state.yn2};
32 
33  const std::size_t NUM_FRAMES{(sample_count + (SAMPLES_PER_FRAME - 1)) /
34  SAMPLES_PER_FRAME}; // Round up.
35  for (std::size_t framei{}; framei < NUM_FRAMES; framei++) {
36  const int frame_header{data[framei * FRAME_LEN]};
37  const int scale{1 << (frame_header & 0xF)};
38  const int idx{(frame_header >> 4) & 0x7};
39 
40  // Coefficients are fixed point with 11 bits fractional part.
41  const s16 coef1{adpcm_coeff[idx * 2 + 0]};
42  const s16 coef2{adpcm_coeff[idx * 2 + 1]};
43 
44  // Decodes an audio sample. One nibble produces one sample.
45  const auto decode_sample = [&](const int nibble) -> s16 {
46  const int xn = nibble * scale;
47  // We first transform everything into 11 bit fixed point, perform the second order
48  // digital filter, then transform back.
49  // 0x400 == 0.5 in 11 bit fixed point.
50  // Filter: y[n] = x[n] + 0.5 + c1 * y[n-1] + c2 * y[n-2]
51  int val = ((xn << 11) + 0x400 + coef1 * yn1 + coef2 * yn2) >> 11;
52  // Clamp to output range.
53  val = std::clamp(val, -32768, 32767);
54  // Advance output feedback.
55  yn2 = yn1;
56  yn1 = val;
57  return (s16)val;
58  };
59 
60  std::size_t outputi{framei * SAMPLES_PER_FRAME};
61  std::size_t datai{framei * FRAME_LEN + 1};
62  for (std::size_t i{}; i < SAMPLES_PER_FRAME && outputi < sample_count; i += 2) {
63  const s16 sample1{decode_sample(SIGNED_NIBBLES[data[datai] >> 4])};
64  ret[outputi].fill(sample1);
65  outputi++;
66 
67  const s16 sample2{decode_sample(SIGNED_NIBBLES[data[datai] & 0xF])};
68  ret[outputi].fill(sample2);
69  outputi++;
70 
71  datai++;
72  }
73  }
74 
75  state.yn1 = yn1;
76  state.yn2 = yn2;
77 
78  return ret;
79 }

81 StereoBuffer16 DecodePCM8(const unsigned num_channels, const u8* const data,
82  const std::size_t sample_count) {
83  ASSERT(num_channels == 1 || num_channels == 2);
84 
85  const auto decode_sample{
86  [](u8 sample) { return static_cast<s16>(static_cast<u16>(sample) << 8); }};
87 
88  StereoBuffer16 ret{sample_count};
89 
90  if (num_channels == 1) {
91  for (std::size_t i{}; i < sample_count; i++) {
92  ret[i].fill(decode_sample(data[i]));
93  }
94  } else {
95  for (std::size_t i{}; i < sample_count; i++) {
96  ret[i][0] = decode_sample(data[i * 2 + 0]);
97  ret[i][1] = decode_sample(data[i * 2 + 1]);
98  }
99  }
100 
101  return ret;
102 }
