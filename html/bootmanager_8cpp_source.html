18 void EmuThread::run() {
19  render_window->MakeCurrent();
20  stop_run = false;
21 
22  while (!stop_run) {
23  if (running) {
24  Core::System::ResultStatus result{Core::System::GetInstance().RunLoop()};
25  if (result == Core::System::ResultStatus::ShutdownRequested) {
26  // Notify frontend we shutdown
27  emit ErrorThrown(result, "");
28  // End emulation execution
29  break;
30  }
31  if (result != Core::System::ResultStatus::Success) {
32  this->SetRunning(false);
33  emit ErrorThrown(result, Core::System::GetInstance().GetStatusDetails());
34  }
35  } else {
36  std::unique_lock<std::mutex> lock{running_mutex};
37  running_cv.wait(lock, [this] { return IsRunning() || stop_run; });
38  }
39  }
40 
41  // Shutdown the core emulation
42  Core::System::GetInstance().Shutdown();
43  render_window->moveContext();
44 }
