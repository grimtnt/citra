50 bool QtCameraSurface::present(const QVideoFrame& frame) {
51  if (!frame.isValid()) {
52  return false;
53  }
54  QVideoFrame cloneFrame{frame};
55  cloneFrame.map(QAbstractVideoBuffer::ReadOnly);
56  const QImage image{cloneFrame.bits(), cloneFrame.width(), cloneFrame.height(),
57  QVideoFrame::imageFormatFromPixelFormat(cloneFrame.pixelFormat())};
58  QMutexLocker locker{&mutex};
59  current_frame = image.mirrored(true, true);
60  locker.unlock();
61  cloneFrame.unmap();
62  return true;

129 std::shared_ptr<QtMultimediaCameraHandler> QtMultimediaCameraHandler::GetHandler(
130  const std::string& camera_name) {
131  if (loaded.count(camera_name)) {
132  return loaded.at(camera_name);
133  }
134  for (int i{}; i < handlers.size(); i++) {
135  if (!status[i]) {
136  LOG_INFO(Service_CAM, "Successfully got handler {}", i);
137  status[i] = true;
138  loaded.emplace(camera_name, handlers[i]);
139  return handlers[i];
140  }
141  }
142  LOG_CRITICAL(Service_CAM, "All handlers taken up");
143  return nullptr;
144 }

146 void QtMultimediaCameraHandler::ReleaseHandler(
147  const std::shared_ptr<Camera::QtMultimediaCameraHandler>& handler) {
148  for (int i{}; i < handlers.size(); i++) {
149  if (handlers[i] == handler) {
150  LOG_INFO(Service_CAM, "Successfully released handler {}", i);
151  status[i] = false;
152  handlers[i]->started = false;
153  for (auto it = loaded.begin(); it != loaded.end(); it++) {
154  if (it->second == handlers[i]) {
155  loaded.erase(it);
156  break;
157  }
158  }
159  break;
160  }
161  }
162 }

164 void QtMultimediaCameraHandler::CreateCamera(const std::string& camera_name) {
165  QList<QCameraInfo> cameras = QCameraInfo::availableCameras();
166  for (const QCameraInfo& cameraInfo : cameras) {
167  if (cameraInfo.deviceName().toStdString() == camera_name)
168  camera = std::make_unique<QCamera>(cameraInfo);
169  }
170  if (!camera) { // no cameras found, using default camera
171  camera = std::make_unique<QCamera>();
172  }
173  settings.setMinimumFrameRate(30);
174  settings.setMaximumFrameRate(30);
175  camera->setViewfinder(&camera_surface);
176  camera->load();
177 }
