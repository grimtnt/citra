40 static std::vector<u8> GenerateSecureDataHeader(u16 data_size, u8 channel, u16 dest_node_id,
41  u16 src_node_id, u16 sequence_number) {
42  SecureDataHeader header{};
43  header.protocol_size = data_size + sizeof(SecureDataHeader);
44  // Note: This size includes everything except the first 4 bytes of the structure,
45  // reinforcing the hypotheses that the first 4 bytes are actually the header of
46  // another container protocol.
47  header.securedata_size = data_size + sizeof(SecureDataHeader) - 4;
48  // Frames sent by the emulated application are never UDS management frames
49  header.is_management = 0;
50  header.data_channel = channel;
51  header.sequence_number = sequence_number;
52  header.dest_node_id = dest_node_id;
53  header.src_node_id = src_node_id;
54 
55  std::vector<u8> buffer(sizeof(header));
56  memcpy(buffer.data(), &header, sizeof(header));
57 
58  return buffer;

81 /*
82  * Generates the key used for encrypting the 802.11 data frames generated by UDS.
83  * @returns The key used for data frames crypto.
84  */
85 static std::array<u8, CryptoPP::AES::BLOCKSIZE> GenerateDataCCMPKey(
86  const std::vector<u8>& passphrase, const NetworkInfo& network_info) {
87  // Calculate the MD5 hash of the input passphrase.
88  std::array<u8, CryptoPP::Weak::MD5::DIGESTSIZE> passphrase_hash{};
89  CryptoPP::Weak::MD5().CalculateDigest(passphrase_hash.data(), passphrase.data(),
90  passphrase.size());
91 
92  std::array<u8, CryptoPP::AES::BLOCKSIZE> ccmp_key{};
93 
94  // The CCMP key is the result of encrypting the MD5 hash of the passphrase with AES-CTR using
95  // keyslot 0x2D.
96  using CryptoPP::AES;
97  std::array<u8, CryptoPP::Weak::MD5::DIGESTSIZE> counter = GetDataCryptoCTR(network_info);
98  std::array<u8, AES::BLOCKSIZE> key = HW::AES::GetNormalKey(HW::AES::KeySlotID::UDSDataKey);
99  CryptoPP::CTR_Mode<AES>::Encryption aes;
100  aes.SetKeyWithIV(key.data(), AES::BLOCKSIZE, counter.data());
101  aes.ProcessData(ccmp_key.data(), passphrase_hash.data(), passphrase_hash.size());
102 
103  return ccmp_key;
104 }

106 /*
107  * Generates the Additional Authenticated Data (AAD) for an UDS 802.11 encrypted data frame.
108  * @returns a buffer with the bytes of the AAD.
109  */
110 static std::vector<u8> GenerateCCMPAAD(const MacAddress& sender, const MacAddress& receiver,
111  const MacAddress& bssid, u16 frame_control) {
112  // Reference: IEEE 802.11-2007
113 
114  // 8.3.3.3.2 Construct AAD (22-30 bytes)
115  // The AAD is constructed from the MPDU header. The AAD does not include the header Duration
116  // field, because the Duration field value can change due to normal IEEE 802.11 operation (e.g.,
117  // a rate change during retransmission). For similar reasons, several subfields in the Frame
118  // Control field are masked to 0.
119  struct {
120  u16_be FC; // MPDU Frame Control field
121  MacAddress A1;
122  MacAddress A2;
123  MacAddress A3;
124  u16_be SC; // MPDU Sequence Control field
125  } aad_struct{};
126 
127  constexpr u16 AADFrameControlMask = 0x8FC7;
128  aad_struct.FC = frame_control & AADFrameControlMask;
129  aad_struct.SC = 0;
130 
131  bool to_ds = (frame_control & (1 << 0)) != 0;
132  bool from_ds = (frame_control & (1 << 1)) != 0;
133  // In the 802.11 standard, ToDS = 1 and FromDS = 1 is a valid configuration,
134  // however, the 3DS doesn't seem to transmit frames with such combination.
135  ASSERT_MSG(to_ds != from_ds, "Invalid combination");
136 
137  // The meaning of the address fields depends on the ToDS and FromDS fields.
138  if (from_ds) {
139  aad_struct.A1 = receiver;
140  aad_struct.A2 = bssid;
141  aad_struct.A3 = sender;
142  }
143 
144  if (to_ds) {
145  aad_struct.A1 = bssid;
146  aad_struct.A2 = sender;
147  aad_struct.A3 = receiver;
148  }
149 
150  std::vector<u8> aad(sizeof(aad_struct));
151  std::memcpy(aad.data(), &aad_struct, sizeof(aad_struct));
152 
153  return aad;
154 }

156 /*
157  * Decrypts the payload of an encrypted 802.11 data frame using the specified key.
158  * @returns The decrypted payload.
159  */
160 static std::vector<u8> DecryptDataFrame(const std::vector<u8>& encrypted_payload,
161  const std::array<u8, CryptoPP::AES::BLOCKSIZE>& ccmp_key,
162  const MacAddress& sender, const MacAddress& receiver,
163  const MacAddress& bssid, u16 sequence_number,
164  u16 frame_control) {
165 
166  // Reference: IEEE 802.11-2007
167 
168  std::vector<u8> aad = GenerateCCMPAAD(sender, receiver, bssid, frame_control);
169 
170  std::vector<u8> packet_number{0,
171  0,
172  0,
173  0,
174  static_cast<u8>((sequence_number >> 8) & 0xFF),
175  static_cast<u8>(sequence_number & 0xFF)};
176 
177  // 8.3.3.3.3 Construct CCM nonce (13 bytes)
178  std::vector<u8> nonce;
179  nonce.push_back(0); // priority
180  nonce.insert(nonce.end(), sender.begin(), sender.end()); // Address 2
181  nonce.insert(nonce.end(), packet_number.begin(), packet_number.end()); // PN
182 
183  try {
184  CryptoPP::CCM<CryptoPP::AES, 8>::Decryption d;
185  d.SetKeyWithIV(ccmp_key.data(), ccmp_key.size(), nonce.data(), nonce.size());
186  d.SpecifyDataLengths(aad.size(), encrypted_payload.size() - 8, 0);
187 
188  CryptoPP::AuthenticatedDecryptionFilter df(
189  d, nullptr,
190  CryptoPP::AuthenticatedDecryptionFilter::MAC_AT_END |
191  CryptoPP::AuthenticatedDecryptionFilter::THROW_EXCEPTION);
192  // put aad
193  df.ChannelPut(CryptoPP::AAD_CHANNEL, aad.data(), aad.size());
194 
195  // put cipher with mac
196  df.ChannelPut(CryptoPP::DEFAULT_CHANNEL, encrypted_payload.data(),
197  encrypted_payload.size() - 8);
198  df.ChannelPut(CryptoPP::DEFAULT_CHANNEL,
199  encrypted_payload.data() + encrypted_payload.size() - 8, 8);
200 
201  df.ChannelMessageEnd(CryptoPP::AAD_CHANNEL);
202  df.ChannelMessageEnd(CryptoPP::DEFAULT_CHANNEL);
203  df.SetRetrievalChannel(CryptoPP::DEFAULT_CHANNEL);
204 
205  std::size_t size = df.MaxRetrievable();
206 
207  std::vector<u8> pdata(size);
208  df.Get(pdata.data(), size);
209  return pdata;
210  } catch (CryptoPP::Exception&) {
211  LOG_ERROR(Service_NWM, "failed to decrypt");
212  }
213 
214  return {};
215 }
