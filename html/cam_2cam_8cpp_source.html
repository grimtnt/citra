76 void Module::CompletionEventCallBack(u64 port_id, s64) {
77  PortConfig& port = ports[port_id];
78  const CameraConfig& camera = cameras[port.camera_id];
79  const auto buffer = port.capture_result.get();
80 
81  if (port.is_trimming) {
82  u32 trim_width;
83  u32 trim_height;
84  const int original_width = camera.contexts[camera.current_context].resolution.width;
85  const int original_height = camera.contexts[camera.current_context].resolution.height;
86  if (port.x1 <= port.x0 || port.y1 <= port.y0 || port.x1 > original_width ||
87  port.y1 > original_height) {
88  LOG_ERROR(Service_CAM, "Invalid trimming coordinates x0={}, y0={}, x1={}, y1={}",
89  port.x0, port.y0, port.x1, port.y1);
90  trim_width = 0;
91  trim_height = 0;
92  } else {
93  trim_width = port.x1 - port.x0;
94  trim_height = port.y1 - port.y0;
95  }
96 
97  u32 trim_size = (port.x1 - port.x0) * (port.y1 - port.y0) * 2;
98  if (port.dest_size != trim_size) {
99  LOG_ERROR(Service_CAM, "The destination size ({}) doesn't match the source ({})!",
100  port.dest_size, trim_size);
101  }
102 
103  const u32 src_offset = port.y0 * original_width + port.x0;
104  const u16* src_ptr = buffer.data() + src_offset;
105  // Note: src_size_left is int because it can be negative if the buffer size doesn't match.
106  int src_size_left = static_cast<int>((buffer.size() - src_offset) * sizeof(u16));
107  VAddr dest_ptr = port.dest;
108  // Note: dest_size_left and line_bytes are int to match the type of src_size_left.
109  int dest_size_left = static_cast<int>(port.dest_size);
110  const int line_bytes = static_cast<int>(trim_width * sizeof(u16));
111 
112  for (u32 y{}; y < trim_height; ++y) {
113  int copy_length = std::min({line_bytes, dest_size_left, src_size_left});
114  if (copy_length <= 0) {
115  break;
116  }
117  Memory::WriteBlock(*port.dest_process, dest_ptr, src_ptr, copy_length);
118  dest_ptr += copy_length;
119  dest_size_left -= copy_length;
120  src_ptr += original_width;
121  src_size_left -= original_width * sizeof(u16);
122  }
123  } else {
124  std::size_t buffer_size = buffer.size() * sizeof(u16);
125  if (port.dest_size != buffer_size) {
126  LOG_ERROR(Service_CAM, "The destination size ({}) doesn't match the source ({})!",
127  port.dest_size, buffer_size);
128  }
129  Memory::WriteBlock(*port.dest_process, port.dest, buffer.data(),
130  std::min<std::size_t>(port.dest_size, buffer_size));
131  }
132 
133  port.is_receiving = false;
134  port.completion_event->Signal();
135 }

137 void Module::StartReceiving(int port_id) {
138  PortConfig& port = ports[port_id];
139  port.is_receiving = true;
140 
141  // launches a capture task asynchronously
142  CameraConfig& camera = cameras[port.camera_id];
143  port.capture_result = std::async(std::launch::async, [&camera, &port, this] {
144  if (is_camera_reload_pending.exchange(false)) {
145  // reinitialize the camera according to new settings
146  camera.impl->StopCapture();
147  LoadCameraImplementation(camera, port.camera_id);
148  camera.impl->StartCapture();
149  }
150  return camera.impl->ReceiveFrame();
151  });
152 
153  // schedules a completion event according to the frame rate. The event will block on the
154  // capture task if it is not finished within the expected time
155  CoreTiming::ScheduleEvent(
156  msToCycles(LATENCY_BY_FRAME_RATE[static_cast<int>(camera.frame_rate)]),
157  completion_event_callback, port_id);
158 }

407 void Module::Interface::GetMaxLines(Kernel::HLERequestContext& ctx) {
408  IPC::RequestParser rp{ctx, 0x0A, 2, 0};
409  const u16 width{rp.Pop<u16>()};
410  const u16 height{rp.Pop<u16>()};
411 
412  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
413 
414  // Note: the result of the algorithm below are hwtested with width < 640 and with height < 480
415  constexpr u32 MIN_TRANSFER_UNIT{256};
416  constexpr u32 MAX_BUFFER_SIZE{2560};
417  if (width * height * 2 % MIN_TRANSFER_UNIT != 0) {
418  rb.Push(ERROR_OUT_OF_RANGE);
419  rb.Skip(1, false);
420  } else {
421  u32 lines{MAX_BUFFER_SIZE / width};
422  if (lines > height) {
423  lines = height;
424  }
425  ResultCode result{RESULT_SUCCESS};
426  while (height % lines != 0 || (lines * width * 2 % MIN_TRANSFER_UNIT != 0)) {
427  --lines;
428  if (lines == 0) {
429  result = ERROR_OUT_OF_RANGE;
430  break;
431  }
432  }
433  rb.Push(result);
434  rb.Push(lines);
435  }
436 
437  LOG_DEBUG(Service_CAM, "called, width={}, height={}", width, height);
438 }
