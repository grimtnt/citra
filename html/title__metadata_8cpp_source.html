36 Loader::ResultStatus TitleMetadata::Load(const std::vector<u8> file_data, std::size_t offset) {
37  std::size_t total_size{static_cast<std::size_t>(file_data.size() - offset)};
38  if (total_size < sizeof(u32_be))
39  return Loader::ResultStatus::Error;
40 
41  memcpy(&signature_type, &file_data[offset], sizeof(u32_be));
42 
43  // Signature lengths are variable, and the body follows the signature
44  u32 signature_size{GetSignatureSize(signature_type)};
45 
46  // The TMD body start position is rounded to the nearest 0x40 after the signature
47  std::size_t body_start{Common::AlignUp(signature_size + sizeof(u32), 0x40)};
48  std::size_t body_end{body_start + sizeof(Body)};
49 
50  if (total_size < body_end)
51  return Loader::ResultStatus::Error;
52 
53  // Read signature + TMD body, then load the amount of ContentChunks specified
54  tmd_signature.resize(signature_size);
55  memcpy(tmd_signature.data(), &file_data[offset + sizeof(u32_be)], signature_size);
56  memcpy(&tmd_body, &file_data[offset + body_start], sizeof(TitleMetadata::Body));
57 
58  std::size_t expected_size{body_start + sizeof(Body) +
59  static_cast<u16>(tmd_body.content_count) * sizeof(ContentChunk)};
60  if (total_size < expected_size) {
61  LOG_ERROR(Service_FS, "Malformed TMD, expected size 0x{:x}, got 0x{:x}!", expected_size,
62  total_size);
63  return Loader::ResultStatus::ErrorInvalidFormat;
64  }
65 
66  for (u16 i{}; i < tmd_body.content_count; i++) {
67  ContentChunk chunk{};
68 
69  memcpy(&chunk, &file_data[offset + body_end + (i * sizeof(ContentChunk))],
70  sizeof(ContentChunk));
71  tmd_chunks.push_back(chunk);
72  }
73 
74  return Loader::ResultStatus::Success;
75 }

77 Loader::ResultStatus TitleMetadata::Save(const std::string& file_path) {
78  FileUtil::IOFile file{file_path, "wb"};
79  if (!file.IsOpen())
80  return Loader::ResultStatus::Error;
81 
82  if (!file.WriteBytes(&signature_type, sizeof(u32_be)))
83  return Loader::ResultStatus::Error;
84 
85  // Signature lengths are variable, and the body follows the signature
86  u32 signature_size{GetSignatureSize(signature_type)};
87 
88  if (!file.WriteBytes(tmd_signature.data(), signature_size))
89  return Loader::ResultStatus::Error;
90 
91  // The TMD body start position is rounded to the nearest 0x40 after the signature
92  std::size_t body_start{Common::AlignUp(signature_size + sizeof(u32), 0x40)};
93  file.Seek(body_start, SEEK_SET);
94 
95  // Update our TMD body values and hashes
96  tmd_body.content_count = static_cast<u16>(tmd_chunks.size());
97 
98  // TODO(shinyquagsire23): Do TMDs with more than one contentinfo exist?
99  // For now we'll just adjust the first index to hold all content chunks
100  // and ensure that no further content info data exists.
101  tmd_body.contentinfo = {};
102  tmd_body.contentinfo[0].index = 0;
103  tmd_body.contentinfo[0].command_count = static_cast<u16>(tmd_chunks.size());
104 
105  CryptoPP::SHA256 chunk_hash{};
106  for (u16 i{}; i < tmd_body.content_count; i++) {
107  chunk_hash.Update(reinterpret_cast<u8*>(&tmd_chunks[i]), sizeof(ContentChunk));
108  }
109  chunk_hash.Final(tmd_body.contentinfo[0].hash.data());
110 
111  CryptoPP::SHA256 contentinfo_hash{};
112  for (std::size_t i{}; i < tmd_body.contentinfo.size(); i++) {
113  chunk_hash.Update(reinterpret_cast<u8*>(&tmd_body.contentinfo[i]), sizeof(ContentInfo));
114  }
115  chunk_hash.Final(tmd_body.contentinfo_hash.data());
116 
117  // Write our TMD body, then write each of our ContentChunks
118  if (file.WriteBytes(&tmd_body, sizeof(TitleMetadata::Body)) != sizeof(TitleMetadata::Body))
119  return Loader::ResultStatus::Error;
120 
121  for (u16 i{}; i < tmd_body.content_count; i++) {
122  ContentChunk chunk{tmd_chunks[i]};
123  if (file.WriteBytes(&chunk, sizeof(ContentChunk)) != sizeof(ContentChunk))
124  return Loader::ResultStatus::Error;
125  }
126 
127  return Loader::ResultStatus::Success;
128 }

200 void TitleMetadata::Print() const {
201  LOG_DEBUG(Service_FS, "{} chunks", static_cast<u32>(tmd_body.content_count));
202 
203  // Content info describes ranges of content chunks
204  LOG_DEBUG(Service_FS, "Content info:");
205  for (std::size_t i{}; i < tmd_body.contentinfo.size(); i++) {
206  if (tmd_body.contentinfo[i].command_count == 0)
207  break;
208 
209  LOG_DEBUG(Service_FS, " Index {:04X}, Command Count {:04X}",
210  static_cast<u32>(tmd_body.contentinfo[i].index),
211  static_cast<u32>(tmd_body.contentinfo[i].command_count));
212  }
213 
214  // For each content info, print their content chunk range
215  for (std::size_t i{}; i < tmd_body.contentinfo.size(); i++) {
216  u16 index{static_cast<u16>(tmd_body.contentinfo[i].index)};
217  u16 count{static_cast<u16>(tmd_body.contentinfo[i].command_count)};
218 
219  if (count == 0)
220  continue;
221 
222  LOG_DEBUG(Service_FS, "Content chunks for content info index {}:", i);
223  for (u16 j = index; j < index + count; j++) {
224  // Don't attempt to print content we don't have
225  if (j > tmd_body.content_count)
226  break;
227 
228  const ContentChunk& chunk{tmd_chunks[j]};
229  LOG_DEBUG(Service_FS, " ID {:08X}, Index {:04X}, Type {:04x}, Size {:016X}",
230  static_cast<u32>(chunk.id), static_cast<u32>(chunk.index),
231  static_cast<u32>(chunk.type), static_cast<u64>(chunk.size));
232  }
233  }
