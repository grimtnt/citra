113 void ConfigureCamera::updateImageSourceUI() {
114  int image_source = ui->image_source->currentIndex();
115  switch (image_source) {
116  case 0: /* blank */
117  case 2: /* system camera */
118  ui->prompt_before_load->setHidden(true);
119  ui->prompt_before_load->setChecked(false);
120  ui->camera_file_label->setHidden(true);
121  ui->camera_file->setHidden(true);

123  ui->toolButton->setHidden(true);
124  break;

126  ui->prompt_before_load->setHidden(false);
127  ui->camera_file_label->setHidden(false);
128  ui->camera_file->setHidden(false);
129  ui->toolButton->setHidden(false);
130  if (camera_config[getSelectedCameraIndex()].empty()) {
131  ui->prompt_before_load->setChecked(true);
132  ui->camera_file->setDisabled(true);
133  ui->toolButton->setDisabled(true);
134  ui->camera_file->clear();
135  } else {
136  ui->camera_file->setDisabled(false);
137  ui->toolButton->setDisabled(false);
138  }
139  break;
140  default:
141  LOG_ERROR(Service_CAM, "Unknown image source {}", image_source);
142  }
143  ui->system_camera_label->setHidden(image_source != 2);
144  ui->system_camera->setHidden(image_source != 2);
145  ui->camera_flip_label->setHidden(image_source == 0);
146  ui->camera_flip->setHidden(image_source == 0);

149 void ConfigureCamera::recordConfig() {
150  std::string implementation = Implementations[ui->image_source->currentIndex()];
151  int image_source = ui->image_source->currentIndex();
152  std::string config;
153  if (image_source == 2) { /* system camera */
154  if (ui->system_camera->currentIndex() == 0) {
155  config = "";
156  } else {
157  config = ui->system_camera->currentText().toStdString();
158  }
159  } else {
160  config = ui->camera_file->text().toStdString();
161  }
162  if (current_selected == CameraPosition::RearBoth) {
163  camera_name[0] = camera_name[2] = implementation;
164  camera_config[0] = camera_config[2] = config;
165  camera_flip[0] = camera_flip[2] = ui->camera_flip->currentIndex();
166  } else if (current_selected != CameraPosition::Null) {
167  int index = static_cast<int>(current_selected);
168  camera_name[index] = implementation;
169  camera_config[index] = config;
170  camera_flip[index] = ui->camera_flip->currentIndex();
171  }
172  current_selected = getCameraSelection();
173 }

175 void ConfigureCamera::startPreviewing() {
176  current_selected = getCameraSelection();
177  recordConfig();
178  int camera_selection = getSelectedCameraIndex();
179  stopPreviewing();
180  // Init preview box
181  ui->preview_box->setHidden(false);
182  ui->preview_button->setHidden(true);
183  preview_width = ui->preview_box->size().width();
184  preview_height = preview_width * 0.75;
185  ui->preview_box->setToolTip(QString("Resolution: ") + QString::number(preview_width) + "*" +
186  QString::number(preview_height));
187  // Load previewing camera
188  previewing_camera = Camera::CreateCameraPreview(
189  camera_name[camera_selection], camera_config[camera_selection], preview_width,
190  preview_height, static_cast<Service::CAM::Flip>(camera_flip[camera_selection]));
191  if (!previewing_camera) {
192  stopPreviewing();
193  return;
194  }
195  previewing_camera->SetResolution(
196  {static_cast<u16>(preview_width), static_cast<u16>(preview_height)});
197  previewing_camera->SetEffect(Service::CAM::Effect::None);
198  previewing_camera->SetFlip(Service::CAM::Flip::None);
199  previewing_camera->SetFormat(Service::CAM::OutputFormat::RGB565);
200  previewing_camera->SetFrameRate(Service::CAM::FrameRate::Rate_30);
201  previewing_camera->StartCapture();
202 
203  timer_id = startTimer(1000 / 30);
204 }

220 void ConfigureCamera::timerEvent(QTimerEvent* event) {
221  if (event->timerId() != timer_id) {
222  return;
223  }
224  if (!previewing_camera) {
225  killTimer(timer_id);
226  timer_id = 0;
227  return;
228  }
229  std::vector<u16> frame{previewing_camera->ReceiveFrame()};
230  int width{ui->preview_box->size().width()};
231  int height{static_cast<int>(width * 0.75)};
232  if (width != preview_width || height != preview_height) {
233  stopPreviewing();
234  return;
235  }
236  QImage image{width, height, QImage::Format::Format_RGB16};
237  std::memcpy(image.bits(), frame.data(), width * height * sizeof(u16));
238  ui->preview_box->setPixmap(QPixmap::fromImage(image));
239 }

241 void ConfigureCamera::setConfiguration() {
242  int index{getSelectedCameraIndex()};
243  for (int i{}; i < Implementations.size(); i++) {
244  if (Implementations[i] == camera_name[index]) {
245  ui->image_source->setCurrentIndex(i);
246  }
247  }
248  if (camera_name[index] == "image") {
249  ui->camera_file->setDisabled(camera_config[index].empty());
250  ui->toolButton->setDisabled(camera_config[index].empty());
251  if (camera_config[index].empty()) {
252  ui->camera_file->clear();
253  }
254  }
255  if (camera_name[index] == "qt") {
256  ui->system_camera->setCurrentIndex(0);
257  if (!camera_config[index].empty()) {
258  ui->system_camera->setCurrentText(QString::fromStdString(camera_config[index]));
259  }
260  } else {
261  ui->camera_file->setText(QString::fromStdString(camera_config[index]));
262  }
263  ui->camera_flip->setCurrentIndex(camera_flip[index]);
264  updateImageSourceUI();
265 }

290 ConfigureCamera::CameraPosition ConfigureCamera::getCameraSelection() {
291  switch (ui->camera_selection->currentIndex()) {
292  case 0: // Front

294  case 1: // Rear

303  default:
304  LOG_ERROR(Frontend, "Unknown camera selection");
305  return CameraPosition::Front;
