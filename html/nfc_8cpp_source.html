51 void Module::Interface::StartTagScanning(Kernel::HLERequestContext& ctx) {
52  IPC::RequestParser rp{ctx, 0x05, 1, 0}; // 0x00050040
53  u16 in_val{rp.Pop<u16>()};
54 
55  ResultCode result = RESULT_SUCCESS;
56 
57  // TODO(shinyquagsire23): Implement NFC tag detection, for now stub result

65 
66  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
