20 class SDMCDelayGenerator : public DelayGenerator {
21 public:
22  u64 GetReadDelayNs(std::size_t length) override {
23  // This is the delay measured on O3DS and O2DS with
24  // https://gist.github.com/B3n30/ac40eac20603f519ff106107f4ac9182
25  // from the results the average of each length was taken.
26  static constexpr u64 slope(183);
27  static constexpr u64 offset(524879);
28  static constexpr u64 minimum(631826);
29  u64 IPCDelayNanoseconds{std::max<u64>(static_cast<u64>(length) * slope + offset, minimum)};
30  return IPCDelayNanoseconds;
31  }
32 };

45 ResultVal<std::unique_ptr<FileBackend>> SDMCArchive::OpenFileBase(const Path& path,
46  const Mode& mode) const {
47  LOG_DEBUG(Service_FS, "called path={} mode={:01X}", path.DebugStr(), mode.hex);
48 
49  const PathParser path_parser{path};
50 
51  if (!path_parser.IsValid()) {
52  LOG_ERROR(Service_FS, "Invalid path {}", path.DebugStr());
53  return ERROR_INVALID_PATH;
54  }
55 
56  if (mode.hex == 0) {
57  LOG_ERROR(Service_FS, "Empty open mode");
58  return ERROR_INVALID_OPEN_FLAGS;
59  }
60 
61  if (mode.create_flag && !mode.write_flag) {
62  LOG_ERROR(Service_FS, "Create flag set but write flag not set");
63  return ERROR_INVALID_OPEN_FLAGS;
64  }
65 
66  const auto full_path{path_parser.BuildHostPath(mount_point)};
67 
68  switch (path_parser.GetHostStatus(mount_point)) {
69  case PathParser::InvalidMountPoint:
70  LOG_CRITICAL(Service_FS, "(unreachable) Invalid mount point {}", mount_point);
71  return ERROR_NOT_FOUND;
72  case PathParser::PathNotFound:
73  case PathParser::FileInPath:
74  LOG_ERROR(Service_FS, "Path not found {}", full_path);
75  return ERROR_NOT_FOUND;
76  case PathParser::DirectoryFound:
77  LOG_ERROR(Service_FS, "{} is not a file", full_path);
78  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY_SDMC;
79  case PathParser::NotFound:
80  if (!mode.create_flag) {
81  LOG_ERROR(Service_FS, "Non-existing file {} can't be open without mode create.",
82  full_path);
83  return ERROR_NOT_FOUND;
84  } else {
85  // Create the file
86  FileUtil::CreateEmptyFile(full_path);
87  }
88  break;
89  case PathParser::FileFound:
90  break; // Expected 'success' case
91  }
92 
93  FileUtil::IOFile file{full_path, mode.write_flag ? "r+b" : "rb"};
94  if (!file.IsOpen()) {
95  LOG_CRITICAL(Service_FS, "(unreachable) Unknown error opening {}", full_path);
96  return ERROR_NOT_FOUND;
97  }
98 
99  std::unique_ptr<DelayGenerator> delay_generator{std::make_unique<SDMCDelayGenerator>()};
100  auto disk_file{std::make_unique<DiskFile>(std::move(file), mode, std::move(delay_generator))};
101  return MakeResult<std::unique_ptr<FileBackend>>(std::move(disk_file));
102 }

167 template <typename T>
168 static ResultCode DeleteDirectoryHelper(const Path& path, const std::string& mount_point,
169  T deleter) {
170  const PathParser path_parser{path};
171 
172  if (!path_parser.IsValid()) {
173  LOG_ERROR(Service_FS, "Invalid path {}", path.DebugStr());
174  return ERROR_INVALID_PATH;
175  }
176 
177  if (path_parser.IsRootDirectory())
178  return ERROR_NOT_FOUND;
179 
180  const auto full_path{path_parser.BuildHostPath(mount_point)};
181 
182  switch (path_parser.GetHostStatus(mount_point)) {
183  case PathParser::InvalidMountPoint:
184  LOG_CRITICAL(Service_FS, "(unreachable) Invalid mount point {}", mount_point);
185  return ERROR_NOT_FOUND;
186  case PathParser::PathNotFound:
187  case PathParser::NotFound:
188  LOG_ERROR(Service_FS, "Path not found {}", full_path);
189  return ERROR_NOT_FOUND;
190  case PathParser::FileInPath:
191  case PathParser::FileFound:
192  LOG_ERROR(Service_FS, "Unexpected file in path {}", full_path);
193  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY_SDMC;
194  case PathParser::DirectoryFound:
195  break; // Expected 'success' case
196  }
197 
198  if (deleter(full_path)) {
199  return RESULT_SUCCESS;
200  }
201 
202  LOG_ERROR(Service_FS, "Directory not empty {}", full_path);
203  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY_SDMC;
204 }

215 ResultCode SDMCArchive::CreateFile(const FileSys::Path& path, u64 size) const {
216  const PathParser path_parser{path};
217 
218  if (!path_parser.IsValid()) {
219  LOG_ERROR(Service_FS, "Invalid path {}", path.DebugStr());
220  return ERROR_INVALID_PATH;
221  }
222 
223  const auto full_path{path_parser.BuildHostPath(mount_point)};
224 
225  switch (path_parser.GetHostStatus(mount_point)) {
226  case PathParser::InvalidMountPoint:
227  LOG_CRITICAL(Service_FS, "(unreachable) Invalid mount point {}", mount_point);
228  return ERROR_NOT_FOUND;
229  case PathParser::PathNotFound:
230  case PathParser::FileInPath:
231  LOG_ERROR(Service_FS, "Path not found {}", full_path);
232  return ERROR_NOT_FOUND;
233  case PathParser::DirectoryFound:
234  LOG_ERROR(Service_FS, "{} already exists", full_path);
235  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY_SDMC;
236  case PathParser::FileFound:
237  LOG_ERROR(Service_FS, "{} already exists", full_path);
238  return ERROR_ALREADY_EXISTS;
239  case PathParser::NotFound:
240  break; // Expected 'success' case
241  }
242 
243  if (size == 0) {
244  FileUtil::CreateEmptyFile(full_path);
245  return RESULT_SUCCESS;
246  }
247 
248  FileUtil::IOFile file{full_path, "wb"};
249  // Creates a sparse file (or a normal file on filesystems without the concept of sparse files)
250  // We do this by seeking to the right size, then writing a single null byte.
251  if (file.Seek(size - 1, SEEK_SET) && file.WriteBytes("", 1) == 1) {
252  return RESULT_SUCCESS;
253  }
254 
255  LOG_ERROR(Service_FS, "Too large file");
256  return ResultCode(ErrorDescription::TooLarge, ErrorModule::FS, ErrorSummary::OutOfResource,
257  ErrorLevel::Info);
258 }
