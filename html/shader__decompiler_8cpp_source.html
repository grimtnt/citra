78  const Subroutine& AddSubroutine(u32 begin, u32 end) {
79  auto iter = subroutines.find(Subroutine{begin, end});
80  if (iter != subroutines.end())
81  return *iter;
82 
83  Subroutine subroutine{begin, end};
84  subroutine.exit_method = Scan(begin, end, subroutine.labels);
85  if (subroutine.exit_method == ExitMethod::Undetermined)
86  throw DecompileFail("Recursive function detected");
87  return *subroutines.insert(std::move(subroutine)).first;
88  }

91  static ExitMethod ParallelExit(ExitMethod a, ExitMethod b) {
92  if (a == ExitMethod::Undetermined) {
93  return b;
94  }
95  if (b == ExitMethod::Undetermined) {
96  return a;
97  }
98  if (a == b) {
99  return a;
100  }
101  return ExitMethod::Conditional;
102  }
103 
105  static ExitMethod SeriesExit(ExitMethod a, ExitMethod b) {
106  // This should be handled before evaluating b.
107  DEBUG_ASSERT(a != ExitMethod::AlwaysEnd);
108 
109  if (a == ExitMethod::Undetermined) {
110  return ExitMethod::Undetermined;
111  }
112 
113  if (a == ExitMethod::AlwaysReturn) {
114  return b;
115  }
116 
117  if (b == ExitMethod::Undetermined || b == ExitMethod::AlwaysEnd) {
118  return ExitMethod::AlwaysEnd;
119  }
120 
121  return ExitMethod::Conditional;
122  }
123 
125  ExitMethod Scan(u32 begin, u32 end, std::set<u32>& labels) {
126  auto [iter, inserted] =
127  exit_method_map.emplace(std::make_pair(begin, end), ExitMethod::Undetermined);
128  ExitMethod& exit_method = iter->second;
129  if (!inserted)
130  return exit_method;
131 
132  for (u32 offset = begin; offset != end && offset != PROGRAM_END; ++offset) {
133  const Instruction instr = {program_code[offset]};
134  switch (instr.opcode.Value()) {
135  case OpCode::Id::END: {
136  return exit_method = ExitMethod::AlwaysEnd;
137  }
138  case OpCode::Id::JMPC:
139  case OpCode::Id::JMPU: {
140  labels.insert(instr.flow_control.dest_offset);
141  ExitMethod no_jmp = Scan(offset + 1, end, labels);
142  ExitMethod jmp = Scan(instr.flow_control.dest_offset, end, labels);
143  return exit_method = ParallelExit(no_jmp, jmp);
144  }
145  case OpCode::Id::CALL: {
146  auto& call = AddSubroutine(instr.flow_control.dest_offset,
147  instr.flow_control.dest_offset +
148  instr.flow_control.num_instructions);
149  if (call.exit_method == ExitMethod::AlwaysEnd)
150  return exit_method = ExitMethod::AlwaysEnd;
151  ExitMethod after_call = Scan(offset + 1, end, labels);
152  return exit_method = SeriesExit(call.exit_method, after_call);
153  }
154  case OpCode::Id::LOOP: {
155  auto& loop = AddSubroutine(offset + 1, instr.flow_control.dest_offset + 1);
156  if (loop.exit_method == ExitMethod::AlwaysEnd)
157  return exit_method = ExitMethod::AlwaysEnd;
158  ExitMethod after_loop = Scan(instr.flow_control.dest_offset + 1, end, labels);
159  return exit_method = SeriesExit(loop.exit_method, after_loop);
160  }
161  case OpCode::Id::CALLC:
162  case OpCode::Id::CALLU: {
163  auto& call = AddSubroutine(instr.flow_control.dest_offset,
164  instr.flow_control.dest_offset +
165  instr.flow_control.num_instructions);
166  ExitMethod after_call = Scan(offset + 1, end, labels);
167  return exit_method = SeriesExit(
168  ParallelExit(call.exit_method, ExitMethod::AlwaysReturn), after_call);
169  }
170  case OpCode::Id::IFU:
171  case OpCode::Id::IFC: {
172  auto& if_sub = AddSubroutine(offset + 1, instr.flow_control.dest_offset);
173  ExitMethod else_method;
174  if (instr.flow_control.num_instructions != 0) {
175  auto& else_sub = AddSubroutine(instr.flow_control.dest_offset,
176  instr.flow_control.dest_offset +
177  instr.flow_control.num_instructions);
178  else_method = else_sub.exit_method;
179  } else {
180  else_method = ExitMethod::AlwaysReturn;
181  }
182 
183  ExitMethod both = ParallelExit(if_sub.exit_method, else_method);
184  if (both == ExitMethod::AlwaysEnd)
185  return exit_method = ExitMethod::AlwaysEnd;
186  ExitMethod after_call =
187  Scan(instr.flow_control.dest_offset + instr.flow_control.num_instructions, end,
188  labels);
189  return exit_method = SeriesExit(both, after_call);
190  }
191  }
192  }
193  return exit_method = ExitMethod::AlwaysReturn;
194  }

219 std::string GetSelectorSrc(const SwizzlePattern& pattern) {
220  std::string out{};
221  for (std::size_t i{}; i < 4; ++i) {

273  static std::string EvaluateCondition(Instruction::FlowControlType flow_control) {
274  using Op = Instruction::FlowControlType::Op;
275 
276  std::string result_x{flow_control.refx.Value() ? "conditional_code.x"
277  : "!conditional_code.x"};
278  std::string result_y{flow_control.refy.Value() ? "conditional_code.y"
279  : "!conditional_code.y"};
280 
281  switch (flow_control.op) {
282  case Op::JustX:
283  return result_x;
284  case Op::JustY:
285  return result_y;
286  case Op::Or:
287  case Op::And: {
288  std::string and_or{flow_control.op == Op::Or ? "any" : "all"};
289  std::string bvec{};
290  if (flow_control.refx.Value() && flow_control.refy.Value()) {
291  bvec = "conditional_code";
292  } else if (!flow_control.refx.Value() && !flow_control.refy.Value()) {
293  bvec = "not(conditional_code)";
294  } else {
295  bvec = "bvec2(" + result_x + ", " + result_y + ")";
296  }
297  return and_or + "(" + bvec + ")";
298  }
299  default:
300  UNREACHABLE();
301  return "";
302  }
303  }

306  std::string GetSourceRegister(const SourceRegister& source_reg,
307  u32 address_register_index) const {
308  u32 index{static_cast<u32>(source_reg.GetIndex())};
309  std::string index_str{std::to_string(index)};
310 
311  switch (source_reg.GetRegisterType()) {
312  case RegisterType::Input:
313  return inputreg_getter(index);
314  case RegisterType::Temporary:
315  return "reg_tmp" + index_str;
316  case RegisterType::FloatUniform:
317  if (address_register_index != 0) {
318  index_str +=
319  std::string(" + address_registers.") + "xyz"[address_register_index - 1];
320  }
321  return "uniforms.f[" + index_str + "]";
322  default:
323  UNREACHABLE();
324  return "";
325  }
326  }

344  std::string GetUniformBool(u32 index) const {
345  if (is_gs && index == 15) {
346  // In PICA geometry shader, b15 is set to true after every geometry shader invocation.
347  // Accessing b15 usually indicates that the program relies on register value
348  // preservation across invocation (and therefore it uses b15 to determine whether to
349  // initialize the registers), which cannot be implemented in GL shaders.
350  throw DecompileFail("b15 access in geometry shader");
351  }
352  return "uniforms.b[" + std::to_string(index) + "]";
353  }

378  void SetDest(const SwizzlePattern& swizzle, const std::string& reg, const std::string& value,
379  u32 dest_num_components, u32 value_num_components) {
380  u32 dest_mask_num_components{};
381  std::string dest_mask_swizzle{"."};
382 
383  for (u32 i{}; i < dest_num_components; ++i) {
384  if (swizzle.DestComponentEnabled(static_cast<int>(i))) {
385  dest_mask_swizzle += "xyzw"[i];
386  ++dest_mask_num_components;
387  }
388  }
389 
390  if (reg.empty() || dest_mask_num_components == 0) {
391  return;
392  }
393  DEBUG_ASSERT(value_num_components >= dest_num_components || value_num_components == 1);
394 
395  std::string dest{reg + (dest_num_components != 1 ? dest_mask_swizzle : "")};
396 
397  std::string src{value};
398  if (value_num_components == 1) {
399  if (dest_mask_num_components != 1) {
400  src = "vec" + std::to_string(dest_mask_num_components) + "(" + value + ")";
401  }
402  } else if (value_num_components != dest_mask_num_components) {
403  src = "(" + value + ")" + dest_mask_swizzle;
404  }
405 
406  shader.AddLine(dest + " = " + src + ";");
407  }

416  u32 CompileInstr(u32 offset) {
417  const Instruction instr{program_code[offset]};
418 
419  std::size_t swizzle_offset{instr.opcode.Value().GetInfo().type == OpCode::Type::MultiplyAdd
420  ? instr.mad.operand_desc_id
421  : instr.common.operand_desc_id};
422  const SwizzlePattern swizzle{swizzle_data[swizzle_offset]};
423 
424  shader.AddLine("// " + std::to_string(offset) + ": " + instr.opcode.Value().GetInfo().name);
425 
426  switch (instr.opcode.Value().GetInfo().type) {
427  case OpCode::Type::Arithmetic: {
428  const bool is_inverted{
429  (0 != (instr.opcode.Value().GetInfo().subtype & OpCode::Info::SrcInversed))};
430 
431  std::string src1{swizzle.negate_src1 ? "-" : ""};
432  src1 += GetSourceRegister(instr.common.GetSrc1(is_inverted),
433  !is_inverted * instr.common.address_register_index);
434  src1 += "." + GetSelectorSrc1(swizzle);
435 
436  std::string src2{swizzle.negate_src2 ? "-" : ""};
437  src2 += GetSourceRegister(instr.common.GetSrc2(is_inverted),
438  is_inverted * instr.common.address_register_index);
439  src2 += "." + GetSelectorSrc2(swizzle);
440 
441  std::string dest_reg{GetDestRegister(instr.common.dest.Value())};
442 
443  switch (instr.opcode.Value().EffectiveOpCode()) {
444  case OpCode::Id::ADD: {
445  SetDest(swizzle, dest_reg, src1 + " + " + src2, 4, 4);
446  break;
447  }
448 
449  case OpCode::Id::MUL: {
450  if (sanitize_mul) {
451  SetDest(swizzle, dest_reg, "sanitize_mul(" + src1 + ", " + src2 + ")", 4, 4);
452  } else {
453  SetDest(swizzle, dest_reg, src1 + " * " + src2, 4, 4);
454  }
455  break;
456  }
457 
458  case OpCode::Id::FLR: {
459  SetDest(swizzle, dest_reg, "floor(" + src1 + ")", 4, 4);
460  break;
461  }
462 
463  case OpCode::Id::MAX: {
464  SetDest(swizzle, dest_reg, "max(" + src1 + ", " + src2 + ")", 4, 4);
465  break;
466  }
467 
468  case OpCode::Id::MIN: {
469  SetDest(swizzle, dest_reg, "min(" + src1 + ", " + src2 + ")", 4, 4);
470  break;
471  }
472 
473  case OpCode::Id::DP3:
474  case OpCode::Id::DP4:
475  case OpCode::Id::DPH:
476  case OpCode::Id::DPHI: {
477  OpCode::Id opcode{instr.opcode.Value().EffectiveOpCode()};
478  std::string dot{};
479  if (opcode == OpCode::Id::DP3) {
480  if (sanitize_mul) {
481  dot = "dot(vec3(sanitize_mul(" + src1 + ", " + src2 + ")), vec3(1.0))";
482  } else {
483  dot = "dot(vec3(" + src1 + "), vec3(" + src2 + "))";
484  }
485  } else {
486  std::string src1_{(opcode == OpCode::Id::DPH || opcode == OpCode::Id::DPHI)
487  ? "vec4(" + src1 + ".xyz, 1.0)"
488  : src1};
489  if (sanitize_mul) {
490  dot = "dot(sanitize_mul(" + src1_ + ", " + src2 + "), vec4(1.0))";
491  } else {
492  dot = "dot(" + src1 + ", " + src2 + ")";
493  }
494  }
495 
496  SetDest(swizzle, dest_reg, dot, 4, 1);
497  break;
498  }
499 
500  case OpCode::Id::RCP: {
501  SetDest(swizzle, dest_reg, "(1.0 / " + src1 + ".x)", 4, 1);
502  break;
503  }
504 
505  case OpCode::Id::RSQ: {
506  SetDest(swizzle, dest_reg, "inversesqrt(" + src1 + ".x)", 4, 1);
507  break;
508  }
509 
510  case OpCode::Id::MOVA: {
511  SetDest(swizzle, "address_registers", "ivec2(" + src1 + ")", 2, 2);
512  break;
513  }
514 
515  case OpCode::Id::MOV: {
516  SetDest(swizzle, dest_reg, src1, 4, 4);
517  break;
518  }
519 
520  case OpCode::Id::SGE:
521  case OpCode::Id::SGEI: {
522  SetDest(swizzle, dest_reg, "vec4(greaterThanEqual(" + src1 + "," + src2 + "))", 4,
523  4);
524  break;
525  }
526 
527  case OpCode::Id::SLT:
528  case OpCode::Id::SLTI: {
529  SetDest(swizzle, dest_reg, "vec4(lessThan(" + src1 + "," + src2 + "))", 4, 4);
530  break;
531  }
532 
533  case OpCode::Id::CMP: {
534  using CompareOp = Instruction::Common::CompareOpType::Op;
535  const std::map<CompareOp, std::pair<std::string, std::string>> cmp_ops{
536  {CompareOp::Equal, {"==", "equal"}},
537  {CompareOp::NotEqual, {"!=", "notEqual"}},
538  {CompareOp::LessThan, {"<", "lessThan"}},
539  {CompareOp::LessEqual, {"<=", "lessThanEqual"}},
540  {CompareOp::GreaterThan, {">", "greaterThan"}},
541  {CompareOp::GreaterEqual, {">=", "greaterThanEqual"}}};
542 
543  const CompareOp op_x{instr.common.compare_op.x.Value()};
544  const CompareOp op_y{instr.common.compare_op.y.Value()};
545 
546  if (cmp_ops.find(op_x) == cmp_ops.end()) {
547  LOG_ERROR(HW_GPU, "Unknown compare mode {:x}", static_cast<int>(op_x));
548  } else if (cmp_ops.find(op_y) == cmp_ops.end()) {
549  LOG_ERROR(HW_GPU, "Unknown compare mode {:x}", static_cast<int>(op_y));
550  } else if (op_x != op_y) {
551  shader.AddLine("conditional_code.x = " + src1 + ".x " +
552  cmp_ops.find(op_x)->second.first + " " + src2 + ".x;");
553  shader.AddLine("conditional_code.y = " + src1 + ".y " +
554  cmp_ops.find(op_y)->second.first + " " + src2 + ".y;");
555  } else {
556  shader.AddLine("conditional_code = " + cmp_ops.find(op_x)->second.second +
557  "(vec2(" + src1 + "), vec2(" + src2 + "));");
558  }
559  break;
560  }
561 
562  case OpCode::Id::EX2: {
563  SetDest(swizzle, dest_reg, "exp2(" + src1 + ".x)", 4, 1);
564  break;
565  }
566 
567  case OpCode::Id::LG2: {
568  SetDest(swizzle, dest_reg, "log2(" + src1 + ".x)", 4, 1);
569  break;
570  }
571 
572  default: {
573  LOG_ERROR(HW_GPU, "Unhandled arithmetic instruction: 0x{:02x} ({}): 0x{:08x}",
574  (int)instr.opcode.Value().EffectiveOpCode(),
575  instr.opcode.Value().GetInfo().name, instr.hex);
576  throw DecompileFail("Unhandled instruction");
577  break;
578  }
579  }
580 
581  break;
582  }
583 
584  case OpCode::Type::MultiplyAdd: {
585  if ((instr.opcode.Value().EffectiveOpCode() == OpCode::Id::MAD) ||
586  (instr.opcode.Value().EffectiveOpCode() == OpCode::Id::MADI)) {
587  bool is_inverted{(instr.opcode.Value().EffectiveOpCode() == OpCode::Id::MADI)};
588 
589  std::string src1{swizzle.negate_src1 ? "-" : ""};
590  src1 += GetSourceRegister(instr.mad.GetSrc1(is_inverted), 0);
591  src1 += "." + GetSelectorSrc1(swizzle);
592 
593  std::string src2{swizzle.negate_src2 ? "-" : ""};
594  src2 += GetSourceRegister(instr.mad.GetSrc2(is_inverted),
595  !is_inverted * instr.mad.address_register_index);
596  src2 += "." + GetSelectorSrc2(swizzle);
597 
598  std::string src3{swizzle.negate_src3 ? "-" : ""};
599  src3 += GetSourceRegister(instr.mad.GetSrc3(is_inverted),
600  is_inverted * instr.mad.address_register_index);
601  src3 += "." + GetSelectorSrc3(swizzle);
602 
603  std::string dest_reg{
604  (instr.mad.dest.Value() < 0x10)
605  ? outputreg_getter(static_cast<u32>(instr.mad.dest.Value().GetIndex()))
606  : (instr.mad.dest.Value() < 0x20)
607  ? "reg_tmp" + std::to_string(instr.mad.dest.Value().GetIndex())
608  : ""};
609 
610  if (sanitize_mul) {
611  SetDest(swizzle, dest_reg, "sanitize_mul(" + src1 + ", " + src2 + ") + " + src3,
612  4, 4);
613  } else {
614  SetDest(swizzle, dest_reg, src1 + " * " + src2 + " + " + src3, 4, 4);
615  }
616  } else {
617  LOG_ERROR(HW_GPU, "Unhandled multiply-add instruction: 0x{:02x} ({}): 0x{:08x}",
618  (int)instr.opcode.Value().EffectiveOpCode(),
619  instr.opcode.Value().GetInfo().name, instr.hex);
620  throw DecompileFail("Unhandled instruction");
621  }
622  break;
623  }
624 
625  default: {
626  switch (instr.opcode.Value()) {
627  case OpCode::Id::END: {
628  shader.AddLine("return true;");
629  offset = PROGRAM_END - 1;
630  break;
631  }
632 
633  case OpCode::Id::JMPC:
634  case OpCode::Id::JMPU: {
635  std::string condition{};
636  if (instr.opcode.Value() == OpCode::Id::JMPC) {
637  condition = EvaluateCondition(instr.flow_control);
638  } else {
639  bool invert_test{static_cast<bool>(instr.flow_control.num_instructions & 1)};
640  condition = (invert_test ? "!" : "") +
641  GetUniformBool(instr.flow_control.bool_uniform_id);
642  }
643 
644  shader.AddLine("if (" + condition + ") {");
645  ++shader.scope;
646  shader.AddLine("{ jmp_to = " + std::to_string(instr.flow_control.dest_offset) +
647  "u; break; }");
648 
649  --shader.scope;
650  shader.AddLine("}");
651  break;
652  }
653 
654  case OpCode::Id::CALL:
655  case OpCode::Id::CALLC:
656  case OpCode::Id::CALLU: {
657  std::string condition{};
658  if (instr.opcode.Value() == OpCode::Id::CALLC) {
659  condition = EvaluateCondition(instr.flow_control);
660  } else if (instr.opcode.Value() == OpCode::Id::CALLU) {
661  condition = GetUniformBool(instr.flow_control.bool_uniform_id);
662  }
663 
664  shader.AddLine(condition.empty() ? "{" : "if (" + condition + ") {");
665  ++shader.scope;
666 
667  auto& call_sub{GetSubroutine(instr.flow_control.dest_offset,
668  instr.flow_control.dest_offset +
669  instr.flow_control.num_instructions)};
670 
671  CallSubroutine(call_sub);
672  if (instr.opcode.Value() == OpCode::Id::CALL &&
673  call_sub.exit_method == ExitMethod::AlwaysEnd) {
674  offset = PROGRAM_END - 1;
675  }
676 
677  --shader.scope;
678  shader.AddLine("}");
679  break;
680  }
681 
682  case OpCode::Id::NOP: {
683  break;
684  }
685 
686  case OpCode::Id::IFC:
687  case OpCode::Id::IFU: {
688  std::string condition{};
689  if (instr.opcode.Value() == OpCode::Id::IFC) {
690  condition = EvaluateCondition(instr.flow_control);
691  } else {
692  condition = GetUniformBool(instr.flow_control.bool_uniform_id);
693  }
694 
695  const u32 if_offset{offset + 1};
696  const u32 else_offset{instr.flow_control.dest_offset};
697  const u32 endif_offset{instr.flow_control.dest_offset +
698  instr.flow_control.num_instructions};
699 
700  shader.AddLine("if (" + condition + ") {");
701  ++shader.scope;
702 
703  auto& if_sub{GetSubroutine(if_offset, else_offset)};
704  CallSubroutine(if_sub);
705  offset = else_offset - 1;
706 
707  if (instr.flow_control.num_instructions != 0) {
708  --shader.scope;
709  shader.AddLine("} else {");
710  ++shader.scope;
711 
712  auto& else_sub{GetSubroutine(else_offset, endif_offset)};
713  CallSubroutine(else_sub);
714  offset = endif_offset - 1;
715 
716  if (if_sub.exit_method == ExitMethod::AlwaysEnd &&
717  else_sub.exit_method == ExitMethod::AlwaysEnd) {
718  offset = PROGRAM_END - 1;
719  }
720  }
721 
722  --shader.scope;
723  shader.AddLine("}");
724  break;
725  }
726 
727  case OpCode::Id::LOOP: {
728  std::string int_uniform{"uniforms.i[" +
729  std::to_string(instr.flow_control.int_uniform_id) + "]"};
730 
731  shader.AddLine("address_registers.z = int(" + int_uniform + ".y);");
732 
733  std::string loop_var{"loop" + std::to_string(offset)};
734  shader.AddLine("for (uint " + loop_var + " = 0u; " + loop_var +
735  " <= " + int_uniform + ".x; address_registers.z += int(" +
736  int_uniform + ".z), ++" + loop_var + ") {");
737  ++shader.scope;
738 
739  auto& loop_sub{GetSubroutine(offset + 1, instr.flow_control.dest_offset + 1)};
740  CallSubroutine(loop_sub);
741  offset = instr.flow_control.dest_offset;
742 
743  --shader.scope;
744  shader.AddLine("}");
745 
746  if (loop_sub.exit_method == ExitMethod::AlwaysEnd) {
747  offset = PROGRAM_END - 1;
748  }
749 
750  break;
751  }
752 
753  case OpCode::Id::EMIT: {
754  if (is_gs) {
755  shader.AddLine("emit();");
756  }
757  break;
758  }
759 
760  case OpCode::Id::SETEMIT: {
761  if (is_gs) {
762  ASSERT(instr.setemit.vertex_id < 3);
763  shader.AddLine("setemit(" + std::to_string(instr.setemit.vertex_id) + "u, " +
764  ((instr.setemit.prim_emit != 0) ? "true" : "false") + ", " +
765  ((instr.setemit.winding != 0) ? "true" : "false") + ");");
766  }
767  break;
768  }
769 
770  default: {
771  LOG_ERROR(HW_GPU, "Unhandled instruction: 0x{:02x} ({}): 0x{:08x}",
772  (int)instr.opcode.Value().EffectiveOpCode(),
773  instr.opcode.Value().GetInfo().name, instr.hex);
774  throw DecompileFail("Unhandled instruction");
775  break;
776  }
777  }
778 
779  break;
780  }
781  }
782  return offset + 1;
783  }

792  u32 CompileRange(u32 begin, u32 end) {
793  u32 program_counter{};
794  for (program_counter = begin; program_counter < (begin > end ? PROGRAM_END : end);) {
795  program_counter = CompileInstr(program_counter);
796  }
797  return program_counter;
798  }

800  void Generate() {
801  if (sanitize_mul) {
802  shader.AddLine("vec4 sanitize_mul(vec4 lhs, vec4 rhs) {");
803  ++shader.scope;
804  shader.AddLine("vec4 product = lhs * rhs;");
805  shader.AddLine("return mix(product, mix(mix(vec4(0.0), product, isnan(rhs)), product, "
806  "isnan(lhs)), isnan(product));");
807  --shader.scope;
808  shader.AddLine("}\n");
809  }
810 
811  // Add declarations for registers
812  shader.AddLine("bvec2 conditional_code = bvec2(false);");
813  shader.AddLine("ivec3 address_registers = ivec3(0);");
814  for (int i{}; i < 16; ++i) {
815  shader.AddLine("vec4 reg_tmp" + std::to_string(i) + " = vec4(0.0, 0.0, 0.0, 1.0);");
816  }
817  shader.AddLine("");
818 
819  // Add declarations for all subroutines
820  for (const auto& subroutine : subroutines) {
821  shader.AddLine("bool " + subroutine.GetName() + "();");
822  }
823  shader.AddLine("");
824 
825  // Add the main entry point
826  shader.AddLine("bool exec_shader() {");
827  ++shader.scope;
828  CallSubroutine(GetSubroutine(main_offset, PROGRAM_END));
829  --shader.scope;
830  shader.AddLine("}\n");
831 
832  // Add definitions for all subroutines
833  for (const auto& subroutine : subroutines) {
834  std::set<u32> labels{subroutine.labels};
835 
836  shader.AddLine("bool " + subroutine.GetName() + "() {");
837  ++shader.scope;
838 
839  if (labels.empty()) {
840  if (CompileRange(subroutine.begin, subroutine.end) != PROGRAM_END) {
841  shader.AddLine("return false;");
842  }
843  } else {
844  labels.insert(subroutine.begin);
845  shader.AddLine("uint jmp_to = " + std::to_string(subroutine.begin) + "u;");
846  shader.AddLine("while (true) {");
847  ++shader.scope;
848 
849  shader.AddLine("switch (jmp_to) {");
850 
851  for (auto label : labels) {
852  shader.AddLine("case " + std::to_string(label) + "u: {");
853  ++shader.scope;
854 
855  auto next_it{labels.lower_bound(label + 1)};
856  u32 next_label{next_it == labels.end() ? subroutine.end : *next_it};
857 
858  u32 compile_end{CompileRange(label, next_label)};
859  if (compile_end > next_label && compile_end != PROGRAM_END) {
860  // This happens only when there is a label inside a IF/LOOP block
861  shader.AddLine("{ jmp_to = " + std::to_string(compile_end) + "u; break; }");
862  labels.emplace(compile_end);
863  }
864 
865  --shader.scope;
866  shader.AddLine("}");
867  }
868 
869  shader.AddLine("default: return false;");
870  shader.AddLine("}");
871 
872  --shader.scope;
873  shader.AddLine("}");
874 
875  shader.AddLine("return false;");
876  }
877 
878  --shader.scope;
879  shader.AddLine("}\n");
880 
881  DEBUG_ASSERT(shader.scope == 0);
882  }
883  }
