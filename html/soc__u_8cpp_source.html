188  union Events {
189  u32 hex;
190  BitField<0, 1, u32> pollin;
191  BitField<1, 1, u32> pollpri;
192  BitField<2, 1, u32> pollhup;
193  BitField<3, 1, u32> pollerr;
194  BitField<4, 1, u32> pollout;
195  BitField<5, 1, u32> pollnval;

197  Events& operator=(const Events& other) {
198  hex = other.hex;
199  return *this;

204  static Events TranslateTo3DS(u32 input_event) {
205  Events ev = {};
206  if (input_event & POLLIN)
207  ev.pollin.Assign(1);
208  if (input_event & POLLPRI)
209  ev.pollpri.Assign(1);
210  if (input_event & POLLHUP)
211  ev.pollhup.Assign(1);
212  if (input_event & POLLERR)
213  ev.pollerr.Assign(1);
214  if (input_event & POLLOUT)
215  ev.pollout.Assign(1);
216  if (input_event & POLLNVAL)
217  ev.pollnval.Assign(1);
218  return ev;
219  }

223  static u32 TranslateToPlatform(Events input_event) {
224  u32 ret{};
225  if (input_event.pollin)
226  ret |= POLLIN;
227  if (input_event.pollpri)
228  ret |= POLLPRI;
229  if (input_event.pollhup)
230  ret |= POLLHUP;
231  if (input_event.pollerr)
232  ret |= POLLERR;
233  if (input_event.pollout)
234  ret |= POLLOUT;
235  if (input_event.pollnval)
236  ret |= POLLNVAL;

263 union CTRSockAddr {
265  struct {
266  u8 len;
267  u8 sa_family;
268  u8 sa_data[0x1A];
269  } raw;
270 
272  struct CTRSockAddrIn {
273  u8 len;
274  u8 sin_family;
275  u16 sin_port;
276  u32 sin_addr;
277  } in;
278 
280  static sockaddr ToPlatform(CTRSockAddr const& ctr_addr) {
281  sockaddr result{};
282  result.sa_family = ctr_addr.raw.sa_family;
283  memset(result.sa_data, 0, sizeof(result.sa_data));
284 
285  // We can not guarantee ABI compatibility between platforms so we copy the fields manually
286  switch (result.sa_family) {
287  case AF_INET: {
288  sockaddr_in* result_in{reinterpret_cast<sockaddr_in*>(&result)};
289  result_in->sin_port = ctr_addr.in.sin_port;
290  result_in->sin_addr.s_addr = ctr_addr.in.sin_addr;
291  memset(result_in->sin_zero, 0, sizeof(result_in->sin_zero));
292  break;
293  }
294  default:
295  ASSERT_MSG(false, "Unhandled address family (sa_family) in CTRSockAddr::ToPlatform");
296  break;
297  }
298  return result;
299  }
300 
302  static CTRSockAddr FromPlatform(sockaddr const& addr) {
303  CTRSockAddr result{};
304  result.raw.sa_family = static_cast<u8>(addr.sa_family);
305  // We can not guarantee ABI compatibility between platforms so we copy the fields manually
306  switch (result.raw.sa_family) {
307  case AF_INET: {
308  sockaddr_in const* addr_in{reinterpret_cast<sockaddr_in const*>(&addr)};
309  result.raw.len = sizeof(CTRSockAddrIn);
310  result.in.sin_port = addr_in->sin_port;
311  result.in.sin_addr = addr_in->sin_addr.s_addr;
312  break;
313  }
314  default:
315  ASSERT_MSG(false, "Unhandled address family (sa_family) in CTRSockAddr::ToPlatform");
316  break;
317  }
318  return result;
319  }

328 void SOC_U::Socket(Kernel::HLERequestContext& ctx) {
329  IPC::RequestParser rp{ctx, 0x02, 3, 2};
330  u32 domain{rp.Pop<u32>()}; // Address family
331  u32 type{rp.Pop<u32>()};
332  u32 protocol{rp.Pop<u32>()};
333  rp.PopPID();
334 
335  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
336 
337  // Only 0 is allowed according to 3dbrew, using 0 will let the OS decide which protocol to use
338  if (protocol != 0) {
339  rb.Push(UnimplementedFunction(ErrorModule::SOC)); // TODO(Subv): Correct error code
340  rb.Skip(1, false);
341  return;
342  }
343 
344  if (domain != AF_INET) {
345  rb.Push(UnimplementedFunction(ErrorModule::SOC)); // TODO(Subv): Correct error code
346  rb.Skip(1, false);
347  return;
348  }
349 
350  if (type != SOCK_DGRAM && type != SOCK_STREAM) {
351  rb.Push(UnimplementedFunction(ErrorModule::SOC)); // TODO(Subv): Correct error code
352  rb.Skip(1, false);
353  return;
354  }
355 
356  u32 ret = static_cast<u32>(::socket(domain, type, protocol));
357 
358  if ((s32)ret != SOCKET_ERROR_VALUE)
359  open_sockets[ret] = {ret, true};
360 
361  if ((s32)ret == SOCKET_ERROR_VALUE)
362  ret = TranslateError(GET_ERRNO);
363 
364  rb.Push(RESULT_SUCCESS);
365  rb.Push(ret);
366 }

390 void SOC_U::Fcntl(Kernel::HLERequestContext& ctx) {
391  IPC::RequestParser rp{ctx, 0x13, 3, 2};
392  u32 socket_handle{rp.Pop<u32>()};
393  u32 ctr_cmd{rp.Pop<u32>()};
394  u32 ctr_arg{rp.Pop<u32>()};
395  rp.PopPID();
396 
397  u32 posix_ret{}; // TODO: Check what hardware returns for F_SETFL (unspecified by POSIX)
398  SCOPE_EXIT({
399  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
400  rb.Push(RESULT_SUCCESS);
401  rb.Push(posix_ret);
402  });
403 
404  if (ctr_cmd == 3) { // F_GETFL
405 #ifdef _WIN32
406  posix_ret = 0;
407  auto iter{open_sockets.find(socket_handle)};
408  if (iter != open_sockets.end() && iter->second.blocking == false)
409  posix_ret |= 4; // O_NONBLOCK
410 #else
411  int ret{::fcntl(socket_handle, F_GETFL, 0)};
412  if (ret == SOCKET_ERROR_VALUE) {
413  posix_ret = TranslateError(GET_ERRNO);
414  return;
415  }
416  posix_ret = 0;
417  if (ret & O_NONBLOCK)
418  posix_ret |= 4; // O_NONBLOCK
419 #endif
420  } else if (ctr_cmd == 4) { // F_SETFL
421 #ifdef _WIN32
422  unsigned long tmp{static_cast<unsigned long>((ctr_arg & 4 /* O_NONBLOCK */) ? 1 : 0)};
423  int ret{ioctlsocket(socket_handle, FIONBIO, &tmp)};
424  if (ret == SOCKET_ERROR_VALUE) {
425  posix_ret = TranslateError(GET_ERRNO);
426  return;
427  }
428  auto iter = open_sockets.find(socket_handle);
429  if (iter != open_sockets.end())
430  iter->second.blocking = (tmp == 0);
431 #else
432  int flags{::fcntl(socket_handle, F_GETFL, 0)};
433  if (flags == SOCKET_ERROR_VALUE) {
434  posix_ret = TranslateError(GET_ERRNO);
435  return;
436  }
437 
438  flags &= ~O_NONBLOCK;
439  if (ctr_arg & 4) // O_NONBLOCK
440  flags |= O_NONBLOCK;
441 
442  int ret{::fcntl(socket_handle, F_SETFL, flags)};
443  if (ret == SOCKET_ERROR_VALUE) {
444  posix_ret = TranslateError(GET_ERRNO);
445  return;
446  }
447 #endif
448  } else {
449  LOG_ERROR(Service_SOC, "Unsupported command ({}) in fcntl call", ctr_cmd);
450  posix_ret = TranslateError(EINVAL); // TODO: Find the correct error
451  return;
452  }
453 }

567 void SOC_U::RecvFromOther(Kernel::HLERequestContext& ctx) {
568  IPC::RequestParser rp{ctx, 0x7, 4, 4};
569  u32 socket_handle{rp.Pop<u32>()};
570  u32 len{rp.Pop<u32>()};
571  u32 flags{rp.Pop<u32>()};
572  u32 addr_len{rp.Pop<u32>()};
573  rp.PopPID();
574  auto& buffer{rp.PopMappedBuffer()};
575  CTRSockAddr ctr_src_addr{};
576  std::vector<u8> output_buff(len);
577  std::vector<u8> addr_buff(sizeof(ctr_src_addr));
578  sockaddr src_addr{};
579  socklen_t src_addr_len{sizeof(src_addr)};
580  s32 ret{-1};
581  if (addr_len > 0) {
582  ret = ::recvfrom(socket_handle, reinterpret_cast<char*>(output_buff.data()), len, flags,
583  &src_addr, &src_addr_len);
584  if (ret >= 0 && src_addr_len > 0) {
585  ctr_src_addr = CTRSockAddr::FromPlatform(src_addr);
586  std::memcpy(addr_buff.data(), &ctr_src_addr, sizeof(ctr_src_addr));
587  }
588  } else {
589  ret = ::recvfrom(socket_handle, reinterpret_cast<char*>(output_buff.data()), len, flags,
590  NULL, 0);
591  addr_buff.resize(0);
592  }
593  if (ret == SOCKET_ERROR_VALUE) {
594  ret = TranslateError(GET_ERRNO);
595  } else {
596  buffer.Write(output_buff.data(), 0, ret);
597  }
598  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 4)};
599  rb.Push(RESULT_SUCCESS);
600  rb.Push(ret);
601  rb.PushStaticBuffer(addr_buff, 0);
602  rb.PushMappedBuffer(buffer);
603 }

605 void SOC_U::RecvFrom(Kernel::HLERequestContext& ctx) {
606  // TODO(Subv): Calling this function on a blocking socket will block the emu thread,
607  // preventing graceful shutdown when closing the emulator, this can be fixed by always
608  // performing nonblocking operations and spinlock until the data is available
609  IPC::RequestParser rp{ctx, 0x08, 4, 2};
610  u32 socket_handle{rp.Pop<u32>()};
611  u32 len{rp.Pop<u32>()};
612  u32 flags{rp.Pop<u32>()};
613  u32 addr_len{rp.Pop<u32>()};
614  rp.PopPID();
615 
616  CTRSockAddr ctr_src_addr{};
617  std::vector<u8> output_buff(len);
618  std::vector<u8> addr_buff(sizeof(ctr_src_addr));
619  sockaddr src_addr{};
620  socklen_t src_addr_len{sizeof(src_addr)};
621 
622  s32 ret{-1};
623  if (addr_len > 0) {
624  // Only get src addr if input addr available
625  ret = ::recvfrom(socket_handle, reinterpret_cast<char*>(output_buff.data()), len, flags,
626  &src_addr, &src_addr_len);
627  if (ret >= 0 && src_addr_len > 0) {
628  ctr_src_addr = CTRSockAddr::FromPlatform(src_addr);
629  std::memcpy(addr_buff.data(), &ctr_src_addr, sizeof(ctr_src_addr));
630  }
631  } else {
632  ret = ::recvfrom(socket_handle, reinterpret_cast<char*>(output_buff.data()), len, flags,
633  NULL, 0);
634  addr_buff.resize(0);
635  }
636 
637  s32 total_received{ret};
638  if (ret == SOCKET_ERROR_VALUE) {
639  ret = TranslateError(GET_ERRNO);
640  total_received = 0;
641  }
642 
643  // Write only the data we received to avoid overwriting parts of the buffer with zeros
644  output_buff.resize(total_received);
645 
646  IPC::ResponseBuilder rb{rp.MakeBuilder(3, 4)};
647  rb.Push(RESULT_SUCCESS);
648  rb.Push(ret);
649  rb.Push(total_received);
650  rb.PushStaticBuffer(output_buff, 0);
651  rb.PushStaticBuffer(addr_buff, 1);

654 void SOC_U::Poll(Kernel::HLERequestContext& ctx) {
655  IPC::RequestParser rp{ctx, 0x14, 2, 4};
656  u32 nfds{rp.Pop<u32>()};
657  s32 timeout{rp.Pop<s32>()};
658  rp.PopPID();
659  auto input_fds{rp.PopStaticBuffer()};
660 
661  std::vector<CTRPollFD> ctr_fds(nfds);
662  std::memcpy(ctr_fds.data(), input_fds.data(), nfds * sizeof(CTRPollFD));
663 
664  // The 3ds_pollfd and the pollfd structures may be different (Windows/Linux have different
665  // sizes)
666  // so we have to copy the data
667  std::vector<pollfd> platform_pollfd(nfds);
668  std::transform(ctr_fds.begin(), ctr_fds.end(), platform_pollfd.begin(), CTRPollFD::ToPlatform);
669 
670  s32 ret = ::poll(platform_pollfd.data(), nfds, timeout);
671 
672  // Now update the output pollfd structure
673  std::transform(platform_pollfd.begin(), platform_pollfd.end(), ctr_fds.begin(),
674  CTRPollFD::FromPlatform);
675 
676  std::vector<u8> output_fds(nfds * sizeof(CTRPollFD));
677  std::memcpy(output_fds.data(), ctr_fds.data(), nfds * sizeof(CTRPollFD));
678 
679  if (ret == SOCKET_ERROR_VALUE)
680  ret = TranslateError(GET_ERRNO);
681 
682  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 2)};
683  rb.Push(RESULT_SUCCESS);
684  rb.Push(ret);
685  rb.PushStaticBuffer(output_fds, 0);
686 }
