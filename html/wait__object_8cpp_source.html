35 SharedPtr<Thread> WaitObject::GetHighestPriorityReadyThread() {
36  Thread* candidate{};
37  u32 candidate_priority{THREADPRIO_LOWEST + 1};
38 
39  for (const auto& thread : waiting_threads) {
40  // The list of waiting threads must not contain threads that are not waiting to be awakened.
41  ASSERT_MSG(thread->status == THREADSTATUS_WAIT_SYNCH_ANY ||
42  thread->status == THREADSTATUS_WAIT_SYNCH_ALL ||
43  thread->status == THREADSTATUS_WAIT_HLE_EVENT,
44  "Inconsistent thread statuses in waiting_threads");
45 
46  if (thread->current_priority >= candidate_priority)
47  continue;
48 
49  if (ShouldWait(thread.get()))
50  continue;
51 
52  // A thread is ready to run if it's either in THREADSTATUS_WAIT_SYNCH_ANY or
53  // in THREADSTATUS_WAIT_SYNCH_ALL and the rest of the objects it is waiting on are ready.
54  bool ready_to_run{true};
55  if (thread->status == THREADSTATUS_WAIT_SYNCH_ALL) {
56  ready_to_run = std::none_of(thread->wait_objects.begin(), thread->wait_objects.end(),
57  [&thread](const SharedPtr<WaitObject>& object) {
58  return object->ShouldWait(thread.get());
59  });
60  }
61 
62  if (ready_to_run) {
63  candidate = thread.get();
64  candidate_priority = thread->current_priority;
65  }
66  }
67 
68  return candidate;
69 }

71 void WaitObject::WakeupAllWaitingThreads() {
72  while (auto thread{GetHighestPriorityReadyThread()}) {
73  if (!thread->IsSleepingOnWaitAll()) {
74  Acquire(thread.get());
75  } else {
76  for (auto& object : thread->wait_objects) {
77  object->Acquire(thread.get());
78  }
79  }
80 
81  // Invoke the wakeup callback before clearing the wait objects
82  if (thread->wakeup_callback)
83  thread->wakeup_callback(ThreadWakeupReason::Signal, thread, this);
84 
85  for (auto& object : thread->wait_objects)
86  object->RemoveWaitingThread(thread.get());
87  thread->wait_objects.clear();
88 
89  thread->ResumeFromWait();
90  }
