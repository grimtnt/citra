26 static void ConvertYUVToRGB(InputFormat input_format, const u8* input_Y, const u8* input_U,
27  const u8* input_V, ImageTile output[], unsigned int width,
28  unsigned int height, const CoefficientSet& coefficients) {
29 
30  for (unsigned int y{}; y < height; ++y) {
31  for (unsigned int x{}; x < width; ++x) {
32  s32 Y{};
33  s32 U{};
34  s32 V{};
35  switch (input_format) {
36  case InputFormat::YUV422_Indiv8:
37  case InputFormat::YUV422_Indiv16:
38  Y = input_Y[y * width + x];
39  U = input_U[(y * width + x) / 2];
40  V = input_V[(y * width + x) / 2];
41  break;
42  case InputFormat::YUV420_Indiv8:
43  case InputFormat::YUV420_Indiv16:
44  Y = input_Y[y * width + x];
45  U = input_U[((y / 2) * width + x) / 2];
46  V = input_V[((y / 2) * width + x) / 2];
47  break;
48  case InputFormat::YUYV422_Interleaved:
49  Y = input_Y[(y * width + x) * 2];
50  U = input_Y[(y * width + (x / 2) * 2) * 2 + 1];
51  V = input_Y[(y * width + (x / 2) * 2) * 2 + 3];
52  break;
53  }
54 
55  // This conversion process is bit-exact with hardware, as far as could be tested.
56  auto& c = coefficients;
57  s32 cY = c[0] * Y;
58 
59  s32 r = cY + c[1] * V;
60  s32 g = cY - c[2] * V - c[3] * U;
61  s32 b = cY + c[4] * U;
62 
63  const s32 rounding_offset = 0x18;
64  r = (r >> 3) + c[5] + rounding_offset;
65  g = (g >> 3) + c[6] + rounding_offset;
66  b = (b >> 3) + c[7] + rounding_offset;
67 
68  unsigned int tile = x / 8;
69  unsigned int tile_x = x % 8;
70  u32* out = &output[tile][y * 8 + tile_x];
71  *out = ((u32)std::clamp(r >> 5, 0, 0xFF) << 24) |
72  ((u32)std::clamp(g >> 5, 0, 0xFF) << 16) |
73  ((u32)std::clamp(b >> 5, 0, 0xFF) << 8);
74  }
75  }

81 static void ReceiveData(u8* output, ConversionBuffer& buf, std::size_t amount_of_data) {
82  const u8* input = Memory::GetPointer(buf.address);
83 
84  std::size_t output_unit = buf.transfer_unit / N;
85  ASSERT(amount_of_data % output_unit == 0);
86 
87  while (amount_of_data > 0) {
88  for (std::size_t i{}; i < output_unit; ++i) {
89  output[i] = input[i * N];
90  }
91 
92  output += output_unit;
93  input += buf.transfer_unit + buf.gap;
94 
95  buf.address += buf.transfer_unit + buf.gap;
96  buf.image_size -= buf.transfer_unit;
97  amount_of_data -= output_unit;
98  }
99 }

103 static void SendData(const u32* input, ConversionBuffer& buf, int amount_of_data,
104  OutputFormat output_format, u8 alpha) {
105 
106  u8* output = Memory::GetPointer(buf.address);
107 
108  while (amount_of_data > 0) {
109  u8* unit_end = output + buf.transfer_unit;
110  while (output < unit_end) {
111  u32 color = *input++;
112  Math::Vec4<u8> col_vec{(u8)(color >> 24), (u8)(color >> 16), (u8)(color >> 8), alpha};
113 
114  switch (output_format) {
115  case OutputFormat::RGBA8:
116  Color::EncodeRGBA8(col_vec, output);
117  output += 4;
118  break;
119  case OutputFormat::RGB8:
120  Color::EncodeRGB8(col_vec, output);
121  output += 3;
122  break;
123  case OutputFormat::RGB5A1:
124  Color::EncodeRGB5A1(col_vec, output);
125  output += 2;
126  break;
127  case OutputFormat::RGB565:
128  Color::EncodeRGB565(col_vec, output);
129  output += 2;
130  break;
131  }
132 
133  amount_of_data -= 1;
134  }
135 
136  output += buf.gap;
137  buf.address += buf.transfer_unit + buf.gap;
138  buf.image_size -= buf.transfer_unit;
139  }

261 void PerformConversion(ConversionConfiguration& cvt) {
262  ASSERT(cvt.input_line_width % 8 == 0);
263  ASSERT(cvt.block_alignment != BlockAlignment::Block8x8 || cvt.input_lines % 8 == 0);
264  // Tiles per row
265  std::size_t num_tiles = cvt.input_line_width / 8;
266  ASSERT(num_tiles <= MAX_TILES);
267 
268  // Buffer used as a CDMA source/target.
269  std::unique_ptr<u8[]> data_buffer(new u8[cvt.input_line_width * 8 * 4]);
270  // Intermediate storage for decoded 8x8 image tiles. Always stored as RGB32.
271  std::unique_ptr<ImageTile[]> tiles(new ImageTile[num_tiles]);
272  ImageTile tmp_tile;
273 
274  // LUT used to remap writes to a tile. Used to allow linear or swizzled output without
275  // requiring two different code paths.
276  const u8* tile_remap{};
277  switch (cvt.block_alignment) {
278  case BlockAlignment::Linear:
279  tile_remap = linear_lut;
280  break;
281  case BlockAlignment::Block8x8:
282  tile_remap = morton_lut;
283  break;
284  }
285 
286  for (unsigned int y{}; y < cvt.input_lines; y += 8) {
287  unsigned int row_height = std::min(cvt.input_lines - y, 8u);
288 
289  // Total size in pixels of incoming data required for this strip.
290  const std::size_t row_data_size = row_height * cvt.input_line_width;
291 
292  u8* input_Y = data_buffer.get();
293  u8* input_U = input_Y + 8 * cvt.input_line_width;
294  u8* input_V = input_U + 8 * cvt.input_line_width / 2;
295 
296  switch (cvt.input_format) {
297  case InputFormat::YUV422_Indiv8:
298  ReceiveData<1>(input_Y, cvt.src_Y, row_data_size);
299  ReceiveData<1>(input_U, cvt.src_U, row_data_size / 2);
300  ReceiveData<1>(input_V, cvt.src_V, row_data_size / 2);
301  break;
302  case InputFormat::YUV420_Indiv8:
303  ReceiveData<1>(input_Y, cvt.src_Y, row_data_size);
304  ReceiveData<1>(input_U, cvt.src_U, row_data_size / 4);
305  ReceiveData<1>(input_V, cvt.src_V, row_data_size / 4);
306  break;
307  case InputFormat::YUV422_Indiv16:
308  ReceiveData<2>(input_Y, cvt.src_Y, row_data_size);
309  ReceiveData<2>(input_U, cvt.src_U, row_data_size / 2);
310  ReceiveData<2>(input_V, cvt.src_V, row_data_size / 2);
311  break;
312  case InputFormat::YUV420_Indiv16:
313  ReceiveData<2>(input_Y, cvt.src_Y, row_data_size);
314  ReceiveData<2>(input_U, cvt.src_U, row_data_size / 4);
315  ReceiveData<2>(input_V, cvt.src_V, row_data_size / 4);
316  break;
317  case InputFormat::YUYV422_Interleaved:
318  input_U = nullptr;
319  input_V = nullptr;
320  ReceiveData<1>(input_Y, cvt.src_YUYV, row_data_size * 2);
321  break;
322  }
323 
324  // Note(yuriks): If additional optimization is required, input_format can be moved to a
325  // template parameter, so that its dispatch can be moved to outside the inner loop.
326  ConvertYUVToRGB(cvt.input_format, input_Y, input_U, input_V, tiles.get(),
327  cvt.input_line_width, row_height, cvt.coefficients);
328 
329  u32* output_buffer = reinterpret_cast<u32*>(data_buffer.get());
330 
331  for (std::size_t i{}; i < num_tiles; ++i) {
332  int image_strip_width{};
333  int output_stride{};
334 
335  switch (cvt.rotation) {
336  case Rotation::None:
337  RotateTile0(tiles[i], tmp_tile, row_height, tile_remap);
338  image_strip_width = cvt.input_line_width;
339  output_stride = 8;
340  break;
341  case Rotation::Clockwise_90:
342  RotateTile90(tiles[i], tmp_tile, row_height, tile_remap);
343  image_strip_width = 8;
344  output_stride = 8 * row_height;
345  break;
346  case Rotation::Clockwise_180:
347  // For 180 and 270 degree rotations we also invert the order of tiles in the strip,
348  // since the rotates are done individually on each tile.
349  RotateTile180(tiles[num_tiles - i - 1], tmp_tile, row_height, tile_remap);
350  image_strip_width = cvt.input_line_width;
351  output_stride = 8;
352  break;
353  case Rotation::Clockwise_270:
354  RotateTile270(tiles[num_tiles - i - 1], tmp_tile, row_height, tile_remap);
355  image_strip_width = 8;
356  output_stride = 8 * row_height;
357  break;
358  }
359 
360  switch (cvt.block_alignment) {
361  case BlockAlignment::Linear:
362  WriteTileToOutput(output_buffer, tmp_tile, row_height, image_strip_width);
363  output_buffer += output_stride;
364  break;
365  case BlockAlignment::Block8x8:
366  WriteTileToOutput(output_buffer, tmp_tile, 8, 8);
367  output_buffer += TILE_SIZE;
368  break;
369  }
370  }
371 
372  // Note(yuriks): If additional optimization is required, output_format can be moved to a
373  // template parameter, so that its dispatch can be moved to outside the inner loop.
374  SendData(reinterpret_cast<u32*>(data_buffer.get()), cvt.dst, (int)row_data_size,
375  cvt.output_format, (u8)cvt.alpha);
376  }
377 }
