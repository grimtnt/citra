94 void Room::RoomImpl::ServerLoop() {
95  while (state != State::Closed) {
96  ENetEvent event;
97  if (enet_host_service(server, &event, 50) > 0) {
98  switch (event.type) {
99  case ENET_EVENT_TYPE_RECEIVE:
100  switch (event.packet->data[0]) {
101  case IdJoinRequest:
102  HandleJoinRequest(&event);
103  break;
104  case IdWifiPacket:
105  HandleWifiPacket(&event);
106  break;
107  }
108  enet_packet_destroy(event.packet);
109  break;
110  case ENET_EVENT_TYPE_DISCONNECT:
111  HandleClientDisconnection(event.peer);
112  break;
113  case ENET_EVENT_TYPE_NONE:
114  case ENET_EVENT_TYPE_CONNECT:
115  break;
116  }
117  }
118  }
119  // Close the connection to all members:
120  SendCloseMessage();
121 }

127 void Room::RoomImpl::HandleJoinRequest(const ENetEvent* event) {
128  Packet packet{};
129  packet.Append(event->packet->data, event->packet->dataLength);
130  packet.IgnoreBytes(sizeof(u8)); // Ignore the message type
131 
132  MacAddress preferred_mac;
133  packet >> preferred_mac;
134 
135  if (preferred_mac != NoPreferredMac) {
136  // Verify if the preferred mac is available
137  if (!IsMacAddressAvailable(preferred_mac)) {
138  SendMacCollision(event->peer);
139  return;
140  }
141  } else {
142  // Assign a MAC address of this client automatically
143  preferred_mac = GenerateMacAddress();
144  }
145 
146  // At this point the client is ready to be added to the room.
147  Member member{};
148  member.mac_address = preferred_mac;
149  member.peer = event->peer;
150 
151  {
152  std::lock_guard<std::mutex> lock{member_mutex};
153  members.push_back(std::move(member));
154  }
155 
156  SendJoinSuccess(event->peer, preferred_mac);
157 }

185 void Room::RoomImpl::SendCloseMessage() {
186  Packet packet{};
187  packet << static_cast<u8>(IdCloseRoom);
188  std::lock_guard<std::mutex> lock{member_mutex};
189  if (!members.empty()) {
190  ENetPacket* enet_packet{
191  enet_packet_create(packet.GetData(), packet.GetDataSize(), ENET_PACKET_FLAG_RELIABLE)};
192  for (auto& member : members) {

202 MacAddress Room::RoomImpl::GenerateMacAddress() {
203  MacAddress result_mac{
204  NintendoOUI}; // The first three bytes of each MAC address will be the NintendoOUI
205  std::uniform_int_distribution<> dis(0x00, 0xFF); // Random byte between 0 and 0xFF
206  do {
207  for (std::size_t i{3}; i < result_mac.size(); ++i) {
208  result_mac[i] = dis(random_gen);
209  }
210  } while (!IsMacAddressAvailable(result_mac));
211  return result_mac;
212 }

214 void Room::RoomImpl::HandleWifiPacket(const ENetEvent* event) {
215  Packet in_packet{};
216  in_packet.Append(event->packet->data, event->packet->dataLength);
217  in_packet.IgnoreBytes(sizeof(u8)); // Message type
218  in_packet.IgnoreBytes(sizeof(u8)); // WifiPacket Type
219  in_packet.IgnoreBytes(sizeof(u8)); // WifiPacket Channel
220  in_packet.IgnoreBytes(sizeof(MacAddress)); // WifiPacket Transmitter Address
221  MacAddress destination_address;
222  in_packet >> destination_address;
223 
224  Packet out_packet{};
225  out_packet.Append(event->packet->data, event->packet->dataLength);
226  ENetPacket* enet_packet{enet_packet_create(out_packet.GetData(), out_packet.GetDataSize(),
227  ENET_PACKET_FLAG_RELIABLE)};
228 
229  if (destination_address == BroadcastMac) { // Send the data to everyone except the sender
230  std::lock_guard<std::mutex> lock{member_mutex};
231  bool sent_packet{};
232  for (const auto& member : members) {
233  if (member.peer != event->peer) {
234  sent_packet = true;
235  enet_peer_send(member.peer, 0, enet_packet);
236  }
237  }
238 
239  if (!sent_packet) {
240  enet_packet_destroy(enet_packet);
241  }
242  } else { // Send the data only to the destination client
243  std::lock_guard<std::mutex> lock{member_mutex};
244  auto member{std::find_if(members.begin(), members.end(),
245  [destination_address](const Member& member) -> bool {
246  return member.mac_address == destination_address;
247  })};
248  if (member != members.end()) {
249  enet_peer_send(member->peer, 0, enet_packet);
250  } else {
251  LOG_ERROR(Network,
252  "Attempting to send to unknown MAC address: "
253  "{:02X}:{:02X}:{:02X}:{:02X}:{:02X}:{:02X}",
254  destination_address[0], destination_address[1], destination_address[2],
255  destination_address[3], destination_address[4], destination_address[5]);
256  enet_packet_destroy(enet_packet);
257  }
258  }
259  enet_host_flush(server);
260 }

262 void Room::RoomImpl::HandleClientDisconnection(ENetPeer* client) {
263  // Remove the client from the members list.
264  {
265  std::lock_guard<std::mutex> lock{member_mutex};
266  members.erase(
267  std::remove_if(members.begin(), members.end(),
268  [client](const Member& member) { return member.peer == client; }),
269  members.end());
270  }
271 
272  enet_peer_disconnect(client, 0);
273 }
