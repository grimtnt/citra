25 bool VirtualMemoryArea::CanBeMergedWith(const VirtualMemoryArea& next) const {
26  ASSERT(base + size == next.base);
27  if (permissions != next.permissions || meminfo_state != next.meminfo_state ||
28  type != next.type) {
29  return false;
30  }
31  if (type == VMAType::AllocatedMemoryBlock &&
32  (backing_block != next.backing_block || offset + size != next.offset)) {
33  return false;
34  }
35  if (type == VMAType::BackingMemory && backing_memory + size != next.backing_memory) {
36  return false;
37  }
38  if (type == VMAType::MMIO && paddr + size != next.paddr) {
39  return false;
40  }
41  return true;
42 }

96 ResultVal<VAddr> VMManager::MapMemoryBlockToBase(VAddr base, u32 region_size,
97  std::shared_ptr<std::vector<u8>> block,
98  std::size_t offset, u32 size, MemoryState state) {
99 
100  // Find the first Free VMA.
101  VMAHandle vma_handle{std::find_if(vma_map.begin(), vma_map.end(), [&](const auto& vma) {
102  if (vma.second.type != VMAType::Free)
103  return false;
104 
105  VAddr vma_end = vma.second.base + vma.second.size;
106  return vma_end > base && vma_end >= base + size;
107  })};
108 
109  VAddr target{std::max(base, vma_handle->second.base)};
110 
111  // Do not try to allocate the block if there are no available addresses within the desired
112  // region.
113  if (vma_handle == vma_map.end() || target + size > base + region_size) {
114  return ResultCode(ErrorDescription::OutOfMemory, ErrorModule::Kernel,
115  ErrorSummary::OutOfResource, ErrorLevel::Permanent);
116  }
117 
118  auto result{MapMemoryBlock(target, block, offset, size, state)};
119 
120  if (result.Failed())
121  return result.Code();
122 
123  return MakeResult<VAddr>(target);
124 }

162 ResultCode VMManager::ChangeMemoryState(VAddr target, u32 size, MemoryState expected_state,
163  VMAPermission expected_perms, MemoryState new_state,
164  VMAPermission new_perms) {
165  VAddr target_end{target + size};
166  VMAIter begin_vma{StripIterConstness(FindVMA(target))};
167  VMAIter i_end{vma_map.lower_bound(target_end)};
168 
169  if (begin_vma == vma_map.end())
170  return ERR_INVALID_ADDRESS;
171 
172  for (auto i = begin_vma; i != i_end; ++i) {
173  auto& vma{i->second};
174  if (vma.meminfo_state != expected_state) {
175  return ERR_INVALID_ADDRESS_STATE;
176  }
177  u32 perms{static_cast<u32>(expected_perms)};
178  if ((static_cast<u32>(vma.permissions) & perms) != perms) {
179  return ERR_INVALID_ADDRESS_STATE;
180  }
181  }
182 
183  CASCADE_RESULT(auto vma, CarveVMARange(target, size));
184  ASSERT(vma->second.size == size);
185 
186  vma->second.permissions = new_perms;
187  vma->second.meminfo_state = new_state;
188  UpdatePageTableForVMA(vma->second);
189 
190  MergeAdjacent(vma);
191 
192  return RESULT_SUCCESS;
193 }

211 ResultCode VMManager::UnmapRange(VAddr target, u32 size) {
212  CASCADE_RESULT(VMAIter vma, CarveVMARange(target, size));
213  const VAddr target_end{target + size};
214 
215  const VMAIter end{vma_map.end()};
216  // The comparison against the end of the range must be done using addresses since VMAs can be
217  // merged during this process, causing invalidation of the iterators.
218  while (vma != end && vma->second.base < target_end) {
219  vma = std::next(Unmap(vma));
220  }
221 
222  ASSERT(FindVMA(target)->second.size >= size);
223  return RESULT_SUCCESS;
224 }

279 ResultVal<VMManager::VMAIter> VMManager::CarveVMA(VAddr base, u32 size) {
280  ASSERT_MSG((size & Memory::PAGE_MASK) == 0, "non-page aligned size: {:#10X}", size);
281  ASSERT_MSG((base & Memory::PAGE_MASK) == 0, "non-page aligned base: {:#010X}", base);
282 
283  VMAIter vma_handle{StripIterConstness(FindVMA(base))};
284  if (vma_handle == vma_map.end()) {
285  // Target address is outside the range managed by the kernel
286  return ERR_INVALID_ADDRESS;
287  }
288 
289  const VirtualMemoryArea& vma{vma_handle->second};
290  if (vma.type != VMAType::Free) {
291  // Region is already allocated
292  return ERR_INVALID_ADDRESS_STATE;
293  }
294 
295  const VAddr start_in_vma{base - vma.base};
296  const VAddr end_in_vma{start_in_vma + size};
297 
298  if (end_in_vma > vma.size) {
299  // Requested allocation doesn't fit inside VMA
300  return ERR_INVALID_ADDRESS_STATE;
301  }
302 
303  if (end_in_vma != vma.size) {
304  // Split VMA at the end of the allocated region
305  SplitVMA(vma_handle, end_in_vma);
306  }
307  if (start_in_vma != 0) {
308  // Split VMA at the start of the allocated region
309  vma_handle = SplitVMA(vma_handle, start_in_vma);
310  }
311 
312  return MakeResult<VMAIter>(vma_handle);
313 }

315 ResultVal<VMManager::VMAIter> VMManager::CarveVMARange(VAddr target, u32 size) {
316  ASSERT_MSG((size & Memory::PAGE_MASK) == 0, "non-page aligned size: {:#10X}", size);
317  ASSERT_MSG((target & Memory::PAGE_MASK) == 0, "non-page aligned base: {:#010X}", target);
318 
319  const VAddr target_end{target + size};
320  ASSERT(target_end >= target);
321  ASSERT(target_end <= MAX_ADDRESS);
322  ASSERT(size > 0);
323 
324  VMAIter begin_vma{StripIterConstness(FindVMA(target))};
325  const VMAIter i_end{vma_map.lower_bound(target_end)};
326  if (std::any_of(begin_vma, i_end,
327  [](const auto& entry) { return entry.second.type == VMAType::Free; })) {
328  return ERR_INVALID_ADDRESS_STATE;
329  }
330 
331  if (target != begin_vma->second.base) {
332  begin_vma = SplitVMA(begin_vma, target - begin_vma->second.base);
333  }
334 
335  VMAIter end_vma{StripIterConstness(FindVMA(target_end))};
336  if (end_vma != vma_map.end() && target_end != end_vma->second.base) {
337  end_vma = SplitVMA(end_vma, target_end - end_vma->second.base);
338  }
339 
340  return MakeResult<VMAIter>(begin_vma);
341 }

343 VMManager::VMAIter VMManager::SplitVMA(VMAIter vma_handle, u32 offset_in_vma) {
344  VirtualMemoryArea& old_vma{vma_handle->second};
345  VirtualMemoryArea new_vma{old_vma}; // Make a copy of the VMA
346 
347  // For now, don't allow no-op VMA splits (trying to split at a boundary) because it's probably
348  // a bug. This restriction might be removed later.
349  ASSERT(offset_in_vma < old_vma.size);
350  ASSERT(offset_in_vma > 0);
351 
352  old_vma.size = offset_in_vma;
353  new_vma.base += offset_in_vma;
354  new_vma.size -= offset_in_vma;
355 
356  switch (new_vma.type) {
357  case VMAType::Free:
358  break;
359  case VMAType::AllocatedMemoryBlock:
360  new_vma.offset += offset_in_vma;
361  break;
362  case VMAType::BackingMemory:
363  new_vma.backing_memory += offset_in_vma;
364  break;
365  case VMAType::MMIO:
366  new_vma.paddr += offset_in_vma;
367  break;
368  }
369 
370  ASSERT(old_vma.CanBeMergedWith(new_vma));
371 
372  return vma_map.emplace_hint(std::next(vma_handle), new_vma.base, new_vma);
373 }

375 VMManager::VMAIter VMManager::MergeAdjacent(VMAIter iter) {
376  const VMAIter next_vma{std::next(iter)};
377  if (next_vma != vma_map.end() && iter->second.CanBeMergedWith(next_vma->second)) {
378  iter->second.size += next_vma->second.size;
379  vma_map.erase(next_vma);
380  }
381 
382  if (iter != vma_map.begin()) {
383  VMAIter prev_vma{std::prev(iter)};
384  if (prev_vma->second.CanBeMergedWith(iter->second)) {
385  prev_vma->second.size += iter->second.size;
386  vma_map.erase(iter);
387  iter = prev_vma;
388  }
389  }
390 
391  return iter;
392 }

394 void VMManager::UpdatePageTableForVMA(const VirtualMemoryArea& vma) {
395  switch (vma.type) {
396  case VMAType::Free:
397  Memory::UnmapRegion(page_table, vma.base, vma.size);
398  break;
399  case VMAType::AllocatedMemoryBlock:
400  Memory::MapMemoryRegion(page_table, vma.base, vma.size,
401  vma.backing_block->data() + vma.offset);
402  break;
403  case VMAType::BackingMemory:
404  Memory::MapMemoryRegion(page_table, vma.base, vma.size, vma.backing_memory);
405  break;
406  case VMAType::MMIO:
407  Memory::MapIoRegion(page_table, vma.base, vma.size, vma.mmio_handler);
408  break;
409  }
