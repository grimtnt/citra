165 void ExtraHID::HandleConfigureHIDPollingRequest(const std::vector<u8>& request) {
166  if (request.size() != 3) {
167  LOG_ERROR(Service_IR, "Wrong request size ({}): {}", request.size(),
168  Common::ArrayToString(request.data(), request.size()));
169  return;
170  }
171 
172  // Change HID input polling interval
173  CoreTiming::UnscheduleEvent(hid_polling_callback_id, 0);
174  hid_period = request[1];
175  CoreTiming::ScheduleEvent(msToCycles(hid_period), hid_polling_callback_id);
176 }

178 void ExtraHID::HandleReadCalibrationDataRequest(const std::vector<u8>& request_buf) {
179  struct ReadCalibrationDataRequest {
180  RequestID request_id;
181  u8 expected_response_time;
182  u16_le offset;
183  u16_le size;
184  };
185  static_assert(sizeof(ReadCalibrationDataRequest) == 6,
186  "ReadCalibrationDataRequest has wrong size");
187 
188  if (request_buf.size() != sizeof(ReadCalibrationDataRequest)) {
189  LOG_ERROR(Service_IR, "Wrong request size ({}): {}", request_buf.size(),
190  Common::ArrayToString(request_buf.data(), request_buf.size()));
191  return;
192  }
193 
194  ReadCalibrationDataRequest request;
195  std::memcpy(&request, request_buf.data(), sizeof(request));
196 
197  const u16 offset = Common::AlignDown(request.offset, 16);
198  const u16 size = Common::AlignDown(request.size, 16);
199 
200  if (offset + size > calibration_data.size()) {
201  LOG_ERROR(Service_IR, "Read beyond the end of calibration data! (offset={}, size={})",
202  offset, size);
203  return;
204  }
205 
206  std::vector<u8> response(5);
207  response[0] = static_cast<u8>(ResponseID::ReadCalibrationData);
208  std::memcpy(&response[1], &request.offset, sizeof(request.offset));
209  std::memcpy(&response[3], &request.size, sizeof(request.size));
210  response.insert(response.end(), calibration_data.begin() + offset,
211  calibration_data.begin() + offset + size);
212  Send(response);
213 }

215 void ExtraHID::OnReceive(const std::vector<u8>& data) {
216  switch (static_cast<RequestID>(data[0])) {
217  case RequestID::ConfigureHIDPolling:
218  HandleConfigureHIDPollingRequest(data);
219  break;
220  case RequestID::ReadCalibrationData:
221  HandleReadCalibrationDataRequest(data);
222  break;
223  default:
224  LOG_ERROR(Service_IR, "Unknown request: {}",
225  Common::ArrayToString(data.data(), data.size()));
226  break;
227  }
228 }

230 void ExtraHID::SendHIDStatus() {
231  if (is_device_reload_pending.exchange(false))
232  LoadInputDevices();
233 
234  constexpr int C_STICK_CENTER{0x800};
235  // TODO(wwylele): this value is not accurately measured. We currently assume that the axis can
236  // take values in the whole range of a 12-bit integer.
237  constexpr int C_STICK_RADIUS{0x7FF};
238 
239  float x, y;
240  std::tie(x, y) = c_stick->GetStatus();
241 
242  ExtraHIDResponse response;
243  response.c_stick.header.Assign(static_cast<u8>(ResponseID::PollHID));
244  response.c_stick.c_stick_x.Assign(static_cast<u32>(C_STICK_CENTER + C_STICK_RADIUS * x));
245  response.c_stick.c_stick_y.Assign(static_cast<u32>(C_STICK_CENTER + C_STICK_RADIUS * y));
246  response.buttons.battery_level.Assign(0x1F);
247  response.buttons.zl_not_held.Assign(!zl->GetStatus());
248  response.buttons.zr_not_held.Assign(!zr->GetStatus());
249  response.buttons.r_not_held.Assign(1);
250  response.unknown = 0;
251 
252  Core::Movie::GetInstance().HandleExtraHidResponse(response);
253 
254  std::vector<u8> response_buffer(sizeof(response));
255  memcpy(response_buffer.data(), &response, sizeof(response));
256  Send(response_buffer);
257 }
