72 static u64 GetTitleIdForApplet(AppletId id) {
73  ASSERT_MSG(id != AppletId::None, "Invalid applet id");
74 
75  auto itr{std::find_if(applet_titleids.begin(), applet_titleids.end(),
76  [id](const AppletTitleData& data) {
77  return data.applet_ids[0] == id || data.applet_ids[1] == id;
78  })};
79 
80  ASSERT_MSG(itr != applet_titleids.end(), "Unknown applet id {:#05X}", static_cast<u32>(id));
81 
82  return itr->title_ids[CFG::GetCurrentModule()->GetRegionValue()];
83 }

85 AppletManager::AppletSlotData* AppletManager::GetAppletSlotData(AppletId id) {
86  auto GetSlot{[this](AppletSlot slot) -> AppletSlotData* {
87  return &applet_slots[static_cast<std::size_t>(slot)];
88  }};
89 
90  if (id == AppletId::Application) {
91  auto* slot = GetSlot(AppletSlot::Application);
92  if (slot->applet_id != AppletId::None)
93  return slot;
94 
95  return nullptr;
96  }
97 
98  if (id == AppletId::AnySystemApplet) {
99  auto* system_slot{GetSlot(AppletSlot::SystemApplet)};
100  if (system_slot->applet_id != AppletId::None)
101  return system_slot;
102 
103  // The Home Menu is also a system applet, but it lives in its own slot to be able to run
104  // concurrently with other system applets.
105  auto* home_slot{GetSlot(AppletSlot::HomeMenu)};
106  if (home_slot->applet_id != AppletId::None)
107  return home_slot;
108 
109  return nullptr;
110  }
111 
112  if (id == AppletId::AnyLibraryApplet || id == AppletId::AnySysLibraryApplet) {
113  auto* slot{GetSlot(AppletSlot::LibraryApplet)};
114  if (slot->applet_id == AppletId::None)
115  return nullptr;
116 
117  u32 applet_pos{slot->attributes.applet_pos};
118 
119  if (id == AppletId::AnyLibraryApplet && applet_pos == static_cast<u32>(AppletPos::Library))
120  return slot;
121 
122  if (id == AppletId::AnySysLibraryApplet &&
123  applet_pos == static_cast<u32>(AppletPos::SysLibrary))
124  return slot;
125 
126  return nullptr;
127  }
128 
129  if (id == AppletId::HomeMenu || id == AppletId::AlternateMenu) {
130  auto* slot{GetSlot(AppletSlot::HomeMenu)};
131  if (slot->applet_id != AppletId::None)
132  return slot;
133 
134  return nullptr;
135  }
136 
137  for (auto& slot : applet_slots) {
138  if (slot.applet_id == id)
139  return &slot;
140  }
141 
142  return nullptr;
143 }

145 AppletManager::AppletSlotData* AppletManager::GetAppletSlotData(AppletAttributes attributes) {
146  // Mapping from AppletPos to AppletSlot
147  static constexpr std::array<AppletSlot, 6> applet_position_slots = {
148  AppletSlot::Application, AppletSlot::LibraryApplet, AppletSlot::SystemApplet,
149  AppletSlot::LibraryApplet, AppletSlot::Error, AppletSlot::LibraryApplet};
150 
151  u32 applet_pos{attributes.applet_pos};
152  if (applet_pos >= applet_position_slots.size())
153  return nullptr;
154 
155  AppletSlot slot{applet_position_slots[applet_pos]};
156 
157  if (slot == AppletSlot::Error)
158  return nullptr;
159 
160  // The Home Menu is a system applet, however, it has its own applet slot so that it can run
161  // concurrently with other system applets.
162  if (slot == AppletSlot::SystemApplet && attributes.is_home_menu)
163  return &applet_slots[static_cast<std::size_t>(AppletSlot::HomeMenu)];
164 
165  return &applet_slots[static_cast<std::size_t>(slot)];
166 }

168 void AppletManager::CancelAndSendParameter(const MessageParameter& parameter) {
169  next_parameter = parameter;
170  // Signal the event to let the receiver know that a new parameter is ready to be read
171  auto* const slot_data{GetAppletSlotData(static_cast<AppletId>(parameter.destination_id))};
172  if (slot_data == nullptr) {
173  LOG_DEBUG(Service_APT, "No applet was registered with the id {:03X}",
174  static_cast<u32>(parameter.destination_id));
175  return;
176  }
177 
178  slot_data->parameter_event->Signal();

181 ResultCode AppletManager::SendParameter(const MessageParameter& parameter) {
182  // A new parameter can not be sent if the previous one hasn't been consumed yet
183  if (next_parameter) {
184  return ResultCode(ErrCodes::ParameterPresent, ErrorModule::Applet,
185  ErrorSummary::InvalidState, ErrorLevel::Status);
186  }
187  CancelAndSendParameter(parameter);
188  if (auto dest_applet{HLE::Applets::Applet::Get(parameter.destination_id)}) {
189  return dest_applet->ReceiveParameter(parameter);
190  } else {
191  return RESULT_SUCCESS;
192  }
193 }

195 ResultVal<MessageParameter> AppletManager::GlanceParameter(AppletId app_id) {
196  if (!next_parameter) {
197  return ResultCode(ErrorDescription::NoData, ErrorModule::Applet, ErrorSummary::InvalidState,
198  ErrorLevel::Status);
199  }
200 
201  if (next_parameter->destination_id != app_id) {
202  return ResultCode(ErrorDescription::NotFound, ErrorModule::Applet, ErrorSummary::NotFound,
203  ErrorLevel::Status);
204  }
205 
206  MessageParameter parameter{*next_parameter};
207 
208  // Note: The NS module always clears the DSPSleep and DSPWakeup signals even in GlanceParameter.
209  if (next_parameter->signal == SignalType::DspSleep ||
210  next_parameter->signal == SignalType::DspWakeup)
211  next_parameter = boost::none;
212 
213  return MakeResult<MessageParameter>(parameter);
214 }

225 bool AppletManager::CancelParameter(bool check_sender, AppletId sender_appid, bool check_receiver,
226  AppletId receiver_appid) {
227  bool cancellation_success = true;
228 
229  if (!next_parameter) {
230  cancellation_success = false;
231  } else {
232  if (check_sender && next_parameter->sender_id != sender_appid)
233  cancellation_success = false;
234 
235  if (check_receiver && next_parameter->destination_id != receiver_appid)
236  cancellation_success = false;
237  }
238 
239  if (cancellation_success)
240  next_parameter = boost::none;
241 
242  return cancellation_success;
243 }

245 ResultVal<AppletManager::InitializeResult> AppletManager::Initialize(AppletId app_id,
246  AppletAttributes attributes) {
247  auto* const slot_data = GetAppletSlotData(attributes);
248 
249  // Note: The real NS service does not check if the attributes value is valid before accessing
250  // the data in the array
251  ASSERT_MSG(slot_data, "Invalid application attributes");
252 
253  if (slot_data->registered) {
254  return ResultCode(ErrorDescription::AlreadyExists, ErrorModule::Applet,
255  ErrorSummary::InvalidState, ErrorLevel::Status);
256  }
257 
258  slot_data->applet_id = static_cast<AppletId>(app_id);
259  slot_data->attributes.raw = attributes.raw;
260 
261  if (slot_data->applet_id == AppletId::Application ||
262  slot_data->applet_id == AppletId::HomeMenu) {
263  // Initialize the APT parameter to wake up the application.
264  next_parameter.emplace();
265  next_parameter->signal = SignalType::Wakeup;
266  next_parameter->sender_id = AppletId::None;
267  next_parameter->destination_id = app_id;
268  // Not signaling the parameter event will cause the application (or Home Menu) to hang
269  // during startup. In the real console, it is usually the Kernel and Home Menu who cause NS
270  // to signal the HomeMenu and Application parameter events, respectively.
271  slot_data->parameter_event->Signal();
272  }
273 
274  return MakeResult<InitializeResult>(
275  {slot_data->notification_event, slot_data->parameter_event});
276 }

291 bool AppletManager::IsRegistered(AppletId app_id) {
292  const auto* slot_data = GetAppletSlotData(app_id);
293 
294  // Check if an LLE applet was registered first, then fallback to HLE applets
295  bool is_registered = slot_data && slot_data->registered;
296 
297  if (!is_registered) {
298  if (app_id == AppletId::AnyLibraryApplet) {
299  is_registered = HLE::Applets::IsLibraryAppletRunning();
300  } else if (auto applet = HLE::Applets::Applet::Get(app_id)) {
301  // The applet exists, set it as registered.
302  is_registered = true;
303  }
304  }
305 
306  return is_registered;

309 ResultCode AppletManager::PrepareToStartLibraryApplet(AppletId applet_id) {
310  // The real APT service returns an error if there's a pending APT parameter when this function
311  // is called.
312  if (next_parameter) {
313  return ResultCode(ErrCodes::ParameterPresent, ErrorModule::Applet,
314  ErrorSummary::InvalidState, ErrorLevel::Status);
315  }
316 
317  const auto& slot = applet_slots[static_cast<std::size_t>(AppletSlot::LibraryApplet)];
318 
319  if (slot.registered) {
320  return ResultCode(ErrorDescription::AlreadyExists, ErrorModule::Applet,
321  ErrorSummary::InvalidState, ErrorLevel::Status);

323 
324  auto process = NS::LaunchTitleImpl(FS::MediaType::NAND, GetTitleIdForApplet(applet_id));
325  if (process) {
326  return RESULT_SUCCESS;
327  }
328 
329  // If we weren't able to load the native applet title, try to fallback to an HLE implementation.
330  auto applet = HLE::Applets::Applet::Get(applet_id);
331  if (applet) {
332  LOG_WARNING(Service_APT, "applet has already been started id={:08X}",
333  static_cast<u32>(applet_id));
334  return RESULT_SUCCESS;
335  } else {
336  return HLE::Applets::Applet::Create(applet_id, shared_from_this());
337  }

440  std::vector<u8> buffer) {

442 
443  MessageParameter param;
444  // TODO(Subv): The destination id should be the "current applet slot id", which changes

446  // application, but it could be something else if a system applet is launched.
447  param.destination_id = AppletId::Application;
448  param.sender_id = slot.applet_id;

454 
455  if (library_applet_closing_command != SignalType::WakeupByPause) {
456  // TODO(Subv): Terminate the running applet title
457  slot.Reset();
