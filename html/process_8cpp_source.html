51 void Process::ParseKernelCaps(const u32* kernel_caps, std::size_t len) {
52  for (std::size_t i{}; i < len; ++i) {
53  u32 descriptor{kernel_caps[i]};
54  u32 type{descriptor >> 20};
55 
56  if (descriptor == 0xFFFFFFFF) {
57  // Unused descriptor entry
58  continue;
59  } else if ((type & 0xF00) == 0xE00) { // 0x0FFF
60  // Allowed interrupts list
61  LOG_WARNING(Loader, "ExHeader allowed interrupts list ignored");
62  } else if ((type & 0xF80) == 0xF00) { // 0x07FF
63  // Allowed syscalls mask
64  unsigned int index{((descriptor >> 24) & 7) * 24};
65  u32 bits{descriptor & 0xFFFFFF};
66 
67  while (bits && index < svc_access_mask.size()) {
68  svc_access_mask.set(index, bits & 1);
69  ++index;
70  bits >>= 1;
71  }
72  } else if ((type & 0xFF0) == 0xFE0) { // 0x00FF
73  // Handle table size
74  handle_table_size = descriptor & 0x3FF;
75  } else if ((type & 0xFF8) == 0xFF0) { // 0x007F
76  // Misc. flags
77  flags.raw = descriptor & 0xFFFF;
78  } else if ((type & 0xFFE) == 0xFF8) { // 0x001F
79  // Mapped memory range
80  if (i + 1 >= len || ((kernel_caps[i + 1] >> 20) & 0xFFE) != 0xFF8) {
81  LOG_WARNING(Loader, "Incomplete exheader memory range descriptor ignored.");
82  continue;
83  }
84  u32 end_desc{kernel_caps[i + 1]};
85  ++i; // Skip over the second descriptor on the next iteration
86 
87  AddressMapping mapping{};
88  mapping.address = descriptor << 12;
89  VAddr end_address{end_desc << 12};
90 
91  if (mapping.address < end_address) {
92  mapping.size = end_address - mapping.address;
93  } else {
94  mapping.size = 0;
95  }
96 
97  mapping.read_only = (descriptor & (1 << 20)) != 0;
98  mapping.unk_flag = (end_desc & (1 << 20)) != 0;
99 
100  address_mappings.push_back(mapping);
101  } else if ((type & 0xFFF) == 0xFFE) { // 0x000F
102  // Mapped memory page
103  AddressMapping mapping{};
104  mapping.address = descriptor << 12;
105  mapping.size = Memory::PAGE_SIZE;
106  mapping.read_only = false;
107  mapping.unk_flag = false;
108 
109  address_mappings.push_back(mapping);
110  } else if ((type & 0xFE0) == 0xFC0) { // 0x01FF
111  // Kernel version
112  kernel_version = descriptor & 0xFFFF;
113 
114  int minor{kernel_version & 0xFF};
115  int major{(kernel_version >> 8) & 0xFF};
116  LOG_INFO(Loader, "ExHeader kernel version: {}.{}", major, minor);
117  } else {
118  LOG_ERROR(Loader, "Unhandled kernel caps descriptor: 0x{:08X}", descriptor);
119  }
120  }
121 }

123 void Process::Run(s32 main_thread_priority, u32 stack_size) {
124  memory_region = GetMemoryRegion(flags.memory_region);
125 
126  auto MapSegment{
127  [&](CodeSet::Segment& segment, VMAPermission permissions, MemoryState memory_state) {
128  auto vma{vm_manager
129  .MapMemoryBlock(segment.addr, codeset->memory, segment.offset,
130  segment.size, memory_state)
131  .Unwrap()};
132  vm_manager.Reprotect(vma, permissions);
133  misc_memory_used += segment.size;
134  memory_region->used += segment.size;
135  }};
136 
137  // Map CodeSet segments
138  MapSegment(codeset->CodeSegment(), VMAPermission::ReadExecute, MemoryState::Code);
139  MapSegment(codeset->RODataSegment(), VMAPermission::Read, MemoryState::Code);
140  MapSegment(codeset->DataSegment(), VMAPermission::ReadWrite, MemoryState::Private);
141 
142  // Allocate and map stack
143  vm_manager
144  .MapMemoryBlock(Memory::HEAP_VADDR_END - stack_size,
145  std::make_shared<std::vector<u8>>(stack_size, 0), 0, stack_size,
146  MemoryState::Locked)
147  .Unwrap();
148  misc_memory_used += stack_size;
149  memory_region->used += stack_size;
150 
151  // Map special address mappings
152  MapSharedPages(vm_manager);
153  for (const auto& mapping : address_mappings) {
154  HandleSpecialMapping(vm_manager, mapping);
155  }
156 
157  status = ProcessStatus::Running;
158 
159  vm_manager.LogLayout(Log::Level::Debug);
160  Kernel::SetupMainThread(codeset->entrypoint, main_thread_priority, this);
161 }

179 ResultVal<VAddr> Process::HeapAllocate(VAddr target, u32 size, VMAPermission perms) {
180  if (target < Memory::HEAP_VADDR || target + size > Memory::HEAP_VADDR_END ||
181  target + size < target) {
182  return ERR_INVALID_ADDRESS;
183  }
184 
185  if (heap_memory == nullptr) {
186  // Initialize heap
187  heap_memory = std::make_shared<std::vector<u8>>();
188  heap_start = heap_end = target;
189  }
190 
191  // If necessary, expand backing vector to cover new heap extents.
192  if (target < heap_start) {
193  heap_memory->insert(begin(*heap_memory), heap_start - target, 0);
194  heap_start = target;
195  vm_manager.RefreshMemoryBlockMappings(heap_memory.get());
196  }
197  if (target + size > heap_end) {
198  heap_memory->insert(end(*heap_memory), (target + size) - heap_end, 0);
199  heap_end = target + size;
200  vm_manager.RefreshMemoryBlockMappings(heap_memory.get());
201  }
202  ASSERT(heap_end - heap_start == heap_memory->size());
203 
204  CASCADE_RESULT(auto vma, vm_manager.MapMemoryBlock(target, heap_memory, target - heap_start,
205  size, MemoryState::Private));
206  vm_manager.Reprotect(vma, perms);
207 
208  heap_used += size;
209  memory_region->used += size;
210 
211  return MakeResult<VAddr>(heap_end - size);
212 }

214 ResultCode Process::HeapFree(VAddr target, u32 size) {
215  if (target < Memory::HEAP_VADDR || target + size > Memory::HEAP_VADDR_END ||
216  target + size < target) {
217  return ERR_INVALID_ADDRESS;
218  }
219 
220  if (size == 0) {
221  return RESULT_SUCCESS;
222  }
223 
224  ResultCode result = vm_manager.UnmapRange(target, size);
225  if (result.IsError())
226  return result;
227 
228  heap_used -= size;
229  memory_region->used -= size;
230 
231  return RESULT_SUCCESS;

234 ResultVal<VAddr> Process::LinearAllocate(VAddr target, u32 size, VMAPermission perms) {
235  auto& linheap_memory{memory_region->linear_heap_memory};
236 
237  VAddr heap_end{GetLinearHeapBase() + (u32)linheap_memory->size()};
238  // Games and homebrew only ever seem to pass 0 here (which lets the kernel decide the address),
239  // but explicit addresses are also accepted and respected.
240  if (target == 0) {
241  target = heap_end;
242  }
243 
244  if (target < GetLinearHeapBase() || target + size > GetLinearHeapLimit() || target > heap_end ||
245  target + size < target) {
246 
247  return ERR_INVALID_ADDRESS;
248  }
249 
250  // Expansion of the linear heap is only allowed if you do an allocation immediately at its
251  // end. It's possible to free gaps in the middle of the heap and then reallocate them later,
252  // but expansions are only allowed at the end.
253  if (target == heap_end) {
254  linheap_memory->insert(linheap_memory->end(), size, 0);
255  vm_manager.RefreshMemoryBlockMappings(linheap_memory.get());
256  }
257 
258  // TODO(yuriks): As is, this lets processes map memory allocated by other processes from the
259  // same region. It is unknown if or how the 3DS kernel checks against this.
260  std::size_t offset{target - GetLinearHeapBase()};
261  CASCADE_RESULT(auto vma, vm_manager.MapMemoryBlock(target, linheap_memory, offset, size,
262  MemoryState::Continuous));
263  vm_manager.Reprotect(vma, perms);
264 
265  linear_heap_used += size;
266  memory_region->used += size;
267 
268  return MakeResult<VAddr>(target);
269 }

271 ResultCode Process::LinearFree(VAddr target, u32 size) {
272  auto& linheap_memory{memory_region->linear_heap_memory};
273 
274  if (target < GetLinearHeapBase() || target + size > GetLinearHeapLimit() ||
275  target + size < target) {
276 
277  return ERR_INVALID_ADDRESS;
278  }
279 
280  if (size == 0) {
281  return RESULT_SUCCESS;
282  }
283 
284  VAddr heap_end{GetLinearHeapBase() + (u32)linheap_memory->size()};
285  if (target + size > heap_end) {
286  return ERR_INVALID_ADDRESS_STATE;
287  }
288 
289  ResultCode result{vm_manager.UnmapRange(target, size)};
290  if (result.IsError())
291  return result;
292 
293  linear_heap_used -= size;
294  memory_region->used -= size;
295 
296  if (target + size == heap_end) {
297  // End of linear heap has been freed, so check what's the last allocated block in it and
298  // reduce the size.
299  auto vma{vm_manager.FindVMA(target)};
300  ASSERT(vma != vm_manager.vma_map.end());
301  ASSERT(vma->second.type == VMAType::Free);
302  VAddr new_end{vma->second.base};
303  if (new_end >= GetLinearHeapBase()) {
304  linheap_memory->resize(new_end - GetLinearHeapBase());
305  }
306  }
307 
308  return RESULT_SUCCESS;
309 }
