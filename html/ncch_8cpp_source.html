35 FileType AppLoader_NCCH::IdentifyType(FileUtil::IOFile& file) {
36  u32 magic{};
37  file.Seek(0x100, SEEK_SET);
38  if (1 != file.ReadArray<u32>(&magic, 1))
39  return FileType::Error;
40 
41  if (MakeMagic('N', 'C', 'S', 'D') == magic)
42  return FileType::CCI;
43 
44  if (MakeMagic('N', 'C', 'C', 'H') == magic)
45  return FileType::CXI;
46 
47  return FileType::Error;
48 }

63 ResultStatus AppLoader_NCCH::LoadExec(Kernel::SharedPtr<Kernel::Process>& process) {
64  using Kernel::CodeSet;
65  using Kernel::SharedPtr;
66 
67  if (!is_loaded)
68  return ResultStatus::ErrorNotLoaded;
69 
70  std::vector<u8> code{};
71  u64_le program_id{};
72  if (ResultStatus::Success == ReadCode(code) &&
73  ResultStatus::Success == ReadProgramId(program_id)) {
74  std::string process_name{Common::StringFromFixedZeroTerminatedBuffer(
75  (const char*)overlay_ncch->exheader_header.codeset_info.name, 8)};
76 
77  SharedPtr<CodeSet> codeset{CodeSet::Create(process_name, program_id)};
78 
79  codeset->CodeSegment().offset = 0;
80  codeset->CodeSegment().addr = overlay_ncch->exheader_header.codeset_info.text.address;
81  codeset->CodeSegment().size =
82  overlay_ncch->exheader_header.codeset_info.text.num_max_pages * Memory::PAGE_SIZE;
83 
84  codeset->RODataSegment().offset =
85  codeset->CodeSegment().offset + codeset->CodeSegment().size;
86  codeset->RODataSegment().addr = overlay_ncch->exheader_header.codeset_info.ro.address;
87  codeset->RODataSegment().size =
88  overlay_ncch->exheader_header.codeset_info.ro.num_max_pages * Memory::PAGE_SIZE;
89 
90  // TODO(yuriks): Not sure if the bss size is added to the page-aligned .data size or just
91  // to the regular size. Playing it safe for now.
92  u32 bss_page_size{(overlay_ncch->exheader_header.codeset_info.bss_size + 0xFFF) & ~0xFFF};
93  code.resize(code.size() + bss_page_size, 0);
94 
95  codeset->DataSegment().offset =
96  codeset->RODataSegment().offset + codeset->RODataSegment().size;
97  codeset->DataSegment().addr = overlay_ncch->exheader_header.codeset_info.data.address;
98  codeset->DataSegment().size =
99  overlay_ncch->exheader_header.codeset_info.data.num_max_pages * Memory::PAGE_SIZE +
100  bss_page_size;
101 
102  codeset->entrypoint = codeset->CodeSegment().addr;
103  codeset->memory = std::make_shared<std::vector<u8>>(std::move(code));
104 
105  process = Kernel::Process::Create(std::move(codeset));
106 
107  // Attach a resource limit to the process based on the resource limit category
108  process->resource_limit =
109  Kernel::ResourceLimit::GetForCategory(static_cast<Kernel::ResourceLimitCategory>(
110  overlay_ncch->exheader_header.arm11_system_local_caps.resource_limit_category));
111 
112  // Set the default CPU core for this process
113  process->ideal_processor =
114  overlay_ncch->exheader_header.arm11_system_local_caps.ideal_processor;
115 
116  // Copy data while converting endianness
117  std::array<u32, ARRAY_SIZE(overlay_ncch->exheader_header.arm11_kernel_caps.descriptors)>
118  kernel_caps;
119  std::copy_n(overlay_ncch->exheader_header.arm11_kernel_caps.descriptors, kernel_caps.size(),
120  begin(kernel_caps));
121  process->ParseKernelCaps(kernel_caps.data(), kernel_caps.size());
122 
123  s32 priority{overlay_ncch->exheader_header.arm11_system_local_caps.priority};
124  u32 stack_size{overlay_ncch->exheader_header.codeset_info.stack_size};
125  process->Run(priority, stack_size);
126  return ResultStatus::Success;
127  }
128  return ResultStatus::Error;
129 }

131 void AppLoader_NCCH::ParseRegionLockoutInfo() {
132  std::vector<u8> smdh_buffer{};
133  if (ReadIcon(smdh_buffer) == ResultStatus::Success && smdh_buffer.size() >= sizeof(SMDH)) {
134  SMDH smdh{};
135  memcpy(&smdh, smdh_buffer.data(), sizeof(SMDH));
136  u32 region_lockout = smdh.region_lockout;
137  constexpr u32 REGION_COUNT = 7;
138  for (u32 region{}; region < REGION_COUNT; ++region) {
139  if (region_lockout & 1) {
140  Service::CFG::GetCurrentModule()->SetPreferredRegionCode(region);
141  break;
142  }
143  region_lockout >>= 1;
144  }
145  }

148 ResultStatus AppLoader_NCCH::Load(Kernel::SharedPtr<Kernel::Process>& process) {
149  u64_le ncch_program_id{};
150 
151  if (is_loaded)
152  return ResultStatus::ErrorAlreadyLoaded;
153 
154  ResultStatus result{base_ncch.Load()};
155  if (result != ResultStatus::Success)
156  return result;
157 
158  ReadProgramId(ncch_program_id);
159  std::string program_id{fmt::format("{:016X}", ncch_program_id)};
160 
161  LOG_INFO(Loader, "Program ID: {}", program_id);
162 
163  update_ncch.OpenFile(Service::AM::GetTitleContentPath(Service::FS::MediaType::SDMC,
164  ncch_program_id | UPDATE_MASK));
165  result = update_ncch.Load();
166  if (result == ResultStatus::Success) {
167  overlay_ncch = &update_ncch;
168  }
169 
170  is_loaded = true; // Set state to loaded
171 
172  result = LoadExec(process); // Load the executable into memory for booting
173  if (ResultStatus::Success != result)
174  return result;
175 
176  Service::FS::RegisterSelfNCCH(*this);
177 
178  ParseRegionLockoutInfo();
179 
180  return ResultStatus::Success;
181 }

220 ResultStatus AppLoader_NCCH::ReadTitle(std::string& title) {
221  std::vector<u8> data{};
222  Loader::SMDH smdh{};
223  ReadIcon(data);
224 
225  if (!Loader::IsValidSMDH(data)) {
226  return ResultStatus::ErrorInvalidFormat;
227  }
228 
229  memcpy(&smdh, data.data(), sizeof(Loader::SMDH));
230 
231  const auto& short_title{smdh.GetShortTitle(SMDH::TitleLanguage::English)};
232  auto title_end{std::find(short_title.begin(), short_title.end(), u'\0')};
233  title = Common::UTF16ToUTF8(std::u16string{short_title.begin(), title_end});
234 
235  return ResultStatus::Success;
236 }
