29 CubebSink::CubebSink(std::string target_device_name) : impl(std::make_unique<Impl>()) {
30  if (cubeb_init(&impl->ctx, "Citra", nullptr) != CUBEB_OK) {
31  LOG_CRITICAL(Audio_Sink, "cubeb_init failed");
32  return;
33  }
34  cubeb_set_log_callback(CUBEB_LOG_NORMAL, &Impl::LogCallback);
35 
36  impl->sample_rate = native_sample_rate;
37 
38  cubeb_stream_params params;
39  params.rate = impl->sample_rate;
40  params.channels = 2;
41  params.layout = CUBEB_LAYOUT_STEREO;
42  params.format = CUBEB_SAMPLE_S16NE;
43  params.prefs = CUBEB_STREAM_PREF_NONE;
44 
45  u32 minimum_latency{100 * impl->sample_rate / 1000}; // Firefox default
46  if (cubeb_get_min_latency(impl->ctx, &params, &minimum_latency) != CUBEB_OK) {
47  LOG_CRITICAL(Audio_Sink, "Error getting minimum latency");
48  }
49 
50  cubeb_devid output_device{};
51  if (target_device_name != auto_device_name && !target_device_name.empty()) {
52  cubeb_device_collection collection;
53  if (cubeb_enumerate_devices(impl->ctx, CUBEB_DEVICE_TYPE_OUTPUT, &collection) != CUBEB_OK) {
54  LOG_WARNING(Audio_Sink, "Audio output device enumeration not supported");
55  } else {
56  const auto collection_end{collection.device + collection.count};
57  const auto device{std::find_if(collection.device, collection_end,
58  [&](const cubeb_device_info& device) {
59  return target_device_name == device.friendly_name;
60  })};
61  if (device != collection_end) {
62  output_device = device->devid;
63  }
64  cubeb_device_collection_destroy(impl->ctx, &collection);
65  }
66  }
67 
68  int stream_err = cubeb_stream_init(impl->ctx, &impl->stream, "CitraAudio", nullptr, nullptr,
69  output_device, &params, std::max(512u, minimum_latency),
70  &Impl::DataCallback, &Impl::StateCallback, impl.get());
71  if (stream_err != CUBEB_OK) {
72  switch (stream_err) {
73  case CUBEB_ERROR:
74  default:
75  LOG_CRITICAL(Audio_Sink, "Error initializing cubeb stream ({})", stream_err);
76  break;
77  case CUBEB_ERROR_INVALID_FORMAT:
78  LOG_CRITICAL(Audio_Sink, "Invalid format when initializing cubeb stream");
79  break;
80  case CUBEB_ERROR_DEVICE_UNAVAILABLE:
81  LOG_CRITICAL(Audio_Sink, "Device unavailable when initializing cubeb stream");
82  break;
83  }
84  return;
85  }
86 
87  if (cubeb_stream_start(impl->stream) != CUBEB_OK) {
88  LOG_CRITICAL(Audio_Sink, "Error starting cubeb stream");
89  return;
90  }
91 }

119 long CubebSink::Impl::DataCallback(cubeb_stream* stream, void* user_data, const void* input_buffer,
120  void* output_buffer, long num_frames) {
121  Impl* impl{static_cast<Impl*>(user_data)};
122  s16* buffer{reinterpret_cast<s16*>(output_buffer)};
123 
124  if (!impl || !impl->cb) {
125  LOG_DEBUG(Audio_Sink, "Emitting zeros");
126  std::memset(output_buffer, 0, num_frames * 2 * sizeof(s16));
127  return num_frames;
128  }
129 
130  impl->cb(buffer, num_frames);
131 
132  return num_frames;
133 }

166 std::vector<std::string> ListCubebSinkDevices() {
167  std::vector<std::string> device_list;
168  cubeb* ctx;
169 
170  if (cubeb_init(&ctx, "CitraEnumerator", nullptr) != CUBEB_OK) {
171  LOG_CRITICAL(Audio_Sink, "cubeb_init failed");
172  return {};
173  }
174 
175  cubeb_device_collection collection;
176  if (cubeb_enumerate_devices(ctx, CUBEB_DEVICE_TYPE_OUTPUT, &collection) != CUBEB_OK) {
177  LOG_WARNING(Audio_Sink, "Audio output device enumeration not supported");
178  } else {
179  for (std::size_t i{}; i < collection.count; i++) {
180  const cubeb_device_info& device{collection.device[i]};
181  if (device.friendly_name) {
182  device_list.emplace_back(device.friendly_name);
183  }
184  }
185  cubeb_device_collection_destroy(ctx, &collection);
186  }
187 
188  cubeb_destroy(ctx);
189  return device_list;
190 }
