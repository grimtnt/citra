18 void OutputVertex::ValidateSemantics(const RasterizerRegs& regs) {
19  unsigned int num_attributes{regs.vs_output_total};
20  ASSERT(num_attributes <= 7);
21  for (std::size_t attrib{}; attrib < num_attributes; ++attrib) {
22  u32 output_register_map{regs.vs_output_attributes[attrib].raw};
23  for (std::size_t comp{}; comp < 4; ++comp) {
24  u32 semantic{(output_register_map >> (8 * comp)) & 0x1F};
25  ASSERT_MSG(semantic < 24 || semantic == RasterizerRegs::VSOutputAttributes::INVALID,
26  "Invalid/unknown semantic id: {}", semantic);
27  }
28  }
29 }

31 OutputVertex OutputVertex::FromAttributeBuffer(const RasterizerRegs& regs,
32  const AttributeBuffer& input) {
33  // Setup output data
34  union {
35  OutputVertex ret{};
36  // Allow us to overflow OutputVertex to avoid branches, since
37  // RasterizerRegs::VSOutputAttributes::INVALID would write to slot 31, which
38  // would be out of bounds otherwise.
39  std::array<float24, 32> vertex_slots_overflow;
40  };
41 
42  // Assert that OutputVertex has enough space for 24 semantic registers
43  static_assert(sizeof(std::array<float24, 24>) == sizeof(ret),
44  "Struct and array have different sizes.");
45 
46  unsigned int num_attributes{regs.vs_output_total & 7};
47  for (std::size_t attrib{}; attrib < num_attributes; ++attrib) {
48  const auto output_register_map{regs.vs_output_attributes[attrib]};
49  vertex_slots_overflow[output_register_map.map_x] = input.attr[attrib][0];
50  vertex_slots_overflow[output_register_map.map_y] = input.attr[attrib][1];
51  vertex_slots_overflow[output_register_map.map_z] = input.attr[attrib][2];
52  vertex_slots_overflow[output_register_map.map_w] = input.attr[attrib][3];
53  }
54 
55  // The hardware takes the absolute and saturates vertex colors like this, *before* doing
56  // interpolation
57  for (unsigned i{}; i < 4; ++i) {
58  float c{std::fabs(ret.color[i].ToFloat32())};
59  ret.color[i] = float24::FromFloat32(c < 1.0f ? c : 1.0f);
60  }
61 
62  return ret;
63 }
