48 void Mutex::Acquire(Thread* thread) {
49  ASSERT_MSG(!ShouldWait(thread), "object unavailable!");
50 
51  // Actually "acquire" the mutex only if we don't already have it
52  if (lock_count == 0) {
53  priority = thread->current_priority;
54  thread->held_mutexes.insert(this);
55  holding_thread = thread;

63 ResultCode Mutex::Release(Thread* thread) {
64  // We can only release the mutex if it's held by the calling thread.
65  if (thread != holding_thread) {
66  if (holding_thread) {
67  LOG_ERROR(
68  Kernel,
69  "Tried to release a mutex (owned by thread id %u) from a different thread id {}",
70  holding_thread->thread_id, thread->thread_id);
71  }
72  return ResultCode(ErrCodes::WrongLockingThread, ErrorModule::Kernel,
73  ErrorSummary::InvalidArgument, ErrorLevel::Permanent);
74  }
75 
76  // Note: It should not be possible for the situation where the mutex has a holding thread with a
77  // zero lock count to occur. The real kernel still checks for this, so we do too.
78  if (lock_count <= 0)
79  return ResultCode(ErrorDescription::InvalidResultValue, ErrorModule::Kernel,
80  ErrorSummary::InvalidState, ErrorLevel::Permanent);
81 
82  lock_count--;
83 
84  // Yield to the next thread only if we've fully released the mutex
85  if (lock_count == 0) {
86  holding_thread->held_mutexes.erase(this);
87  holding_thread->UpdatePriority();
88  holding_thread = nullptr;
89  WakeupAllWaitingThreads();
90  Core::System::GetInstance().PrepareReschedule();
91  }
92 
93  return RESULT_SUCCESS;
94 }

108 void Mutex::UpdatePriority() {
109  if (!holding_thread)
110  return;
111 
112  u32 best_priority = THREADPRIO_LOWEST;
113  for (auto& waiter : GetWaitingThreads()) {
114  if (waiter->current_priority < best_priority)
115  best_priority = waiter->current_priority;
116  }
117 
118  if (best_priority != priority) {
119  priority = best_priority;
120  holding_thread->UpdatePriority();
121  }
122 }
