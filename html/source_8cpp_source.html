28 void Source::MixInto(QuadFrame32& dest, std::size_t intermediate_mix_id) const {
29  if (!state.enabled)
30  return;
31 
32  const std::array<float, 4>& gains{state.gain.at(intermediate_mix_id)};
33  for (std::size_t samplei{}; samplei < samples_per_frame; samplei++) {
34  // Conversion from stereo (current_frame) to quadraphonic (dest) occurs here.
35  dest[samplei][0] += static_cast<s32>(gains[0] * current_frame[samplei][0]);
36  dest[samplei][1] += static_cast<s32>(gains[1] * current_frame[samplei][1]);
37  dest[samplei][2] += static_cast<s32>(gains[2] * current_frame[samplei][0]);
38  dest[samplei][3] += static_cast<s32>(gains[3] * current_frame[samplei][1]);
39  }
40 }

47 void Source::ParseConfig(SourceConfiguration::Configuration& config,
48  const s16_le (&adpcm_coeffs)[16]) {
49  if (!config.dirty_raw) {
50  return;
51  }
52 
53  if (config.reset_flag) {
54  config.reset_flag.Assign(0);
55  Reset();
56  LOG_TRACE(Audio_DSP, "source_id={} reset", source_id);
57  }
58 
59  if (config.partial_reset_flag) {
60  config.partial_reset_flag.Assign(0);
61  state.input_queue = std::priority_queue<Buffer, std::vector<Buffer>, BufferOrder>{};
62  LOG_TRACE(Audio_DSP, "source_id={} partial_reset", source_id);
63  }
64 
65  if (config.enable_dirty) {
66  config.enable_dirty.Assign(0);
67  state.enabled = config.enable != 0;
68  LOG_TRACE(Audio_DSP, "source_id={} enable={}", source_id, state.enabled);
69  }
70 
71  if (config.sync_dirty) {
72  config.sync_dirty.Assign(0);
73  state.sync = config.sync;
74  LOG_TRACE(Audio_DSP, "source_id={} sync={}", source_id, state.sync);
75  }
76 
77  if (config.rate_multiplier_dirty) {
78  config.rate_multiplier_dirty.Assign(0);
79  state.rate_multiplier = config.rate_multiplier;
80  LOG_TRACE(Audio_DSP, "source_id={} rate={}", source_id, state.rate_multiplier);
81 
82  if (state.rate_multiplier <= 0) {
83  LOG_ERROR(Audio_DSP, "Was given an invalid rate multiplier: source_id={} rate={}",
84  source_id, state.rate_multiplier);
85  state.rate_multiplier = 1.0f;
86  // Note: Actual firmware starts producing garbage if this occurs.
87  }
88  }
89 
90  if (config.adpcm_coefficients_dirty) {
91  config.adpcm_coefficients_dirty.Assign(0);
92  std::transform(adpcm_coeffs, adpcm_coeffs + state.adpcm_coeffs.size(),
93  state.adpcm_coeffs.begin(),
94  [](const auto& coeff) { return static_cast<s16>(coeff); });
95  LOG_TRACE(Audio_DSP, "source_id={} adpcm update", source_id);
96  }
97 
98  if (config.gain_0_dirty) {
99  config.gain_0_dirty.Assign(0);
100  std::transform(config.gain[0], config.gain[0] + state.gain[0].size(), state.gain[0].begin(),
101  [](const auto& coeff) { return static_cast<float>(coeff); });
102  LOG_TRACE(Audio_DSP, "source_id={} gain 0 update", source_id);
103  }
104 
105  if (config.gain_1_dirty) {
106  config.gain_1_dirty.Assign(0);
107  std::transform(config.gain[1], config.gain[1] + state.gain[1].size(), state.gain[1].begin(),
108  [](const auto& coeff) { return static_cast<float>(coeff); });
109  LOG_TRACE(Audio_DSP, "source_id={} gain 1 update", source_id);
110  }
111 
112  if (config.gain_2_dirty) {
113  config.gain_2_dirty.Assign(0);
114  std::transform(config.gain[2], config.gain[2] + state.gain[2].size(), state.gain[2].begin(),
115  [](const auto& coeff) { return static_cast<float>(coeff); });
116  LOG_TRACE(Audio_DSP, "source_id={} gain 2 update", source_id);
117  }
118 
119  if (config.filters_enabled_dirty) {
120  config.filters_enabled_dirty.Assign(0);
121  state.filters.Enable(config.simple_filter_enabled.ToBool(),
122  config.biquad_filter_enabled.ToBool());
123  LOG_TRACE(Audio_DSP, "source_id={} enable_simple={} enable_biquad={}", source_id,
124  config.simple_filter_enabled.Value(), config.biquad_filter_enabled.Value());
125  }
126 
127  if (config.simple_filter_dirty) {
128  config.simple_filter_dirty.Assign(0);
129  state.filters.Configure(config.simple_filter);
130  LOG_TRACE(Audio_DSP, "source_id={} simple filter update", source_id);
131  }
132 
133  if (config.biquad_filter_dirty) {
134  config.biquad_filter_dirty.Assign(0);
135  state.filters.Configure(config.biquad_filter);
136  LOG_TRACE(Audio_DSP, "source_id={} biquad filter update", source_id);
137  }
138 
139  if (config.interpolation_dirty) {
140  config.interpolation_dirty.Assign(0);
141  state.interpolation_mode = config.interpolation_mode;
142  LOG_TRACE(Audio_DSP, "source_id={} interpolation_mode={}", source_id,
143  static_cast<std::size_t>(state.interpolation_mode));
144  }
145 
146  if (config.format_dirty || config.embedded_buffer_dirty) {
147  config.format_dirty.Assign(0);
148  state.format = config.format;
149  LOG_TRACE(Audio_DSP, "source_id={} format={}", source_id,
150  static_cast<std::size_t>(state.format));
151  }
152 
153  if (config.mono_or_stereo_dirty || config.embedded_buffer_dirty) {
154  config.mono_or_stereo_dirty.Assign(0);
155  state.mono_or_stereo = config.mono_or_stereo;
156  LOG_TRACE(Audio_DSP, "source_id={} mono_or_stereo={}", source_id,
157  static_cast<std::size_t>(state.mono_or_stereo));
158  }
159 
160  u32_dsp play_position = {};
161  if (config.play_position_dirty && config.play_position != 0) {
162  config.play_position_dirty.Assign(0);
163  play_position = config.play_position;
164  // play_position applies only to the embedded buffer, and defaults to 0 w/o a dirty bit
165  // This will be the starting sample for the first time the buffer is played.
166  }
167 
168  if (config.embedded_buffer_dirty) {
169  config.embedded_buffer_dirty.Assign(0);
170  state.input_queue.emplace(Buffer{
171  config.physical_address,
172  config.length,
173  static_cast<u8>(config.adpcm_ps),
174  {config.adpcm_yn[0], config.adpcm_yn[1]},
175  config.adpcm_dirty.ToBool(),
176  config.is_looping.ToBool(),
177  config.buffer_id,
178  state.mono_or_stereo,
179  state.format,
180  false,
181  play_position,
182  false,
183  });
184  LOG_TRACE(Audio_DSP, "enqueuing embedded addr={:#010x} len={} id={} start={}",
185  config.physical_address, config.length, config.buffer_id,
186  static_cast<u32>(config.play_position));
187  }
188 
189  if (config.loop_related_dirty && config.loop_related != 0) {
190  config.loop_related_dirty.Assign(0);
191  LOG_WARNING(Audio_DSP, "Unhandled complex loop with loop_related={:#010x}",
192  static_cast<u32>(config.loop_related));
193  }
194 
195  if (config.buffer_queue_dirty) {
196  config.buffer_queue_dirty.Assign(0);
197  for (std::size_t i{}; i < 4; i++) {
198  if (config.buffers_dirty & (1 << i)) {
199  const auto& b = config.buffers[i];
200  state.input_queue.emplace(Buffer{
201  b.physical_address,
202  b.length,
203  static_cast<u8>(b.adpcm_ps),
204  {b.adpcm_yn[0], b.adpcm_yn[1]},
205  b.adpcm_dirty != 0,
206  b.is_looping != 0,
207  b.buffer_id,
208  state.mono_or_stereo,
209  state.format,
210  true,
211  {}, // 0 in u32_dsp
212  false,
213  });
214  LOG_TRACE(Audio_DSP, "enqueuing queued {} addr={:#010x} len={} id={}", i,
215  b.physical_address, b.length, b.buffer_id);
216  }
217  }
218  config.buffers_dirty = 0;
219  }
220 
221  if (config.dirty_raw) {
222  LOG_DEBUG(Audio_DSP, "source_id={} remaining_dirty={:x}", source_id, config.dirty_raw);
223  }
224 
225  config.dirty_raw = 0;
226 }

228 void Source::GenerateFrame() {
229  current_frame.fill({});
230 
231  if (state.current_buffer.empty() && !DequeueBuffer()) {
232  state.enabled = false;
233  state.buffer_update = true;
234  state.current_buffer_id = 0;
235  return;
236  }
237 
238  std::size_t frame_position{};
239 
240  state.current_sample_number = state.next_sample_number;
241  while (frame_position < current_frame.size()) {
242  if (state.current_buffer.empty() && !DequeueBuffer()) {
243  break;
244  }
245 
246  switch (state.interpolation_mode) {
247  case InterpolationMode::None:
248  AudioInterp::None(state.interp_state, state.current_buffer, state.rate_multiplier,
249  current_frame, frame_position);
250  break;
251  case InterpolationMode::Linear:
252  AudioInterp::Linear(state.interp_state, state.current_buffer, state.rate_multiplier,
253  current_frame, frame_position);
254  break;
255  case InterpolationMode::Polyphase:
256  // TODO(merry): Implement polyphase interpolation
257  LOG_DEBUG(Audio_DSP, "Polyphase interpolation unimplemented; falling back to linear");
258  AudioInterp::Linear(state.interp_state, state.current_buffer, state.rate_multiplier,
259  current_frame, frame_position);
260  break;
261  default:
262  UNIMPLEMENTED();
263  break;
264  }
265  }
266  state.next_sample_number += static_cast<u32>(frame_position);
267 
268  state.filters.ProcessFrame(current_frame);
269 }

271 bool Source::DequeueBuffer() {
272  ASSERT_MSG(state.current_buffer.empty(),
273  "Shouldn't dequeue; we still have data in current_buffer");
274 
275  if (state.input_queue.empty())
276  return false;
277 
278  Buffer buf{state.input_queue.top()};
279  state.input_queue.pop();
280 
281  if (buf.adpcm_dirty) {
282  state.adpcm_state.yn1 = buf.adpcm_yn[0];
283  state.adpcm_state.yn2 = buf.adpcm_yn[1];
284  }
285 
286  const u8* const memory{Memory::GetPhysicalPointer(buf.physical_address)};
287  if (memory) {
288  const unsigned num_channels{
289  static_cast<unsigned>(buf.mono_or_stereo == MonoOrStereo::Stereo ? 2 : 1)};
290  switch (buf.format) {
291  case Format::PCM8:
292  state.current_buffer = Codec::DecodePCM8(num_channels, memory, buf.length);
293  break;
294  case Format::PCM16:
295  state.current_buffer = Codec::DecodePCM16(num_channels, memory, buf.length);
296  break;
297  case Format::ADPCM:
298  DEBUG_ASSERT(num_channels == 1);
299  state.current_buffer =
300  Codec::DecodeADPCM(memory, buf.length, state.adpcm_coeffs, state.adpcm_state);
301  break;
302  default:
303  UNIMPLEMENTED();
304  break;
305  }
306  } else {
307  LOG_WARNING(Audio_DSP,
308  "source_id={} buffer_id={} length={}: Invalid physical address {:#010x}",
309  source_id, buf.buffer_id, buf.length, buf.physical_address);
310  state.current_buffer.clear();
311  return true;
312  }
313 
314  // the first playthrough starts at play_position, loops start at the beginning of the buffer
315  state.current_sample_number = (!buf.has_played) ? buf.play_position : 0;
316  state.next_sample_number = state.current_sample_number;
317  state.current_buffer_id = buf.buffer_id;
318  state.buffer_update = buf.from_queue && !buf.has_played;
319 
320  if (buf.is_looping) {
321  buf.has_played = true;
322  state.input_queue.push(buf);
323  }
324 
325  LOG_TRACE(Audio_DSP, "source_id={} buffer_id={} from_queue={} current_buffer.size()={}",
326  source_id, buf.buffer_id, buf.from_queue, state.current_buffer.size());
327  return true;
328 }

330 SourceStatus::Status Source::GetCurrentStatus() {
331  SourceStatus::Status ret{};
332 
333  // Applications depend on the correct emulation of
334  // current_buffer_id_dirty and current_buffer_id to synchronise
335  // audio with video.
336  ret.is_enabled = state.enabled;
337  ret.current_buffer_id_dirty = state.buffer_update ? 1 : 0;
338  state.buffer_update = false;
339  ret.current_buffer_id = state.current_buffer_id;
340  ret.buffer_position = state.current_sample_number;
341  ret.sync = state.sync;
342
