34 SharedPtr<Event> HLERequestContext::SleepClientThread(SharedPtr<Thread> thread,
35  const std::string& reason,
36  std::chrono::nanoseconds timeout,
37  WakeupCallback&& callback) {
38  // Put the client thread to sleep until the wait event is signaled or the timeout expires.
39  thread->wakeup_callback = [context = *this, callback](ThreadWakeupReason reason,
40  SharedPtr<Thread> thread,
41  SharedPtr<WaitObject> object) mutable {
42  ASSERT(thread->status == THREADSTATUS_WAIT_HLE_EVENT);
43  callback(thread, context, reason);
44 
45  auto& process = thread->owner_process;
46  // We must copy the entire command buffer *plus* the entire static buffers area, since
47  // the translation might need to read from it in order to retrieve the StaticBuffer
48  // target addresses.
49  std::array<u32_le, IPC::COMMAND_BUFFER_LENGTH + 2 * IPC::MAX_STATIC_BUFFERS> cmd_buff;
50  Memory::ReadBlock(*process, thread->GetCommandBufferAddress(), cmd_buff.data(),
51  cmd_buff.size() * sizeof(u32));
52  context.WriteToOutgoingCommandBuffer(cmd_buff.data(), *process, Kernel::g_handle_table);
53  // Copy the translated command buffer back into the thread's command buffer area.
54  Memory::WriteBlock(*process, thread->GetCommandBufferAddress(), cmd_buff.data(),
55  cmd_buff.size() * sizeof(u32));
56  };
57 
58  auto event = Kernel::Event::Create(Kernel::ResetType::OneShot, "HLE Pause Event: " + reason);
59  thread->status = THREADSTATUS_WAIT_HLE_EVENT;
60  thread->wait_objects = {event};
61  event->AddWaitingThread(thread);
62 
63  if (timeout.count() > 0)
64  thread->WakeAfterDelay(timeout.count());
65 
66  return event;
67 }

98 ResultCode HLERequestContext::PopulateFromIncomingCommandBuffer(const u32_le* src_cmdbuf,
99  Process& src_process,
100  HandleTable& src_table) {
101  IPC::Header header{src_cmdbuf[0]};
102 
103  std::size_t untranslated_size = 1u + header.normal_params_size;
104  std::size_t command_size = untranslated_size + header.translate_params_size;
105  ASSERT(command_size <= IPC::COMMAND_BUFFER_LENGTH); // TODO(yuriks): Return error
106 
107  std::copy_n(src_cmdbuf, untranslated_size, cmd_buf.begin());
108 
109  std::size_t i = untranslated_size;
110  while (i < command_size) {
111  u32 descriptor = cmd_buf[i] = src_cmdbuf[i];
112  i += 1;
113 
114  switch (IPC::GetDescriptorType(descriptor)) {
115  case IPC::DescriptorType::CopyHandle:
116  case IPC::DescriptorType::MoveHandle: {
117  u32 num_handles = IPC::HandleNumberFromDesc(descriptor);
118  ASSERT(i + num_handles <= command_size); // TODO(yuriks): Return error
119  for (u32 j{}; j < num_handles; ++j) {
120  Handle handle = src_cmdbuf[i];
121  SharedPtr<Object> object{};
122  if (handle != 0) {
123  object = src_table.GetGeneric(handle);
124  ASSERT(object != nullptr); // TODO(yuriks): Return error
125  if (descriptor == IPC::DescriptorType::MoveHandle) {
126  src_table.Close(handle);
127  }
128  }
129 
130  cmd_buf[i++] = AddOutgoingHandle(std::move(object));
131  }
132  break;
133  }
134  case IPC::DescriptorType::CallingPid: {
135  cmd_buf[i++] = src_process.process_id;
136  break;
137  }
138  case IPC::DescriptorType::StaticBuffer: {
139  VAddr source_address = src_cmdbuf[i];
140  IPC::StaticBufferDescInfo buffer_info{descriptor};
141 
142  // Copy the input buffer into our own vector and store it.
143  std::vector<u8> data(buffer_info.size);
144  Memory::ReadBlock(src_process, source_address, data.data(), data.size());
145 
146  AddStaticBuffer(buffer_info.buffer_id, std::move(data));
147  cmd_buf[i++] = source_address;
148  break;
149  }
150  case IPC::DescriptorType::MappedBuffer: {
151  u32 next_id = static_cast<u32>(request_mapped_buffers.size());
152  request_mapped_buffers.emplace_back(src_process, descriptor, src_cmdbuf[i], next_id);
153  cmd_buf[i++] = next_id;
154  break;
155  }
156  default:
157  UNIMPLEMENTED_MSG("Unsupported handle translation: {:#010X}", descriptor);
158  }
159  }
160 
161  return RESULT_SUCCESS;
162 }

164 ResultCode HLERequestContext::WriteToOutgoingCommandBuffer(u32_le* dst_cmdbuf, Process& dst_process,
165  HandleTable& dst_table) const {
166  IPC::Header header{cmd_buf[0]};
167 
168  std::size_t untranslated_size = 1u + header.normal_params_size;
169  std::size_t command_size = untranslated_size + header.translate_params_size;
170  ASSERT(command_size <= IPC::COMMAND_BUFFER_LENGTH);
171 
172  std::copy_n(cmd_buf.begin(), untranslated_size, dst_cmdbuf);
173 
174  std::size_t i = untranslated_size;
175  while (i < command_size) {
176  u32 descriptor = dst_cmdbuf[i] = cmd_buf[i];
177  i += 1;
178 
179  switch (IPC::GetDescriptorType(descriptor)) {
180  case IPC::DescriptorType::CopyHandle:
181  case IPC::DescriptorType::MoveHandle: {
182  // HLE services don't use handles, so we treat both CopyHandle and MoveHandle equally
183  u32 num_handles = IPC::HandleNumberFromDesc(descriptor);
184  ASSERT(i + num_handles <= command_size);
185  for (u32 j{}; j < num_handles; ++j) {
186  SharedPtr<Object> object = GetIncomingHandle(cmd_buf[i]);
187  Handle handle{};
188  if (object != nullptr) {
189  // TODO(yuriks): Figure out the proper error handling for if this fails
190  handle = dst_table.Create(object).Unwrap();
191  }
192  dst_cmdbuf[i++] = handle;
193  }
194  break;
195  }
196  case IPC::DescriptorType::StaticBuffer: {
197  IPC::StaticBufferDescInfo buffer_info{descriptor};
198 
199  const auto& data = GetStaticBuffer(buffer_info.buffer_id);
200 
201  // Grab the address that the target thread set up to receive the response static buffer
202  // and write our data there. The static buffers area is located right after the command
203  // buffer area.
204  std::size_t static_buffer_offset = IPC::COMMAND_BUFFER_LENGTH + 2 * buffer_info.buffer_id;
205  IPC::StaticBufferDescInfo target_descriptor{dst_cmdbuf[static_buffer_offset]};
206  VAddr target_address = dst_cmdbuf[static_buffer_offset + 1];
207 
208  Memory::WriteBlock(dst_process, target_address, data.data(), data.size());
209 
210  dst_cmdbuf[i++] = target_address;
211  break;
212  }
213  case IPC::DescriptorType::MappedBuffer: {
214  VAddr addr = request_mapped_buffers[cmd_buf[i]].address;
215  dst_cmdbuf[i++] = addr;
216  break;
217  }
218  default:
219  UNIMPLEMENTED_MSG("Unsupported handle translation: {:#010X}", descriptor);
220  }
221  }
222 
223  return RESULT_SUCCESS;
224 }
