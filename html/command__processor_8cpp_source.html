68 static void WriteUniformFloatReg(ShaderRegs& config, Shader::ShaderSetup& setup,
69  int& float_regs_counter, u32 uniform_write_buffer[4], u32 value) {
70  auto& uniform_setup = config.uniform_setup;
71 
72  // TODO: Does actual hardware indeed keep an intermediate buffer or does
73  // it directly write the values?
74  uniform_write_buffer[float_regs_counter++] = value;
75 
76  // Uniforms are written in a packed format such that four float24 values are encoded in
77  // three 32-bit numbers. We write to internal memory once a full such vector is
78  // written.
79  if ((float_regs_counter >= 4 && uniform_setup.IsFloat32()) ||
80  (float_regs_counter >= 3 && !uniform_setup.IsFloat32())) {
81  float_regs_counter = 0;
82 
83  auto& uniform = setup.uniforms.f[uniform_setup.index];
84 
85  if (uniform_setup.index >= 96) {
86  LOG_ERROR(HW_GPU, "Invalid {} float uniform index {}", GetShaderSetupTypeName(setup),
87  (int)uniform_setup.index);
88  } else {
89 
90  // NOTE: The destination component order indeed is "backwards"
91  if (uniform_setup.IsFloat32()) {
92  for (auto i : {0, 1, 2, 3})
93  uniform[3 - i] = float24::FromFloat32(*(float*)(&uniform_write_buffer[i]));
94  } else {
95  // TODO: Untested
96  uniform.w = float24::FromRaw(uniform_write_buffer[0] >> 8);
97  uniform.z = float24::FromRaw(((uniform_write_buffer[0] & 0xFF) << 16) |
98  ((uniform_write_buffer[1] >> 16) & 0xFFFF));
99  uniform.y = float24::FromRaw(((uniform_write_buffer[1] & 0xFFFF) << 8) |
100  ((uniform_write_buffer[2] >> 24) & 0xFF));
101  uniform.x = float24::FromRaw(uniform_write_buffer[2] & 0xFFFFFF);
102  }
103 
104  LOG_TRACE(HW_GPU, "Set {} float uniform {:x} to ({} {} {} {})",
105  GetShaderSetupTypeName(setup), (int)uniform_setup.index,
106  uniform.x.ToFloat32(), uniform.y.ToFloat32(), uniform.z.ToFloat32(),
107  uniform.w.ToFloat32());
108 
109  // TODO: Verify that this actually modifies the register!
110  uniform_setup.index.Assign(uniform_setup.index + 1);
111  }
112  }

115 static void WritePicaReg(u32 id, u32 value, u32 mask) {
116  auto& regs = g_state.regs;
117 
118  if (id >= Regs::NUM_REGS) {
119  LOG_ERROR(
120  HW_GPU,
121  "Commandlist tried to write to invalid register 0x{:03X} (value: {:08X}, mask: {:X})",
122  id, value, mask);
123  return;
124  }
125 
126  // TODO: Figure out how register masking acts on e.g. vs.uniform_setup.set_value
127  u32 old_value = regs.reg_array[id];
128 
129  const u32 write_mask = expand_bits_to_bytes[mask];
130 
131  regs.reg_array[id] = (old_value & ~write_mask) | (value & write_mask);
132 
133  switch (id) {
134  // Trigger IRQ
135  case PICA_REG_INDEX(trigger_irq):
136  Service::GSP::SignalInterrupt(Service::GSP::InterruptId::P3D);
137  break;
138 
139  case PICA_REG_INDEX(pipeline.triangle_topology):
140  g_state.primitive_assembler.Reconfigure(regs.pipeline.triangle_topology);
141  break;
142 
143  case PICA_REG_INDEX(pipeline.restart_primitive):
144  g_state.primitive_assembler.Reset();
145  break;
146 
147  case PICA_REG_INDEX(pipeline.vs_default_attributes_setup.index):
148  g_state.immediate.current_attribute = 0;
149  g_state.immediate.reset_geometry_pipeline = true;
150  default_attr_counter = 0;
151  break;
152 
153  // Load default vertex input attributes
154  case PICA_REG_INDEX_WORKAROUND(pipeline.vs_default_attributes_setup.set_value[0], 0x233):
155  case PICA_REG_INDEX_WORKAROUND(pipeline.vs_default_attributes_setup.set_value[1], 0x234):
156  case PICA_REG_INDEX_WORKAROUND(pipeline.vs_default_attributes_setup.set_value[2], 0x235): {
157  // TODO: Does actual hardware indeed keep an intermediate buffer or does
158  // it directly write the values?
159  default_attr_write_buffer[default_attr_counter++] = value;
160 
161  // Default attributes are written in a packed format such that four float24 values are
162  // encoded in
163  // three 32-bit numbers. We write to internal memory once a full such vector is
164  // written.
165  if (default_attr_counter >= 3) {
166  default_attr_counter = 0;
167 
168  auto& setup = regs.pipeline.vs_default_attributes_setup;
169 
170  if (setup.index >= 16) {
171  LOG_ERROR(HW_GPU, "Invalid VS default attribute index {}", (int)setup.index);
172  break;
173  }
174 
175  Math::Vec4<float24> attribute;
176 
177  // NOTE: The destination component order indeed is "backwards"
178  attribute.w = float24::FromRaw(default_attr_write_buffer[0] >> 8);
179  attribute.z = float24::FromRaw(((default_attr_write_buffer[0] & 0xFF) << 16) |
180  ((default_attr_write_buffer[1] >> 16) & 0xFFFF));
181  attribute.y = float24::FromRaw(((default_attr_write_buffer[1] & 0xFFFF) << 8) |
182  ((default_attr_write_buffer[2] >> 24) & 0xFF));
183  attribute.x = float24::FromRaw(default_attr_write_buffer[2] & 0xFFFFFF);
184 
185  LOG_TRACE(HW_GPU, "Set default VS attribute {:x} to ({} {} {} {})", (int)setup.index,
186  attribute.x.ToFloat32(), attribute.y.ToFloat32(), attribute.z.ToFloat32(),
187  attribute.w.ToFloat32());
188 
189  // TODO: Verify that this actually modifies the register!
190  if (setup.index < 15) {
191  g_state.input_default_attributes.attr[setup.index] = attribute;
192  setup.index++;
193  } else {
194  // Put each attribute into an immediate input buffer. When all specified immediate
195  // attributes are present, the Vertex Shader is invoked and everything is sent to
196  // the primitive assembler.
197 
198  auto& immediate_input = g_state.immediate.input_vertex;
199  auto& immediate_attribute_id = g_state.immediate.current_attribute;
200 
201  immediate_input.attr[immediate_attribute_id] = attribute;
202 
203  if (immediate_attribute_id < regs.pipeline.max_input_attrib_index) {
204  immediate_attribute_id += 1;
205  } else {
206  immediate_attribute_id = 0;
207 
208  Shader::OutputVertex::ValidateSemantics(regs.rasterizer);
209 
210  auto* shader_engine = Shader::GetEngine();
211  shader_engine->SetupBatch(g_state.vs, regs.vs.main_offset);
212 
213  // Send to vertex shader
214  Shader::UnitState shader_unit;
215  Shader::AttributeBuffer output{};
216 
217  shader_unit.LoadInput(regs.vs, immediate_input);
218  shader_engine->Run(g_state.vs, shader_unit);
219  shader_unit.WriteOutput(regs.vs, output);
220 
221  // Send to geometry pipeline
222  if (g_state.immediate.reset_geometry_pipeline) {
223  g_state.geometry_pipeline.Reconfigure();
224  g_state.immediate.reset_geometry_pipeline = false;
225  }
226  ASSERT(!g_state.geometry_pipeline.NeedIndexInput());
227  g_state.geometry_pipeline.Setup(shader_engine);
228  g_state.geometry_pipeline.SubmitVertex(output);
229 
230  // TODO: If drawing after every immediate mode triangle kills performance,
231  // change it to flush triangles whenever a drawing config register changes
232  // See: https://github.com/citra-emu/citra/pull/2866#issuecomment-327011550
233  VideoCore::g_renderer->GetRasterizer()->DrawTriangles();
234  }
235  }
236  }
237  break;
238  }

261 
262  bool accelerate_draw{Settings::values.use_hw_shader && primitive_assembler.IsEmpty()};
263 
264  if (regs.pipeline.use_gs == PipelineRegs::UseGS::No) {
265  switch (primitive_assembler.GetTopology()) {
266  case PipelineRegs::TriangleTopology::Shader:
267  case PipelineRegs::TriangleTopology::List:

275  }
276  } else {
277  if (Settings::values.shaders_accurate_gs) {
278  accelerate_draw = false;
279  }
280  }
281

283  VideoCore::g_renderer->GetRasterizer()->AccelerateDrawBatch(is_indexed)) {
284  break;

287  // Processes information about internal vertex attributes to figure out how a vertex is
288  // loaded.
289  // Later, these can be compiled and cached.
290  const u32 base_address{regs.pipeline.vertex_attributes.GetPhysicalBaseAddress()};
291  VertexLoader loader{regs.pipeline};
292  Shader::OutputVertex::ValidateSemantics(regs.rasterizer);
293 
294  // Load vertices
295  const auto& index_info{regs.pipeline.index_array};
296  const u8* index_address_8{Memory::GetPhysicalPointer(base_address + index_info.offset)};
297  if (!index_address_8)
298  break;
299  const u16* index_address_16{reinterpret_cast<const u16*>(index_address_8)};
300  bool index_u16{index_info.format != 0};
301 
302  // Simple circular-replacement vertex cache
303  // The size has been tuned for optimal balance between hit-rate and the cost of lookup
304  const std::size_t VERTEX_CACHE_SIZE{32};
305  std::array<bool, VERTEX_CACHE_SIZE> vertex_cache_valid{};
306  std::array<u16, VERTEX_CACHE_SIZE> vertex_cache_ids{};
307  std::array<Shader::AttributeBuffer, VERTEX_CACHE_SIZE> vertex_cache;
308  Shader::AttributeBuffer vs_output{};
309 
310  unsigned int vertex_cache_pos{};
311 
312  auto* shader_engine{Shader::GetEngine()};
313  Shader::UnitState shader_unit{};
314 
315  shader_engine->SetupBatch(g_state.vs, regs.vs.main_offset);
316 
317  g_state.geometry_pipeline.Reconfigure();
318  g_state.geometry_pipeline.Setup(shader_engine);
319  if (g_state.geometry_pipeline.NeedIndexInput())
320  ASSERT(is_indexed);
321 
322  for (unsigned int index{}; index < regs.pipeline.num_vertices; ++index) {
323  // Indexed rendering doesn't use the start offset
324  unsigned int vertex{is_indexed
325  ? (index_u16 ? index_address_16[index] : index_address_8[index])
326  : (index + regs.pipeline.vertex_offset)};
327 
328  bool vertex_cache_hit{};
329 
330  if (is_indexed) {
331  if (g_state.geometry_pipeline.NeedIndexInput()) {
332  g_state.geometry_pipeline.SubmitIndex(vertex);
333  continue;
334  }
335 
336  for (unsigned int i{}; i < VERTEX_CACHE_SIZE; ++i) {
337  if (vertex_cache_valid[i] && vertex == vertex_cache_ids[i]) {
338  vs_output = vertex_cache[i];
339  vertex_cache_hit = true;
340  break;
341  }
342  }
343  }
344 
345  if (!vertex_cache_hit) {
346  // Initialize data for the current vertex
347  Shader::AttributeBuffer input;
348  loader.LoadVertex(base_address, index, vertex, input);
349 
350  shader_unit.LoadInput(regs.vs, input);
351  shader_engine->Run(g_state.vs, shader_unit);
352  shader_unit.WriteOutput(regs.vs, vs_output);
353 
354  if (is_indexed) {
355  vertex_cache[vertex_cache_pos] = vs_output;
356  vertex_cache_valid[vertex_cache_pos] = true;
357  vertex_cache_ids[vertex_cache_pos] = vertex;
358  vertex_cache_pos = (vertex_cache_pos + 1) % VERTEX_CACHE_SIZE;
359  }
360  }
361 
362  // Send to geometry pipeline
363  g_state.geometry_pipeline.SubmitVertex(vs_output);
364  }
365 
366  VideoCore::g_renderer->GetRasterizer()->DrawTriangles();
367  break;
368  }

581 void ProcessCommandList(const u32* list, u32 size) {
582  g_state.cmd_list.head_ptr = g_state.cmd_list.current_ptr = list;
583  g_state.cmd_list.length = size / sizeof(u32);
584 
585  while (g_state.cmd_list.current_ptr < g_state.cmd_list.head_ptr + g_state.cmd_list.length) {
586  // Align read pointer to 8 bytes
587  if ((g_state.cmd_list.head_ptr - g_state.cmd_list.current_ptr) % 2 != 0)
588  ++g_state.cmd_list.current_ptr;
589 
590  u32 value{*g_state.cmd_list.current_ptr++};
591  const CommandHeader header{*g_state.cmd_list.current_ptr++};
592 
593  WritePicaReg(header.cmd_id, value, header.parameter_mask);
594 
595  for (unsigned i{}; i < header.extra_data_length; ++i) {
596  u32 cmd{header.cmd_id + (header.group_commands ? i + 1 : 0)};
597  WritePicaReg(cmd, *g_state.cmd_list.current_ptr++, header.parameter_mask);
598  }
599  }
