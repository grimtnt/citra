55 struct CheatLine {
56  explicit CheatLine(std::string line) {
57  line = std::string(line.c_str()); // remove '/0' characters if any.
58  line = Common::Trim(line);
59  constexpr int cheat_length = 17;
60  if (line.length() != cheat_length) {
61  type = CheatType::Null;
62  cheat_line = line;
63  return;
64  }
65  try {
66  std::string type_temp = line.substr(0, 1);
67 
68  // 0xD types have extra subtype value, i.e. 0xDA
69  std::string sub_type_temp;
70  if (type_temp == "D" || type_temp == "d")
71  sub_type_temp = line.substr(1, 1);
72  type = static_cast<CheatType>(std::stoi(type_temp + sub_type_temp, 0, 16));
73  address = std::stoi(line.substr(1, 8), 0, 16);
74  value = std::stoi(line.substr(10, 8), 0, 16);
75  cheat_line = line;
76  } catch (const std::exception&) {
77  type = CheatType::Null;
78  cheat_line = line;
79  return;
80  }
81  }
82 
83  CheatType type;
84  u32 address;
85  u32 value;
86  std::string cheat_line;
87 };
