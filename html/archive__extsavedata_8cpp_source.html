42  ResultVal<std::size_t> Write(u64 offset, std::size_t length, bool flush,
43  const u8* buffer) override {
44  if (offset > size) {
45  return ERR_WRITE_BEYOND_END;
46  } else if (offset == size) {
47  return MakeResult<std::size_t>(0);

49

53 
54  return DiskFile::Write(offset, length, flush, buffer);

91  ResultVal<std::unique_ptr<FileBackend>> OpenFile(const Path& path,
92  const Mode& mode) const override {
93  LOG_DEBUG(Service_FS, "called path={} mode={:01X}", path.DebugStr(), mode.hex);
94 
95  const PathParser path_parser{path};
96 
97  if (!path_parser.IsValid()) {
98  LOG_ERROR(Service_FS, "Invalid path {}", path.DebugStr());
99  return ERROR_INVALID_PATH;
100  }
101 
102  if (mode.hex == 0) {
103  LOG_ERROR(Service_FS, "Empty open mode");
104  return ERROR_UNSUPPORTED_OPEN_FLAGS;
105  }
106 
107  if (mode.create_flag) {
108  LOG_ERROR(Service_FS, "Create flag is not supported");
109  return ERROR_UNSUPPORTED_OPEN_FLAGS;
110  }
111 
112  const auto full_path{path_parser.BuildHostPath(mount_point)};
113 
114  switch (path_parser.GetHostStatus(mount_point)) {
115  case PathParser::InvalidMountPoint:
116  LOG_CRITICAL(Service_FS, "(unreachable) Invalid mount point {}", mount_point);
117  return ERROR_FILE_NOT_FOUND;
118  case PathParser::PathNotFound:
119  LOG_ERROR(Service_FS, "Path not found {}", full_path);
120  return ERROR_PATH_NOT_FOUND;
121  case PathParser::FileInPath:
122  case PathParser::DirectoryFound:
123  LOG_ERROR(Service_FS, "Unexpected file or directory in {}", full_path);
124  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY;
125  case PathParser::NotFound:
126  LOG_ERROR(Service_FS, "{} not found", full_path);
127  return ERROR_FILE_NOT_FOUND;
128  case PathParser::FileFound:
129  break; // Expected 'success' case
130  }
131 
132  FileUtil::IOFile file{full_path, "r+b"};
133  if (!file.IsOpen()) {
134  LOG_CRITICAL(Service_FS, "(unreachable) Unknown error opening {}", full_path);
135  return ERROR_FILE_NOT_FOUND;
136  }
137 
138  Mode rwmode{};
139  rwmode.write_flag.Assign(1);
140  rwmode.read_flag.Assign(1);
141  std::unique_ptr<DelayGenerator> delay_generator{
142  std::make_unique<ExtSaveDataDelayGenerator>()};
143  auto disk_file{
144  std::make_unique<FixSizeDiskFile>(std::move(file), rwmode, std::move(delay_generator))};
145  return MakeResult<std::unique_ptr<FileBackend>>(std::move(disk_file));
146  }
147 
148  ResultCode CreateFile(const Path& path, u64 size) const override {

150  LOG_ERROR(Service_FS, "Zero-size file is not supported");
151  return ERROR_UNSUPPORTED_OPEN_FLAGS;
152  }
153  return SaveDataArchive::CreateFile(path, size);
154  }
