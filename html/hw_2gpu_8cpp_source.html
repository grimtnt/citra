69 static void MemoryFill(const Regs::MemoryFillConfig& config) {
70  const PAddr start_addr{config.GetStartAddress()};
71  const PAddr end_addr{config.GetEndAddress()};
72 
73  // TODO: do hwtest with these cases
74  if (!Memory::IsValidPhysicalAddress(start_addr)) {
75  LOG_CRITICAL(HW_GPU, "invalid start address {:#010X}", start_addr);
76  return;
77  }
78 
79  if (!Memory::IsValidPhysicalAddress(end_addr)) {
80  LOG_CRITICAL(HW_GPU, "invalid end address {:#010X}", end_addr);
81  return;
82  }
83 
84  if (end_addr <= start_addr) {
85  LOG_CRITICAL(HW_GPU, "invalid memory range from {:#010X} to {:#010X}", start_addr,
86  end_addr);
87  return;
88  }
89 
90  u8* start{Memory::GetPhysicalPointer(start_addr)};
91  u8* end{Memory::GetPhysicalPointer(end_addr)};
92 
93  if (VideoCore::g_renderer->GetRasterizer()->AccelerateFill(config))
94  return;
95 
96  Memory::RasterizerInvalidateRegion(config.GetStartAddress(),
97  config.GetEndAddress() - config.GetStartAddress());
98 
99  if (config.fill_24bit) {
100  // fill with 24-bit values
101  for (u8* ptr = start; ptr < end; ptr += 3) {
102  ptr[0] = config.value_24bit_r;
103  ptr[1] = config.value_24bit_g;
104  ptr[2] = config.value_24bit_b;
105  }
106  } else if (config.fill_32bit) {
107  // fill with 32-bit values
108  if (end > start) {
109  u32 value = config.value_32bit;
110  std::size_t len = (end - start) / sizeof(u32);
111  for (std::size_t i{}; i < len; ++i)
112  memcpy(&start[i * sizeof(u32)], &value, sizeof(u32));
113  }
114  } else {
115  // fill with 16-bit values
116  u16 value_16bit = config.value_16bit.Value();
117  for (u8* ptr = start; ptr < end; ptr += sizeof(u16))
118  memcpy(ptr, &value_16bit, sizeof(u16));
119  }
120 }

122 static void DisplayTransfer(const Regs::DisplayTransferConfig& config) {
123  const PAddr src_addr = config.GetPhysicalInputAddress();
124  const PAddr dst_addr = config.GetPhysicalOutputAddress();
125 
126  // TODO: do hwtest with these cases
127  if (!Memory::IsValidPhysicalAddress(src_addr)) {
128  LOG_CRITICAL(HW_GPU, "invalid input address {:#010X}", src_addr);
129  return;
130  }
131 
132  if (!Memory::IsValidPhysicalAddress(dst_addr)) {
133  LOG_CRITICAL(HW_GPU, "invalid output address {:#010X}", dst_addr);
134  return;
135  }
136 
137  if (config.input_width == 0) {
138  LOG_CRITICAL(HW_GPU, "zero input width");
139  return;
140  }
141 
142  if (config.input_height == 0) {
143  LOG_CRITICAL(HW_GPU, "zero input height");
144  return;
145  }
146 
147  if (config.output_width == 0) {
148  LOG_CRITICAL(HW_GPU, "zero output width");
149  return;
150  }
151 
152  if (config.output_height == 0) {
153  LOG_CRITICAL(HW_GPU, "zero output height");
154  return;
155  }
156 
157  if (VideoCore::g_renderer->GetRasterizer()->AccelerateDisplayTransfer(config))
158  return;
159 
160  u8* src_pointer = Memory::GetPhysicalPointer(src_addr);
161  u8* dst_pointer = Memory::GetPhysicalPointer(dst_addr);
162 
163  if (config.scaling > config.ScaleXY) {
164  LOG_CRITICAL(HW_GPU, "Unimplemented display transfer scaling mode {}",
165  config.scaling.Value());
166  UNIMPLEMENTED();
167  return;
168  }
169 
170  if (config.input_linear && config.scaling != config.NoScale) {
171  LOG_CRITICAL(HW_GPU, "Scaling is only implemented on tiled input");
172  UNIMPLEMENTED();
173  return;
174  }
175 
176  int horizontal_scale = config.scaling != config.NoScale ? 1 : 0;
177  int vertical_scale = config.scaling == config.ScaleXY ? 1 : 0;
178 
179  u32 output_width = config.output_width >> horizontal_scale;
180  u32 output_height = config.output_height >> vertical_scale;
181 
182  u32 input_size =
183  config.input_width * config.input_height * GPU::Regs::BytesPerPixel(config.input_format);
184  u32 output_size = output_width * output_height * GPU::Regs::BytesPerPixel(config.output_format);
185 
186  Memory::RasterizerFlushRegion(config.GetPhysicalInputAddress(), input_size);
187  Memory::RasterizerInvalidateRegion(config.GetPhysicalOutputAddress(), output_size);
188 
189  for (u32 y{}; y < output_height; ++y) {
190  for (u32 x{}; x < output_width; ++x) {
191  Math::Vec4<u8> src_color;
192 
193  // Calculate the [x,y] position of the input image
194  // based on the current output position and the scale
195  u32 input_x = x << horizontal_scale;
196  u32 input_y = y << vertical_scale;
197 
198  u32 output_y;
199  if (config.flip_vertically) {
200  // Flip the y value of the output data,
201  // we do this after calculating the [x,y] position of the input image
202  // to account for the scaling options.
203  output_y = output_height - y - 1;
204  } else {
205  output_y = y;
206  }
207 
208  u32 dst_bytes_per_pixel = GPU::Regs::BytesPerPixel(config.output_format);
209  u32 src_bytes_per_pixel = GPU::Regs::BytesPerPixel(config.input_format);
210  u32 src_offset;
211  u32 dst_offset;
212 
213  if (config.input_linear) {
214  if (!config.dont_swizzle) {
215  // Interpret the input as linear and the output as tiled
216  u32 coarse_y = output_y & ~7;
217  u32 stride = output_width * dst_bytes_per_pixel;
218 
219  src_offset = (input_x + input_y * config.input_width) * src_bytes_per_pixel;
220  dst_offset = VideoCore::GetMortonOffset(x, output_y, dst_bytes_per_pixel) +
221  coarse_y * stride;
222  } else {
223  // Both input and output are linear
224  src_offset = (input_x + input_y * config.input_width) * src_bytes_per_pixel;
225  dst_offset = (x + output_y * output_width) * dst_bytes_per_pixel;
226  }
227  } else {
228  if (!config.dont_swizzle) {
229  // Interpret the input as tiled and the output as linear
230  u32 coarse_y = input_y & ~7;
231  u32 stride = config.input_width * src_bytes_per_pixel;
232 
233  src_offset = VideoCore::GetMortonOffset(input_x, input_y, src_bytes_per_pixel) +
234  coarse_y * stride;
235  dst_offset = (x + output_y * output_width) * dst_bytes_per_pixel;
236  } else {
237  // Both input and output are tiled
238  u32 out_coarse_y = output_y & ~7;
239  u32 out_stride = output_width * dst_bytes_per_pixel;
240 
241  u32 in_coarse_y = input_y & ~7;
242  u32 in_stride = config.input_width * src_bytes_per_pixel;
243 
244  src_offset = VideoCore::GetMortonOffset(input_x, input_y, src_bytes_per_pixel) +
245  in_coarse_y * in_stride;
246  dst_offset = VideoCore::GetMortonOffset(x, output_y, dst_bytes_per_pixel) +
247  out_coarse_y * out_stride;
248  }
249  }
250 
251  if (!dst_pointer)
252  break;
253 
254  const u8* src_pixel = src_pointer + src_offset;
255  src_color = DecodePixel(config.input_format, src_pixel);
256  if (config.scaling == config.ScaleX) {
257  Math::Vec4<u8> pixel =
258  DecodePixel(config.input_format, src_pixel + src_bytes_per_pixel);
259  src_color = ((src_color + pixel) / 2).Cast<u8>();
260  } else if (config.scaling == config.ScaleXY) {
261  Math::Vec4<u8> pixel1 =
262  DecodePixel(config.input_format, src_pixel + 1 * src_bytes_per_pixel);
263  Math::Vec4<u8> pixel2 =
264  DecodePixel(config.input_format, src_pixel + 2 * src_bytes_per_pixel);
265  Math::Vec4<u8> pixel3 =
266  DecodePixel(config.input_format, src_pixel + 3 * src_bytes_per_pixel);
267  src_color = (((src_color + pixel1) + (pixel2 + pixel3)) / 4).Cast<u8>();
268  }
269 
270  u8* dst_pixel = dst_pointer + dst_offset;
271  switch (config.output_format) {
272  case Regs::PixelFormat::RGBA8:
273  Color::EncodeRGBA8(src_color, dst_pixel);
274  break;
275 
276  case Regs::PixelFormat::RGB8:
277  Color::EncodeRGB8(src_color, dst_pixel);
278  break;
279 
280  case Regs::PixelFormat::RGB565:
281  Color::EncodeRGB565(src_color, dst_pixel);
282  break;
283 
284  case Regs::PixelFormat::RGB5A1:
285  Color::EncodeRGB5A1(src_color, dst_pixel);
286  break;
287 
288  case Regs::PixelFormat::RGBA4:
289  Color::EncodeRGBA4(src_color, dst_pixel);
290  break;
291 
292  default:
293  LOG_ERROR(HW_GPU, "Unknown destination framebuffer format {:x}",
294  static_cast<u32>(config.output_format.Value()));
295  break;
296  }
297  }
298  }

301 static void TextureCopy(const Regs::DisplayTransferConfig& config) {
302  const PAddr src_addr = config.GetPhysicalInputAddress();
303  const PAddr dst_addr = config.GetPhysicalOutputAddress();
304 
305  // TODO: do hwtest with invalid addresses
306  if (!Memory::IsValidPhysicalAddress(src_addr)) {
307  LOG_CRITICAL(HW_GPU, "invalid input address {:#010X}", src_addr);
308  return;
309  }
310 
311  if (!Memory::IsValidPhysicalAddress(dst_addr)) {
312  LOG_CRITICAL(HW_GPU, "invalid output address {:#010X}", dst_addr);
313  return;
314  }
315 
316  if (VideoCore::g_renderer->GetRasterizer()->AccelerateTextureCopy(config))
317  return;
318 
319  u8* src_pointer = Memory::GetPhysicalPointer(src_addr);
320  u8* dst_pointer = Memory::GetPhysicalPointer(dst_addr);
321 
322  u32 remaining_size = Common::AlignDown(config.texture_copy.size, 16);
323 
324  if (remaining_size == 0) {
325  LOG_CRITICAL(HW_GPU, "zero size. Real hardware freezes on this.");
326  return;
327  }
328 
329  u32 input_gap = config.texture_copy.input_gap * 16;
330  u32 output_gap = config.texture_copy.output_gap * 16;
331 
332  // Zero gap means contiguous input/output even if width = 0. To avoid infinite loop below, width
333  // is assigned with the total size if gap = 0.
334  u32 input_width = input_gap == 0 ? remaining_size : config.texture_copy.input_width * 16;
335  u32 output_width = output_gap == 0 ? remaining_size : config.texture_copy.output_width * 16;
336 
337  if (input_width == 0) {
338  LOG_CRITICAL(HW_GPU, "zero input width. Real hardware freezes on this.");
339  return;
340  }
341 
342  if (output_width == 0) {
343  LOG_CRITICAL(HW_GPU, "zero output width. Real hardware freezes on this.");
344  return;
345  }
346 
347  std::size_t contiguous_input_size =
348  config.texture_copy.size / input_width * (input_width + input_gap);
349  Memory::RasterizerFlushRegion(config.GetPhysicalInputAddress(),
350  static_cast<u32>(contiguous_input_size));
351 
352  std::size_t contiguous_output_size =
353  config.texture_copy.size / output_width * (output_width + output_gap);
354  // Only need to flush output if it has a gap
355  const auto FlushInvalidate_fn = (output_gap != 0) ? Memory::RasterizerFlushAndInvalidateRegion
356  : Memory::RasterizerInvalidateRegion;
357  FlushInvalidate_fn(config.GetPhysicalOutputAddress(), static_cast<u32>(contiguous_output_size));
358 
359  u32 remaining_input = input_width;
360  u32 remaining_output = output_width;
361  while (remaining_size > 0) {
362  u32 copy_size = std::min({remaining_input, remaining_output, remaining_size});
363 
364  std::memcpy(dst_pointer, src_pointer, copy_size);
365  src_pointer += copy_size;
366  dst_pointer += copy_size;
367 
368  remaining_input -= copy_size;
369  remaining_output -= copy_size;
370  remaining_size -= copy_size;
371 
372  if (remaining_input == 0) {
373  remaining_input = input_width;
374  src_pointer += input_gap;
375  }
376  if (remaining_output == 0) {
377  remaining_output = output_width;
378  dst_pointer += output_gap;
379  }
380  }
381 }

383 template <typename T>
384 inline void Write(u32 addr, const T data) {
385  addr -= HW::VADDR_GPU;
386  u32 index = addr / 4;
387 
388  // Writes other than u32 are untested, so I'd rather have them abort than silently fail
389  if (index >= Regs::NumIds() || !std::is_same<T, u32>::value) {
390  LOG_ERROR(HW_GPU, "unknown Write{} {:#010X} @ {:#010X}", sizeof(data) * 8, (u32)data, addr);
391  return;
392  }
393 
394  g_regs[index] = static_cast<u32>(data);
395 
396  switch (index) {
397 
398  // Memory fills are triggered once the fill value is written.
399  case GPU_REG_INDEX_WORKAROUND(memory_fill_config[0].trigger, 0x00004 + 0x3):
400  case GPU_REG_INDEX_WORKAROUND(memory_fill_config[1].trigger, 0x00008 + 0x3): {
401  const bool is_second_filler = (index != GPU_REG_INDEX(memory_fill_config[0].trigger));
402  auto& config = g_regs.memory_fill_config[is_second_filler];
403 
404  if (config.trigger) {
405  MemoryFill(config);
406  LOG_TRACE(HW_GPU, "MemoryFill from {:#010X} to {:#010X}", config.GetStartAddress(),
407  config.GetEndAddress());
408 
409  // It seems that it won't signal interrupt if "address_start" is zero.
410  // TODO: hwtest this
411  if (config.GetStartAddress() != 0) {
412  if (!is_second_filler) {
413  Service::GSP::SignalInterrupt(Service::GSP::InterruptId::PSC0);
414  } else {
415  Service::GSP::SignalInterrupt(Service::GSP::InterruptId::PSC1);
416  }
417  }
418 
419  // Reset "trigger" flag and set the "finish" flag
420  // NOTE: This was confirmed to happen on hardware even if "address_start" is zero.
421  config.trigger.Assign(0);
422  config.finished.Assign(1);
423  }
424  break;
425  }
426 
427  case GPU_REG_INDEX(display_transfer_config.trigger): {
428  const auto& config = g_regs.display_transfer_config;
429  if (config.trigger & 1) {
430  if (config.is_texture_copy) {
431  TextureCopy(config);
432  LOG_TRACE(HW_GPU,
433  "TextureCopy: {:#X} bytes from {:#010X}({}+{})-> "
434  "{:#010X}({}+{}), flags {:#010X}",
435  config.texture_copy.size, config.GetPhysicalInputAddress(),
436  config.texture_copy.input_width * 16, config.texture_copy.input_gap * 16,
437  config.GetPhysicalOutputAddress(), config.texture_copy.output_width * 16,
438  config.texture_copy.output_gap * 16, config.flags);
439  } else {
440  DisplayTransfer(config);
441  LOG_TRACE(HW_GPU,
442  "DisplayTransfer: {:#010X}({}x{})-> "
443  "{:#010X}({}x{}), dst format {:x}, flags {:#010X}",
444  config.GetPhysicalInputAddress(), config.input_width.Value(),
445  config.input_height.Value(), config.GetPhysicalOutputAddress(),
446  config.output_width.Value(), config.output_height.Value(),
447  static_cast<u32>(config.output_format.Value()), config.flags);
448  }
449 
450  g_regs.display_transfer_config.trigger = 0;
451  Service::GSP::SignalInterrupt(Service::GSP::InterruptId::PPF);
452  }
453  break;
454  }
455 
456  // Seems like writing to this register triggers processing
457  case GPU_REG_INDEX(command_processor_config.trigger): {
458  const auto& config = g_regs.command_processor_config;
459  if (config.trigger & 1) {
460  u32* buffer = (u32*)Memory::GetPhysicalPointer(config.GetPhysicalAddress());
461  Pica::CommandProcessor::ProcessCommandList(buffer, config.size);
462  g_regs.command_processor_config.trigger = 0;
463  }
464  break;
465  }
466 
467  default:
468  break;
469  }
470 }

485 static void VBlankCallback(u64 userdata, s64 cycles_late) {
486  VideoCore::g_renderer->SwapBuffers();
487 
488  // Signal to GSP that GPU interrupt has occurred
489  // TODO(yuriks): hwtest to determine if PDC0 is for the Top screen and PDC1 for the Sub
490  // screen, or if both use the same interrupts and these two instead determine the
491  // beginning and end of the VBlank period. If needed, split the interrupt firing into
492  // two different intervals.
493  Service::GSP::SignalInterrupt(Service::GSP::InterruptId::PDC0);
494  Service::GSP::SignalInterrupt(Service::GSP::InterruptId::PDC1);
495 
496  // Reschedule recurrent event
497  CoreTiming::ScheduleEvent(frame_ticks - cycles_late, vblank_event);
498 }
