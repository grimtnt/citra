51 ResultVal<std::unique_ptr<FileBackend>> NCCHArchive::OpenFile(const Path& path,
52  const Mode& mode) const {
53  if (path.GetType() != LowPathType::Binary) {
54  LOG_ERROR(Service_FS, "Path need to be Binary");
55  return ERROR_INVALID_PATH;
56  }
57 
58  std::vector<u8> binary = path.AsBinary();
59  if (binary.size() != sizeof(NCCHFilePath)) {
60  LOG_ERROR(Service_FS, "Wrong path size {}", binary.size());
61  return ERROR_INVALID_PATH;
62  }
63 
64  NCCHFilePath openfile_path{};
65  std::memcpy(&openfile_path, binary.data(), sizeof(NCCHFilePath));
66 
67  std::string file_path{
68  Service::AM::GetTitleContentPath(media_type, title_id, openfile_path.content_index)};
69  auto ncch_container{NCCHContainer(file_path)};
70 
71  Loader::ResultStatus result{};
72  std::unique_ptr<FileBackend> file{};
73 
74  // NCCH RomFS
75  NCCHFilePathType filepath_type = static_cast<NCCHFilePathType>(openfile_path.filepath_type);
76  if (filepath_type == NCCHFilePathType::RomFS) {
77  std::shared_ptr<RomFSReader> romfs_file{};
78 
79  result = ncch_container.ReadRomFS(romfs_file);
80  std::unique_ptr<DelayGenerator> delay_generator{std::make_unique<RomFSDelayGenerator>()};
81  file = std::make_unique<IVFCFile>(std::move(romfs_file), std::move(delay_generator));
82  } else if (filepath_type == NCCHFilePathType::Code ||
83  filepath_type == NCCHFilePathType::ExeFS) {
84  std::vector<u8> buffer{};
85 
86  // Load NCCH .code or icon/banner/logo
87  result = ncch_container.LoadSectionExeFS(openfile_path.exefs_filepath.data(), buffer);
88  std::unique_ptr<DelayGenerator> delay_generator{std::make_unique<ExeFSDelayGenerator>()};
89  file = std::make_unique<NCCHFile>(std::move(buffer), std::move(delay_generator));
90  } else {
91  LOG_ERROR(Service_FS, "Unknown NCCH archive type {}!", openfile_path.filepath_type);
92  result = Loader::ResultStatus::Error;
93  }
94 
95  if (result != Loader::ResultStatus::Success) {
96  // High Title ID of the archive: The category (https://3dbrew.org/wiki/Title_list).
97  constexpr u32 shared_data_archive{0x0004009B};
98  constexpr u32 system_data_archive{0x000400DB};
99 
100  // Low Title IDs.
101  constexpr u32 mii_data{0x00010202};
102  constexpr u32 region_manifest{0x00010402};
103  constexpr u32 ng_word_list{0x00010302};
104  constexpr u32 shared_font{0x00014002};
105 
106  u32 high{static_cast<u32>(title_id >> 32)};
107  u32 low{static_cast<u32>(title_id & 0xFFFFFFFF)};
108 
109  LOG_DEBUG(Service_FS, "Full Path: {}. Category: 0x{:X}. Path: 0x{:X}.", path.DebugStr(),
110  high, low);
111 
112  std::string archive_name{};
113  if (high == shared_data_archive) {
114  if (low == mii_data)
115  archive_name = "Mii Data";
116  else if (low == region_manifest)
117  archive_name = "Region manifest";
118  else if (low == shared_font) {
119  LOG_WARNING(
120  Service_FS,
121  "Shared Font file missing. Loading open source replacement from memory");
122  std::vector<u8> shared_font_file{};
123  shared_font_file.assign(SHARED_FONT_DATA, SHARED_FONT_DATA + SHARED_FONT_DATA_len);
124  u64 romfs_offset{};
125  u64 romfs_size{shared_font_file.size()};
126  std::unique_ptr<DelayGenerator> delay_generator{
127  std::make_unique<RomFSDelayGenerator>()};
128  file = std::make_unique<IVFCFileInMemory>(std::move(shared_font_file), romfs_offset,
129  romfs_size, std::move(delay_generator));
130  return MakeResult<std::unique_ptr<FileBackend>>(std::move(file));
131  }
132  } else if (high == system_data_archive) {
133  if (low == ng_word_list) {
134  LOG_WARNING(
135  Service_FS,
136  "Bad Word List file missing. Loading open source replacement from memory");
137  std::vector<u8> bad_word_list_file{};
138  bad_word_list_file.assign(BAD_WORD_LIST_DATA,
139  BAD_WORD_LIST_DATA + BAD_WORD_LIST_DATA_len);
140  u64 romfs_offset{};
141  u64 romfs_size{bad_word_list_file.size()};
142  std::unique_ptr<DelayGenerator> delay_generator{
143  std::make_unique<RomFSDelayGenerator>()};
144  file =
145  std::make_unique<IVFCFileInMemory>(std::move(bad_word_list_file), romfs_offset,
146  romfs_size, std::move(delay_generator));
147  return MakeResult<std::unique_ptr<FileBackend>>(std::move(file));
148  }
149  }
150 
151  if (!archive_name.empty()) {
152  LOG_ERROR(Service_FS, "Failed to get a handle for shared data archive: {}. ",
153  archive_name);
154  Core::System::GetInstance().SetStatus(Core::System::ResultStatus::ErrorSystemFiles,
155  archive_name.c_str());
156  }
157  return ERROR_NOT_FOUND;
158  }
159 
160  return MakeResult<std::unique_ptr<FileBackend>>(std::move(file));
161 }

228 ResultVal<std::size_t> NCCHFile::Read(const u64 offset, const std::size_t length, u8* buffer) const {
229  LOG_TRACE(Service_FS, "called offset={}, length={}", offset, length);
230  std::size_t length_left{static_cast<std::size_t>(data_size - offset)};
231  std::size_t read_length{static_cast<std::size_t>(std::min(length, length_left))};
232 
233  std::size_t available_size{static_cast<std::size_t>(file_buffer.size() - offset)};
234  std::size_t copy_size{std::min(length, available_size)};
235  memcpy(buffer, file_buffer.data() + offset, copy_size);
236 
237  return MakeResult<std::size_t>(copy_size);
238 }
