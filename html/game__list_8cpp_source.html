34 // EventFilter in order to process systemkeys while editing the searchfield
35 bool GameListSearchField::KeyReleaseEater::eventFilter(QObject* obj, QEvent* event) {
36  // If it isn't a KeyRelease event then continue with standard event processing
37  if (event->type() != QEvent::KeyRelease)
38  return QObject::eventFilter(obj, event);
39 
40  QKeyEvent* keyEvent = static_cast<QKeyEvent*>(event);
41  QString edit_filter_text = gamelist->search_field->edit_filter->text().toLower();
42 
43  // If the searchfield's text hasn't changed special function keys get checked
44  // If no function key changes the searchfield's text the filter doesn't need to get reloaded
45  if (edit_filter_text == edit_filter_text_old) {
46  switch (keyEvent->key()) {
47  // Escape: Resets the searchfield
48  case Qt::Key_Escape: {
49  if (edit_filter_text_old.isEmpty()) {
50  return QObject::eventFilter(obj, event);
51  } else {
52  gamelist->search_field->edit_filter->clear();
53  edit_filter_text = "";
54  }
55  break;
56  }
57  // Return and Enter
58  // If the enter key gets pressed first checks how many and which entry is visible
59  // If there is only one result launch this game
60  case Qt::Key_Return:
61  case Qt::Key_Enter: {
62  if (gamelist->search_field->visible == 1) {
63  QString file_path = gamelist->getLastFilterResultItem();
64 
65  // To avoid loading error dialog loops while confirming them using enter
66  // Also users usually want to run a diffrent game after closing one
67  gamelist->search_field->edit_filter->clear();
68  edit_filter_text = "";
69  emit gamelist->GameChosen(file_path);
70  } else {
71  return QObject::eventFilter(obj, event);
72  }
73  break;
74  }
75  default:
76  return QObject::eventFilter(obj, event);
77  }
78  }
79  edit_filter_text_old = edit_filter_text;
80  return QObject::eventFilter(obj, event);
81 }

165 static bool ContainsAllWords(const QString& haystack, const QString& userinput) {
166  const QStringList userinput_split =
167  userinput.split(' ', QString::SplitBehavior::SkipEmptyParts);
168 
169  return std::all_of(userinput_split.begin(), userinput_split.end(),
170  [&haystack](const QString& s) { return haystack.contains(s); });
171 }

186 // Event in order to filter the gamelist after editing the searchfield
187 void GameList::onTextChanged(const QString& newText) {
188  int folderCount = tree_view->model()->rowCount();
189  QString edit_filter_text = newText.toLower();
190  QStandardItem* folder;
191  int childrenTotal{};
192 
193  // If the searchfield is empty every item is visible
194  // Otherwise the filter gets applied
195  if (edit_filter_text.isEmpty()) {
196  for (int i{}; i < folderCount; ++i) {
197  folder = item_model->item(i, 0);
198  QModelIndex folder_index = folder->index();
199  int childrenCount = folder->rowCount();
200  for (int j{}; j < childrenCount; ++j) {
201  ++childrenTotal;
202  tree_view->setRowHidden(j, folder_index, false);
203  }
204  }
205  search_field->setFilterResult(childrenTotal, childrenTotal);
206  } else {
207  int result_count{};
208  for (int i{}; i < folderCount; ++i) {
209  folder = item_model->item(i, 0);
210  QModelIndex folder_index{folder->index()};
211  int childrenCount{folder->rowCount()};
212  for (int j{}; j < childrenCount; ++j) {
213  ++childrenTotal;
214  const QStandardItem* child{folder->child(j, 0)};
215  const QString file_path{
216  child->data(GameListItemPath::FullPathRole).toString().toLower()};
217  QString file_name{file_path.mid(file_path.lastIndexOf("/") + 1)};
218  const QString file_title{
219  child->data(GameListItemPath::TitleRole).toString().toLower()};
220  const QString file_programid{
221  child->data(GameListItemPath::ProgramIdRole).toString().toLower()};
222 
223  // Only items which filename in combination with its title contains all words
224  // that are in the searchfield will be visible in the gamelist
225  // The search is case insensitive because of toLower()
226  // I decided not to use Qt::CaseInsensitive in ContainsAllWords to prevent
227  // multiple conversions of edit_filter_text for each game in the gamelist
228  if (ContainsAllWords(file_name.append(' ').append(file_title), edit_filter_text) ||
229  (file_programid.count() == 16 && edit_filter_text.contains(file_programid))) {
230  tree_view->setRowHidden(j, folder_index, false);
231  ++result_count;
232  } else {
233  tree_view->setRowHidden(j, folder_index, true);
234  }
235  search_field->setFilterResult(result_count, childrenTotal);
236  }
237  }
238  }

345 void GameList::ValidateEntry(const QModelIndex& item) {
346  // The click should still register in the first column no matter which column was
347  // clicked
348  int row{item_model->itemFromIndex(item)->row()};
349  const QStandardItem* parent{item_model->itemFromIndex(item.parent())};
350  if (parent == nullptr)
351  parent = item_model->invisibleRootItem();
352  const QStandardItem* child{parent->child(row, COLUMN_NAME)};
353 
354  switch (static_cast<GameListItemType>(child->type())) {
355  case GameListItemType::Game: {
356  QString file_path{child->data(GameListItemPath::FullPathRole).toString()};
357  if (file_path.isEmpty())
358  return;
359  QFileInfo file_info{file_path};
360  if (!file_info.exists() || file_info.isDir())
361  return;
362  // Users usually want to run a different game after closing one
363  search_field->clear();
364  emit GameChosen(file_path);
365  break;
366  }
367  case GameListItemType::AddDir:
368  emit AddDirectory();
369  break;
370  }

373 bool GameList::isEmpty() {
374  for (int i{}; i < item_model->invisibleRootItem()->rowCount(); i++) {
375  const QStandardItem* child{item_model->invisibleRootItem()->child(i)};
376  GameListItemType type{static_cast<GameListItemType>(child->type())};

386 void GameList::DonePopulating(QStringList watch_list) {
387  if (isEmpty()) {
388  emit ShowList(false);
389  } else {
390  item_model->invisibleRootItem()->appendRow(new GameListAddDir());
391  emit ShowList(true);
392  }
393 
394  // Clear out the old directories to watch for changes and add the new ones
395  auto watch_dirs{watcher->directories()};
396  if (!watch_dirs.isEmpty()) {
397  watcher->removePaths(watch_dirs);
398  }
399  // Workaround: Add the watch paths in chunks to allow the gui to refresh
400  // This prevents the UI from stalling when a large number of watch paths are added
401  // Also artificially caps the watcher to a certain number of directories
402  constexpr int LIMIT_WATCH_DIRECTORIES{5000};
403  constexpr int SLICE_SIZE{25};
404  int len = std::min(watch_list.length(), LIMIT_WATCH_DIRECTORIES);
405  for (int i{}; i < len; i += SLICE_SIZE) {
406  watcher->addPaths(watch_list.mid(i, i + SLICE_SIZE));
407  QCoreApplication::processEvents();
408  }
409  tree_view->setEnabled(true);
410  int folderCount{tree_view->model()->rowCount()};
411  int childrenTotal{};
412  for (int i{}; i < folderCount; ++i) {
413  int childrenCount{item_model->item(i, 0)->rowCount()};
414  for (int j{}; j < childrenCount; ++j) {
415  ++childrenTotal;
416  }
417  }
418  search_field->setFilterResult(childrenTotal, childrenTotal);
419  if (childrenTotal > 0) {
420  search_field->setFocus();
421  }
422 }

543 void GameList::PopulateAsync(QList<UISettings::GameDir>& game_dirs) {
544  tree_view->setEnabled(false);
545  // Delete any rows that might already exist if we're repopulating
546  item_model->removeRows(0, item_model->rowCount());
547  search_field->clear();
548 
549  emit ShouldCancelWorker();
550 
551  GameListWorker* worker{new GameListWorker(game_dirs)};
552 
553  connect(worker, &GameListWorker::EntryReady, this, &GameList::AddEntry, Qt::QueuedConnection);
554  connect(worker, &GameListWorker::DirEntryReady, this, &GameList::AddDirEntry,
555  Qt::QueuedConnection);
556  connect(worker, &GameListWorker::Finished, this, &GameList::DonePopulating,
557  Qt::QueuedConnection);
558  // Use DirectConnection here because worker->Cancel() is thread-safe and we want it to
559  // cancel without delay.
560  connect(this, &GameList::ShouldCancelWorker, worker, &GameListWorker::Cancel,
561  Qt::DirectConnection);
562 
563  QThreadPool::globalInstance()->start(worker);
564  current_worker = std::move(worker);
565 }

615 void GameListWorker::AddFstEntriesToGameList(const std::string& dir_path, unsigned int recursion,
616  GameListDir* parent_dir) {
617  const auto callback{[this, recursion, parent_dir](u64* num_entries_out,
618  const std::string& directory,
619  const std::string& virtual_name) -> bool {
620  std::string physical_name{directory + DIR_SEP + virtual_name};
621 
622  if (stop_processing)
623  return false; // Breaks the callback loop.
624 
625  bool is_dir = FileUtil::IsDirectory(physical_name);
626  if (!is_dir && HasSupportedFileExtension(physical_name)) {
627  std::unique_ptr<Loader::AppLoader> loader = Loader::GetLoader(physical_name);
628  if (!loader)
629  return true;
630 
631  u64 program_id{};
632  loader->ReadProgramId(program_id);
633 
634  std::vector<u8> smdh{[program_id, &loader]() -> std::vector<u8> {
635  std::vector<u8> original_smdh;
636  loader->ReadIcon(original_smdh);
637 
638  if (program_id < 0x0004000000000000 || program_id > 0x00040000FFFFFFFF)
639  return original_smdh;
640 
641  std::string update_path = Service::AM::GetTitleContentPath(
642  Service::FS::MediaType::SDMC, program_id + 0x0000000E00000000);
643 
644  if (!FileUtil::Exists(update_path))
645  return original_smdh;
646 
647  std::unique_ptr<Loader::AppLoader> update_loader = Loader::GetLoader(update_path);
648 
649  if (!update_loader)
650  return original_smdh;
651 
652  std::vector<u8> update_smdh;
653  update_loader->ReadIcon(update_smdh);
654  return update_smdh;
655  }()};
656 
657  auto it{compatibility_database.find(program_id)};
658 
659  Compatibility compatibility{Compatibility::NotTested};
660  if (it != compatibility_database.end())
661  compatibility = it->second;
662 
663  emit EntryReady(
664  {
665  new GameListItemPath(QString::fromStdString(physical_name), smdh, program_id),
666  new GameListItemCompat(compatibility),
667  new GameListItemRegion(smdh),
668  new GameListItem(
669  QString::fromStdString(Loader::GetFileTypeString(loader->GetFileType()))),
670  new GameListItemSize(FileUtil::GetSize(physical_name)),
671  },
672  parent_dir);
673 
674  } else if (is_dir && recursion > 0) {
675  watch_list.append(QString::fromStdString(physical_name));
676  AddFstEntriesToGameList(physical_name, recursion - 1, parent_dir);
677  }
678 
679  return true;
680  }};
681 
682  FileUtil::ForeachDirectoryEntry(nullptr, dir_path, callback);
683 }
