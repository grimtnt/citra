455 bool GMainWindow::LoadROM(const QString& filename) {
456  // Shutdown previous session if the emu thread is still active...
457  if (emu_thread != nullptr)
458  ShutdownGame();
459 
460  render_window->InitRenderTarget();
461  render_window->MakeCurrent();
462 
463  if (!gladLoadGL()) {
464  QMessageBox::critical(this, "OpenGL 3.3 Unsupported",
465  "Your GPU may not support OpenGL 3.3, or you do not "
466  "have the latest graphics driver.");
467  return false;
468  }
469 
470  Core::System& system{Core::System::GetInstance()};
471 
472  const Core::System::ResultStatus result{system.Load(*render_window, filename.toStdString())};
473 
474  if (result != Core::System::ResultStatus::Success) {
475  switch (result) {
476  case Core::System::ResultStatus::ErrorGetLoader:
477  LOG_CRITICAL(Frontend, "Failed to obtain loader for {}!", filename.toStdString());
478  QMessageBox::critical(
479  this, "Invalid ROM Format",
480  "Your ROM format is not supported.<br/>Please follow the guides to redump your "
481  "<a "
482  "href='https://github.com/valentinvanelslande/citra/wiki/"
483  "Dumping-Game-Cartridges/'>game "
484  "cartridges</a> or "
485  "<a "
486  "href='https://github.com/valentinvanelslande/citra/wiki/"
487  "Dumping-Installed-Titles/'>installed "
488  "titles</a>.");
489  break;
490 
491  case Core::System::ResultStatus::ErrorSystemMode:
492  LOG_CRITICAL(Frontend, "Failed to load ROM!");
493  QMessageBox::critical(
494  this, "ROM Corrupted",
495  "Your ROM is corrupted. <br/>Please follow the guides to redump your "
496  "<a "
497  "href='https://github.com/valentinvanelslande/citra/wiki/"
498  "Dumping-Game-Cartridges/'>game "
499  "cartridges</a> or "
500  "<a "
501  "href='https://github.com/valentinvanelslande/citra/wiki/"
502  "Dumping-Installed-Titles/'>installed "
503  "titles</a>.");
504  break;
505 
506  case Core::System::ResultStatus::ErrorLoader_ErrorEncrypted: {
507  QMessageBox::critical(
508  this, "ROM Encrypted",
509  "Your ROM is encrypted. <br/>Please follow the guides to redump your "
510  "<a "
511  "href='https://github.com/valentinvanelslande/citra/wiki/"
512  "Dumping-Game-Cartridges/'>game "
513  "cartridges</a> or "
514  "<a "
515  "href='https://github.com/valentinvanelslande/citra/wiki/"
516  "Dumping-Installed-Titles/'>installed "
517  "titles</a>.");
518  break;
519  }
520  case Core::System::ResultStatus::ErrorLoader_ErrorInvalidFormat:
521  QMessageBox::critical(
522  this, "Invalid ROM Format",
523  "Your ROM format is not supported.<br/>Please follow the guides to redump your "
524  "<a "
525  "href='https://github.com/valentinvanelslande/citra/wiki/"
526  "Dumping-Game-Cartridges/'>game "
527  "cartridges</a> or "
528  "<a "
529  "href='https://github.com/valentinvanelslande/citra/wiki/"
530  "Dumping-Installed-Titles/'>installed "
531  "titles</a>.");
532  break;
533 
534  case Core::System::ResultStatus::ErrorVideoCore:
535  QMessageBox::critical(
536  this, "Video Core Error",
537  "An error has occured. Please see the log for more details."
538  "To access the log, Click Open Log Location in the general tab of the "
539  "configuration window.<br/><br/>"
540  "Ensure that you have the latest graphics drivers for your GPU.");
541  break;
542 
543  case Core::System::ResultStatus::ErrorVideoCore_ErrorGenericDrivers:
544  QMessageBox::critical(
545  this, "Video Core Error",
546  "You are running default Windows drivers "
