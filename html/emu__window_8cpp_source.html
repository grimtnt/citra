61 static bool IsWithinTouchscreen(const Layout::FramebufferLayout& layout, unsigned framebuffer_x,
62  unsigned framebuffer_y) {
63  if (Settings::values.factor_3d != 0) {
64  return (framebuffer_y >= layout.bottom_screen.top &&
65  framebuffer_y < layout.bottom_screen.bottom &&
66  framebuffer_x >= layout.bottom_screen.left / 2 &&
67  framebuffer_x < layout.bottom_screen.right / 2);
68  } else {
69  return (framebuffer_y >= layout.bottom_screen.top &&
70  framebuffer_y < layout.bottom_screen.bottom &&
71  framebuffer_x >= layout.bottom_screen.left &&
72  framebuffer_x < layout.bottom_screen.right);
73  }
74 }

86 void EmuWindow::TouchPressed(unsigned framebuffer_x, unsigned framebuffer_y) {
87  if (!IsWithinTouchscreen(framebuffer_layout, framebuffer_x, framebuffer_y))
88  return;
89 
90  std::lock_guard<std::mutex> guard{touch_state->mutex};
91  if (Settings::values.factor_3d != 0) {
92  touch_state->touch_x =
93  static_cast<float>(framebuffer_x - framebuffer_layout.bottom_screen.left / 2) /
94  (framebuffer_layout.bottom_screen.right / 2 -
95  framebuffer_layout.bottom_screen.left / 2);
96  } else {
97  touch_state->touch_x =
98  static_cast<float>(framebuffer_x - framebuffer_layout.bottom_screen.left) /
99  (framebuffer_layout.bottom_screen.right - framebuffer_layout.bottom_screen.left);
100  }
101  touch_state->touch_y =
102  static_cast<float>(framebuffer_y - framebuffer_layout.bottom_screen.top) /
103  (framebuffer_layout.bottom_screen.bottom - framebuffer_layout.bottom_screen.top);
104 
105  touch_state->touch_pressed = true;
106 }
