35  if (!last_path.empty()) {
36  return last_path;
37  }
38  QList<QByteArray> types{QImageReader::supportedImageFormats()};
39  QList<QString> temp_filters{};

41  temp_filters << QString("*." + QString(type));

43 
44  QString filter{QString("Supported image files (%1)").arg(temp_filters.join(" "))};
45  last_path = QFileDialog::getOpenFileName(nullptr, "Open File", ".", filter).toStdString();

49 std::unique_ptr<CameraInterface> StillImageCameraFactory::Create(const std::string& config,
50  const Service::CAM::Flip& flip) {
51  std::string real_config{config};
52  if (config.empty()) {
53  // call GetFilePath() in UI thread (note: StillImageCameraFactory itself is initialized in
54  // UI thread, so we can just pass in "this" here)
55  if (thread() == QThread::currentThread()) {
56  real_config = GetFilePath();
57  } else {
58  QMetaObject::invokeMethod(this, "GetFilePath", Qt::BlockingQueuedConnection,
59  Q_RETURN_ARG(std::string, real_config));
60  }
61  }
62  QImage image{QString::fromStdString(real_config)};
63  if (image.isNull()) {
64  LOG_ERROR(Service_CAM, "Couldn't load image \"{}\"", real_config);
65  }
66  return std::make_unique<StillImageCamera>(image, flip);
