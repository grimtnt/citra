58 void NDM_U::SuspendDaemons(Kernel::HLERequestContext& ctx) {
59  IPC::RequestParser rp{ctx, 0x06, 1, 0};
60  u32 bit_mask{rp.Pop<u32>() & 0xF};
61  daemon_bit_mask =
62  static_cast<DaemonMask>(static_cast<u32>(default_daemon_bit_mask) & ~bit_mask);
63  for (std::size_t index{}; index < daemon_status.size(); ++index) {
