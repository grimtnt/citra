35 DirectionState GetStickDirectionState(s16 circle_pad_x, s16 circle_pad_y) {
36  // 30 degree and 60 degree are angular thresholds for directions
37  constexpr float TAN30{0.577350269f};
38  constexpr float TAN60{1 / TAN30};
39  // a circle pad radius greater than 40 will trigger circle pad direction
40  constexpr int CIRCLE_PAD_THRESHOLD_SQUARE{40 * 40};
41  DirectionState state{false, false, false, false};
42 
43  if (circle_pad_x * circle_pad_x + circle_pad_y * circle_pad_y > CIRCLE_PAD_THRESHOLD_SQUARE) {
44  float t = std::abs(static_cast<float>(circle_pad_y) / circle_pad_x);
45 
46  if (circle_pad_x != 0 && t < TAN60) {
47  if (circle_pad_x > 0)
48  state.right = true;
49  else
50  state.left = true;
51  }
52 
53  if (circle_pad_x == 0 || t > TAN30) {
54  if (circle_pad_y > 0)
55  state.up = true;
56  else
57  state.down = true;
58  }
59  }
60 
61  return state;
62 }

76 void Module::UpdatePadCallback(u64 userdata, s64 cycles_late) {
77  SharedMem* mem{reinterpret_cast<SharedMem*>(shared_mem->GetPointer())};
78 
79  if (is_device_reload_pending.exchange(false))
80  LoadInputDevices();
81 
82  PadState state;
83  using namespace Settings::NativeButton;
84  state.a.Assign(buttons[A - BUTTON_HID_BEGIN]->GetStatus());
85  state.b.Assign(buttons[B - BUTTON_HID_BEGIN]->GetStatus());
86  state.x.Assign(buttons[X - BUTTON_HID_BEGIN]->GetStatus());
87  state.y.Assign(buttons[Y - BUTTON_HID_BEGIN]->GetStatus());
88  state.right.Assign(buttons[Right - BUTTON_HID_BEGIN]->GetStatus());
89  state.left.Assign(buttons[Left - BUTTON_HID_BEGIN]->GetStatus());
90  state.up.Assign(buttons[Up - BUTTON_HID_BEGIN]->GetStatus());
91  state.down.Assign(buttons[Down - BUTTON_HID_BEGIN]->GetStatus());
92  state.l.Assign(buttons[L - BUTTON_HID_BEGIN]->GetStatus());
93  state.r.Assign(buttons[R - BUTTON_HID_BEGIN]->GetStatus());
94  state.start.Assign(buttons[Start - BUTTON_HID_BEGIN]->GetStatus());
95  state.select.Assign(buttons[Select - BUTTON_HID_BEGIN]->GetStatus());
96  if (button_home->GetStatus())
97  Core::System::GetInstance().RequestShutdown();
98 
99  // Get current circle pad position and update circle pad direction
100  s16 circle_pad_x{}, circle_pad_y{};
101  if (use_override_circle_pad) {
102  circle_pad_x = override_circle_x;
103  circle_pad_y = override_circle_y;
104  } else {
105  float circle_pad_x_f, circle_pad_y_f;
106  std::tie(circle_pad_x_f, circle_pad_y_f) = circle_pad->GetStatus();
107  constexpr int MAX_CIRCLEPAD_POS = 0x9C; // Max value for a circle pad position
108  circle_pad_x = static_cast<s16>(circle_pad_x_f * MAX_CIRCLEPAD_POS);
109  circle_pad_y = static_cast<s16>(circle_pad_y_f * MAX_CIRCLEPAD_POS);
110  }
111  Core::Movie::GetInstance().HandlePadAndCircleStatus(state, circle_pad_x, circle_pad_y);
112 
113  const DirectionState direction = GetStickDirectionState(circle_pad_x, circle_pad_y);
114  state.circle_up.Assign(direction.up);
115  state.circle_down.Assign(direction.down);
116  state.circle_left.Assign(direction.left);
117  state.circle_right.Assign(direction.right);
118 
119  if (use_override_pad_state) {
120  state.hex = override_pad_state;
121  }
122  inputs_this_frame.hex = state.hex;
123 
124  mem->pad.current_state.hex = state.hex;
125  mem->pad.index = next_pad_index;
126  next_pad_index = (next_pad_index + 1) % mem->pad.entries.size();
127 
128  // Get the previous Pad state
129  u32 last_entry_index = (mem->pad.index - 1) % mem->pad.entries.size();
130  PadState old_state = mem->pad.entries[last_entry_index].current_state;
131 
132  // Compute bitmask with 1s for bits different from the old state
133  PadState changed = {{(state.hex ^ old_state.hex)}};
134 
135  // Get the current Pad entry
136  PadDataEntry& pad_entry = mem->pad.entries[mem->pad.index];
137 
138  // Update entry properties
139  pad_entry.current_state.hex = state.hex;
140  pad_entry.delta_additions.hex = changed.hex & state.hex;
141  pad_entry.delta_removals.hex = changed.hex & old_state.hex;
142  pad_entry.circle_pad_x = circle_pad_x;
143  pad_entry.circle_pad_y = circle_pad_y;
144 
145  // If we just updated index 0, provide a new timestamp
146  if (mem->pad.index == 0) {
147  mem->pad.index_reset_ticks_previous = mem->pad.index_reset_ticks;
148  mem->pad.index_reset_ticks = (s64)CoreTiming::GetTicks();
149  }
150 
151  mem->touch.index = next_touch_index;
152  next_touch_index = (next_touch_index + 1) % mem->touch.entries.size();
153 
154  // Get the current touch entry
155  TouchDataEntry& touch_entry = mem->touch.entries[mem->touch.index];
156  if (use_override_touch) {
157  touch_entry.x = override_touch_x;
158  touch_entry.y = override_touch_y;
159  touch_entry.valid.Assign(override_touch_valid ? 1 : 0);
160  } else {
161  float x, y;
162  bool pressed;
163  std::tie(x, y, pressed) = touch_device->GetStatus();
164  touch_entry.x = static_cast<u16>(x * Core::kScreenBottomWidth);
165  touch_entry.y = static_cast<u16>(y * Core::kScreenBottomHeight);
166  touch_entry.valid.Assign(pressed ? 1 : 0);
167  }
168 
169  Core::Movie::GetInstance().HandleTouchStatus(touch_entry);
170 
171  // TODO(bunnei): We're not doing anything with offset 0xA8 + 0x18 of HID SharedMemory, which
172  // supposedly is "Touch-screen entry, which contains the raw coordinate data prior to being
173  // converted to pixel coordinates." (http://3dbrew.org/wiki/HID_Shared_Memory#Offset_0xA8).
174 
175  // If we just updated index 0, provide a new timestamp
176  if (mem->touch.index == 0) {
177  mem->touch.index_reset_ticks_previous = mem->touch.index_reset_ticks;
178  mem->touch.index_reset_ticks = (s64)CoreTiming::GetTicks();
179  }
180 
181  // Signal both handles when there's an update to Pad or touch
182  event_pad_or_touch_1->Signal();
183  event_pad_or_touch_2->Signal();
184 
185  // Reschedule recurrent event
186  CoreTiming::ScheduleEvent(pad_update_ticks - cycles_late, pad_update_event);
187 }

189 void Module::UpdateAccelerometerCallback(u64 userdata, s64 cycles_late) {
190  SharedMem* mem{reinterpret_cast<SharedMem*>(shared_mem->GetPointer())};
191 
192  mem->accelerometer.index = next_accelerometer_index;
193  next_accelerometer_index = (next_accelerometer_index + 1) % mem->accelerometer.entries.size();
194  AccelerometerDataEntry& accelerometer_entry{
195  mem->accelerometer.entries[mem->accelerometer.index]};
196  if (use_override_motion) {
197  accelerometer_entry.x = override_motion_x;
198  accelerometer_entry.y = override_motion_y;
199  accelerometer_entry.z = override_motion_z;
200  } else {
201  Math::Vec3<float> accel;
202  std::tie(accel, std::ignore) = motion_device->GetStatus();
203  accel *= accelerometer_coef;
204  // TODO(wwylele): do a time stretch like the one in UpdateGyroscopeCallback
205  // The time stretch formula should be like
206  // stretched_vector = (raw_vector - gravity) * stretch_ratio + gravity
207  accelerometer_entry.x = static_cast<s16>(accel.x);
208  accelerometer_entry.y = static_cast<s16>(accel.y);
209  accelerometer_entry.z = static_cast<s16>(accel.z);
210  }
211 
212  Core::Movie::GetInstance().HandleAccelerometerStatus(accelerometer_entry);
213 
214  // Make up "raw" entry
215  // TODO(wwylele):
216  // From hardware testing, the raw_entry values are approximately, but not exactly, as twice as
217  // corresponding entries (or with a minus sign). It may caused by system calibration to the
218  // accelerometer. Figure out how it works, or, if no game reads raw_entry, the following three
219  // lines can be removed and leave raw_entry unimplemented.
220  mem->accelerometer.raw_entry.x = -2 * accelerometer_entry.x;
221  mem->accelerometer.raw_entry.z = 2 * accelerometer_entry.y;
222  mem->accelerometer.raw_entry.y = -2 * accelerometer_entry.z;
223 
224  // If we just updated index 0, provide a new timestamp
225  if (mem->accelerometer.index == 0) {
226  mem->accelerometer.index_reset_ticks_previous = mem->accelerometer.index_reset_ticks;
227  mem->accelerometer.index_reset_ticks = (s64)CoreTiming::GetTicks();
228  }
229 
230  event_accelerometer->Signal();
231 
232  // Reschedule recurrent event
233  CoreTiming::ScheduleEvent(accelerometer_update_ticks - cycles_late, accelerometer_update_event);
234 }

240 void Module::UpdateGyroscopeCallback(u64 userdata, s64 cycles_late) {
241  SharedMem* mem{reinterpret_cast<SharedMem*>(shared_mem->GetPointer())};
242 
243  mem->gyroscope.index = next_gyroscope_index;
244  next_gyroscope_index = (next_gyroscope_index + 1) % mem->gyroscope.entries.size();
245 
246  GyroscopeDataEntry& gyroscope_entry{mem->gyroscope.entries[mem->gyroscope.index]};
247 
248  if (use_override_motion) {
249  gyroscope_entry.x = override_motion_roll;
250  gyroscope_entry.y = override_motion_pitch;
251  gyroscope_entry.z = override_motion_yaw;
252  } else {
253  Math::Vec3<float> gyro;
254  std::tie(std::ignore, gyro) = motion_device->GetStatus();
255  double stretch{Core::System::GetInstance().perf_stats.GetLastFrameTimeScale()};
256  gyro *= gyroscope_coef * static_cast<float>(stretch);
257  gyroscope_entry.x = static_cast<s16>(gyro.x);
258  gyroscope_entry.y = static_cast<s16>(gyro.y);
259  gyroscope_entry.z = static_cast<s16>(gyro.z);
260  }
261 
262  Core::Movie::GetInstance().HandleGyroscopeStatus(gyroscope_entry);
263 
264  // Make up "raw" entry
265  mem->gyroscope.raw_entry.x = gyroscope_entry.x;
266  mem->gyroscope.raw_entry.z = -gyroscope_entry.y;
267  mem->gyroscope.raw_entry.y = gyroscope_entry.z;
268 
269  // If we just updated index 0, provide a new timestamp
270  if (mem->gyroscope.index == 0) {
271  mem->gyroscope.index_reset_ticks_previous = mem->gyroscope.index_reset_ticks;
272  mem->gyroscope.index_reset_ticks = (s64)CoreTiming::GetTicks();
273  }
274 
275  event_gyroscope->Signal();
276 
277  // Reschedule recurrent event
278  CoreTiming::ScheduleEvent(gyroscope_update_ticks - cycles_late, gyroscope_update_event);
279 }
