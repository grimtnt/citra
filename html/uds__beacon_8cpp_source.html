127 std::vector<u8> GenerateNintendoNetworkInfoTag(const NetworkInfo& network_info) {
128  NetworkInfoTag tag{};
129  tag.header.tag_id = static_cast<u8>(TagId::VendorSpecific);
130  tag.header.length =
131  sizeof(NetworkInfoTag) - sizeof(TagHeader) + network_info.application_data_size;
132  tag.appdata_size = network_info.application_data_size;
133  // Set the hash to zero initially, it will be updated once we calculate it.
134  tag.sha_hash = {};
135 
136  // Ensure the network structure has the correct OUI and OUI type.
137  ASSERT(network_info.oui_type == static_cast<u8>(NintendoTagId::NetworkInfo));
138  ASSERT(network_info.oui_value == NintendoOUI);
139 
140  // This tag contains the network info structure starting at the OUI.
141  std::memcpy(tag.network_info.data(), &network_info.oui_value, tag.network_info.size());
142 
143  // Copy the tag and the data so we can calculate the SHA1 over it.
144  std::vector<u8> buffer(sizeof(tag) + network_info.application_data_size);
145  std::memcpy(buffer.data(), &tag, sizeof(tag));
146  std::memcpy(buffer.data() + sizeof(tag), network_info.application_data.data(),
147  network_info.application_data_size);
148 
149  // Calculate the SHA1 of the contents of the tag.
150  std::array<u8, CryptoPP::SHA1::DIGESTSIZE> hash;
151  CryptoPP::SHA1().CalculateDigest(hash.data(),
152  buffer.data() + offsetof(NetworkInfoTag, network_info),
153  buffer.size() - sizeof(TagHeader));
154 
155  // Copy it directly into the buffer, overwriting the zeros that we had previously placed there.
156  std::memcpy(buffer.data() + offsetof(NetworkInfoTag, sha_hash), hash.data(), hash.size());
157 
158  return buffer;
159 }

180 /*
181  * Serializes the node information into the format needed for network transfer,
182  * and then encrypts it with the NWM key.
183  * @returns The serialized and encrypted node information.
184  */
185 std::vector<u8> GeneratedEncryptedData(const NetworkInfo& network_info, const NodeList& nodes) {
186  std::vector<u8> buffer(sizeof(BeaconData));
187 
188  BeaconData data{};
189  std::memcpy(buffer.data(), &data, sizeof(BeaconData));
190 
191  for (const NodeInfo& node : nodes) {
192  // Serialize each node and convert the data from
193  // host byte-order to Big Endian.
194  BeaconNodeInfo info{};
195  info.friend_code_seed = node.friend_code_seed;
196  info.network_node_id = node.network_node_id;
197  for (int i{}; i < info.username.size(); ++i)
198  info.username[i] = node.username[i];
199 
200  buffer.insert(buffer.end(), reinterpret_cast<u8*>(&info),
201  reinterpret_cast<u8*>(&info) + sizeof(info));
202  }
203 
204  // Calculate the MD5 hash of the data in the buffer, not including the hash field.
205  std::array<u8, CryptoPP::Weak::MD5::DIGESTSIZE> hash;
206  CryptoPP::Weak::MD5().CalculateDigest(hash.data(),
207  buffer.data() + offsetof(BeaconData, bitmask),
208  buffer.size() - sizeof(data.md5_hash));
209 
210  // Copy the hash into the buffer.
211  std::memcpy(buffer.data(), hash.data(), hash.size());
212 
213  // Encrypt the data using AES-CTR and the NWM beacon key.
214  using CryptoPP::AES;
215  std::array<u8, AES::BLOCKSIZE> counter = GetBeaconCryptoCTR(network_info);
216  CryptoPP::CTR_Mode<AES>::Encryption aes;
217  aes.SetKeyWithIV(nwm_beacon_key.data(), AES::BLOCKSIZE, counter.data());
218  aes.ProcessData(buffer.data(), buffer.data(), buffer.size());
219 
220  return buffer;
221 }

266 std::vector<u8> GenerateNintendoSecondEncryptedDataTag(const NetworkInfo& network_info,
267  const NodeList& nodes) {
268  // This tag is only present if the payload is larger than EncryptedDataSizeCutoff (0xFA).
269  if (nodes.size() * sizeof(NodeInfo) <= EncryptedDataSizeCutoff)
270  return {};
271 
272  const std::size_t payload_size{nodes.size() * sizeof(NodeInfo) - EncryptedDataSizeCutoff};
273  const std::size_t tag_length{sizeof(EncryptedDataTag) - sizeof(TagHeader) + payload_size};
274 
275  EncryptedDataTag tag{};
276  tag.header.tag_id = static_cast<u8>(TagId::VendorSpecific);
277  tag.header.length = static_cast<u8>(tag_length);
278  tag.oui_type = static_cast<u8>(NintendoTagId::EncryptedData1);
279  tag.oui = NintendoOUI;
280 
281  std::vector<u8> buffer(sizeof(tag) + payload_size);
282  std::memcpy(buffer.data(), &tag, sizeof(tag));
283 
284  std::vector<u8> encrypted_data = GeneratedEncryptedData(network_info, nodes);
285  std::memcpy(buffer.data() + sizeof(tag), encrypted_data.data() + EncryptedDataSizeCutoff,
286  payload_size);
