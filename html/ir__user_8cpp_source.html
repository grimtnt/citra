17 // This is a header that will present in the ir:USER shared memory if it is initialized with
18 // InitializeIrNopShared service function. Otherwise the shared memory doesn't have this header if
19 // it is initialized with InitializeIrNop service function.
20 struct SharedMemoryHeader {
21  u32_le latest_receive_error_result;
22  u32_le latest_send_error_result;
23  // TODO(wwylele): for these fields below, make them enum when the meaning of values is known.
24  u8 connection_status;

27  u8 machine_id;

36 };

66 class BufferManager {
67 public:
68  BufferManager(Kernel::SharedPtr<Kernel::SharedMemory> shared_memory_, u32 info_offset_,
69  u32 buffer_offset_, u32 max_packet_count_, u32 buffer_size)
70  : shared_memory(shared_memory_), info_offset(info_offset_), buffer_offset(buffer_offset_),
71  max_packet_count(max_packet_count_),
72  max_data_size(buffer_size - sizeof(PacketInfo) * max_packet_count_) {
73  UpdateBufferInfo();
74  }
75 
81  bool Put(const std::vector<u8>& packet) {
82  if (info.packet_count == max_packet_count)
83  return false;
84 
85  u32 write_offset;
86 
87  // finds free space offset in data buffer
88  if (info.packet_count == 0) {
89  write_offset = 0;
90  if (packet.size() > max_data_size)
91  return false;
92  } else {
93  const u32 last_index = (info.end_index + max_packet_count - 1) % max_packet_count;
94  const PacketInfo first = GetPacketInfo(info.begin_index);
95  const PacketInfo last = GetPacketInfo(last_index);
96  write_offset = (last.offset + last.size) % max_data_size;
97  const u32 free_space = (first.offset + max_data_size - write_offset) % max_data_size;
98  if (packet.size() > free_space)
99  return false;
100  }
101 
102  // writes packet info
103  PacketInfo packet_info{write_offset, static_cast<u32>(packet.size())};
104  SetPacketInfo(info.end_index, packet_info);
105 
106  // writes packet data
107  for (std::size_t i{}; i < packet.size(); ++i) {
108  *GetDataBufferPointer((write_offset + i) % max_data_size) = packet[i];
109  }
110 
111  // updates buffer info
112  info.end_index++;
113  info.end_index %= max_packet_count;
114  info.packet_count++;
115  UpdateBufferInfo();
116  return true;
117  }
118 
124  bool Release(u32 count) {
125  if (info.packet_count < count)
126  return false;
127 
128  info.packet_count -= count;
129  info.begin_index += count;
130  info.begin_index %= max_packet_count;
131  UpdateBufferInfo();
132  return true;
133  }
134

184 void IR_USER::PutToReceive(const std::vector<u8>& payload) {
185  LOG_TRACE(Service_IR, "called, data={}", Common::ArrayToString(payload.data(), payload.size()));
186  std::size_t size = payload.size();
187 
188  std::vector<u8> packet;
189 
190  // Builds packet header. For the format info:
191  // https://www.3dbrew.org/wiki/IRUSER_Shared_Memory#Packet_structure
192 
193  // fixed value
194  packet.push_back(0xA5);
195  // destination network ID
196  u8 network_id = *(shared_memory->GetPointer(offsetof(SharedMemoryHeader, network_id)));
197  packet.push_back(network_id);
198 
199  // puts the size info.
200  // The highest bit of the first byte is unknown, which is set to zero here. The second highest
201  // bit is a flag that determines whether the size info is in extended form. If the packet size
202  // can be represent within 6 bits, the short form (1 byte) of size info is chosen, the size is
203  // put to the lower bits of this byte, and the flag is clear. If the packet size cannot be
204  // represent within 6 bits, the extended form (2 bytes) is chosen, the lower 8 bits of the size
205  // is put to the second byte, the higher bits of the size is put to the lower bits of the first
206  // byte, and the flag is set. Note that the packet size must be within 14 bits due to this
207  // format restriction, or it will overlap with the flag bit.
208  if (size < 0x40) {
209  packet.push_back(static_cast<u8>(size));
210  } else if (size < 0x4000) {
211  packet.push_back(static_cast<u8>(size >> 8) | 0x40);
212  packet.push_back(static_cast<u8>(size));
213  } else {
214  ASSERT(false);
215  }
216 
217  // puts the payload
218  packet.insert(packet.end(), payload.begin(), payload.end());
219 
220  // calculates CRC and puts to the end
221  packet.push_back(boost::crc<8, 0x07, 0, 0, false, false>(packet.data(), packet.size()));
222 
223  if (receive_buffer->Put(packet)) {
224  receive_event->Signal();
225  } else {
226  LOG_ERROR(Service_IR, "receive buffer is full!");
227  }
228 }

230 void IR_USER::InitializeIrNopShared(Kernel::HLERequestContext& ctx) {
231  IPC::RequestParser rp{ctx, 0x18, 6, 2};
232  const u32 shared_buff_size{rp.Pop<u32>()};
233  const u32 recv_buff_size{rp.Pop<u32>()};
234  const u32 recv_buff_packet_count{rp.Pop<u32>()};
235  const u32 send_buff_size{rp.Pop<u32>()};
236  const u32 send_buff_packet_count{rp.Pop<u32>()};
237  const u8 baud_rate{rp.Pop<u8>()};
238  shared_memory = rp.PopObject<Kernel::SharedMemory>();
239 
240  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
241 
242  shared_memory->name = "IR_USER: shared memory";
243 
244  receive_buffer = std::make_unique<BufferManager>(shared_memory, 0x10, 0x20,
245  recv_buff_packet_count, recv_buff_size);
246  SharedMemoryHeader shared_memory_init{};
247  shared_memory_init.initialized = 1;
248  std::memcpy(shared_memory->GetPointer(), &shared_memory_init, sizeof(SharedMemoryHeader));
249 
250  rb.Push(RESULT_SUCCESS);
251 
252  LOG_INFO(Service_IR,
253  "called, shared_buff_size={}, recv_buff_size={}, "
254  "recv_buff_packet_count={}, send_buff_size={}, "
255  "send_buff_packet_count={}, baud_rate={}",
256  shared_buff_size, recv_buff_size, recv_buff_packet_count, send_buff_size,
257  send_buff_packet_count, baud_rate);
258 }

260 void IR_USER::RequireConnection(Kernel::HLERequestContext& ctx) {
261  IPC::RequestParser rp{ctx, 0x06, 1, 0};
262  const u8 device_id{rp.Pop<u8>()};
263 
264  u8* shared_memory_ptr = shared_memory->GetPointer();
265  if (device_id == 1) {
266  // These values are observed on a New 3DS. The meaning of them is unclear.
267  // TODO (wwylele): should assign network_id a (random?) number
268  shared_memory_ptr[offsetof(SharedMemoryHeader, connection_status)] = 2;
269  shared_memory_ptr[offsetof(SharedMemoryHeader, connection_role)] = 2;
270  shared_memory_ptr[offsetof(SharedMemoryHeader, connected)] = 1;
271 
272  connected_device = extra_hid.get();
273  connected_device->OnConnect();
274  conn_status_event->Signal();
275  } else {
276  LOG_WARNING(Service_IR, "unknown device id {}. Won't connect.", device_id);
277  shared_memory_ptr[offsetof(SharedMemoryHeader, connection_status)] = 1;
278  shared_memory_ptr[offsetof(SharedMemoryHeader, trying_to_connect_status)] = 2;
279  }
280 
281  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
282  rb.Push(RESULT_SUCCESS);
283 
284  LOG_INFO(Service_IR, "called, device_id = {}", device_id);
285 }
