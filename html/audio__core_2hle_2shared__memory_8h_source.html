20 // The application-accessible region of DSP memory consists of two parts. Both are marked as IO and
21 // have Read/Write permissions.
22 //
23 // First Region: 0x1FF50000 (Size: 0x8000)
24 // Second Region: 0x1FF70000 (Size: 0x8000)
25 //
26 // The DSP reads from each region alternately based on the frame counter for each region much like a
27 // double-buffer. The frame counter is located as the very last u16 of each region and is
28 // incremented each audio tick.
29 
30 constexpr u32 region0_offset = 0x50000;
31 constexpr u32 region1_offset = 0x70000;

45 struct u32_dsp {
46  u32_dsp() = default;
47  operator u32() const {
48  return Convert(storage);

64 // There are 15 structures in each memory region. A table of them in the order they appear in memory
65 // is presented below:
66 //
67 // # First Region DSP Address Purpose Control
68 // 5 0x8400 DSP Status DSP
69 // 9 0x8410 DSP Debug Info DSP
70 // 6 0x8540 Final Mix Samples DSP
71 // 2 0x8680 Source Status [24] DSP
72 // 8 0x8710 Compressor Table Application
73 // 4 0x9430 DSP Configuration Application
74 // 7 0x9492 Intermediate Mix Samples DSP + App
75 // 1 0x9E92 Source Configuration [24] Application
76 // 3 0xA792 Source ADPCM Coefficients [24] Application
77 // 10 0xA912 Surround Sound Related
78 // 11 0xAA12 Surround Sound Related
79 // 12 0xAAD2 Surround Sound Related
80 // 13 0xAC52 Surround Sound Related
81 // 14 0xAC5C Surround Sound Related
82 // 0 0xBFFF Frame Counter Application
83 //
84 // #: This refers to the order in which they appear in the DspPipe::Audio DSP pipe.
85 // See also: HLE::PipeRead.
86 //
87 // Note that the above addresses do vary slightly between audio firmwares observed; the addresses
88 // are not fixed in stone. The addresses above are only an examplar; they're what this
89 // implementation does and provides to applications.
90 //
91 // Application requests the DSP service to convert DSP addresses into ARM11 virtual addresses using
92 // the ConvertProcessAddressFromDspDram service call. Applications seem to derive the addresses for
93 // the second region via:
94 // second_region_dsp_addr = first_region_dsp_addr | 0x10000
95 //
96 // Applications maintain most of its own audio state, the memory region is used mainly for
97 // communication and not storage of state.
98 //
99 // In the documentation below, filter and effect transfer functions are specified in the z domain.
100 // (If you are more familiar with the Laplace transform, z = exp(sT). The z domain is the digital
101 // frequency domain, just like how the s domain is the analog frequency domain.)
102 
103 #define INSERT_PADDING_DSPWORDS(num_words) INSERT_PADDING_BYTES(2 * (num_words))
