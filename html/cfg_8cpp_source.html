148 void Module::Interface::GetCountryCodeID(Kernel::HLERequestContext& ctx) {
149  IPC::RequestParser rp{ctx, 0x0A, 1, 0};
150  u16 country_code{rp.Pop<u16>()};
151  u16 country_code_id{};
152 
153  // The following algorithm will fail if the first country code isn't 0.
154  DEBUG_ASSERT(country_codes[0] == 0);
155 
156  for (u16 id{}; id < country_codes.size(); ++id) {
157  if (country_codes[id] == country_code) {
158  country_code_id = id;
159  break;
160  }
161  }
162 
163  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
164  if (0 == country_code_id) {
165  LOG_ERROR(Service_CFG, "requested country code name={}{} is invalid", country_code & 0xff,
166  country_code >> 8);
167  rb.Push(ResultCode(ErrorDescription::NotFound, ErrorModule::Config,
168  ErrorSummary::WrongArgument, ErrorLevel::Permanent));
169  rb.Push<u16>(0x00FF);
170  return;
171  }
172 
173  rb.Push(RESULT_SUCCESS);
174  rb.Push<u16>(country_code_id);
175 }

192 void Module::Interface::GenHashConsoleUnique(Kernel::HLERequestContext& ctx) {
193  IPC::RequestParser rp{ctx, 0x03, 1, 0};
194  const u32 app_id_salt{rp.Pop<u32>() & 0x000FFFFF};
195 
196  IPC::ResponseBuilder rb{rp.MakeBuilder(3, 0)};
197 
198  std::array<u8, 12> buffer;
199  const ResultCode result = cfg->GetConfigInfoBlock(ConsoleUniqueID2BlockID, 8, 2, buffer.data());
200  rb.Push(result);
201  if (result.IsSuccess()) {
202  std::memcpy(&buffer[8], &app_id_salt, sizeof(u32));
203  std::array<u8, CryptoPP::SHA256::DIGESTSIZE> hash;
204  CryptoPP::SHA256().CalculateDigest(hash.data(), buffer.data(), sizeof(buffer));
205  u32 low, high;
206  memcpy(&low, &hash[hash.size() - 8], sizeof(u32));
207  memcpy(&high, &hash[hash.size() - 4], sizeof(u32));
208  rb.Push(low);
209  rb.Push(high);
210  } else {
211  rb.Push<u32>(0);
212  rb.Push<u32>(0);
213  }
214 
215  LOG_DEBUG(Service_CFG, "called app_id_salt=0x{:X}", app_id_salt);
216 }

316 ResultVal<void*> Module::GetConfigInfoBlockPointer(u32 block_id, u32 size, u32 flag) {
317  // Read the header
318  SaveFileConfig* config = reinterpret_cast<SaveFileConfig*>(cfg_config_file_buffer.data());
319 
320  auto itr =
321  std::find_if(std::begin(config->block_entries), std::end(config->block_entries),
322  [&](const SaveConfigBlockEntry& entry) { return entry.block_id == block_id; });
323 
324  if (itr == std::end(config->block_entries)) {
325  LOG_ERROR(Service_CFG, "Config block 0x{:X} with flags {} and size {} was not found",
326  block_id, flag, size);
327  return ResultCode(ErrorDescription::NotFound, ErrorModule::Config,
328  ErrorSummary::WrongArgument, ErrorLevel::Permanent);
329  }
330 
331  if ((itr->flags & flag) == 0) {
332  LOG_ERROR(Service_CFG, "Invalid flag {} for config block 0x{:X} with size {}", flag,
333  block_id, size);
334  return ResultCode(ErrorDescription::NotAuthorized, ErrorModule::Config,
335  ErrorSummary::WrongArgument, ErrorLevel::Permanent);
336  }
337 
338  if (itr->size != size) {
339  LOG_ERROR(Service_CFG, "Invalid size {} for config block 0x{:X} with flags {}", size,
340  block_id, flag);
341  return ResultCode(ErrorDescription::InvalidSize, ErrorModule::Config,
342  ErrorSummary::WrongArgument, ErrorLevel::Permanent);
343  }
344 
345  void* pointer;
346 
347  // The data is located in the block header itself if the size is less than 4 bytes
348  if (itr->size <= 4)
349  pointer = &itr->offset_or_data;
350  else
351  pointer = &cfg_config_file_buffer[itr->offset_or_data];
352 
353  return MakeResult<void*>(pointer);
354 }

371 ResultCode Module::CreateConfigInfoBlk(u32 block_id, u16 size, u16 flags, const void* data) {
372  SaveFileConfig* config{reinterpret_cast<SaveFileConfig*>(cfg_config_file_buffer.data())};
373  if (config->total_entries >= CONFIG_FILE_MAX_BLOCK_ENTRIES)
374  return ResultCode(-1); // TODO(Subv): Find the right error code
375 
376  // Insert the block header with offset 0 for now
377  config->block_entries[config->total_entries] = {block_id, 0, size, flags};
378  if (size > 4) {
379  u32 offset{config->data_entries_offset};
380  // Perform a search to locate the next offset for the new data
381  // use the offset and size of the previous block to determine the new position
382  for (int i{config->total_entries - 1}; i >= 0; --i) {
383  // Ignore the blocks that don't have a separate data offset
384  if (config->block_entries[i].size > 4) {
385  offset = config->block_entries[i].offset_or_data + config->block_entries[i].size;
386  break;
387  }
388  }
389 
390  config->block_entries[config->total_entries].offset_or_data = offset;
391 
392  // Write the data at the new offset
393  memcpy(&cfg_config_file_buffer[offset], data, size);
394  } else {
395  // The offset_or_data field in the header contains the data itself if it's 4 bytes or less
396  memcpy(&config->block_entries[config->total_entries].offset_or_data, data, size);
397  }
398 
399  ++config->total_entries;
400  return RESULT_SUCCESS;
401 }

428 ResultCode Module::FormatConfig() {
429  ResultCode res{DeleteConfigNANDSaveFile()};
430  // The delete command fails if the file doesn't exist, so we have to check that too
431  if (!res.IsSuccess() && res != FileSys::ERROR_FILE_NOT_FOUND) {
432  return res;
433  }
434  // Delete the old data
435  cfg_config_file_buffer.fill(0);
436  // Create the header
437  SaveFileConfig* config{reinterpret_cast<SaveFileConfig*>(cfg_config_file_buffer.data())};
438  // This value is hardcoded, taken from 3dbrew, verified by hardware, it's always the same value
439  config->data_entries_offset = 0x455C;
440 
441  // Insert the default blocks
442  u8 zero_buffer[0xC0] = {};
443 
444  // 0x00030001 - Unknown
445  res = CreateConfigInfoBlk(0x00030001, 0x8, 0xE, zero_buffer);
446  if (!res.IsSuccess())
447  return res;
448 
449  res = CreateConfigInfoBlk(StereoCameraSettingsBlockID, sizeof(STEREO_CAMERA_SETTINGS), 0xE,
450  STEREO_CAMERA_SETTINGS.data());
451  if (!res.IsSuccess())
452  return res;
453 
454  res = CreateConfigInfoBlk(SoundOutputModeBlockID, sizeof(SOUND_OUTPUT_MODE), 0xE,
455  &SOUND_OUTPUT_MODE);
456  if (!res.IsSuccess())
457  return res;
458 
459  u32 random_number{};
460  u64 console_id{};
461  GenerateConsoleUniqueId(random_number, console_id);
462 
463  u64_le console_id_le = console_id;
464  res = CreateConfigInfoBlk(ConsoleUniqueID1BlockID, sizeof(console_id_le), 0xE, &console_id_le);
465  if (!res.IsSuccess())
466  return res;
467 
468  res = CreateConfigInfoBlk(ConsoleUniqueID2BlockID, sizeof(console_id_le), 0xE, &console_id_le);
469  if (!res.IsSuccess())
470  return res;
471 
472  u32_le random_number_le = random_number;
473  res = CreateConfigInfoBlk(ConsoleUniqueID3BlockID, sizeof(random_number_le), 0xE,
474  &random_number_le);
475  if (!res.IsSuccess())
476  return res;
477 
478  res = CreateConfigInfoBlk(UsernameBlockID, sizeof(CONSOLE_USERNAME_BLOCK), 0xE,
479  &CONSOLE_USERNAME_BLOCK);
480  if (!res.IsSuccess())
481  return res;
482 
483  res = CreateConfigInfoBlk(BirthdayBlockID, sizeof(PROFILE_BIRTHDAY), 0xE, &PROFILE_BIRTHDAY);
484  if (!res.IsSuccess())
485  return res;
486 
487  res = CreateConfigInfoBlk(LanguageBlockID, sizeof(CONSOLE_LANGUAGE), 0xE, &CONSOLE_LANGUAGE);
488  if (!res.IsSuccess())
489  return res;
490 
491  res = CreateConfigInfoBlk(CountryInfoBlockID, sizeof(COUNTRY_INFO), 0xE, &COUNTRY_INFO);
492  if (!res.IsSuccess())
493  return res;
494 
495  u16_le country_name_buffer[16][0x40] = {};
496  std::u16string region_name{Common::UTF8ToUTF16("Gensokyo")};
497  for (std::size_t i{}; i < 16; ++i) {
498  std::copy(region_name.cbegin(), region_name.cend(), country_name_buffer[i]);
499  }
500  // 0x000B0001 - Localized names for the profile Country
501  res = CreateConfigInfoBlk(CountryNameBlockID, sizeof(country_name_buffer), 0xE,
502  country_name_buffer);
503  if (!res.IsSuccess())
504  return res;
505  // 0x000B0002 - Localized names for the profile State/Province
506  res = CreateConfigInfoBlk(StateNameBlockID, sizeof(country_name_buffer), 0xE,
507  country_name_buffer);
508  if (!res.IsSuccess())
509  return res;
510 
511  // 0x000B0003 - Unknown, related to country/address (zip code?)
512  res = CreateConfigInfoBlk(0x000B0003, 0x4, 0xE, zero_buffer);
513  if (!res.IsSuccess())
514  return res;
515 
516  // 0x000C0000 - Unknown
517  res = CreateConfigInfoBlk(0x000C0000, 0xC0, 0xE, zero_buffer);
518  if (!res.IsSuccess())
519  return res;
520 
521  // 0x000C0001 - Unknown
522  res = CreateConfigInfoBlk(0x000C0001, 0x14, 0xE, zero_buffer);
523  if (!res.IsSuccess())
524  return res;
525 
526  // 0x000D0000 - Accepted EULA version
527  res = CreateConfigInfoBlk(EULAVersionBlockID, 0x4, 0xE, &MAX_EULA_VERSION);
528  if (!res.IsSuccess())
529  return res;
530 
531  res = CreateConfigInfoBlk(ConsoleModelBlockID, sizeof(CONSOLE_MODEL), 0xC, &CONSOLE_MODEL);
532  if (!res.IsSuccess())
533  return res;
534 
535  // 0x00170000 - Unknown
536  res = CreateConfigInfoBlk(0x00170000, 0x4, 0xE, zero_buffer);
537  if (!res.IsSuccess())
538  return res;
539 
540  // Save the buffer to the file
541  res = UpdateConfigNANDSavegame();
542  if (!res.IsSuccess())
543  return res;
544  return RESULT_SUCCESS;
545 }

547 ResultCode Module::LoadConfigNANDSaveFile() {
548  // Open the SystemSaveData archive 0x00010017
549  FileSys::Path archive_path{cfg_system_savedata_id};
550  auto archive_result{
551  Service::FS::OpenArchive(Service::FS::ArchiveIdCode::SystemSaveData, archive_path)};
552 
553  // If the archive didn't exist, create the files inside
554  if (archive_result.Code() == FileSys::ERR_NOT_FORMATTED) {
555  // Format the archive to create the directories
556  Service::FS::FormatArchive(Service::FS::ArchiveIdCode::SystemSaveData,
557  FileSys::ArchiveFormatInfo(), archive_path);
558 
559  // Open it again to get a valid archive now that the folder exists
560  archive_result =
561  Service::FS::OpenArchive(Service::FS::ArchiveIdCode::SystemSaveData, archive_path);
562  }
563 
564  ASSERT_MSG(archive_result.Succeeded(), "Could not open the CFG SystemSaveData archive!");
565 
566  cfg_system_save_data_archive = *archive_result;
567 
568  FileSys::Path config_path{"/config"};
569  FileSys::Mode open_mode = {};
570  open_mode.read_flag.Assign(1);
571 
572  auto config_result{Service::FS::OpenFileFromArchive(*archive_result, config_path, open_mode)};
573 
574  // Read the file if it already exists
575  if (config_result.Succeeded()) {
576  auto config{std::move(config_result).Unwrap()};
577  config->backend->Read(0, CONFIG_SAVEFILE_SIZE, cfg_config_file_buffer.data());
578  u32 eula_version{};
579  GetConfigInfoBlock(EULAVersionBlockID, 0x4, 0xE, &eula_version);
580  if (eula_version == 0)
581  SetConfigInfoBlock(EULAVersionBlockID, 0x4, 0xE, &MAX_EULA_VERSION);
582  return RESULT_SUCCESS;
583  }
584 
585  return FormatConfig();
586 }
