22 ResultVal<Handle> HandleTable::Create(SharedPtr<Object> obj) {
23  DEBUG_ASSERT(obj != nullptr);
24 
25  u16 slot = next_free_slot;
26  if (slot >= generations.size()) {
27  LOG_ERROR(Kernel, "Unable to allocate Handle, too many slots in use.");
28  return ERR_OUT_OF_HANDLES;
29  }
30  next_free_slot = generations[slot];
31 
32  u16 generation = next_generation++;
33 
34  // Overflow count so it fits in the 15 bits dedicated to the generation in the handle.
35  // CTR-OS doesn't use generation 0, so skip straight to 1.
36  if (next_generation >= (1 << 15))
37  next_generation = 1;
38 
39  generations[slot] = generation;
40  objects[slot] = std::move(obj);
41 
42  Handle handle = generation | (slot << 15);
43  return MakeResult<Handle>(handle);
44 }
