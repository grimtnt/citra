18 void VertexLoader::Setup(const PipelineRegs& regs) {
19  ASSERT_MSG(!is_setup, "VertexLoader is not intended to be setup more than once.");
20 
21  const auto& attribute_config = regs.vertex_attributes;
22  num_total_attributes = attribute_config.GetNumTotalAttributes();
23 
24  boost::fill(vertex_attribute_sources, 0xdeadbeef);
25 
26  for (int i{}; i < 16; i++) {
27  vertex_attribute_is_default[i] = attribute_config.IsDefaultAttribute(i);
28  }
29 
30  // Setup attribute data from loaders
31  for (int loader{}; loader < 12; ++loader) {
32  const auto& loader_config = attribute_config.attribute_loaders[loader];
33 
34  u32 offset{};
35 
36  // TODO: What happens if a loader overwrites a previous one's data?
37  for (unsigned component{}; component < loader_config.component_count; ++component) {
38  if (component >= 12) {
39  LOG_ERROR(HW_GPU,
40  "Overflow in the vertex attribute loader {} trying to load component {}",
41  loader, component);
42  continue;
43  }
44 
45  u32 attribute_index = loader_config.GetComponent(component);
46  if (attribute_index < 12) {
47  offset = Common::AlignUp(offset,
48  attribute_config.GetElementSizeInBytes(attribute_index));
49  vertex_attribute_sources[attribute_index] = loader_config.data_offset + offset;
50  vertex_attribute_strides[attribute_index] =
51  static_cast<u32>(loader_config.byte_count);
52  vertex_attribute_formats[attribute_index] =
53  attribute_config.GetFormat(attribute_index);
54  vertex_attribute_elements[attribute_index] =
55  attribute_config.GetNumElements(attribute_index);
56  offset += attribute_config.GetStride(attribute_index);
57  } else if (attribute_index < 16) {
58  // Attribute ids 12, 13, 14 and 15 signify 4, 8, 12 and 16-byte paddings,
59  // respectively
60  offset = Common::AlignUp(offset, 4);
61  offset += (attribute_index - 11) * 4;
62  } else {
63  UNREACHABLE(); // This is truly unreachable due to the number of bits for each
64  // component
65  }
66  }
67  }
68

70 }

74  ASSERT_MSG(is_setup, "A VertexLoader needs to be setup before loading vertices.");
75 
76  for (int i{}; i < num_total_attributes; ++i) {
77  if (vertex_attribute_elements[i] != 0) {
78  // Load per-vertex data from the loader arrays
79  u32 source_addr{base_address + vertex_attribute_sources[i] +
80  vertex_attribute_strides[i] * vertex};
81 
82  switch (vertex_attribute_formats[i]) {
83  case PipelineRegs::VertexAttributeFormat::Byte: {
84  const s8* srcdata{

86  for (unsigned int comp{}; comp < vertex_attribute_elements[i]; ++comp) {
87  input.attr[i][comp] = float24::FromFloat32(srcdata[comp]);
88  }
89  break;
90  }
91  case PipelineRegs::VertexAttributeFormat::UnsignedByte: {
92  const u8* srcdata{
93  reinterpret_cast<const u8*>(Memory::GetPhysicalPointer(source_addr))};
94  for (unsigned int comp{}; comp < vertex_attribute_elements[i]; ++comp) {
95  input.attr[i][comp] = float24::FromFloat32(srcdata[comp]);
96  }
97  break;
98  }
99  case PipelineRegs::VertexAttributeFormat::Short: {
100  const s16* srcdata{
101  reinterpret_cast<const s16*>(Memory::GetPhysicalPointer(source_addr))};
102  for (unsigned int comp{}; comp < vertex_attribute_elements[i]; ++comp) {
103  input.attr[i][comp] = float24::FromFloat32(srcdata[comp]);
104  }
105  break;
106  }
107  case PipelineRegs::VertexAttributeFormat::Float: {
108  const float* srcdata{
109  reinterpret_cast<const float*>(Memory::GetPhysicalPointer(source_addr))};
110  for (unsigned int comp{}; comp < vertex_attribute_elements[i]; ++comp) {
111  input.attr[i][comp] = float24::FromFloat32(srcdata[comp]);
112  }
113  break;
114  }
115  }
116 
117  // Default attribute values set if array elements have < 4 components. This
118  // is *not* carried over from the default attribute settings even if they're
119  // enabled for this attribute.
120  for (unsigned int comp{vertex_attribute_elements[i]}; comp < 4; ++comp) {
121  input.attr[i][comp] =

123  }
124

130  input.attr[i][0].ToFloat32(), input.attr[i][1].ToFloat32(),
131  input.attr[i][2].ToFloat32(), input.attr[i][3].ToFloat32());
132  } else if (vertex_attribute_is_default[i]) {
133  // Load the default attribute if we're configured to do so
134  input.attr[i] = g_state.input_default_attributes.attr[i];
135  LOG_TRACE(

139  input.attr[i][2].ToFloat32(), input.attr[i][3].ToFloat32());
140  } else {
