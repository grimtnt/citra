65  void Send() {
66  using lup = LUrlParser::clParseURL;
67  namespace hl = httplib;
68  lup parsedUrl = lup::ParseURL(url);
69  std::unique_ptr<hl::Client> cli{};
70  int port;
71  if (parsedUrl.m_Scheme == "http") {
72  if (!parsedUrl.GetPort(&port)) {
73  port = 80;
74  }
75  cli = std::make_unique<hl::Client>(parsedUrl.m_Host.c_str(), port,
76  (timeout == 0) ? 300 : (timeout * std::pow(10, -9)));
77  } else if (parsedUrl.m_Scheme == "https") {
78  if (!parsedUrl.GetPort(&port)) {
79  port = 443;
80  }
81  cli = std::make_unique<hl::SSLClient>(parsedUrl.m_Host.c_str(), port,
82  (timeout == 0) ? 300
83  : (timeout * std::pow(10, -9)));
84  } else {
85  UNREACHABLE_MSG("Invalid scheme!");
86  }
87  if (disable_verify) {
88  cli->set_verify(hl::SSLVerifyMode::None);
89  }
90  hl::Request request;
91  static const std::unordered_map<RequestMethod, std::string> methods_map_strings = {
92  {RequestMethod::Get, "GET"}, {RequestMethod::Post, "POST"},
93  {RequestMethod::Head, "HEAD"}, {RequestMethod::Put, "PUT"},
94  {RequestMethod::Delete, "DELETE"}, {RequestMethod::PostEmpty, "POST"},
95  {RequestMethod::PutEmpty, "PUT"},
96  };
97  static const std::unordered_map<RequestMethod, bool> methods_map_body = {
98  {RequestMethod::Get, false}, {RequestMethod::Post, true},
99  {RequestMethod::Head, false}, {RequestMethod::Put, true},
100  {RequestMethod::Delete, true}, {RequestMethod::PostEmpty, false},
101  {RequestMethod::PutEmpty, false},
102  };
103  request.method = methods_map_strings.find(method)->second;
104  request.path = '/' + parsedUrl.m_Path;
105  request.headers = request_headers;
106  if (methods_map_body.find(method)->second) {
107  request.body = body;
108  }
109  hl::Params params;
110  hl::detail::parse_query_text(parsedUrl.m_Query, params);
111  request.params = params;
112  response = std::make_shared<hl::Response>();
113  cli->send(request, *response);
114  }
