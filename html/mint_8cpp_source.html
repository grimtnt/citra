11 ResultCode Mint::ReceiveParameter(const Service::APT::MessageParameter& parameter) {
12  if (parameter.signal != Service::APT::SignalType::Request) {
13  LOG_ERROR(Applet_Mint, "unsupported signal {}", static_cast<u32>(parameter.signal));
14  UNIMPLEMENTED();
15  // TODO(Subv): Find the right error code
16  return ResultCode(-1);
17  }
18 
19  // The Request message contains a buffer with the size of the framebuffer shared
20  // memory.
21  // Create the SharedMemory that will hold the framebuffer data
22  Service::APT::CaptureBufferInfo capture_info{};
23  ASSERT(sizeof(capture_info) == parameter.buffer.size());
24 
25  memcpy(&capture_info, parameter.buffer.data(), sizeof(capture_info));
26 
27  // TODO: allocated memory never released
28  using Kernel::MemoryPermission;
29  // Allocate a heap block of the required size for this applet.
30  heap_memory = std::make_shared<std::vector<u8>>(capture_info.size);
31  // Create a SharedMemory that directly points to this heap block.
32  framebuffer_memory = Kernel::SharedMemory::CreateForApplet(
33  heap_memory, 0, capture_info.size, MemoryPermission::ReadWrite, MemoryPermission::ReadWrite,
34  "Mint Memory");
35 
36  // Send the response message with the newly created SharedMemory
37  Service::APT::MessageParameter result{};
38  result.signal = Service::APT::SignalType::Response;
39  result.buffer.clear();
40  result.destination_id = Service::APT::AppletId::Application;
41  result.sender_id = id;
42  result.object = framebuffer_memory;
43 
44  SendParameter(result);
45  return RESULT_SUCCESS;
46 }
