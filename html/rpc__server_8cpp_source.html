31 void RPCServer::HandleReadMemory(Packet& packet, u32 address, u32 data_size) {
32  if (data_size > MAX_READ_SIZE) {
33  return;
34  }
35 
36  // Note: Memory read occurs asynchronously from the state of the emulator
37  Memory::ReadBlock(address, packet.GetPacketData().data(), data_size);
38  packet.SetPacketDataSize(data_size);
39  packet.SendReply();
40 }

42 void RPCServer::HandleWriteMemory(Packet& packet, u32 address, const u8* data, u32 data_size) {
43  // Only allow writing to certain memory regions
44  if ((address >= Memory::PROCESS_IMAGE_VADDR && address <= Memory::PROCESS_IMAGE_VADDR_END) ||
45  (address >= Memory::HEAP_VADDR && address <= Memory::HEAP_VADDR_END) ||
46  (address >= Memory::N3DS_EXTRA_RAM_VADDR && address <= Memory::N3DS_EXTRA_RAM_VADDR_END)) {
47  // Note: Memory write occurs asynchronously from the state of the emulator
48  Memory::WriteBlock(address, data, data_size);
49  // If the memory happens to be executable code, make sure the changes become visible
50  Core::GetCPU().InvalidateCacheRange(address, data_size);
51  }
52  packet.SetPacketDataSize(0);
53  packet.SendReply();
54 }

100 bool RPCServer::ValidatePacket(const PacketHeader& packet_header) {
101  if (packet_header.version <= CURRENT_VERSION) {
102  switch (packet_header.packet_type) {
103  case PacketType::ReadMemory:
104  case PacketType::WriteMemory:
105  case PacketType::PadState:
106  case PacketType::TouchState:
107  case PacketType::MotionState:
108  case PacketType::CircleState:
109  case PacketType::SetResolution:
110  case PacketType::SetGame:
111  case PacketType::SetOverrideControls:
112  if (packet_header.packet_size >= (sizeof(u32) * 2)) {
113  return true;
114  }
115  break;
116  default:
117  break;
118  }
119  }
120  return false;
121 }

123 void RPCServer::HandleSingleRequest(std::unique_ptr<Packet> request_packet) {
124  bool success{};
125 
126  if (ValidatePacket(request_packet->GetHeader())) {
127  // Currently, all request types use the address/data_size wire format
128  u32 address{};
129  u32 data_size{};
130  std::memcpy(&address, request_packet->GetPacketData().data(), sizeof(address));
131  std::memcpy(&data_size, request_packet->GetPacketData().data() + sizeof(address),
132  sizeof(data_size));
133  switch (request_packet->GetPacketType()) {
134  case PacketType::ReadMemory: {
135  if (data_size > 0 && data_size <= MAX_READ_SIZE) {
136  HandleReadMemory(*request_packet, address, data_size);
137  success = true;
138  }
139  break;
140  }
141  case PacketType::WriteMemory: {
142  if (data_size > 0 && data_size <= MAX_PACKET_DATA_SIZE - (sizeof(u32) * 2)) {
143  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
144  HandleWriteMemory(*request_packet, address, data, data_size);
145  success = true;
146  }
147  break;
148  }
149  case PacketType::PadState: {
150  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
151  u32 raw{};
152  std::memcpy(&raw, data, sizeof(u32));
153  HandlePadState(*request_packet, raw);
154  success = true;
155  break;
156  }
157  case PacketType::TouchState: {
158  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
159  struct State {
160  s16 x;
161  s16 y;
162  bool valid;
163  };
164  State state{};
165  std::memcpy(&state, data, sizeof(State));
166  HandleTouchState(*request_packet, state.x, state.y, state.valid);
167  success = true;
168  break;
169  }
170  case PacketType::MotionState: {
171  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
172  struct State {
173  s16 x;
174  s16 y;
175  s16 z;
176  s16 roll;
177  s16 pitch;
178  s16 yaw;
179  };
180  State state{};
181  std::memcpy(&state, data, sizeof(State));
182  HandleMotionState(*request_packet, state.x, state.y, state.z, state.roll, state.pitch,
183  state.yaw);
184  success = true;
185  break;
186  }
187  case PacketType::CircleState: {
188  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
189  struct State {
190  s16 x;
191  s16 y;
192  };
193  State state{};
194  std::memcpy(&state, data, sizeof(State));
195  HandleCircleState(*request_packet, state.x, state.y);
196  success = true;
197  break;
198  }
199  case PacketType::SetResolution: {
200  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
201  u16 resolution{};
202  std::memcpy(&resolution, data, sizeof(u16));
203  HandleSetResolution(*request_packet, resolution);
204  success = true;
205  break;
206  }
207  case PacketType::SetGame: {
208  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
209  std::string path{};
210  path.resize(request_packet->GetPacketDataSize() - (sizeof(u32) * 2));
211  std::memcpy(&path[0], data, request_packet->GetPacketDataSize() - (sizeof(u32) * 2));
212  HandleSetGame(*request_packet, path);
213  success = true;
214  break;
215  }
216  case PacketType::SetOverrideControls: {
217  const u8* data{request_packet->GetPacketData().data() + (sizeof(u32) * 2)};
218  struct State {
219  bool pad;
220  bool touch;
221  bool motion;
222  bool circle;
223  };
224  State state{};
225  std::memcpy(&state, data, sizeof(State));
226  HandleSetOverrideControls(*request_packet, state.pad, state.touch, state.motion,
227  state.circle);
228  success = true;
229  break;
230  }
231  default:
232  break;
233  }
234  }
235 
236  if (!success) {
237  // Send an empty reply, so as not to hang the client
238  request_packet->SetPacketDataSize(0);
239  request_packet->SendReply();
240  }
241 }

243 void RPCServer::HandleRequestsLoop() {
244  std::unique_ptr<RPC::Packet> request_packet{};
245 
246  LOG_INFO(RPC, "Request handler started.");
247 
248  while (request_queue.PopWait(request_packet)) {
249  HandleSingleRequest(std::move(request_packet));
250  }
