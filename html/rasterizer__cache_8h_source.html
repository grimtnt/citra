94  static constexpr unsigned int GetFormatBpp(PixelFormat format) {

97  24, // RGB8

99  16, // RGB565

102  16, // RG8

111  0,

113  32, // D24S8

148  static bool CheckFormatsBlittable(PixelFormat pixel_format_a, PixelFormat pixel_format_b) {
149  SurfaceType a_type = GetFormatType(pixel_format_a);
150  SurfaceType b_type = GetFormatType(pixel_format_b);
151 
152  if ((a_type == SurfaceType::Color || a_type == SurfaceType::Texture) &&
153  (b_type == SurfaceType::Color || b_type == SurfaceType::Texture)) {
154  return true;
155  }
156 
157  if (a_type == SurfaceType::Depth && b_type == SurfaceType::Depth) {
158  return true;
159  }
160 
161  if (a_type == SurfaceType::DepthStencil && b_type == SurfaceType::DepthStencil) {
162  return true;
163  }
164 
165  return false;
166  }
167 
168  static constexpr SurfaceType GetFormatType(PixelFormat pixel_format) {
169  if ((unsigned int)pixel_format < 5) {
170  return SurfaceType::Color;
171  }
172 
173  if ((unsigned int)pixel_format < 14) {
174  return SurfaceType::Texture;
175  }
176 
177  if (pixel_format == PixelFormat::D16 || pixel_format == PixelFormat::D24) {
178  return SurfaceType::Depth;
179  }
180 
181  if (pixel_format == PixelFormat::D24S8) {
182  return SurfaceType::DepthStencil;
183  }
184 
185  return SurfaceType::Invalid;
186  }

190  void UpdateParams() {
191  if (stride == 0) {
192  stride = width;
193  }
194  type = GetFormatType(pixel_format);
195  size = !is_tiled ? BytesInPixels(stride * (height - 1) + width)
196  : BytesInPixels(stride * 8 * (height / 8 - 1) + width * 8);
197  end = addr + size;
198  }
