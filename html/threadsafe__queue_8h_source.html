44  template <typename Arg>
45  void Push(Arg&& t) {
46  // create the element, add it to the queue
47  write_ptr->current = std::move(t);
48  // set the next pointer to a new element ptr
49  // then advance the write pointer
50  ElementPtr* new_ptr{new ElementPtr()};
51  write_ptr->next.store(new_ptr, std::memory_order_release);
52  write_ptr = new_ptr;
53  if (NeedSize)
54  size++;
55  cv.notify_one();
56  }
57 
58  void Finalize() {
59  // Create a new next, the queue wont be empty but the optional will contain no value
60  ElementPtr* new_ptr{new ElementPtr()};
61  write_ptr->next.store(new_ptr, std::memory_order_release);
62  write_ptr = new_ptr;
63  if (NeedSize)
64  size++;
65  cv.notify_one();
66  }

68  void Pop() {
69  if (NeedSize)
70  size--;
71  ElementPtr* tmpptr{read_ptr};
72  // advance the read pointer
73  read_ptr = tmpptr->next.load();
74  // set the next element to nullptr to stop the recursive deletion
75  tmpptr->next.store(nullptr);
76  delete tmpptr; // this also deletes the element
77  }
78 
79  bool Pop(T& t) {
80  if (Empty())
81  return false;
82 
83  ElementPtr* tmpptr{read_ptr};
84 
85  // If the finialize msg was pushed return false
86  if (!tmpptr->current)
87  return false;
88 
89  if (NeedSize)
90  size--;
91 
92  read_ptr = tmpptr->next.load(std::memory_order_acquire);
93  t = std::move(tmpptr->current.value());
94  tmpptr->next.store(nullptr);
95  delete tmpptr;
96  return true;
97  }

99  bool PopWait(T& t) {
100  if (Empty()) {
101  std::unique_lock<std::mutex> lock(cv_mutex);
102  cv.wait(lock, [this]() { return !Empty(); });
103  }
104  return Pop(t);
105  }
