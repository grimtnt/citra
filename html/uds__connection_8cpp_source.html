57 std::vector<u8> GenerateAssocResponseFrame(AssocStatus status, u16 association_id, u32 network_id) {
58  AssociationResponseFrame frame{};
59  frame.capabilities = DefaultExtraCapabilities;
60  frame.status_code = static_cast<u16>(status);
61  // The association id is ORed with this magic value (0xC000)
62  constexpr u16 AssociationIdMagic = 0xC000;
63  frame.assoc_id = association_id | AssociationIdMagic;
