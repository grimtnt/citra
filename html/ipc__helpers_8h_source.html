365 template <unsigned int N>
366 std::array<u32, N> RequestParser::PopHLEHandles() {
367  u32 handle_descriptor{Pop<u32>()};
368  ASSERT_MSG(IsHandleDescriptor(handle_descriptor),
369  "Tried to pop handle(s) but the descriptor is not a handle descriptor");
370  ASSERT_MSG(N == HandleNumberFromDesc(handle_descriptor),
371  "Number of handles doesn't match the descriptor");
372 
373  std::array<u32, N> handles{};
374  for (u32& handle : handles) {
375  handle = Pop<u32>();
376  }
377  return handles;
378 }
