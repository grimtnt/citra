83 template <typename InputDeviceType>
84 std::unique_ptr<InputDeviceType> CreateDevice(const std::string& params) {
85  const Common::ParamPackage package{params};
86  const std::string engine{package.Get("engine", "null")};
87  const auto& factory_list{Impl::FactoryList<InputDeviceType>::list};
88  const auto pair{factory_list.find(engine)};
89  if (pair == factory_list.end()) {
90  if (engine != "null") {
91  LOG_ERROR(Input, "Unknown engine name: {}", engine);
92  }
93  return std::make_unique<InputDeviceType>();
94  }
95  return pair->second->Create(package);
