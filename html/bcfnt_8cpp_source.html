10 void RelocateSharedFont(Kernel::SharedPtr<Kernel::SharedMemory> shared_font, VAddr new_address) {
11  static const u32 SharedFontStartOffset = 0x80;
12  const u8* cfnt_ptr = shared_font->GetPointer(SharedFontStartOffset);
13 
14  CFNT cfnt;
15  memcpy(&cfnt, cfnt_ptr, sizeof(cfnt));
16 
17  u32 assumed_cmap_offset{};
18  u32 assumed_cwdh_offset{};
19  u32 assumed_tglp_offset{};
20  u32 first_cmap_offset{};
21  u32 first_cwdh_offset{};
22  u32 first_tglp_offset{};
23 
24  // First discover the location of sections so that the rebase offset can be auto-detected
25  u32 current_offset = SharedFontStartOffset + cfnt.header_size;
26  for (unsigned block{}; block < cfnt.num_blocks; ++block) {
27  const u8* data = shared_font->GetPointer(current_offset);
28 
29  SectionHeader section_header;
30  memcpy(&section_header, data, sizeof(section_header));
31 
32  if (first_cmap_offset == 0 && memcmp(section_header.magic, "CMAP", 4) == 0) {
33  first_cmap_offset = current_offset;
34  } else if (first_cwdh_offset == 0 && memcmp(section_header.magic, "CWDH", 4) == 0) {
35  first_cwdh_offset = current_offset;
36  } else if (first_tglp_offset == 0 && memcmp(section_header.magic, "TGLP", 4) == 0) {
37  first_tglp_offset = current_offset;
38  } else if (memcmp(section_header.magic, "FINF", 4) == 0) {
39  BCFNT::FINF finf;
40  memcpy(&finf, data, sizeof(finf));
41 
42  assumed_cmap_offset = finf.cmap_offset - sizeof(SectionHeader);
43  assumed_cwdh_offset = finf.cwdh_offset - sizeof(SectionHeader);
44  assumed_tglp_offset = finf.tglp_offset - sizeof(SectionHeader);
45  }
46 
47  current_offset += section_header.section_size;
48  }
49 
50  u32 previous_base = assumed_cmap_offset - first_cmap_offset;
51  ASSERT(previous_base == assumed_cwdh_offset - first_cwdh_offset);
52  ASSERT(previous_base == assumed_tglp_offset - first_tglp_offset);
53 
54  u32 offset = new_address - previous_base;
55 
56  // Reset pointer back to start of sections and do the actual rebase
57  current_offset = SharedFontStartOffset + cfnt.header_size;
58  for (unsigned block{}; block < cfnt.num_blocks; ++block) {
59  u8* data = shared_font->GetPointer(current_offset);
60

62  memcpy(&section_header, data, sizeof(section_header));
63 
64  if (memcmp(section_header.magic, "FINF", 4) == 0) {
65  BCFNT::FINF finf;
66  memcpy(&finf, data, sizeof(finf));
67

69  finf.cmap_offset += offset;
70  finf.cwdh_offset += offset;
71  finf.tglp_offset += offset;
72 
73  memcpy(data, &finf, sizeof(finf));
74  } else if (memcmp(section_header.magic, "CMAP", 4) == 0) {
75  BCFNT::CMAP cmap;

77 
78  // Relocate the offsets in the CMAP section

81 
82  memcpy(data, &cmap, sizeof(cmap));
83  } else if (memcmp(section_header.magic, "CWDH", 4) == 0) {
84  BCFNT::CWDH cwdh;
85  memcpy(&cwdh, data, sizeof(cwdh));
86 
87  // Relocate the offsets in the CWDH section
88  if (cwdh.next_cwdh_offset != 0)
89  cwdh.next_cwdh_offset += offset;
90 
91  memcpy(data, &cwdh, sizeof(cwdh));
92  } else if (memcmp(section_header.magic, "TGLP", 4) == 0) {
93  BCFNT::TGLP tglp;
94  memcpy(&tglp, data, sizeof(tglp));
95 
96  // Relocate the offsets in the TGLP section
97  tglp.sheet_data_offset += offset;
98 
99  memcpy(data, &tglp, sizeof(tglp));
100  }
101 
102  current_offset += section_header.section_size;
103  }
104 }
