16 class SaveDataDelayGenerator : public DelayGenerator {
17 public:
18  u64 GetReadDelayNs(std::size_t length) override {
19  // The delay was measured on O3DS and O2DS with
20  // https://gist.github.com/B3n30/ac40eac20603f519ff106107f4ac9182
21  // from the results the average of each length was taken.
22  static constexpr u64 slope(183);
23  static constexpr u64 offset(524879);
24  static constexpr u64 minimum(631826);
25  u64 IPCDelayNanoseconds{std::max<u64>(static_cast<u64>(length) * slope + offset, minimum)};
26  return IPCDelayNanoseconds;
27  }
28 };

30 ResultVal<std::unique_ptr<FileBackend>> SaveDataArchive::OpenFile(const Path& path,
31  const Mode& mode) const {
32  LOG_DEBUG(Service_FS, "called path={} mode={:01X}", path.DebugStr(), mode.hex);
33 
34  const PathParser path_parser{path};
35 
36  if (!path_parser.IsValid()) {
37  LOG_ERROR(Service_FS, "Invalid path {}", path.DebugStr());
38  return ERROR_INVALID_PATH;
39  }
40 
41  if (mode.hex == 0) {
42  LOG_ERROR(Service_FS, "Empty open mode");
43  return ERROR_UNSUPPORTED_OPEN_FLAGS;
44  }
45 
46  if (mode.create_flag && !mode.write_flag) {
47  LOG_ERROR(Service_FS, "Create flag set but write flag not set");
48  return ERROR_UNSUPPORTED_OPEN_FLAGS;
49  }
50 
51  const auto full_path{path_parser.BuildHostPath(mount_point)};
52 
53  switch (path_parser.GetHostStatus(mount_point)) {
54  case PathParser::InvalidMountPoint:
55  LOG_CRITICAL(Service_FS, "(unreachable) Invalid mount point {}", mount_point);
56  return ERROR_FILE_NOT_FOUND;
57  case PathParser::PathNotFound:
58  LOG_ERROR(Service_FS, "Path not found {}", full_path);
59  return ERROR_PATH_NOT_FOUND;
60  case PathParser::FileInPath:
61  case PathParser::DirectoryFound:
62  LOG_ERROR(Service_FS, "Unexpected file or directory in {}", full_path);
63  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY;
64  case PathParser::NotFound:
65  if (!mode.create_flag) {
66  LOG_ERROR(Service_FS, "Non-existing file {} can't be open without mode create.",
67  full_path);
68  return ERROR_FILE_NOT_FOUND;
69  } else {
70  // Create the file
71  FileUtil::CreateEmptyFile(full_path);
72  }
73  break;
74  case PathParser::FileFound:
75  break; // Expected 'success' case
76  }
77 
78  FileUtil::IOFile file{full_path, mode.write_flag ? "r+b" : "rb"};
79  if (!file.IsOpen()) {
80  LOG_CRITICAL(Service_FS, "(unreachable) Unknown error opening {}", full_path);
81  return ERROR_FILE_NOT_FOUND;
82  }
83 
84  std::unique_ptr<DelayGenerator> delay_generator{std::make_unique<SaveDataDelayGenerator>()};
85  auto disk_file{std::make_unique<DiskFile>(std::move(file), mode, std::move(delay_generator))};
86  return MakeResult<std::unique_ptr<FileBackend>>(std::move(disk_file));
87 }

153 static ResultCode DeleteDirectoryHelper(const Path& path, const std::string& mount_point,
154  T deleter) {

156 
157  if (!path_parser.IsValid()) {
158  LOG_ERROR(Service_FS, "Invalid path {}", path.DebugStr());
159  return ERROR_INVALID_PATH;
160  }
161 
162  if (path_parser.IsRootDirectory())
163  return ERROR_DIRECTORY_NOT_EMPTY;
164 
165  const auto full_path{path_parser.BuildHostPath(mount_point)};
166 
167  switch (path_parser.GetHostStatus(mount_point)) {
168  case PathParser::InvalidMountPoint:
169  LOG_CRITICAL(Service_FS, "(unreachable) Invalid mount point {}", mount_point);
170  return ERROR_PATH_NOT_FOUND;
171  case PathParser::PathNotFound:
172  case PathParser::NotFound:
173  LOG_ERROR(Service_FS, "Path not found {}", full_path);
174  return ERROR_PATH_NOT_FOUND;
175  case PathParser::FileInPath:
176  case PathParser::FileFound:
177  LOG_ERROR(Service_FS, "Unexpected file or directory {}", full_path);
178  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY;
179  case PathParser::DirectoryFound:
180  break; // Expected 'success' case
181  }
182 
183  if (deleter(full_path)) {
184  return RESULT_SUCCESS;
185  }
186 
187  LOG_ERROR(Service_FS, "Directory not empty {}", full_path);
188  return ERROR_DIRECTORY_NOT_EMPTY;

200 ResultCode SaveDataArchive::CreateFile(const FileSys::Path& path, u64 size) const {
201  const PathParser path_parser{path};
202 
203  if (!path_parser.IsValid()) {
204  LOG_ERROR(Service_FS, "Invalid path {}", path.DebugStr());
205  return ERROR_INVALID_PATH;
206  }
207 
208  const auto full_path{path_parser.BuildHostPath(mount_point)};
209 
210  switch (path_parser.GetHostStatus(mount_point)) {
211  case PathParser::InvalidMountPoint:
212  LOG_CRITICAL(Service_FS, "(unreachable) Invalid mount point {}", mount_point);
213  return ERROR_FILE_NOT_FOUND;
214  case PathParser::PathNotFound:
215  LOG_ERROR(Service_FS, "Path not found {}", full_path);
216  return ERROR_PATH_NOT_FOUND;
217  case PathParser::FileInPath:
218  LOG_ERROR(Service_FS, "Unexpected file in path {}", full_path);
219  return ERROR_UNEXPECTED_FILE_OR_DIRECTORY;
220  case PathParser::DirectoryFound:
221  case PathParser::FileFound:
222  LOG_ERROR(Service_FS, "{} already exists", full_path);
223  return ERROR_FILE_ALREADY_EXISTS;
224  case PathParser::NotFound:
225  break; // Expected 'success' case
226  }
227 
228  if (size == 0) {
229  FileUtil::CreateEmptyFile(full_path);
230  return RESULT_SUCCESS;
231  }
232 
233  FileUtil::IOFile file{full_path, "wb"};
234  // Creates a sparse file (or a normal file on filesystems without the concept of sparse files)
235  // We do this by seeking to the right size, then writing a single null byte.
236  if (file.Seek(size - 1, SEEK_SET) && file.WriteBytes("", 1) == 1) {
237  return RESULT_SUCCESS;
238  }
239 
240  LOG_ERROR(Service_FS, "Too large file");
241  return ResultCode(ErrorDescription::TooLarge, ErrorModule::FS, ErrorSummary::OutOfResource,
242  ErrorLevel::Info);
243 }
