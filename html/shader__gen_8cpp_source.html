311 static std::string SampleTexture(const PicaFSConfig& config, unsigned texture_unit) {
312  const auto& state = config.state;
313  switch (texture_unit) {
314  case 0:
315  // Only unit 0 respects the texturing type
316  switch (state.texture0_type) {
317  case TexturingRegs::TextureConfig::Texture2D:
318  return "texture(tex0, texcoord0)";
319  case TexturingRegs::TextureConfig::Projection2D:
320  return "textureProj(tex0, vec3(texcoord0, texcoord0_w))";
321  case TexturingRegs::TextureConfig::TextureCube:
322  return "texture(tex_cube, vec3(texcoord0, texcoord0_w))";
323  case TexturingRegs::TextureConfig::Shadow2D:
324  return "shadowTexture(texcoord0, texcoord0_w)";
325  case TexturingRegs::TextureConfig::ShadowCube:
326  return "shadowTextureCube(texcoord0, texcoord0_w)";
327  case TexturingRegs::TextureConfig::Disabled:
328  return "vec4(0.0)";
329  default:
330  LOG_CRITICAL(HW_GPU, "Unhandled texture type {:x}",
331  static_cast<int>(state.texture0_type));
332  UNIMPLEMENTED();
333  return "texture(tex0, texcoord0)";
334  }
335  case 1:
336  return "texture(tex1, texcoord1)";
337  case 2:
338  if (state.texture2_use_coord1)
339  return "texture(tex2, texcoord1)";
340  else
341  return "texture(tex2, texcoord2)";
342  case 3:
343  if (state.proctex.enable) {
344  return "ProcTex()";
345  } else {
346  LOG_DEBUG(Render, "Using Texture3 without enabling it");
347  return "vec4(0.0)";
348  }
349  default:
350  UNREACHABLE();
351  return "";
352  }
353 }

356 static void AppendSource(std::string& out, const PicaFSConfig& config,
357  TevStageConfig::Source source, const std::string& index_name) {
358  const auto& state = config.state;
359  using Source = TevStageConfig::Source;
360  switch (source) {
361  case Source::PrimaryColor:
362  out += "rounded_primary_color";
363  break;
364  case Source::PrimaryFragmentColor:
365  out += "primary_fragment_color";
366  break;
367  case Source::SecondaryFragmentColor:
368  out += "secondary_fragment_color";
369  break;
370  case Source::Texture0:
371  out += SampleTexture(config, 0);
372  break;
373  case Source::Texture1:
374  out += SampleTexture(config, 1);
375  break;
376  case Source::Texture2:
377  out += SampleTexture(config, 2);
378  break;
379  case Source::Texture3:
380  out += SampleTexture(config, 3);
381  break;
382  case Source::PreviousBuffer:
383  out += "combiner_buffer";
384  break;
385  case Source::Constant:
386  ((out += "const_color[") += index_name) += ']';
387  break;
388  case Source::Previous:
389  out += "last_tex_env_out";
390  break;
391  default:
392  out += "vec4(0.0)";
393  LOG_CRITICAL(Render, "Unknown source op {}", static_cast<u32>(source));
394  break;
395  }

506 static void AppendColorCombiner(std::string& out, TevStageConfig::Operation operation,
507  const std::string& variable_name) {
508  out += "clamp(";
509  using Operation = TevStageConfig::Operation;
510  switch (operation) {
511  case Operation::Replace:
512  out += variable_name + "[0]";
513  break;
514  case Operation::Modulate:
515  out += variable_name + "[0] * " + variable_name + "[1]";
516  break;
517  case Operation::Add:
518  out += variable_name + "[0] + " + variable_name + "[1]";
519  break;
520  case Operation::AddSigned:
521  out += variable_name + "[0] + " + variable_name + "[1] - vec3(0.5)";
522  break;
523  case Operation::Lerp:
524  out += variable_name + "[0] * " + variable_name + "[2] + " + variable_name +
525  "[1] * (vec3(1.0) - " + variable_name + "[2])";
526  break;
527  case Operation::Subtract:
528  out += variable_name + "[0] - " + variable_name + "[1]";
529  break;
530  case Operation::MultiplyThenAdd:
531  out += variable_name + "[0] * " + variable_name + "[1] + " + variable_name + "[2]";
532  break;
533  case Operation::AddThenMultiply:
534  out += "min(" + variable_name + "[0] + " + variable_name + "[1], vec3(1.0)) * " +
535  variable_name + "[2]";
536  break;
537  case Operation::Dot3_RGB:
538  case Operation::Dot3_RGBA:
539  out += "vec3(dot(" + variable_name + "[0] - vec3(0.5), " + variable_name +
540  "[1] - vec3(0.5)) * 4.0)";
541  break;
542  default:
543  out += "vec3(0.0)";
544  LOG_CRITICAL(Render, "Unknown color combiner operation: {}", static_cast<u32>(operation));
545  break;
546  }
547  out += ", vec3(0.0), vec3(1.0))"; // Clamp result to 0.0, 1.0
548 }

551 static void AppendAlphaCombiner(std::string& out, TevStageConfig::Operation operation,
552  const std::string& variable_name) {
553  out += "clamp(";
554  using Operation = TevStageConfig::Operation;
555  switch (operation) {
556  case Operation::Replace:
557  out += variable_name + "[0]";
558  break;
559  case Operation::Modulate:
560  out += variable_name + "[0] * " + variable_name + "[1]";
561  break;
562  case Operation::Add:
563  out += variable_name + "[0] + " + variable_name + "[1]";
564  break;
565  case Operation::AddSigned:
566  out += variable_name + "[0] + " + variable_name + "[1] - 0.5";
567  break;
568  case Operation::Lerp:
569  out += variable_name + "[0] * " + variable_name + "[2] + " + variable_name +
570  "[1] * (1.0 - " + variable_name + "[2])";
571  break;
572  case Operation::Subtract:
573  out += variable_name + "[0] - " + variable_name + "[1]";
574  break;
575  case Operation::MultiplyThenAdd:
576  out += variable_name + "[0] * " + variable_name + "[1] + " + variable_name + "[2]";
577  break;
578  case Operation::AddThenMultiply:
579  out += "min(" + variable_name + "[0] + " + variable_name + "[1], 1.0) * " + variable_name +
580  "[2]";
581  break;
582  default:
583  out += "0.0";
584  LOG_CRITICAL(Render, "Unknown alpha combiner operation: {}", static_cast<u32>(operation));
585  break;
586  }
587  out += ", 0.0, 1.0)";

678 static void WriteLighting(std::string& out, const PicaFSConfig& config) {
679  const auto& lighting{config.state.lighting};
680 
681  // Define lighting globals
682  out += "vec4 diffuse_sum = vec4(0.0, 0.0, 0.0, 1.0);\n"
683  "vec4 specular_sum = vec4(0.0, 0.0, 0.0, 1.0);\n"
684  "vec3 light_vector = vec3(0.0);\n"
685  "vec3 refl_value = vec3(0.0);\n"
686  "vec3 spot_dir = vec3(0.0);\n"
687  "vec3 half_vector = vec3(0.0);\n"
688  "float dot_product = 0.0;\n"
689  "float clamp_highlights = 1.0;\n"
690  "float geo_factor = 1.0;\n";
691 
692  // Compute fragment normals and tangents
693  auto Perturbation{[&]() {
694  return "2.0 * (" + SampleTexture(config, lighting.bump_selector) + ").rgb - 1.0";
695  }};
696  if (lighting.bump_mode == LightingRegs::LightingBumpMode::NormalMap) {
697  // Bump mapping is enabled using a normal map
698  out += "vec3 surface_normal = " + Perturbation() + ";\n";
699 
700  // Recompute Z-component of perturbation if 'renorm' is enabled, this provides a higher
701  // precision result
702  if (lighting.bump_renorm) {
703  std::string val =
704  "(1.0 - (surface_normal.x*surface_normal.x + surface_normal.y*surface_normal.y))";
705  out += "surface_normal.z = sqrt(max(" + val + ", 0.0));\n";
706  }
707 
708  // The tangent vector is not perturbed by the normal map and is just a unit vector.
709  out += "vec3 surface_tangent = vec3(1.0, 0.0, 0.0);\n";
710  } else if (lighting.bump_mode == LightingRegs::LightingBumpMode::TangentMap) {
711  // Bump mapping is enabled using a tangent map
712  out += "vec3 surface_tangent = " + Perturbation() + ";\n";
713  // Mathematically, recomputing Z-component of the tangent vector won't affect the relevant
714  // computation below, which is also confirmed on 3DS. So we don't bother recomputing here
715  // even if 'renorm' is enabled.
716 
717  // The normal vector is not perturbed by the tangent map and is just a unit vector.
718  out += "vec3 surface_normal = vec3(0.0, 0.0, 1.0);\n";
719  } else {
720  // No bump mapping - surface local normal and tangent are just unit vectors
721  out += "vec3 surface_normal = vec3(0.0, 0.0, 1.0);\n";
722  out += "vec3 surface_tangent = vec3(1.0, 0.0, 0.0);\n";
723  }
724 
725  // Rotate the surface-local normal by the interpolated normal quaternion to convert it to
726  // eyespace.
727  out += "vec4 normalized_normquat = normalize(normquat);\n";
728  out += "vec3 normal = quaternion_rotate(normalized_normquat, surface_normal);\n";
729  out += "vec3 tangent = quaternion_rotate(normalized_normquat, surface_tangent);\n";
730 
731  if (lighting.enable_shadow) {
732  std::string shadow_texture = SampleTexture(config, lighting.shadow_selector);
733  if (lighting.shadow_invert) {
734  out += "vec4 shadow = vec4(1.0) - " + shadow_texture + ";\n";
735  } else {
736  out += "vec4 shadow = " + shadow_texture + ";\n";
737  }
738  } else {
739  out += "vec4 shadow = vec4(1.0);\n";
740  }
741 
742  // Samples the specified lookup table for specular lighting
743  auto GetLutValue{[&lighting](LightingRegs::LightingSampler sampler, unsigned light_num,
744  LightingRegs::LightingLutInput input, bool abs) {
745  std::string index{};
746  switch (input) {
747  case LightingRegs::LightingLutInput::NH:
748  index = "dot(normal, normalize(half_vector))";
749  break;
750 
751  case LightingRegs::LightingLutInput::VH:
752  index = std::string("dot(normalize(view), normalize(half_vector))");
753  break;
754 
755  case LightingRegs::LightingLutInput::NV:
756  index = std::string("dot(normal, normalize(view))");
757  break;
758 
759  case LightingRegs::LightingLutInput::LN:
760  index = std::string("dot(light_vector, normal)");
761  break;
762 
763  case LightingRegs::LightingLutInput::SP:
764  index = std::string("dot(light_vector, spot_dir)");
765  break;
766 
767  case LightingRegs::LightingLutInput::CP:
768  // CP input is only available with configuration 7
769  if (lighting.config == LightingRegs::LightingConfig::Config7) {
770  // Note: even if the normal vector is modified by normal map, which is not the
771  // normal of the tangent plane anymore, the half angle vector is still projected
772  // using the modified normal vector.
773  std::string half_angle_proj =
774  "normalize(half_vector) - normal * dot(normal, normalize(half_vector))";
775  // Note: the half angle vector projection is confirmed not normalized before the dot
776  // product. The result is in fact not cos(phi) as the name suggested.
777  index = "dot(" + half_angle_proj + ", tangent)";
778  } else {
779  index = "0.0";
780  }
781  break;
782 
783  default:
784  LOG_CRITICAL(HW_GPU, "Unknown lighting LUT input {}", (int)input);
785  UNIMPLEMENTED();
786  index = "0.0";
787  break;
788  }
789 
790  std::string sampler_string{std::to_string(static_cast<unsigned>(sampler))};
791 
792  if (abs) {
793  // LUT index is in the range of (0.0, 1.0)
794  index = lighting.light[light_num].two_sided_diffuse ? "abs(" + index + ")"
795  : "max(" + index + ", 0.0)";
796  return "LookupLightingLUTUnsigned(" + sampler_string + ", " + index + ")";
797  } else {
798  // LUT index is in the range of (-1.0, 1.0)
799  return "LookupLightingLUTSigned(" + sampler_string + ", " + index + ")";
800  }
801  }};
802 
803  // Write the code to emulate each enabled light
804  for (unsigned light_index{}; light_index < lighting.src_num; ++light_index) {
805  const auto& light_config{lighting.light[light_index]};
806  std::string light_src{"light_src[" + std::to_string(light_config.num) + "]"};
807 
808  // Compute light vector (directional or positional)
809  if (light_config.directional)
810  out += "light_vector = normalize(" + light_src + ".position);\n";
811  else
812  out += "light_vector = normalize(" + light_src + ".position + view);\n";
813 
814  out += "spot_dir = " + light_src + ".spot_direction;\n";
815  out += "half_vector = normalize(view) + light_vector;\n";
816 
817  // Compute dot product of light_vector and normal, adjust if lighting is one-sided or
818  // two-sided
819  out += std::string("dot_product = ") + (light_config.two_sided_diffuse
820  ? "abs(dot(light_vector, normal));\n"
821  : "max(dot(light_vector, normal), 0.0);\n");
822 
823  // If enabled, clamp specular component if lighting result is zero
824  if (lighting.clamp_highlights) {
825  out += "clamp_highlights = sign(dot_product);\n";
826  }
827 
828  // If enabled, compute spot light attenuation value
829  std::string spot_atten = "1.0";
830  if (light_config.spot_atten_enable &&
831  LightingRegs::IsLightingSamplerSupported(
832  lighting.config, LightingRegs::LightingSampler::SpotlightAttenuation)) {
833  std::string value{
834  GetLutValue(LightingRegs::SpotlightAttenuationSampler(light_config.num),
835  light_config.num, lighting.lut_sp.type, lighting.lut_sp.abs_input)};
836  spot_atten = "(" + std::to_string(lighting.lut_sp.scale) + " * " + value + ")";
837  }
838 
839  // If enabled, compute distance attenuation value
840  std::string dist_atten{"1.0"};
841  if (light_config.dist_atten_enable) {
842  std::string index = "clamp(" + light_src + ".dist_atten_scale * length(-view - " +
843  light_src + ".position) + " + light_src +
844  ".dist_atten_bias, 0.0, 1.0)";
845  auto sampler{LightingRegs::DistanceAttenuationSampler(light_config.num)};
846  dist_atten = "LookupLightingLUTUnsigned(" +
847  std::to_string(static_cast<unsigned>(sampler)) + "," + index + ")";
848  }
849 
850  if (light_config.geometric_factor_0 || light_config.geometric_factor_1) {
851  out += "geo_factor = dot(half_vector, half_vector);\n"
852  "geo_factor = geo_factor == 0.0 ? 0.0 : min("
853  "dot_product / geo_factor, 1.0);\n";
854  }
855 
856  // Specular 0 component
857  std::string d0_lut_value{"1.0"};
858  if (lighting.lut_d0.enable &&
859  LightingRegs::IsLightingSamplerSupported(
860  lighting.config, LightingRegs::LightingSampler::Distribution0)) {
861  // Lookup specular "distribution 0" LUT value
862  std::string value{GetLutValue(LightingRegs::LightingSampler::Distribution0,
863  light_config.num, lighting.lut_d0.type,
864  lighting.lut_d0.abs_input)};
865  d0_lut_value = "(" + std::to_string(lighting.lut_d0.scale) + " * " + value + ")";
866  }
867  std::string specular_0{"(" + d0_lut_value + " * " + light_src + ".specular_0)"};
868  if (light_config.geometric_factor_0) {
869  specular_0 = "(" + specular_0 + " * geo_factor)";
870  }
871 
872  // If enabled, lookup ReflectRed value, otherwise, 1.0 is used
873  if (lighting.lut_rr.enable &&
874  LightingRegs::IsLightingSamplerSupported(lighting.config,
875  LightingRegs::LightingSampler::ReflectRed)) {
876  std::string value{GetLutValue(LightingRegs::LightingSampler::ReflectRed,
877  light_config.num, lighting.lut_rr.type,
878  lighting.lut_rr.abs_input)};
879  value = "(" + std::to_string(lighting.lut_rr.scale) + " * " + value + ")";
880  out += "refl_value.r = " + value + ";\n";
881  } else {
882  out += "refl_value.r = 1.0;\n";
883  }
884 
885  // If enabled, lookup ReflectGreen value, otherwise, ReflectRed value is used
886  if (lighting.lut_rg.enable &&
887  LightingRegs::IsLightingSamplerSupported(lighting.config,
888  LightingRegs::LightingSampler::ReflectGreen)) {
889  std::string value{GetLutValue(LightingRegs::LightingSampler::ReflectGreen,
890  light_config.num, lighting.lut_rg.type,
891  lighting.lut_rg.abs_input)};
892  value = "(" + std::to_string(lighting.lut_rg.scale) + " * " + value + ")";
893  out += "refl_value.g = " + value + ";\n";
894  } else {
895  out += "refl_value.g = refl_value.r;\n";
896  }
897 
898  // If enabled, lookup ReflectBlue value, otherwise, ReflectRed value is used
899  if (lighting.lut_rb.enable &&
900  LightingRegs::IsLightingSamplerSupported(lighting.config,
901  LightingRegs::LightingSampler::ReflectBlue)) {
902  std::string value{GetLutValue(LightingRegs::LightingSampler::ReflectBlue,
903  light_config.num, lighting.lut_rb.type,
904  lighting.lut_rb.abs_input)};
905  value = "(" + std::to_string(lighting.lut_rb.scale) + " * " + value + ")";
906  out += "refl_value.b = " + value + ";\n";
907  } else {
908  out += "refl_value.b = refl_value.r;\n";
909  }
910 
911  // Specular 1 component
912  std::string d1_lut_value{"1.0"};
913  if (lighting.lut_d1.enable &&
914  LightingRegs::IsLightingSamplerSupported(
915  lighting.config, LightingRegs::LightingSampler::Distribution1)) {
916  // Lookup specular "distribution 1" LUT value
917  std::string value{GetLutValue(LightingRegs::LightingSampler::Distribution1,
918  light_config.num, lighting.lut_d1.type,
919  lighting.lut_d1.abs_input)};
920  d1_lut_value = "(" + std::to_string(lighting.lut_d1.scale) + " * " + value + ")";
921  }
922  std::string specular_1{"(" + d1_lut_value + " * refl_value * " + light_src +

924  if (light_config.geometric_factor_1) {
925  specular_1 = "(" + specular_1 + " * geo_factor)";
926  }
927 
928  // Fresnel
929  // Note: only the last entry in the light slots applies the Fresnel factor
930  if (light_index == lighting.src_num - 1 && lighting.lut_fr.enable &&
931  LightingRegs::IsLightingSamplerSupported(lighting.config,
932  LightingRegs::LightingSampler::Fresnel)) {

935  lighting.lut_fr.type, lighting.lut_fr.abs_input)};
936  value = "(" + std::to_string(lighting.lut_fr.scale) + " * " + value + ")";
937 
938  // Enabled for diffuse lighting alpha component
939  if (lighting.enable_primary_alpha) {
940  out += "diffuse_sum.a = " + value + ";\n";
941  }
942 
943  // Enabled for the specular lighting alpha component

945  out += "specular_sum.a = " + value + ";\n";
946  }
947  }
948 
949  bool shadow_primary_enable{lighting.shadow_primary && light_config.shadow_enable};
950  bool shadow_secondary_enable{lighting.shadow_secondary && light_config.shadow_enable};
951  std::string shadow_primary{shadow_primary_enable ? " * shadow.rgb" : ""};
952  std::string shadow_secondary{shadow_secondary_enable ? " * shadow.rgb" : ""};

956  ".ambient) * " + dist_atten + " * " + spot_atten + shadow_primary + ";\n";

958  // Compute secondary fragment color (specular lighting) function

971  }
972  }

976  out += "primary_fragment_color = clamp(diffuse_sum, vec4(0.0), vec4(1.0));\n";
