34 void ZMQServer::WorkerLoop() {
35  zmq::message_t request;
36  while (running) {
37  try {
38  if (zmq_socket->recv(&request, 0)) {
39  if (request.size() >= MIN_PACKET_SIZE && request.size() <= MAX_PACKET_SIZE) {
40  u8* request_buffer{static_cast<u8*>(request.data())};
41  PacketHeader header;
42  std::memcpy(&header, request_buffer, sizeof(header));
43  if ((request.size() - MIN_PACKET_SIZE) == header.packet_size) {
44  u8* data{request_buffer + MIN_PACKET_SIZE};
45  std::function<void(Packet&)> send_reply_callback{
46  std::bind(&ZMQServer::SendReply, this, std::placeholders::_1)};
47  std::unique_ptr<Packet> new_packet{
48  std::make_unique<Packet>(header, data, send_reply_callback)};
49 
50  // Send the request to the upper layer for handling
51  new_request_callback(std::move(new_packet));
52  }
53  }
54  }
55  } catch (...) {
56  LOG_WARNING(RPC, "Failed to receive data on ZeroMQ socket");
57  }
58  }
59 
60  // Destroying the socket must be done by this thread.
61  zmq_socket.reset();
62 }

64 void ZMQServer::SendReply(Packet& reply_packet) {
65  if (running) {
66  auto reply_buffer{
67  std::make_unique<u8[]>(MIN_PACKET_SIZE + reply_packet.GetPacketDataSize())};
68  auto reply_header{reply_packet.GetHeader()};
69 
70  std::memcpy(reply_buffer.get(), &reply_header, sizeof(reply_header));
71  std::memcpy(reply_buffer.get() + (4 * sizeof(u32)), reply_packet.GetPacketData().data(),
72  reply_packet.GetPacketDataSize());
73 
74  zmq_socket->send(reply_buffer.get(), MIN_PACKET_SIZE + reply_packet.GetPacketDataSize());
75 
76  LOG_INFO(RPC, "Sent reply version({}) id=({}) type=({}) size=({})",
77  reply_packet.GetVersion(), reply_packet.GetId(),
78  static_cast<u32>(reply_packet.GetPacketType()), reply_packet.GetPacketDataSize());
79  }
