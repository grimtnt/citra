46 void MemoryInit(u32 mem_type) {
47  if (Service::CFG::IsNewModeEnabled()) {
48  if (Settings::values.memory_developer_mode) {
49  mem_type = 7;
50  } else if (mem_type <= 5) {
51  mem_type = 6;
52  }
53  }
54 
55  ASSERT(mem_type != 1);
56 
57  // The kernel allocation regions (APPLICATION, SYSTEM and BASE) are laid out in sequence, with
58  // the sizes specified in the memory_region_sizes table.
59  VAddr base{};
60  for (int i{}; i < 3; ++i) {
61  memory_regions[i].base = base;
62  memory_regions[i].size = memory_region_sizes[mem_type][i];
63  memory_regions[i].used = 0;
64  memory_regions[i].linear_heap_memory = std::make_shared<std::vector<u8>>();
65  // Reserve enough space for this region of FCRAM.
66  // We do not want this block of memory to be relocated when allocating from it.
67  memory_regions[i].linear_heap_memory->reserve(memory_regions[i].size);
68 
69  base += memory_regions[i].size;
70  }
71 
72  // We must've allocated the entire FCRAM by the end
73  if (mem_type > 5) {
74  ASSERT(base == Memory::FCRAM_N3DS_SIZE);
75  } else {
76  ASSERT(base == Memory::FCRAM_SIZE);
77  }
78 
79  using ConfigMem::config_mem;
80  config_mem.app_mem_type = mem_type;
81  // app_mem_malloc does not always match the configured size for memory_region[0]: in case the
82  // n3DS type override is in effect it reports the size the game expects, not the real one.
83  config_mem.app_mem_alloc = memory_region_sizes[mem_type][0];
84  config_mem.sys_mem_alloc = memory_regions[1].size;
85  config_mem.base_mem_alloc = memory_regions[2].size;
86 }

88 void MemoryShutdown() {
89  for (auto& region : memory_regions) {
90  region.base = 0;
91  region.size = 0;
92  region.used = 0;
93  region.linear_heap_memory = nullptr;
94  }

110 void HandleSpecialMapping(VMManager& address_space, const AddressMapping& mapping) {
111  using namespace Memory;
112 
113  struct MemoryArea {
114  VAddr vaddr_base;
115  PAddr paddr_base;
116  u32 size;
117  };
118 
119  // The order of entries in this array is important. The VRAM and IO VAddr ranges overlap, and
120  // VRAM must be tried first.
121  static constexpr MemoryArea memory_areas[] = {
122  {VRAM_VADDR, VRAM_PADDR, VRAM_SIZE},
123  {IO_AREA_VADDR, IO_AREA_PADDR, IO_AREA_SIZE},
124  {DSP_RAM_VADDR, DSP_RAM_PADDR, DSP_RAM_SIZE},
125  {N3DS_EXTRA_RAM_VADDR, N3DS_EXTRA_RAM_PADDR, N3DS_EXTRA_RAM_SIZE - 0x20000},
126  };
127 
128  VAddr mapping_limit = mapping.address + mapping.size;
129  if (mapping_limit < mapping.address) {
130  LOG_CRITICAL(Loader, "Mapping size overflowed: address=0x{:08X}, size=0x{:08X}",
131  mapping.address, mapping.size);
132  return;
133  }
134 
135  auto area =
136  std::find_if(std::begin(memory_areas), std::end(memory_areas), [&](const auto& area) {
137  return mapping.address >= area.vaddr_base &&
138  mapping_limit <= area.vaddr_base + area.size;
139  });
140  if (area == std::end(memory_areas)) {
141  LOG_ERROR(
142  Loader,
143  "Unhandled special mapping: address=0x{:08X} size=0x{:X} read_only={}, unk_flag={}",
144  mapping.address, mapping.size, mapping.read_only, mapping.unk_flag);
145  return;
146  }
147 
148  u32 offset_into_region = mapping.address - area->vaddr_base;
149  if (area->paddr_base == IO_AREA_PADDR) {
150  LOG_ERROR(Loader, "MMIO mappings are not supported yet. phys_addr=0x{:08X}",
151  area->paddr_base + offset_into_region);
152  return;
153  }
154 
155  u8* target_pointer = Memory::GetPhysicalPointer(area->paddr_base + offset_into_region);
156 
157  // TODO(yuriks): This flag seems to have some other effect, but it's unknown what
158  MemoryState memory_state = mapping.unk_flag ? MemoryState::Static : MemoryState::IO;
159 
160  auto vma =
161  address_space.MapBackingMemory(mapping.address, target_pointer, mapping.size, memory_state)
162  .Unwrap();
163  address_space.Reprotect(vma,
164  mapping.read_only ? VMAPermission::Read : VMAPermission::ReadWrite);
165 }
