77 void FrameLimiter::DoFrameLimiting(u64 current_system_time_us) {
78  if (!Settings::values.use_frame_limit) {
79  return;
80  }
81 
82  auto now{Clock::now()};
83  double sleep_scale{Settings::values.frame_limit / 100.0};
84 
85  // Max lag caused by slow frames. Shouldn't be more than the length of a frame at the current
86  // speed percent or it will clamp too much and prevent this from properly limiting to that
87  // percent. High values means it'll take longer after a slow frame to recover and start limiting
88  const microseconds max_lag_time_us{duration_cast<microseconds>(
89  std::chrono::duration<double, std::chrono::microseconds::period>(25ms / sleep_scale))};
90  frame_limiting_delta_err += duration_cast<microseconds>(
91  std::chrono::duration<double, std::chrono::microseconds::period>(
92  (current_system_time_us - previous_system_time_us) / sleep_scale));
93  frame_limiting_delta_err -= duration_cast<microseconds>(now - previous_walltime);
94  frame_limiting_delta_err =
95  std::clamp(frame_limiting_delta_err, -max_lag_time_us, max_lag_time_us);
96 
97  if (frame_limiting_delta_err > microseconds::zero()) {
98  std::this_thread::sleep_for(frame_limiting_delta_err);
99  auto now_after_sleep = Clock::now();
100  frame_limiting_delta_err -= duration_cast<microseconds>(now_after_sleep - now);
101  now = now_after_sleep;
102  }
103 
104  previous_system_time_us = current_system_time_us;
105  previous_walltime = now;
106 }
