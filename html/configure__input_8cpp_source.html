25 static QString getKeyName(int key_code) {
26  switch (key_code) {
27  case Qt::Key_Shift:

29  case Qt::Key_Control:
30  return "Ctrl";
31  case Qt::Key_Alt:
32  return "Alt";

34  return "";

37  }

51 static QString ButtonToText(const Common::ParamPackage& param) {
52  if (!param.Has("engine")) {
53  return "[not set]";
54  } else if (param.Get("engine", "") == "keyboard") {
55  return getKeyName(param.Get("code", 0));
56  } else if (param.Get("engine", "") == "sdl") {
57  if (param.Has("hat")) {
58  return QString("Hat %1 %2")
59  .arg(QString::fromStdString(param.Get("hat", "")),
60  QString::fromStdString(param.Get("direction", "")));
61  }
62  if (param.Has("axis")) {
63  return QString("Axis %1% 2")
64  .arg(QString::fromStdString(param.Get("axis", "")),
65  QString::fromStdString(param.Get("direction", "")));
66  }
67  if (param.Has("button")) {
68  return QString("Button %1").arg(QString::fromStdString(param.Get("button", "")));
69  }
70  return QString();
71  } else {
72  return "[unknown]";
73  }
74 };

76 static QString AnalogToText(const Common::ParamPackage& param, const std::string& dir) {
77  if (!param.Has("engine")) {
78  return "[not set]";
79  } else if (param.Get("engine", "") == "analog_from_button") {
80  return ButtonToText(Common::ParamPackage{param.Get(dir, "")});
81  } else if (param.Get("engine", "") == "sdl") {
82  if (dir == "modifier") {
83  return "[unused]";
84  }
85 
86  if (dir == "left" || dir == "right") {
87  return QString("Axis %1").arg(QString::fromStdString(param.Get("axis_x", "")));
88  } else if (dir == "up" || dir == "down") {
89  return QString("Axis %1").arg(QString::fromStdString(param.Get("axis_y", "")));
90  }
91  return QString();
92  } else {
93  return "[unknown]";
94  }
95 };

293 void ConfigureInput::handleClick(QPushButton* button,
294  std::function<void(const Common::ParamPackage&)> new_input_setter,
295  InputCommon::Polling::DeviceType type) {
296  button->setText("[press key]");
297  button->setFocus();
298 
299  input_setter = new_input_setter;
300 
301  device_pollers = InputCommon::Polling::GetPollers(type);
302 
303  // Keyboard keys can only be used as button devices
304  want_keyboard_keys = type == InputCommon::Polling::DeviceType::Button;
305 
306  for (auto& poller : device_pollers) {
307  poller->Start();
308  }
309 
310  grabKeyboard();
311  grabMouse();
312  timeout_timer->start(5000); // Cancel after 5 seconds
313  poll_timer->start(200); // Check for new inputs every 200ms
314 }

316 void ConfigureInput::setPollingResult(const Common::ParamPackage& params, bool abort) {
317  releaseKeyboard();
318  releaseMouse();
319  timeout_timer->stop();
320  poll_timer->stop();
321  for (auto& poller : device_pollers) {
322  poller->Stop();
323  }
324 
325  if (!abort) {
326  (*input_setter)(params);
327  }
328 
329  updateButtonLabels();
330  input_setter = boost::none;
331 }

333 void ConfigureInput::keyPressEvent(QKeyEvent* event) {
334  if (!input_setter || !event)
335  return;
336 
337  if (event->key() != Qt::Key_Escape) {
338  if (want_keyboard_keys) {
339  setPollingResult(Common::ParamPackage{InputCommon::GenerateKeyboardParam(event->key())},
340  false);
341  } else {
342  // Escape key wasn't pressed and we don't want any keyboard keys, so don't stop polling
343  return;
344  }
345  }
346  setPollingResult({}, true);
