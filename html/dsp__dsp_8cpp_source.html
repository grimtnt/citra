24 void DSP_DSP::RecvData(Kernel::HLERequestContext& ctx) {
25  IPC::RequestParser rp{ctx, 0x01, 1, 0};
26  const u32 register_number{rp.Pop<u32>()};
27 
28  ASSERT_MSG(register_number == 0, "Unknown register_number {}", register_number);
29 
30  // Application reads this after requesting DSP shutdown, to verify the DSP has indeed shutdown
31  // or slept.
32 
33  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
34  rb.Push(RESULT_SUCCESS);
35 
36  switch (Core::DSP().GetDspState()) {
37  case AudioCore::DspState::On:
38  rb.Push<u32>(0);
39  break;
40  case AudioCore::DspState::Off:
41  case AudioCore::DspState::Sleeping:
42  rb.Push<u32>(1);
43  break;
44  default:
45  UNREACHABLE();
46  break;
47  }
48 
49  LOG_DEBUG(Service_DSP, "register_number={}", register_number);
50 }

89 void DSP_DSP::WriteProcessPipe(Kernel::HLERequestContext& ctx) {
90  IPC::RequestParser rp{ctx, 0x0D, 2, 2};
91  const u32 channel{rp.Pop<u32>()};
92  const u32 size{rp.Pop<u32>()};
93  auto buffer{rp.PopStaticBuffer()};
94 
95  const DspPipe pipe{static_cast<DspPipe>(channel)};
96 
97  // This behaviour was confirmed by RE.
98  // The likely reason for this is that games tend to pass in garbage at these bytes
99  // because they read random bytes off the stack.
100  switch (pipe) {
101  case DspPipe::Audio:
102  ASSERT(buffer.size() >= 4);
103  buffer[2] = 0;
104  buffer[3] = 0;
105  break;
106  case DspPipe::Binary:
107  ASSERT(buffer.size() >= 8);
108  buffer[4] = 1;
109  buffer[5] = 0;
110  buffer[6] = 0;
111  buffer[7] = 0;
112  break;
113  }
114 
115  Core::DSP().PipeWrite(pipe, buffer);
116 
117  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
118  rb.Push(RESULT_SUCCESS);
119 
120  LOG_DEBUG(Service_DSP, "channel={}, size=0x{:X}, buffer_size={:X}", channel, size,
121  buffer.size());
122 }

124 void DSP_DSP::ReadPipe(Kernel::HLERequestContext& ctx) {
125  IPC::RequestParser rp{ctx, 0x0E, 3, 0};
126  const u32 channel{rp.Pop<u32>()};
127  const u32 peer{rp.Pop<u32>()};
128  const u16 size{rp.Pop<u16>()};
129 
130  const DspPipe pipe{static_cast<DspPipe>(channel)};
131  const u16 pipe_readable_size{static_cast<u16>(Core::DSP().GetPipeReadableSize(pipe))};
132 
133  std::vector<u8> pipe_buffer;
134  if (pipe_readable_size >= size)
135  pipe_buffer = Core::DSP().PipeRead(pipe, size);
136  else {
137  // No more data is in pipe. Hardware hangs in this case. we emulate the hang
138  for (;;)
139  ;
140  }
141 
142  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 2)};
143  rb.Push(RESULT_SUCCESS);
144  rb.PushStaticBuffer(std::move(pipe_buffer), 0);
145 
146  LOG_DEBUG(Service_DSP, "channel={}, peer={}, size=0x{:04X}, pipe_readable_size=0x{:04X}",
147  channel, peer, size, pipe_readable_size);
148 }

189  IPC::RequestParser rp{ctx, 0x11, 3, 2};
190  const u32 size{rp.Pop<u32>()};
191  const u32 prog_mask{rp.Pop<u32>()};
192  const u32 data_mask{rp.Pop<u32>()};
193  auto& buffer{rp.PopMappedBuffer()};
194 
195  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 2)};
196  rb.Push(RESULT_SUCCESS);
197  rb.Push(true);
198  rb.PushMappedBuffer(buffer);
199 
200  // TODO(bunnei): Implement real DSP firmware loading
201 
202  std::vector<u8> component_data(size);
203  buffer.Read(component_data.data(), 0, size);
204 
205  LOG_INFO(Service_DSP, "Firmware hash: {:#018x}",
206  Common::ComputeHash64(component_data.data(), component_data.size()));
207  // Some versions of the firmware have the location of DSP structures listed here.
208  if (size > 0x37C) {
209  LOG_INFO(Service_DSP, "Structures hash: {:#018x}",
210  Common::ComputeHash64(component_data.data() + 0x340, 60));
211  }
212  LOG_WARNING(Service_DSP, "(STUBBED) called size=0x{:X}, prog_mask=0x{:08X}, data_mask=0x{:08X}",

251 void DSP_DSP::RegisterInterruptEvents(Kernel::HLERequestContext& ctx) {
252  IPC::RequestParser rp{ctx, 0x15, 2, 2};
253  const u32 interrupt{rp.Pop<u32>()};
254  const u32 channel{rp.Pop<u32>()};
255  auto event{rp.PopObject<Kernel::Event>()};
256 
257  ASSERT_MSG(interrupt < NUM_INTERRUPT_TYPE && channel < AudioCore::num_dsp_pipe,
258  "Invalid type or pipe: interrupt = {}, channel = {}", interrupt, channel);
259 
260  const InterruptType type{static_cast<InterruptType>(interrupt)};
261  const DspPipe pipe{static_cast<DspPipe>(channel)};
262 
263  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
264 
265  if (event) {
266  if (HasTooManyEventsRegistered()) {
267  LOG_INFO(Service_DSP,
268  "Ran out of space to register interrupts (Attempted to register "
269  "interrupt={}, channel={}, event={})",
270  interrupt, channel, event->GetName());
271  rb.Push(ResultCode(ErrorDescription::InvalidResultValue, ErrorModule::DSP,
272  ErrorSummary::OutOfResource, ErrorLevel::Status));
273  return;
274  } else {
275  GetInterruptEvent(type, pipe) = event;
276  LOG_INFO(Service_DSP, "Registered interrupt={}, channel={}, event={}", interrupt,
277  channel, event->GetName());
278  }
279  } else {
280  GetInterruptEvent(type, pipe) = nullptr;
281  LOG_INFO(Service_DSP, "Unregistered interrupt={}, channel={}", interrupt, channel);
282  }
283  rb.Push(RESULT_SUCCESS);
284 }

336 // DSP Interrupts:
337 // The audio-pipe interrupt occurs every frame tick. Userland programs normally have a thread
338 // that's waiting for an interrupt event. Immediately after this interrupt event, userland
339 // normally updates the state in the next region and increments the relevant frame counter by two.
340 void DSP_DSP::SignalInterrupt(InterruptType type, DspPipe pipe) {
341  LOG_DEBUG(Service_DSP, "called, type={}, pipe={}", static_cast<u32>(type),
342  static_cast<u32>(pipe));
343  const auto& event{GetInterruptEvent(type, pipe)};
344  if (event)
345  event->Signal();
346 }

348 Kernel::SharedPtr<Kernel::Event>& DSP_DSP::GetInterruptEvent(InterruptType type, DspPipe pipe) {
349  switch (type) {
350  case InterruptType::Zero:
351  return interrupt_zero;
352  case InterruptType::One:
353  return interrupt_one;
354  case InterruptType::Pipe: {
355  const std::size_t pipe_index{static_cast<std::size_t>(pipe)};
356  ASSERT(pipe_index < AudioCore::num_dsp_pipe);
357  return pipes[pipe_index];
358  }
359  }
360  UNREACHABLE_MSG("Invalid interrupt type = {}", static_cast<std::size_t>(type));
361 }

363 bool DSP_DSP::HasTooManyEventsRegistered() const {
364  std::size_t number{static_cast<std::size_t>(
365  std::count_if(pipes.begin(), pipes.end(), [](const auto& evt) { return evt != nullptr; }))};
366 
367  if (interrupt_zero != nullptr)
368  number++;
369  if (interrupt_one != nullptr)
370  number++;
371 
372  LOG_DEBUG(Service_DSP, "Number of events registered = {}", number);
373  return number >= max_number_of_interrupt_events;
374 }
