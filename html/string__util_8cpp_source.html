275 template <typename T>
276 static std::string CodeToUTF8(const char* fromcode, const std::basic_string<T>& input) {
277  iconv_t const conv_desc = iconv_open("UTF-8", fromcode);
278  if ((iconv_t)(-1) == conv_desc) {
279  LOG_ERROR(Common, "Iconv initialization failure [{}]: {}", fromcode, strerror(errno));
280  iconv_close(conv_desc);
281  return {};
282  }
283 
284  const std::size_t in_bytes = sizeof(T) * input.size();
285  // Multiply by 4, which is the max number of bytes to encode a codepoint
286  const std::size_t out_buffer_size = 4 * in_bytes;
287 
288  std::string out_buffer(out_buffer_size, '\0');
289 
290  auto src_buffer = &input[0];
291  std::size_t src_bytes = in_bytes;
292  auto dst_buffer = &out_buffer[0];
293  std::size_t dst_bytes = out_buffer.size();
294 
295  while (0 != src_bytes) {
296  std::size_t const iconv_result =
297  iconv(conv_desc, (char**)(&src_buffer), &src_bytes, &dst_buffer, &dst_bytes);
298 
299  if (static_cast<std::size_t>(-1) == iconv_result) {
300  if (EILSEQ == errno || EINVAL == errno) {
301  // Try to skip the bad character
302  if (0 != src_bytes) {
303  --src_bytes;
304  ++src_buffer;
305  }
306  } else {
307  LOG_ERROR(Common, "iconv failure [{}]: {}", fromcode, strerror(errno));
308  break;
309  }
310  }
311  }
312 
313  std::string result;
314  out_buffer.resize(out_buffer_size - dst_bytes);
315  out_buffer.swap(result);
316 
317  iconv_close(conv_desc);
318 
319  return result;
320 }

393 const char* TrimSourcePath(const char* path, const char* root) {
394  const char* p = path;
395 
396  while (*p != '\0') {
397  const char* next_slash = p;
398  while (*next_slash != '\0' && *next_slash != '/' && *next_slash != '\\') {
399  ++next_slash;
400  }
401 
402  bool is_src = Common::ComparePartialString(p, next_slash, root);
403  p = next_slash;
404 
405  if (*p != '\0') {
406  ++p;
407  }
408  if (is_src) {
409  path = p;
410  }
411  }
412  return path;
413 }
