33 // A read-only file created from a block of data. It only allows you to read the entire file at
34 // once, in a single read operation.
35 class ExeFSSectionFile final : public FileBackend {
36 public:
37  explicit ExeFSSectionFile(std::shared_ptr<std::vector<u8>> data_) : data(std::move(data_)) {}
38 
39  ResultVal<std::size_t> Read(u64 offset, std::size_t length, u8* buffer) const override {
40  if (offset != 0) {
41  LOG_ERROR(Service_FS, "offset must be zero!");
42  return ERROR_UNSUPPORTED_OPEN_FLAGS;
43  }
44 
45  if (length != data->size()) {
46  LOG_ERROR(Service_FS, "size must match the file size!");
47  return ERROR_INCORRECT_EXEFS_READ_SIZE;
48  }
49 
50  std::memcpy(buffer, data->data(), data->size());
51  return MakeResult<std::size_t>(data->size());
52  }
53 
54  ResultVal<std::size_t> Write(u64 offset, std::size_t length, bool flush, const u8* buffer) override {
55  LOG_ERROR(Service_FS, "The file is read-only!");
56  return ERROR_UNSUPPORTED_OPEN_FLAGS;
57  }
58 
59  u64 GetSize() const override {
60  return data->size();
61  }
62 
63  bool SetSize(u64 size) const override {
64  return false;
65  }
66 
67  bool Close() const override {
68  return true;
69  }
70 
71  void Flush() const override {}
72 
73 private:
74  std::shared_ptr<std::vector<u8>> data;
75 };

80 public:

87  ResultVal<std::unique_ptr<FileBackend>> OpenFile(const Path& path, const Mode&) const override {
88  // Note: SelfNCCHArchive doesn't check the open mode.
89 
90  if (path.GetType() != LowPathType::Binary) {
91  LOG_ERROR(Service_FS, "Path need to be Binary");
92  return ERROR_INVALID_PATH;
93  }
94 
95  std::vector<u8> binary = path.AsBinary();
96  if (binary.size() != sizeof(SelfNCCHFilePath)) {
97  LOG_ERROR(Service_FS, "Wrong path size {}", binary.size());
98  return ERROR_INVALID_PATH;
99  }
100 
101  SelfNCCHFilePath file_path{};
102  std::memcpy(&file_path, binary.data(), sizeof(SelfNCCHFilePath));
103 
104  switch (static_cast<SelfNCCHFilePathType>(file_path.type)) {
105  case SelfNCCHFilePathType::UpdateRomFS:
106  return OpenUpdateRomFS();
107 
108  case SelfNCCHFilePathType::RomFS:
109  return OpenRomFS();
110 
111  case SelfNCCHFilePathType::Code:
112  LOG_ERROR(Service_FS, "Reading the code section is not supported!");
113  return ERROR_COMMAND_NOT_ALLOWED;
114 
115  case SelfNCCHFilePathType::ExeFS: {
116  const auto& raw{file_path.exefs_filename};
117  auto end{std::find(raw.begin(), raw.end(), '\0')};
118  std::string filename{raw.begin(), end};
119  return OpenExeFS(filename);
120  }
121  default:
122  LOG_ERROR(Service_FS, "Unknown file type {}!", static_cast<u32>(file_path.type));
123  return ERROR_INVALID_PATH;
124  }
125  }

172  ResultVal<std::unique_ptr<FileBackend>> OpenRomFS() const {
173  if (ncch_data.romfs_file) {
174  std::unique_ptr<DelayGenerator> delay_generator{
175  std::make_unique<RomFSDelayGenerator>()};
176  return MakeResult<std::unique_ptr<FileBackend>>(
177  std::make_unique<IVFCFile>(ncch_data.romfs_file, std::move(delay_generator)));
178  } else {
179  LOG_INFO(Service_FS, "Unable to read RomFS");
180  return ERROR_ROMFS_NOT_FOUND;
181  }
182  }
183 
184  ResultVal<std::unique_ptr<FileBackend>> OpenUpdateRomFS() const {
185  if (ncch_data.update_romfs_file) {
186  std::unique_ptr<DelayGenerator> delay_generator{
187  std::make_unique<RomFSDelayGenerator>()};
188  return MakeResult<std::unique_ptr<FileBackend>>(std::make_unique<IVFCFile>(
189  ncch_data.update_romfs_file, std::move(delay_generator)));
190  } else {
191  LOG_INFO(Service_FS, "Unable to read update RomFS");
192  return ERROR_ROMFS_NOT_FOUND;
193  }
194  }
195 
196  ResultVal<std::unique_ptr<FileBackend>> OpenExeFS(const std::string& filename) const {
197  if (filename == "icon") {
198  if (ncch_data.icon) {
199  return MakeResult<std::unique_ptr<FileBackend>>(
200  std::make_unique<ExeFSSectionFile>(ncch_data.icon));
201  }
202 
203  LOG_WARNING(Service_FS, "Unable to read icon");
204  return ERROR_EXEFS_SECTION_NOT_FOUND;
205  }
206 
207  if (filename == "logo") {
208  if (ncch_data.logo) {
209  return MakeResult<std::unique_ptr<FileBackend>>(
210  std::make_unique<ExeFSSectionFile>(ncch_data.logo));
211  }
212 
213  LOG_WARNING(Service_FS, "Unable to read logo");
214  return ERROR_EXEFS_SECTION_NOT_FOUND;
215  }
216 
217  if (filename == "banner") {
218  if (ncch_data.banner) {
219  return MakeResult<std::unique_ptr<FileBackend>>(
220  std::make_unique<ExeFSSectionFile>(ncch_data.banner));
221  }
222 
223  LOG_WARNING(Service_FS, "Unable to read banner");
224  return ERROR_EXEFS_SECTION_NOT_FOUND;
225  }
226 
227  LOG_ERROR(Service_FS, "Unknown ExeFS section {}!", filename);
228  return ERROR_INVALID_PATH;
229  }

234 void ArchiveFactory_SelfNCCH::Register(Loader::AppLoader& app_loader) {
235  u64 program_id{};
236  if (app_loader.ReadProgramId(program_id) != Loader::ResultStatus::Success) {
237  LOG_WARNING(
238  Service_FS,
239  "Could not read program id when registering with SelfNCCH, this might be a 3dsx file");
240  }
241 
242  LOG_DEBUG(Service_FS, "Registering program {:016X} with the SelfNCCH archive factory",
243  program_id);
244 
245  if (ncch_data.find(program_id) != ncch_data.end()) {
246  LOG_WARNING(Service_FS,
247  "Registering program {:016X} with SelfNCCH will override existing mapping",
248  program_id);
249  }
250 
251  NCCHData& data{ncch_data[program_id]};
252 
253  std::shared_ptr<RomFSReader> romfs_file_{};
254  if (Loader::ResultStatus::Success == app_loader.ReadRomFS(romfs_file_)) {
255  data.romfs_file = std::move(romfs_file_);
256  }
257 
258  std::shared_ptr<RomFSReader> update_romfs_file{};
259  if (Loader::ResultStatus::Success == app_loader.ReadUpdateRomFS(update_romfs_file)) {
260  data.update_romfs_file = std::move(update_romfs_file);
261  }
262 
263  std::vector<u8> buffer{};
264 
265  if (Loader::ResultStatus::Success == app_loader.ReadIcon(buffer))
266  data.icon = std::make_shared<std::vector<u8>>(std::move(buffer));
267 
268  buffer.clear();
269  if (Loader::ResultStatus::Success == app_loader.ReadLogo(buffer))
270  data.logo = std::make_shared<std::vector<u8>>(std::move(buffer));
271 
272  buffer.clear();
273  if (Loader::ResultStatus::Success == app_loader.ReadBanner(buffer))
274  data.banner = std::make_shared<std::vector<u8>>(std::move(buffer));
275 }
