279 class GameListItemPath : public GameListItem {
280 public:
281  static const int FullPathRole = Qt::UserRole + 1;
282  static const int TitleRole = Qt::UserRole + 2;
283  static const int ProgramIdRole = Qt::UserRole + 3;
284 
285  GameListItemPath() : GameListItem() {}
286  GameListItemPath(const QString& game_path, const std::vector<u8>& smdh_data, u64 program_id)
287  : GameListItem() {
288  setData(game_path, FullPathRole);
289  setData(qulonglong(program_id), ProgramIdRole);
290 
291  if (!Loader::IsValidSMDH(smdh_data)) {
292  // SMDH is not valid, set a default icon
293  setData(GetDefaultIcon(true), Qt::DecorationRole);
294  return;
295  }
296 
297  Loader::SMDH smdh;
298  memcpy(&smdh, smdh_data.data(), sizeof(Loader::SMDH));
299 
300  // Get icon from SMDH
301  setData(GetQPixmapFromSMDH(smdh, true), Qt::DecorationRole);
302 
303  // Get title from SMDH
304  setData(GetQStringShortTitleFromSMDH(smdh, Loader::SMDH::TitleLanguage::English),
305  TitleRole);
306  }
307 
308  int type() const override {
309  return static_cast<int>(GameListItemType::Game);
310  }
311 
312  QVariant data(int role) const override {
313  if (role == Qt::DisplayRole) {
314  std::string path, filename, extension;
315  std::string sdmc_root = Settings::values.sd_card_directory.empty()
316  ? FileUtil::GetUserPath(D_SDMC_IDX)
317  : Settings::values.sd_card_directory + "/";
318  Common::SplitPath(data(FullPathRole).toString().toStdString(), &path, &filename,
319  &extension);
320  QString title = data(TitleRole).toString();
321  QString second_name = QString::fromStdString(filename + extension);
322  static QRegExp installed_pattern(
323  QString::fromStdString(
324  sdmc_root +
325  "Nintendo "
326  "3DS/00000000000000000000000000000000/00000000000000000000000000000000/" +
327  "title/0004000(0|2|e)/[0-9a-f]{8}/content/")
328  .replace("\\", "\\\\"));
329  static QRegExp system_pattern(
330  QString::fromStdString(FileUtil::GetUserPath(D_NAND_IDX) +
331  "00000000000000000000000000000000/"
332  "title/00040010/[0-9a-f]{8}/content/")
333  .replace("\\", "\\\\"));
334  if (installed_pattern.exactMatch(QString::fromStdString(path)) ||
335  system_pattern.exactMatch(QString::fromStdString(path))) {
336  // Use a different mechanism for system / installed titles showing program ID
337  second_name = QString("%1-%2")
338  .arg(data(ProgramIdRole).toULongLong(), 16, 16, QChar('0'))
339  .toUpper()
340  .arg(QString::fromStdString(filename));
341  }
342  return title + (title.isEmpty() ? "" : "\n ") + second_name;
343  } else {
344  return GameListItem::data(role);
345  }
346  }
347 };
