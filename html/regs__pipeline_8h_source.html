181  struct {
182  // There are two channels that can be used to configure the next command buffer, which can
183  // be then executed by writing to the "trigger" registers. There are two reasons why a game
184  // might use this feature:
185  // 1) With this, an arbitrary number of additional command buffers may be executed in
186  // sequence without requiring any intervention of the CPU after the initial one is
187  // kicked off.
188  // 2) Games can configure these registers to provide a command list subroutine mechanism.
189 
190  // TODO: verify the bit length of these two fields
191  // According to 3dbrew, the bit length of them are 21 and 29, respectively
192  BitField<0, 20, u32> size[2];
