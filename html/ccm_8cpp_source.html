20 // and override with the non-standard part.

27 public:

29  lword footer_length) override {

45 std::vector<u8> EncryptSignCCM(const std::vector<u8>& pdata, const CCMNonce& nonce,
46  std::size_t slot_id) {
47  if (!IsNormalKeyAvailable(slot_id)) {
48  LOG_ERROR(HW_AES, "Key slot {} not available. Will use zero key.", slot_id);
49  }
50  const AESKey normal = GetNormalKey(slot_id);
51  std::vector<u8> cipher(pdata.size() + CCM_MAC_SIZE);
52 
53  try {
54  CCM_3DSVariant::Encryption e;
55  e.SetKeyWithIV(normal.data(), AES_BLOCK_SIZE, nonce.data(), CCM_NONCE_SIZE);
56  e.SpecifyDataLengths(0, pdata.size(), 0);
57  CryptoPP::ArraySource as(pdata.data(), pdata.size(), true,
58  new CryptoPP::AuthenticatedEncryptionFilter(
59  e, new CryptoPP::ArraySink(cipher.data(), cipher.size())));
60  } catch (const CryptoPP::Exception& e) {
61  LOG_ERROR(HW_AES, "FAILED with: {}", e.what());
62  }
63  return cipher;
64 }

66 std::vector<u8> DecryptVerifyCCM(const std::vector<u8>& cipher, const CCMNonce& nonce,
67  std::size_t slot_id) {
68  if (!IsNormalKeyAvailable(slot_id)) {
69  LOG_ERROR(HW_AES, "Key slot {} not available. Will use zero key.", slot_id);
70  }
71  const AESKey normal = GetNormalKey(slot_id);
72  const std::size_t pdata_size = cipher.size() - CCM_MAC_SIZE;
73  std::vector<u8> pdata(pdata_size);
74 
75  try {
76  CCM_3DSVariant::Decryption d;
77  d.SetKeyWithIV(normal.data(), AES_BLOCK_SIZE, nonce.data(), CCM_NONCE_SIZE);
78  d.SpecifyDataLengths(0, pdata_size, 0);
79  CryptoPP::AuthenticatedDecryptionFilter df(
80  d, new CryptoPP::ArraySink(pdata.data(), pdata_size));
81  CryptoPP::ArraySource as(cipher.data(), cipher.size(), true, new CryptoPP::Redirector(df));
82  if (!df.GetLastResult()) {
83  LOG_ERROR(HW_AES, "FAILED");
84  return {};
85  }
86  } catch (const CryptoPP::Exception& e) {
87  LOG_ERROR(HW_AES, "FAILED with: {}", e.what());
88  return {};
89  }
90  return pdata;
91 }
