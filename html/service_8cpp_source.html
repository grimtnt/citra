121 static std::string MakeFunctionString(const char* name, const char* port_name,
122  const u32* cmd_buff) {
123  // Number of params == bits 0-5 + bits 6-11
124  int num_params{static_cast<int>((cmd_buff[0] & 0x3F) + ((cmd_buff[0] >> 6) & 0x3F))};
125 
126  std::string function_string{fmt::format("function '{}': port={}", name, port_name)};
127  for (int i{1}; i <= num_params; ++i) {
128  function_string += fmt::format(", cmd_buff[{}]=0x{:X}", i, cmd_buff[i]);
129  }
130  return function_string;
131 }

162 void ServiceFrameworkBase::ReportUnimplementedFunction(u32* cmd_buf, const FunctionInfoBase* info) {
163  IPC::Header header{cmd_buf[0]};
164  int num_params{static_cast<int>(header.normal_params_size + header.translate_params_size)};
165  std::string function_name{info == nullptr ? fmt::format("{:#08x}", cmd_buf[0]) : info->name};
166 
167  fmt::memory_buffer buf{};
168  fmt::format_to(buf, "function '{}': port='{}' cmd_buf={{[0]={:#x}", function_name, service_name,
169  cmd_buf[0]);
170  for (int i{1}; i <= num_params; ++i) {
171  fmt::format_to(buf, ", [{}]={:#x}", i, cmd_buf[i]);
172  }
173  buf.push_back('}');
174 
175  LOG_ERROR(Service, "unknown / unimplemented {}", fmt::to_string(buf));
176  // TODO(bunnei): Hack - ignore error
177  cmd_buf[1] = 0;
178 }

180 void ServiceFrameworkBase::HandleSyncRequest(SharedPtr<ServerSession> server_session) {
181  u32* cmd_buf = Kernel::GetCommandBuffer();
182 
183  u32 header_code{cmd_buf[0]};
184  auto itr{handlers.find(header_code)};
185  const FunctionInfoBase* info{itr == handlers.end() ? nullptr : &itr->second};
186  if (info == nullptr || info->handler_callback == nullptr) {
187  return ReportUnimplementedFunction(cmd_buf, info);
188  }
189 
190  // TODO(yuriks): The kernel should be the one handling this as part of translation after
191  // everything else is migrated
192  Kernel::HLERequestContext context(std::move(server_session));
193  context.PopulateFromIncomingCommandBuffer(cmd_buf, *Kernel::g_current_process,
194  Kernel::g_handle_table);
195 
196  LOG_TRACE(Service, "{}", MakeFunctionString(info->name, GetServiceName().c_str(), cmd_buf));
197  handler_invoker(this, info->handler_callback, context);
198 
199  auto thread = Kernel::GetCurrentThread();
200  ASSERT(thread->status == THREADSTATUS_RUNNING || thread->status == THREADSTATUS_WAIT_HLE_EVENT);
201  // Only write the response immediately if the thread is still running. If the HLE
202  // handler put the thread to sleep then the writing of the command buffer will be
203  // deferred to the wakeup callback.
204  if (thread->status == THREADSTATUS_RUNNING) {
205  context.WriteToOutgoingCommandBuffer(cmd_buf, *Kernel::g_current_process,
206  Kernel::g_handle_table);
207  }
208 }

218 static bool AttemptLLE(const ServiceModuleInfo& service_module) {
219  if (!Settings::values.lle_modules.at(service_module.name))
220  return false;
221  std::unique_ptr<Loader::AppLoader> loader{
222  Loader::GetLoader(AM::GetTitleContentPath(FS::MediaType::NAND, service_module.title_id))};
223  if (!loader) {
224  LOG_ERROR(Service,
225  "Service module \"{}\" could not be loaded; Defaulting to HLE implementation.",
226  service_module.name);
227  return false;
228  }
229  SharedPtr<Kernel::Process> process;
230  loader->Load(process);
231  LOG_DEBUG(Service, "Service module \"{}\" has been successfully loaded.", service_module.name);
232  return true;
233 }
