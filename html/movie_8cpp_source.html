347 Movie::ValidationResult Movie::ValidateHeader(const CTMHeader& header, u64 program_id) const {
348  if (header_magic_bytes != header.filetype) {
349  LOG_ERROR(Movie, "Playback file does not have valid header");
350  return ValidationResult::Invalid;
351  }
352 
353  std::string revision{
354  Common::ArrayToString(header.revision.data(), header.revision.size(), 21, false)};
355  revision = Common::ToLower(revision);
356 
357  if (!program_id)
358  Core::System::GetInstance().GetAppLoader().ReadProgramId(program_id);
359  if (program_id != header.program_id) {
360  LOG_WARNING(Movie, "This movie was recorded using a ROM with a different program id");
361  return ValidationResult::GameDismatch;
362  }
363 
364  if (revision != Common::g_scm_rev) {
365  LOG_WARNING(Movie,
366  "This movie was created on a different version of Citra, playback may desync");
367  return ValidationResult::RevisionDismatch;
368  }
369 
370  return ValidationResult::OK;
371 }

373 void Movie::SaveMovie() {
374  LOG_INFO(Movie, "Saving recorded movie to '{}'", record_movie_file);
375  FileUtil::IOFile save_record{record_movie_file, "wb"};
376 
377  if (!save_record.IsGood()) {
378  LOG_ERROR(Movie, "Unable to open file to save movie");
379  return;
380  }
381 
382  CTMHeader header = {};
383  header.filetype = header_magic_bytes;
384 
385  Core::System::GetInstance().GetAppLoader().ReadProgramId(header.program_id);
386 
387  std::string rev_bytes{};
388  CryptoPP::StringSource(Common::g_scm_rev, true,
389  new CryptoPP::HexDecoder(new CryptoPP::StringSink(rev_bytes)));
390  std::memcpy(header.revision.data(), rev_bytes.data(), sizeof(CTMHeader::revision));
391 
392  save_record.WriteBytes(&header, sizeof(CTMHeader));
393  save_record.WriteBytes(recorded_input.data(), recorded_input.size());
394 
395  if (!save_record.IsGood()) {
396  LOG_ERROR(Movie, "Error saving movie");
397  }
398 }

400 void Movie::StartPlayback(const std::string& movie_file,
401  std::function<void()> completion_callback) {
402  LOG_INFO(Movie, "Loading Movie for playback");
403  FileUtil::IOFile save_record{movie_file, "rb"};
404  const u64 size{save_record.GetSize()};
405 
406  if (save_record.IsGood() && size > sizeof(CTMHeader)) {
407  CTMHeader header{};
408  save_record.ReadArray(&header, 1);
409  if (ValidateHeader(header) != ValidationResult::Invalid) {
410  play_mode = PlayMode::Playing;
411  recorded_input.resize(size - sizeof(CTMHeader));
412  save_record.ReadArray(recorded_input.data(), recorded_input.size());
413  current_byte = 0;
414  playback_completion_callback = completion_callback;
415  }
416  } else {
417  LOG_ERROR(Movie, "Failed to playback movie: Unable to open '{}'", movie_file);
418  }
419 }
