59 void MultiplayerState::OnNetworkStateChanged(const Network::RoomMember::State& state) {
60  LOG_DEBUG(Frontend, "Network State: {}", Network::GetStateStr(state));
61  bool is_connected{};
62  switch (state) {
63  case Network::RoomMember::State::LostConnection:
64  NetworkMessage::ShowError(NetworkMessage::LOST_CONNECTION);
65  break;
66  case Network::RoomMember::State::CouldNotConnect:
67  NetworkMessage::ShowError(NetworkMessage::UNABLE_TO_CONNECT);
68  break;
69  case Network::RoomMember::State::MacCollision:
70  NetworkMessage::ShowError(NetworkMessage::MAC_COLLISION);
71  break;
72  case Network::RoomMember::State::Error:
73  NetworkMessage::ShowError(NetworkMessage::UNABLE_TO_CONNECT);
74  break;
75  case Network::RoomMember::State::Joined:
76  is_connected = true;
77  break;
78  }
79  if (is_connected) {
80  status_icon->setPixmap(QIcon::fromTheme("connected").pixmap(16));
81  leave_room->setEnabled(true);
82  } else {
83  status_icon->setPixmap(QIcon::fromTheme("disconnected").pixmap(16));
84  leave_room->setEnabled(false);
85  }
86 
87  current_state = state;
88 }
