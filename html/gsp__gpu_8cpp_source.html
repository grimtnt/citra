112 static ResultCode WriteHWRegs(u32 base_address, u32 size_in_bytes, const std::vector<u8>& data) {
113  // This magic number is verified to be done by the gsp module
114  const u32 max_size_in_bytes = 0x80;
115 
116  if (base_address & 3 || base_address >= 0x420000) {
117  LOG_ERROR(Service_GSP,
118  "Write address was out of range or misaligned! (address=0x{:08x}, size=0x{:08x})",
119  base_address, size_in_bytes);
120  return ERR_REGS_OUTOFRANGE_OR_MISALIGNED;
121  } else if (size_in_bytes <= max_size_in_bytes) {
122  if (size_in_bytes & 3) {
123  LOG_ERROR(Service_GSP, "Misaligned size 0x{:08x}", size_in_bytes);
124  return ERR_REGS_MISALIGNED;
125  } else {
126  std::size_t offset{};
127  while (size_in_bytes > 0) {
128  u32 value;
129  std::memcpy(&value, &data[offset], sizeof(u32));
130  WriteSingleHWReg(base_address, value);
131 
132  size_in_bytes -= 4;
133  offset += 4;
134  base_address += 4;
135  }
136  return RESULT_SUCCESS;
137  }
138 
139  } else {
140  LOG_ERROR(Service_GSP, "Out of range size 0x{:08x}", size_in_bytes);
141  return ERR_REGS_INVALID_SIZE;
142  }
143 }

155 static ResultCode WriteHWRegsWithMask(u32 base_address, u32 size_in_bytes,
156  const std::vector<u8>& data, const std::vector<u8>& masks) {
157  // This magic number is verified to be done by the gsp module
158  const u32 max_size_in_bytes = 0x80;
159 
160  if (base_address & 3 || base_address >= 0x420000) {
161  LOG_ERROR(Service_GSP,
162  "Write address was out of range or misaligned! (address=0x{:08x}, size=0x{:08x})",
163  base_address, size_in_bytes);
164  return ERR_REGS_OUTOFRANGE_OR_MISALIGNED;
165  } else if (size_in_bytes <= max_size_in_bytes) {
166  if (size_in_bytes & 3) {
167  LOG_ERROR(Service_GSP, "Misaligned size 0x{:08x}", size_in_bytes);
168  return ERR_REGS_MISALIGNED;
169  } else {
170  std::size_t offset{};
171  while (size_in_bytes > 0) {
172  const u32 reg_address = base_address + REGS_BEGIN;
173 
174  u32 reg_value;
175  HW::Read<u32>(reg_value, reg_address);
176 
177  u32 value, mask;
178  std::memcpy(&value, &data[offset], sizeof(u32));
179  std::memcpy(&mask, &masks[offset], sizeof(u32));
180 
181  // Update the current value of the register only for set mask bits
182  reg_value = (reg_value & ~mask) | (value & mask);
183 
184  WriteSingleHWReg(base_address, reg_value);
185 
186  size_in_bytes -= 4;
187  offset += 4;
188  base_address += 4;
189  }
190  return RESULT_SUCCESS;
191  }
192 
193  } else {
194  LOG_ERROR(Service_GSP, "Out of range size 0x{:08x}", size_in_bytes);
195  return ERR_REGS_INVALID_SIZE;
196  }
197 }

221 void GSP_GPU::ReadHWRegs(Kernel::HLERequestContext& ctx) {
222  IPC::RequestParser rp{ctx, 0x4, 2, 0};
223  u32 reg_addr{rp.Pop<u32>()};
224  u32 input_size{rp.Pop<u32>()};
225 
226  static constexpr u32 MaxReadSize = 0x80;
227  u32 size = std::min(input_size, MaxReadSize);
228 
229  if ((reg_addr % 4) != 0 || reg_addr >= 0x420000) {
230  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
231  rb.Push(ERR_REGS_OUTOFRANGE_OR_MISALIGNED);
232  LOG_ERROR(Service_GSP, "Invalid address 0x{:08x}", reg_addr);
233  return;
234  }
235 
236  // size should be word-aligned
237  if ((size % 4) != 0) {
238  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
239  rb.Push(ERR_REGS_MISALIGNED);
240  LOG_ERROR(Service_GSP, "Invalid size 0x{:08x}", size);
241  return;
242  }
243 
244  std::vector<u8> buffer(size);
245  for (u32 offset{}; offset < size; ++offset) {
246  HW::Read<u8>(buffer[offset], REGS_BEGIN + reg_addr + offset);
247  }
248 
249  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 2)};
250  rb.Push(RESULT_SUCCESS);
251  rb.PushStaticBuffer(std::move(buffer), 0);
252 }

254 ResultCode SetBufferSwap(u32 screen_id, const FrameBufferInfo& info) {
255  u32 base_address = 0x400000;
256  PAddr phys_address_left = Memory::VirtualToPhysicalAddress(info.address_left);
257  PAddr phys_address_right = Memory::VirtualToPhysicalAddress(info.address_right);
258  if (info.active_fb == 0) {
259  WriteSingleHWReg(base_address + 4 * static_cast<u32>(GPU_REG_INDEX(
260  framebuffer_config[screen_id].address_left1)),
261  phys_address_left);
262  WriteSingleHWReg(base_address + 4 * static_cast<u32>(GPU_REG_INDEX(
263  framebuffer_config[screen_id].address_right1)),
264  phys_address_right);
265  } else {
266  WriteSingleHWReg(base_address + 4 * static_cast<u32>(GPU_REG_INDEX(
267  framebuffer_config[screen_id].address_left2)),
268  phys_address_left);
269  WriteSingleHWReg(base_address + 4 * static_cast<u32>(GPU_REG_INDEX(
270  framebuffer_config[screen_id].address_right2)),
271  phys_address_right);
272  }
273  WriteSingleHWReg(base_address +
274  4 * static_cast<u32>(GPU_REG_INDEX(framebuffer_config[screen_id].stride)),
275  info.stride);
276  WriteSingleHWReg(base_address + 4 * static_cast<u32>(GPU_REG_INDEX(
277  framebuffer_config[screen_id].color_format)),
278  info.format);
279  WriteSingleHWReg(
280  base_address + 4 * static_cast<u32>(GPU_REG_INDEX(framebuffer_config[screen_id].active_fb)),
281  info.shown_fb);
282 
283  if (screen_id == 0) {
284  Core::System::GetInstance().perf_stats.EndGameFrame();
285  }
286 
287  return RESULT_SUCCESS;
288 }

367 void GSP_GPU::SignalInterruptForThread(InterruptId interrupt_id, u32 thread_id) {
368  SessionData* session_data = FindRegisteredThreadData(thread_id);
369  if (session_data == nullptr)
370  return;
371 
372  auto interrupt_event = session_data->interrupt_event;
373  if (interrupt_event == nullptr) {
374  LOG_WARNING(Service_GSP, "cannot synchronize until GSP event has been created!");
375  return;
376  }
377  InterruptRelayQueue* interrupt_relay_queue = GetInterruptRelayQueue(shared_memory, thread_id);
378  u8 next = interrupt_relay_queue->index;
379  next += interrupt_relay_queue->number_interrupts;
380  next = next % 0x34; // 0x34 is the number of interrupt slots
381 
382  interrupt_relay_queue->number_interrupts += 1;
383 
384  interrupt_relay_queue->slot[next] = interrupt_id;
385  interrupt_relay_queue->error_code = 0x0; // No error
386 
387  // Update framebuffer information if requested
388  // TODO(yuriks): Confirm where this code should be called. It is definitely updated without
389  // executing any GSP commands, only waiting on the event.
390  // TODO(Subv): The real GSP module triggers PDC0 after updating both the top and bottom
391  // screen, it is currently unknown what PDC1 does.
392  int screen_id =
393  (interrupt_id == InterruptId::PDC0) ? 0 : (interrupt_id == InterruptId::PDC1) ? 1 : -1;
394  if (screen_id != -1) {
395  FrameBufferUpdate* info = GetFrameBufferInfo(thread_id, screen_id);
396  if (info->is_dirty) {
397  GSP::SetBufferSwap(screen_id, info->framebuffer_info[info->index]);
398  info->is_dirty.Assign(false);
399  }
400  }
401  interrupt_event->Signal();
402 }

410 void GSP_GPU::SignalInterrupt(InterruptId interrupt_id) {
411  if (nullptr == shared_memory) {
412  LOG_WARNING(Service_GSP, "cannot synchronize until GSP shared memory has been created!");
413  return;
414  }
415 
416  // The PDC0 and PDC1 interrupts are fired even if the GPU right hasn't been acquired.
417  // Normal interrupts are only signaled for the active thread (ie, the thread that has the GPU
418  // right), but the PDC0/1 interrupts are signaled for every registered thread.
419  if (interrupt_id == InterruptId::PDC0 || interrupt_id == InterruptId::PDC1) {
420  for (u32 thread_id{}; thread_id < MaxGSPThreads; ++thread_id) {
421  SignalInterruptForThread(interrupt_id, thread_id);
422  }
423  return;
424  }
425 
426  // For normal interrupts, don't do anything if no process has acquired the GPU right.
427  if (active_thread_id == -1)
428  return;
429 
430  SignalInterruptForThread(interrupt_id, active_thread_id);
431 }

434 static void ExecuteCommand(const Command& command, u32 thread_id) {
435  // Utility function to convert register ID to address
436  static auto WriteGPURegister = [](u32 id, u32 data) {
437  GPU::Write<u32>(0x1EF00000 + 4 * id, data);
438  };
439 
440  switch (command.id) {
441 
442  // GX request DMA - typically used for copying memory from GSP heap to VRAM
443  case CommandId::REQUEST_DMA: {
444  // TODO: Consider attempting rasterizer-accelerated surface blit if that usage is ever
445  // possible/likely
446  Memory::RasterizerFlushVirtualRegion(command.dma_request.source_address,
447  command.dma_request.size, Memory::FlushMode::Flush);
448  Memory::RasterizerFlushVirtualRegion(command.dma_request.dest_address,
449  command.dma_request.size,
450  Memory::FlushMode::Invalidate);
451 
452  // TODO(Subv): These memory accesses should not go through the application's memory mapping.
453  // They should go through the GSP module's memory mapping.
454  Memory::CopyBlock(command.dma_request.dest_address, command.dma_request.source_address,
455  command.dma_request.size);
456  SignalInterrupt(InterruptId::DMA);
457  break;
458  }
459  // TODO: This will need some rework in the future. (why?)
460  case CommandId::SUBMIT_GPU_CMDLIST: {
461  auto& params = command.submit_gpu_cmdlist;
462 
463  if (params.do_flush) {
464  // This flag flushes the command list (params.address, params.size) from the cache.
465  // Command lists are not processed by the hardware renderer, so we don't need to
466  // actually flush them in Citra.
467  }
468 
469  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(command_processor_config.address)),
470  Memory::VirtualToPhysicalAddress(params.address) >> 3);
471  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(command_processor_config.size)),
472  params.size);
473 
474  // TODO: Not sure if we are supposed to always write this .. seems to trigger processing
475  // though
476  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(command_processor_config.trigger)), 1);
477 
478  // TODO(yuriks): Figure out the meaning of the `flags` field.
479 
480  break;
481  }
482 
483  // It's assumed that the two "blocks" behave equivalently.
484  // Presumably this is done simply to allow two memory fills to run in parallel.
485  case CommandId::SET_MEMORY_FILL: {
486  auto& params = command.memory_fill;
487 
488  if (params.start1 != 0) {
489  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[0].address_start)),
490  Memory::VirtualToPhysicalAddress(params.start1) >> 3);
491  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[0].address_end)),
492  Memory::VirtualToPhysicalAddress(params.end1) >> 3);
493  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[0].value_32bit)),
494  params.value1);
495  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[0].control)),
496  params.control1);
497  }
498 
499  if (params.start2 != 0) {
500  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[1].address_start)),
501  Memory::VirtualToPhysicalAddress(params.start2) >> 3);
502  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[1].address_end)),
503  Memory::VirtualToPhysicalAddress(params.end2) >> 3);
504  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[1].value_32bit)),
505  params.value2);
506  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(memory_fill_config[1].control)),
507  params.control2);
508  }
509  break;
510  }
511 
512  case CommandId::SET_DISPLAY_TRANSFER: {
513  auto& params = command.display_transfer;
514  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(display_transfer_config.input_address)),
515  Memory::VirtualToPhysicalAddress(params.in_buffer_address) >> 3);
516  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(display_transfer_config.output_address)),
517  Memory::VirtualToPhysicalAddress(params.out_buffer_address) >> 3);
518  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(display_transfer_config.input_size)),
519  params.in_buffer_size);
520  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(display_transfer_config.output_size)),
521  params.out_buffer_size);
522  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(display_transfer_config.flags)),
523  params.flags);
524  WriteGPURegister(static_cast<u32>(GPU_REG_INDEX(display_transfer_config.trigger)), 1);
525  break;
526  }
527 
528  case CommandId::SET_TEXTURE_COPY: {
529  auto& params = command.texture_copy;
530  WriteGPURegister((u32)GPU_REG_INDEX(display_transfer_config.input_address),
531  Memory::VirtualToPhysicalAddress(params.in_buffer_address) >> 3);
532  WriteGPURegister((u32)GPU_REG_INDEX(display_transfer_config.output_address),
533  Memory::VirtualToPhysicalAddress(params.out_buffer_address) >> 3);
534  WriteGPURegister((u32)GPU_REG_INDEX(display_transfer_config.texture_copy.size),
535  params.size);
536  WriteGPURegister((u32)GPU_REG_INDEX(display_transfer_config.texture_copy.input_size),
537  params.in_width_gap);
538  WriteGPURegister((u32)GPU_REG_INDEX(display_transfer_config.texture_copy.output_size),
539  params.out_width_gap);
540  WriteGPURegister((u32)GPU_REG_INDEX(display_transfer_config.flags), params.flags);
541 
542  // NOTE: Actual GSP ORs 1 with current register instead of overwriting. Doesn't seem to
543  // matter.
544  WriteGPURegister((u32)GPU_REG_INDEX(display_transfer_config.trigger), 1);
545  break;
546  }
547 
548  case CommandId::CACHE_FLUSH: {
549  // NOTE: Rasterizer flushing handled elsewhere in CPU read/write and other GPU handlers
550  // Use command.cache_flush.regions to implement this handler
551  break;
552  }
553 
554  default:
555  LOG_ERROR(Service_GSP, "unknown command 0x{:08X}", (int)command.id.Value());
556  }
557 }

576 void GSP_GPU::TriggerCmdReqQueue(Kernel::HLERequestContext& ctx) {
577  IPC::RequestParser rp{ctx, 0xC, 0, 0};
578 
579  // Iterate through each thread's command queue...
580  for (unsigned thread_id{}; thread_id < 0x4; ++thread_id) {
581  CommandBuffer* command_buffer = (CommandBuffer*)GetCommandBuffer(shared_memory, thread_id);
582 
583  // Iterate through each command...
584  for (unsigned i{}; i < command_buffer->number_commands; ++i) {
