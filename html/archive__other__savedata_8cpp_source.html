23 template <typename T>
24 ResultVal<std::tuple<MediaType, u64>> ParsePath(const Path& path, T program_id_reader) {
25  if (path.GetType() != LowPathType::Binary) {
26  LOG_ERROR(Service_FS, "Wrong path type {}", static_cast<int>(path.GetType()));
27  return ERROR_INVALID_PATH;
28  }
29 
30  std::vector<u8> vec_data{path.AsBinary()};
31 
32  if (vec_data.size() != 12) {
33  LOG_ERROR(Service_FS, "Wrong path length {}", vec_data.size());
34  return ERROR_INVALID_PATH;
35  }
36 
37  const u32* data{reinterpret_cast<const u32*>(vec_data.data())};
38  auto media_type{static_cast<MediaType>(data[0])};
39 
40  if (media_type != MediaType::SDMC && media_type != MediaType::GameCard) {
41  LOG_ERROR(Service_FS, "Unsupported media type {}", static_cast<u32>(media_type));
42 
43  // Note: this is strange, but the error code was verified with a real 3DS
44  return ERROR_UNSUPPORTED_OPEN_FLAGS;
45  }
46 
47  return MakeResult<std::tuple<MediaType, u64>>(media_type, program_id_reader(data));
48 }

50 ResultVal<std::tuple<MediaType, u64>> ParsePathPermitted(const Path& path) {
51  return ParsePath(path,
52  [](const u32* data) -> u64 { return (data[1] << 8) | 0x0004000000000000ULL; });
53 }
54 
55 ResultVal<std::tuple<MediaType, u64>> ParsePathGeneral(const Path& path) {
56  return ParsePath(
57  path, [](const u32* data) -> u64 { return data[1] | (static_cast<u64>(data[2]) << 32); });
58 }
