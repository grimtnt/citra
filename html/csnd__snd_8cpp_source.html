13  IPC::RequestParser rp{ctx, 0x01, 5, 0};
14  const u32 size = Common::AlignUp(rp.Pop<u32>(), Memory::PAGE_SIZE);
15  const u32 offset0{rp.Pop<u32>()};
16  const u32 offset1{rp.Pop<u32>()};
17  const u32 offset2{rp.Pop<u32>()};
18  const u32 offset3{rp.Pop<u32>()};
19 
20  using Kernel::MemoryPermission;
21  mutex = Kernel::Mutex::Create(false, "CSND:mutex");
22  shared_memory = Kernel::SharedMemory::Create(nullptr, size, MemoryPermission::ReadWrite,
23  MemoryPermission::ReadWrite, 0,
24  Kernel::MemoryRegion::BASE, "CSND:SharedMemory");
25 
26  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 3)};
27  rb.Push(RESULT_SUCCESS);

30  LOG_WARNING(Service_CSND,

32  "offset0=0x{:08X} offset1=0x{:08X} offset2=0x{:08X} offset3=0x{:08X}",
33  size, offset0, offset1, offset2, offset3);
34 }

50 void CSND_SND::ExecuteCommands(Kernel::HLERequestContext& ctx) {
51  IPC::RequestParser rp{ctx, 0x03, 1, 0};
52  const u32 addr{rp.Pop<u32>()};
53 
54  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
55  if (!shared_memory) {
56  rb.Push<u32>(1);
57  LOG_ERROR(Service_CSND, "called, shared memory not allocated");
58  } else {
59  u8* ptr = shared_memory->GetPointer(addr);
60  Type0Command command;
61 
62  std::memcpy(&command, ptr, sizeof(Type0Command));
63  command.finished |= 1;
64  std::memcpy(ptr, &command, sizeof(Type0Command));
65 
66  rb.Push(RESULT_SUCCESS);
67  }
68 
69  LOG_WARNING(Service_CSND, "(STUBBED) called, addr=0x{:08X}", addr);
70 }

91 void CSND_SND::AcquireCapUnit(Kernel::HLERequestContext& ctx) {
92  IPC::RequestParser rp{ctx, 0x7, 0, 0};
93 
94  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
95  if (capture_units[0] && capture_units[1]) {
96  LOG_WARNING(Service_CSND, "No more capture units available");
97  rb.Push(ResultCode(ErrorDescription::InvalidResultValue, ErrorModule::CSND,
98  ErrorSummary::OutOfResource, ErrorLevel::Status));
99  rb.Skip(1, false);
100  return;
101  }
102  rb.Push(RESULT_SUCCESS);
103 
104  if (capture_units[0]) {
105  capture_units[1] = true;
106  rb.Push<u32>(1);
107  } else {
108  capture_units[0] = true;
109  rb.Push<u32>(0);
110  }
111 
112  LOG_WARNING(Service_CSND, "(STUBBED) called");
113 }
