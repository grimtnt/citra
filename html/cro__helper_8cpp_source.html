66  u32 addend, u32 symbol_address, u32 target_future_address) {
67 
68  switch (relocation_type) {
69  case RelocationType::Nothing:
70  break;
71  case RelocationType::AbsoluteAddress:
72  case RelocationType::AbsoluteAddress2:

81  case RelocationType::ArmBranch:
82  case RelocationType::ModifyArmBranch:
83  case RelocationType::AlignedRelativeAddress:
84  // TODO(wwylele): implement other types

86  break;
87  default:
88  return CROFormatError(0x22);

116 ResultCode CROHelper::ApplyRelocationBatch(VAddr batch, u32 symbol_address, bool reset) {
117  if (symbol_address == 0 && !reset)
118  return CROFormatError(0x10);
119 
120  VAddr relocation_address{batch};
121  while (true) {
122  RelocationEntry relocation;
123  Memory::ReadBlock(relocation_address, &relocation, sizeof(RelocationEntry));
124 
125  VAddr relocation_target = SegmentTagToAddress(relocation.target_position);
126  if (relocation_target == 0) {
127  return CROFormatError(0x12);
128  }
129 
130  ResultCode result = ApplyRelocation(relocation_target, relocation.type, relocation.addend,
131  symbol_address, relocation_target);
132  if (result.IsError()) {
133  LOG_ERROR(Service_LDR, "Error applying relocation {:08X}", result.raw);
134  return result;
135  }
136 
137  if (relocation.is_batch_end)
138  break;
139 
140  relocation_address += sizeof(RelocationEntry);
141  }
142 
143  RelocationEntry relocation;
144  Memory::ReadBlock(batch, &relocation, sizeof(RelocationEntry));
145  relocation.is_batch_resolved = reset ? 0 : 1;
146  Memory::WriteBlock(batch, &relocation, sizeof(RelocationEntry));
147  return RESULT_SUCCESS;
148 }

150 VAddr CROHelper::FindExportNamedSymbol(const std::string& name) const {
151  if (!GetField(ExportTreeNum))
152  return 0;
153 
154  std::size_t len{name.size()};
155  ExportTreeEntry entry;
156  GetEntry(0, entry);
157  ExportTreeEntry::Child next;
158  next.raw = entry.left.raw;
159  u32 found_id{};
160 
161  while (true) {
162  GetEntry(next.next_index, entry);
163 
164  if (next.is_end) {
165  found_id = entry.export_table_index;
166  break;
167  }
168 
169  u16 test_byte{static_cast<u16>(entry.test_bit >> 3)};
170  u16 test_bit_in_byte{static_cast<u16>(entry.test_bit & 7)};
171 
172  if (test_byte >= len) {
173  next.raw = entry.left.raw;
174  } else if ((name[test_byte] >> test_bit_in_byte) & 1) {
175  next.raw = entry.right.raw;
176  } else {
177  next.raw = entry.left.raw;
178  }
179  }
180 
181  u32 export_named_symbol_num = GetField(ExportNamedSymbolNum);
182 
183  if (found_id >= export_named_symbol_num)
184  return 0;
185 
186  u32 export_strings_size{GetField(ExportStringsSize)};
187  ExportNamedSymbolEntry symbol_entry;
188  GetEntry(found_id, symbol_entry);
189 
190  if (Memory::ReadCString(symbol_entry.name_offset, export_strings_size) != name)
191  return 0;
192 
193  return SegmentTagToAddress(symbol_entry.symbol_position);
194 }

196 ResultCode CROHelper::RebaseHeader(u32 cro_size) {
197  ResultCode error{CROFormatError(0x11)};
198 
199  // verifies magic
200  if (GetField(Magic) != MAGIC_CRO0)
201  return error;
202 
203  // verifies not registered
204  if (GetField(NextCRO) != 0 || GetField(PreviousCRO) != 0)
205  return error;
206 
207  // This seems to be a hard limit set by the RO module
208  if (GetField(FileSize) > 0x10000000 || GetField(BssSize) > 0x10000000)
209  return error;
210 
211  // verifies not fixed
212  if (GetField(FixedSize) != 0)
213  return error;
214 
215  if (GetField(CodeOffset) < CRO_HEADER_SIZE)
216  return error;
217 
218  // verifies that all offsets are in the correct order
219  constexpr std::array<HeaderField, 18> OFFSET_ORDER = {{
220  CodeOffset,
221  ModuleNameOffset,
222  SegmentTableOffset,
223  ExportNamedSymbolTableOffset,
224  ExportTreeTableOffset,
225  ExportIndexedSymbolTableOffset,
226  ExportStringsOffset,
227  ImportModuleTableOffset,
228  ExternalRelocationTableOffset,
229  ImportNamedSymbolTableOffset,
230  ImportIndexedSymbolTableOffset,
231  ImportAnonymousSymbolTableOffset,
232  ImportStringsOffset,
233  StaticAnonymousSymbolTableOffset,
234  InternalRelocationTableOffset,
235  StaticRelocationTableOffset,
236  DataOffset,
237  FileSize,
238  }};
239 
240  u32 prev_offset = GetField(OFFSET_ORDER[0]);
241  u32 cur_offset;
242  for (std::size_t i = 1; i < OFFSET_ORDER.size(); ++i) {
243  cur_offset = GetField(OFFSET_ORDER[i]);
244  if (cur_offset < prev_offset)
245  return error;
246  prev_offset = cur_offset;
247  }
248 
249  // rebases offsets
250  u32 offset = GetField(NameOffset);
251  if (offset != 0)
252  SetField(NameOffset, offset + module_address);
253 
254  for (int field = CodeOffset; field < Fix0Barrier; field += 2) {
255  HeaderField header_field = static_cast<HeaderField>(field);
256  offset = GetField(header_field);
257  if (offset != 0)
258  SetField(header_field, offset + module_address);
259  }
260 
261  // verifies everything is not beyond the buffer
262  u32 file_end = module_address + cro_size;
263  for (int field = CodeOffset, i{}; field < Fix0Barrier; field += 2, ++i) {
264  HeaderField offset_field = static_cast<HeaderField>(field);
265  HeaderField size_field = static_cast<HeaderField>(field + 1);
266  if (GetField(offset_field) + GetField(size_field) * ENTRY_SIZE[i] > file_end)
267  return error;
268  }
269 
270  return RESULT_SUCCESS;
271 }

274  u32 data_segment_size, VAddr bss_segment_address,

276

1238 ResultCode CROHelper::Link(VAddr crs_address, bool link_on_load_bug_fix) {
1239  ResultCode result = RESULT_SUCCESS;
1240 
1241  {
1242  VAddr data_segment_address;
1243  if (link_on_load_bug_fix) {
1244  // this is a bug fix introduced by 7.2.0-17's LoadCRO_New
1245  // The bug itself is:
1246  // If a relocation target is in .data segment, it will relocate to the
1247  // user-specified buffer. But if this is linking during loading,
1248  // the .data segment hasn't been transfer from CRO to the buffer,
1249  // thus the relocation will be overwritten by data transfer.
1250  // To fix this bug, we need temporarily restore the old .data segment
1251  // offset and apply imported symbols.
1252 
1253  // RO service seems assuming segment_index == segment_type,
1254  // so we do the same
1255  if (GetField(SegmentNum) >= 2) { // means we have .data segment
1256  SegmentEntry entry;
1257  GetEntry(2, entry);
1258  ASSERT(entry.type == SegmentType::Data);
1259  data_segment_address = entry.offset;
1260  entry.offset = GetField(DataOffset);
1261  SetEntry(2, entry);
1262  }
1263  }
1264  SCOPE_EXIT({
1265  // Restore the new .data segment address after importing
1266  if (link_on_load_bug_fix) {
1267  if (GetField(SegmentNum) >= 2) {
1268  SegmentEntry entry;
1269  GetEntry(2, entry);
1270  entry.offset = data_segment_address;
1271  SetEntry(2, entry);
1272  }
1273  }
1274  });
1275 
1276  // Imports named symbols from other modules
1277  result = ApplyImportNamedSymbol(crs_address);
1278  if (result.IsError()) {
1279  LOG_ERROR(Service_LDR, "Error applying symbol import {:08X}", result.raw);
1280  return result;
1281  }
1282 
1283  // Imports indexed and anonymous symbols from other modules
1284  result = ApplyModuleImport(crs_address);
1285  if (result.IsError()) {
1286  LOG_ERROR(Service_LDR, "Error applying module import {:08X}", result.raw);
1287  return result;
1288  }
1289  }
