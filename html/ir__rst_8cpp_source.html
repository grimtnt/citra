50 void IR_RST::UpdateCallback(u64 userdata, s64 cycles_late) {
51  SharedMem* mem{reinterpret_cast<SharedMem*>(shared_memory->GetPointer())};
52 
53  if (is_device_reload_pending.exchange(false))
54  LoadInputDevices();
55 
56  PadState state;
57  state.zl.Assign(zl_button->GetStatus());
58  state.zr.Assign(zr_button->GetStatus());
59 
60  // Get current c-stick position and update c-stick direction
61  float c_stick_x_f, c_stick_y_f;
62  std::tie(c_stick_x_f, c_stick_y_f) = c_stick->GetStatus();
63  constexpr int MAX_CSTICK_RADIUS = 0x9C; // Max value for a c-stick radius
64  s16 c_stick_x{static_cast<s16>(c_stick_x_f * MAX_CSTICK_RADIUS)};
65  s16 c_stick_y{static_cast<s16>(c_stick_y_f * MAX_CSTICK_RADIUS)};
66 
67  Core::Movie::GetInstance().HandleIrRst(state, c_stick_x, c_stick_y);
68 
69  if (!raw_c_stick) {
70  const HID::DirectionState direction{HID::GetStickDirectionState(c_stick_x, c_stick_y)};
71  state.c_stick_up.Assign(direction.up);
72  state.c_stick_down.Assign(direction.down);
73  state.c_stick_left.Assign(direction.left);
74  state.c_stick_right.Assign(direction.right);
75  }
76 
77  // TODO (wwylele): implement raw C-stick data for raw_c_stick = true
78 
79  const u32 last_entry_index{mem->index};
80  mem->index = next_pad_index;
81  next_pad_index = (next_pad_index + 1) % mem->entries.size();
82 
83  // Get the previous Pad state
84  PadState old_state{mem->entries[last_entry_index].current_state};
85 
86  // Compute bitmask with 1s for bits different from the old state
87  PadState changed{state.hex ^ old_state.hex};
88 
89  // Get the current Pad entry
90  PadDataEntry& pad_entry{mem->entries[mem->index]};
91 
92  // Update entry properties
93  pad_entry.current_state.hex = state.hex;
94  pad_entry.delta_additions.hex = changed.hex & state.hex;
95  pad_entry.delta_removals.hex = changed.hex & old_state.hex;
96  pad_entry.c_stick_x = c_stick_x;
97  pad_entry.c_stick_y = c_stick_y;
98 
99  // If we just updated index 0, provide a new timestamp
100  if (mem->index == 0) {
101  mem->index_reset_ticks_previous = mem->index_reset_ticks;
102  mem->index_reset_ticks = CoreTiming::GetTicks();
103  }
104 
105  update_event->Signal();
106 
107  // Reschedule recurrent event
108  CoreTiming::ScheduleEvent(msToCycles(update_period) - cycles_late, update_callback_id);
109 }

118 void IR_RST::Initialize(Kernel::HLERequestContext& ctx) {
119  IPC::RequestParser rp{ctx, 0x02, 2, 0};
120  update_period = static_cast<int>(rp.Pop<u32>());
121  raw_c_stick = rp.Pop<bool>();
122 
123  if (raw_c_stick)
124  LOG_ERROR(Service_IR, "raw C-stick data is not implemented!");
125 
126  next_pad_index = 0;
127  is_device_reload_pending.store(true);
128  CoreTiming::ScheduleEvent(msToCycles(update_period), update_callback_id);
129 
130  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
131  rb.Push(RESULT_SUCCESS);
132 
133  LOG_DEBUG(Service_IR, "called. update_period={}, raw_c_stick={}", update_period, raw_c_stick);
134 }
