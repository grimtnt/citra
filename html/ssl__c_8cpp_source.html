25 void SSL_C::GenerateRandomData(Kernel::HLERequestContext& ctx) {
26  IPC::RequestParser rp{ctx, 0x11, 1, 2};
27  u32 size{rp.Pop<u32>()};
28  auto buffer{rp.PopMappedBuffer()};
29 
30  // Fill the output buffer with random data.
31  u32 data{};
32  u32 i{};
33  while (i < size) {
34  if ((i % 4) == 0) {
35  // The random number generator returns 4 bytes worth of data, so generate new random
36  // data when i == 0 and when i is divisible by 4
37  data = rand_gen();
38  }
39 
40  if (size > 4) {
41  // Use up the entire 4 bytes of the random data for as long as possible
42  buffer.Write(&data, i, 4);
43  i += 4;
44  } else if (size == 2) {
45  buffer.Write(&data, i, 2);
46  i += 2;
47  } else {
48  buffer.Write(&data, i, 1);
49  i++;
50  }
51  }
52 
53  // Stub, return success
54  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 2)};
55  rb.Push(RESULT_SUCCESS);
56  rb.PushMappedBuffer(buffer);
57 }
