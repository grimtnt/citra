120 template <typename InIt>
121 bool ComparePartialString(InIt begin, InIt end, const char* other) {
122  for (; begin != end && *other != '\0'; ++begin, ++other) {
123  if (*begin != *other) {
124  return false;
125  }
126  }
127  // Only return true if both strings finished at the same point
128  return (begin == end) == (*other == '\0');
129 }
