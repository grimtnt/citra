44 ResultCode Applet::Create(Service::APT::AppletId id,
45  std::weak_ptr<Service::APT::AppletManager> manager) {
46  switch (id) {
47  case Service::APT::AppletId::SoftwareKeyboard1:
48  case Service::APT::AppletId::SoftwareKeyboard2:
49  applets[id] = std::make_shared<SoftwareKeyboard>(id, std::move(manager));
50  break;
51  case Service::APT::AppletId::Ed1:
52  case Service::APT::AppletId::Ed2:
53  applets[id] = std::make_shared<MiiSelector>(id, std::move(manager));
54  break;
55  case Service::APT::AppletId::Error:
56  case Service::APT::AppletId::Error2:
57  applets[id] = std::make_shared<ErrEula>(id, std::move(manager));
58  break;
59  case Service::APT::AppletId::Mint:
60  case Service::APT::AppletId::Mint2:
61  applets[id] = std::make_shared<Mint>(id, std::move(manager));
62  break;
63  default:
64  LOG_ERROR(Applet, "Could not create applet {}", static_cast<u32>(id));
65  // TODO(Subv): Find the right error code
66  return ResultCode(ErrorDescription::NotFound, ErrorModule::Applet,
67  ErrorSummary::NotSupported, ErrorLevel::Permanent);
68  }
69 
70  return RESULT_SUCCESS;
71 }

81 static void AppletUpdateEvent(u64 applet_id, s64 cycles_late) {
82  Service::APT::AppletId id{static_cast<Service::APT::AppletId>(applet_id)};
83  std::shared_ptr<Applet> applet{Applet::Get(id)};
84  ASSERT_MSG(applet != nullptr, "Applet doesn't exist! applet_id={:08X}", static_cast<u32>(id));
85 
86  applet->Update();
87 
88  // If the applet is still running after the last update, reschedule the event
89  if (applet->IsRunning()) {
90  CoreTiming::ScheduleEvent(usToCycles(applet_update_interval_us) - cycles_late,
91  applet_update_event, applet_id);
92  } else {
93  // Otherwise the applet has terminated, in which case we should clean it up
94  applets[id] = nullptr;
95  }
96 }
