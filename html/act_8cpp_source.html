45 void Module::Interface::GetAccountDataBlock(Kernel::HLERequestContext& ctx) {
46  IPC::RequestParser rp{ctx, 0x0006, 3, 2};
47  u8 unk{rp.Pop<u8>()};
48  u32 size{rp.Pop<u32>()};
49  BlkID id{rp.PopEnum<BlkID>()};
50  auto buffer{rp.PopMappedBuffer()};
51  switch (id) {
52  case BlkID::NNID: {
53  std::string nnid = Common::UTF16ToUTF8(CFG::GetCurrentModule()->GetUsername());
54  nnid.resize(0x11);
55  nnid = Common::ReplaceAll(nnid, " ", "_");
56  buffer.Write(nnid.c_str(), 0, nnid.length());
57  break;
58  }
59  case BlkID::Unknown6: {
60  u32 value = 1;
61  buffer.Write(&value, 0, sizeof(u32));
62  break;
63  }
64  case BlkID::U16MiiName: {
65  std::u16string username = CFG::GetCurrentModule()->GetUsername();
66  buffer.Write(username.c_str(), 0, username.length());
67  break;
68  }
69  case BlkID::PrincipalID: {
70  u32 principal_id = 0xDEADBEEF;
71  buffer.Write(&principal_id, 0, sizeof(u32));
72  break;
73  }
74  case BlkID::CountryName: {
75  u8 country_code = CFG::GetCurrentModule()->GetCountryCode();
76  u16 country_name = CFG::country_codes[country_code];
77  buffer.Write(&country_name, 0, sizeof(u16));
78  break;
79  }
80  case BlkID::Age: {
81  u16 age = 0;
82  buffer.Write(&age, 0, sizeof(u16));
83  break;
84  }
85  case BlkID::Birthday: {
86  Birthday birthday = {};
87  buffer.Write(&birthday, 0, sizeof(Birthday));
88  break;
89  }
90  case BlkID::InfoStruct: {
91  InfoBlock info = {};
92  std::u16string username = CFG::GetCurrentModule()->GetUsername();
93  username.copy(info.MachinUserName, username.length());
94  buffer.Write(&info, 0, username.length());
95  break;
96  }
97  default: {
98  UNIMPLEMENTED();
99  break;
100  }
101  }
102 
103  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 2)};
104  rb.Push(RESULT_SUCCESS);
105  rb.PushMappedBuffer(buffer);
106 
107  LOG_WARNING(Service_ACT, "(STUBBED) called, unk=0x{:02X}, size=0x{:X}, id=0x{:X}", unk, size,
108  static_cast<u32>(id));
109 }
