62 u32 DiskDirectory::Read(const u32 count, Entry* entries) {
63  u32 entries_read{};
64 
65  while (entries_read < count && children_iterator != directory.children.cend()) {
66  const FileUtil::FSTEntry& file{*children_iterator};
67  const std::string& filename{file.virtualName};
68  Entry& entry{entries[entries_read]};
69 
70  LOG_TRACE(Service_FS, "File {}: size={} dir={}", filename, file.size, file.isDirectory);
71 
72  // TODO(Link Mauve): use a proper conversion to UTF-16.
73  for (std::size_t j{}; j < FILENAME_LENGTH; ++j) {
74  entry.filename[j] = filename[j];
75  if (!filename[j])
76  break;
77  }
78 
79  FileUtil::SplitFilename83(filename, entry.short_name, entry.extension);
80 
81  entry.is_directory = file.isDirectory;
82  entry.is_hidden = (filename[0] == '.');
83  entry.is_read_only = 0;
84  entry.file_size = file.size;
85 
86  // We emulate a SD card where the archive bit has never been cleared, as it would be on
87  // most user SD cards.
88  // Some homebrews (blargSNES for instance) are known to mistakenly use the archive bit as a
89  // file bit.
90  entry.is_archive = !file.isDirectory;
91 
92  ++entries_read;
93  ++children_iterator;
94  }
95  return entries_read;
