31 void Mixers::ParseConfig(DspConfiguration& config) {
32  if (!config.dirty_raw) {
33  return;
34  }
35 
36  if (config.mixer1_enabled_dirty) {
37  config.mixer1_enabled_dirty.Assign(0);
38  state.mixer1_enabled = config.mixer1_enabled != 0;
39  LOG_TRACE(Audio_DSP, "mixers mixer1_enabled = {}", config.mixer1_enabled);
40  }
41 
42  if (config.mixer2_enabled_dirty) {
43  config.mixer2_enabled_dirty.Assign(0);
44  state.mixer2_enabled = config.mixer2_enabled != 0;
45  LOG_TRACE(Audio_DSP, "mixers mixer2_enabled = {}", config.mixer2_enabled);
46  }
47 
48  if (config.volume_0_dirty) {
49  config.volume_0_dirty.Assign(0);
50  state.intermediate_mixer_volume[0] = config.volume[0];
51  LOG_TRACE(Audio_DSP, "mixers volume[0] = {}", config.volume[0]);
52  }
53 
54  if (config.volume_1_dirty) {
55  config.volume_1_dirty.Assign(0);
56  state.intermediate_mixer_volume[1] = config.volume[1];
57  LOG_TRACE(Audio_DSP, "mixers volume[1] = {}", config.volume[1]);
58  }
59 
60  if (config.volume_2_dirty) {
61  config.volume_2_dirty.Assign(0);
62  state.intermediate_mixer_volume[2] = config.volume[2];
63  LOG_TRACE(Audio_DSP, "mixers volume[2] = {}", config.volume[2]);
64  }
65 
66  if (config.output_format_dirty) {
67  config.output_format_dirty.Assign(0);
68  state.output_format = config.output_format;
69  LOG_TRACE(Audio_DSP, "mixers output_format = {}",
70  static_cast<std::size_t>(config.output_format));
71  }
72 
73  if (config.headphones_connected_dirty) {
74  config.headphones_connected_dirty.Assign(0);
75  // Do nothing. (Note: Whether headphones are connected does affect coefficients used for
76  // surround sound.)
77  LOG_TRACE(Audio_DSP, "mixers headphones_connected={}", config.headphones_connected);
78  }
79 
80  if (config.dirty_raw) {
81  LOG_DEBUG(Audio_DSP, "mixers remaining_dirty={:x}", config.dirty_raw);
82  }
83 
84  config.dirty_raw = 0;
85 }

97 void Mixers::DownmixAndMixIntoCurrentFrame(float gain, const QuadFrame32& samples) {
98  // TODO(merry): Limiter. (Currently we're performing final mixing assuming a disabled limiter.)
99 
100  switch (state.output_format) {
101  case OutputFormat::Mono:
102  std::transform(
103  current_frame.begin(), current_frame.end(), samples.begin(), current_frame.begin(),
104  [gain](const std::array<s16, 2>& accumulator,
105  const std::array<s32, 4>& sample) -> std::array<s16, 2> {
106  // Downmix to mono
107  s16 mono{ClampToS16(static_cast<s32>(
108  (gain * sample[0] + gain * sample[1] + gain * sample[2] + gain * sample[3]) /
109  2))};
110  // Mix into current frame
111  return AddAndClampToS16(accumulator, {mono, mono});
112  });
113  return;
114 
115  case OutputFormat::Surround:
116  // TODO(merry): Implement surround sound.
117  // fallthrough
118 
119  case OutputFormat::Stereo:
120  std::transform(
121  current_frame.begin(), current_frame.end(), samples.begin(), current_frame.begin(),
122  [gain](const std::array<s16, 2>& accumulator,
123  const std::array<s32, 4>& sample) -> std::array<s16, 2> {
124  // Downmix to stereo
125  s16 left{ClampToS16(static_cast<s32>(gain * sample[0] + gain * sample[2]))};
126  s16 right{ClampToS16(static_cast<s32>(gain * sample[1] + gain * sample[3]))};
127  // Mix into current frame
128  return AddAndClampToS16(accumulator, {left, right});
129  });
130  return;
131  }
132 
133  UNREACHABLE_MSG("Invalid output_format {}", static_cast<std::size_t>(state.output_format));
134 }

136 void Mixers::AuxReturn(const IntermediateMixSamples& read_samples) {
137  // NOTE: read_samples.mix{1,2}.pcm32 annoyingly have their dimensions in reverse order to
138  // QuadFrame32.
139 
140  if (state.mixer1_enabled) {
141  for (std::size_t sample{}; sample < samples_per_frame; sample++) {
142  for (std::size_t channel{}; channel < 4; channel++) {
143  state.intermediate_mix_buffer[1][sample][channel] =
144  read_samples.mix1.pcm32[channel][sample];
145  }
146  }
147  }
148 
149  if (state.mixer2_enabled) {
150  for (std::size_t sample{}; sample < samples_per_frame; sample++) {
151  for (std::size_t channel{}; channel < 4; channel++) {
152  state.intermediate_mix_buffer[2][sample][channel] =
153  read_samples.mix2.pcm32[channel][sample];
154  }
155  }
156  }
157 }

159 void Mixers::AuxSend(IntermediateMixSamples& write_samples,
160  const std::array<QuadFrame32, 3>& input) {
161  // NOTE: read_samples.mix{1,2}.pcm32 annoyingly have their dimensions in reverse order to
162  // QuadFrame32.
163 
164  state.intermediate_mix_buffer[0] = input[0];
165 
166  if (state.mixer1_enabled) {
167  for (std::size_t sample{}; sample < samples_per_frame; sample++) {
168  for (std::size_t channel{}; channel < 4; channel++) {
169  write_samples.mix1.pcm32[channel][sample] = input[1][sample][channel];
170  }
171  }
172  } else {
173  state.intermediate_mix_buffer[1] = input[1];
174  }
175 
176  if (state.mixer2_enabled) {
177  for (std::size_t sample{}; sample < samples_per_frame; sample++) {
178  for (std::size_t channel{}; channel < 4; channel++) {
179  write_samples.mix2.pcm32[channel][sample] = input[2][sample][channel];
180  }
181  }
182  } else {
183  state.intermediate_mix_buffer[2] = input[2];
184  }
185 }
