17 ResultCode TranslateCommandBuffer(SharedPtr<Thread> src_thread, SharedPtr<Thread> dst_thread,
18  VAddr src_address, VAddr dst_address, bool reply) {
19 
20  auto& src_process = src_thread->owner_process;
21  auto& dst_process = dst_thread->owner_process;
22 
23  IPC::Header header;
24  // TODO(Subv): Replace by Memory::Read32 when possible.
25  Memory::ReadBlock(*src_process, src_address, &header.raw, sizeof(header.raw));
26 
27  std::size_t untranslated_size = 1u + header.normal_params_size;
28  std::size_t command_size = untranslated_size + header.translate_params_size;
29 
30  // Note: The real kernel does not check that the command length fits into the IPC buffer area.
31  ASSERT(command_size <= IPC::COMMAND_BUFFER_LENGTH);
32 
33  std::array<u32, IPC::COMMAND_BUFFER_LENGTH> cmd_buf;
34  Memory::ReadBlock(*src_process, src_address, cmd_buf.data(), command_size * sizeof(u32));
35 
36  std::size_t i{untranslated_size};
37  while (i < command_size) {
38  u32 descriptor = cmd_buf[i];
39  i += 1;
40 
41  switch (IPC::GetDescriptorType(descriptor)) {
42  case IPC::DescriptorType::CopyHandle:
43  case IPC::DescriptorType::MoveHandle: {
44  u32 num_handles = IPC::HandleNumberFromDesc(descriptor);
45  // Note: The real kernel does not check that the number of handles fits into the command
46  // buffer before writing them, only after finishing.
47  if (i + num_handles > command_size) {
48  return ResultCode(ErrCodes::CommandTooLarge, ErrorModule::OS,
49  ErrorSummary::InvalidState, ErrorLevel::Status);
50  }
51 
52  for (u32 j{}; j < num_handles; ++j) {
53  Handle handle = cmd_buf[i];
54  SharedPtr<Object> object{}; // Perform pseudo-handle detection here because by the
55  // time this function is called, the current thread and
56  // process are no longer the ones which created this IPC
57  // request, but the ones that are handling it.
58  if (handle == CurrentThread) {
59  object = src_thread;
60  } else if (handle == CurrentProcess) {
61  object = src_process;
62  } else if (handle != 0) {
63  object = g_handle_table.GetGeneric(handle);
64  if (descriptor == IPC::DescriptorType::MoveHandle) {
65  g_handle_table.Close(handle);
66  }
67  }
68 
69  if (object == nullptr) {
70  // Note: The real kernel sets invalid translated handles to 0 in the target
71  // command buffer.
72  cmd_buf[i++] = 0;
73  continue;
74  }
75 
76  auto result = g_handle_table.Create(std::move(object));
77  cmd_buf[i++] = result.ValueOr(0);
78  }
79  break;
80  }
81  case IPC::DescriptorType::CallingPid: {
82  cmd_buf[i++] = src_process->process_id;
83  break;
84  }
85  case IPC::DescriptorType::StaticBuffer: {
86  IPC::StaticBufferDescInfo bufferInfo{descriptor};
87  VAddr static_buffer_src_address = cmd_buf[i];
88 
89  std::vector<u8> data(bufferInfo.size);
90  Memory::ReadBlock(*src_process, static_buffer_src_address, data.data(), data.size());
91 
92  // Grab the address that the target thread set up to receive the response static buffer
93  // and write our data there. The static buffers area is located right after the command
94  // buffer area.
95  struct StaticBuffer {
96  IPC::StaticBufferDescInfo descriptor;
97  VAddr address;
98  };
99 
100  static_assert(sizeof(StaticBuffer) == 8, "StaticBuffer struct has incorrect size.");
101 
102  StaticBuffer target_buffer;
103 
104  u32 static_buffer_offset = IPC::COMMAND_BUFFER_LENGTH * sizeof(u32) +
105  sizeof(StaticBuffer) * bufferInfo.buffer_id;
106  Memory::ReadBlock(*dst_process, dst_address + static_buffer_offset, &target_buffer,
107  sizeof(target_buffer));
108 
109  Memory::WriteBlock(*dst_process, target_buffer.address, data.data(), data.size());
110 
111  cmd_buf[i++] = target_buffer.address;
112  break;
113  }
114  case IPC::DescriptorType::MappedBuffer: {
115  IPC::MappedBufferDescInfo descInfo{descriptor};
116  VAddr source_address = cmd_buf[i];
117 
118  u32 size = static_cast<u32>(descInfo.size);
119  IPC::MappedBufferPermissions permissions = descInfo.perms;
120 
121  VAddr page_start = Common::AlignDown(source_address, Memory::PAGE_SIZE);
122  u32 page_offset = source_address - page_start;
123  u32 num_pages =
124  Common::AlignUp(page_offset + size, Memory::PAGE_SIZE) >> Memory::PAGE_BITS;
125 
126  ASSERT(num_pages >= 1);
127 
128  if (reply) {
129  // TODO(Subv): Scan the target's command buffer to make sure that there was a
130  // MappedBuffer descriptor in the original request. The real kernel panics if you
131  // try to reply with an unsolicited MappedBuffer.
132 
133  // Unmap the buffers. Readonly buffers do not need to be copied over to the target
134  // process again because they were (presumably) not modified. This behavior is
135  // consistent with the real kernel.
136  if (permissions == IPC::MappedBufferPermissions::R) {
137  ResultCode result = src_process->vm_manager.UnmapRange(
138  page_start, num_pages * Memory::PAGE_SIZE);
139  ASSERT(result == RESULT_SUCCESS);
140  }
141 
142  ASSERT_MSG(permissions == IPC::MappedBufferPermissions::R,
143  "Unmapping Write MappedBuffers is unimplemented");
144  i += 1;
145  break;
146  }
147 
148  VAddr target_address{};
149 
150  auto IsPageAligned = [](VAddr address) -> bool {
151  return (address & Memory::PAGE_MASK) == 0;
152  };
153 
154  // TODO(Subv): Support more than 1 page and aligned page mappings
155  ASSERT_MSG(
156  num_pages == 1 &&
157  (!IsPageAligned(source_address) || !IsPageAligned(source_address + size)),
158  "MappedBuffers of more than one page or aligned transfers are not implemented");
159 
160  // TODO(Subv): Perform permission checks.
161 
162  // TODO(Subv): Leave a page of Reserved memory before the first page and after the last
163  // page.
164 
165  if (!IsPageAligned(source_address) ||
166  (num_pages == 1 && !IsPageAligned(source_address + size))) {
167  // If the address of the source buffer is not page-aligned or if the buffer doesn't
168  // fill an entire page, then we have to allocate a page of memory in the target
169  // process and copy over the data from the input buffer. This allocated buffer will
170  // be copied back to the source process and deallocated when the server replies to
171  // the request via ReplyAndReceive.
172 
173  auto buffer = std::make_shared<std::vector<u8>>(Memory::PAGE_SIZE);
174 
175  // Number of bytes until the next page.
176  std::size_t difference_to_page =
177  Common::AlignUp(source_address, Memory::PAGE_SIZE) - source_address;
178  // If the data fits in one page we can just copy the required size instead of the
179  // entire page.
180  std::size_t read_size = num_pages == 1 ? static_cast<std::size_t>(size) : difference_to_page;
181 
182  Memory::ReadBlock(*src_process, source_address, buffer->data() + page_offset,
183  read_size);
184 
185  // Map the page into the target process' address space.
186  target_address =
187  dst_process->vm_manager
188  .MapMemoryBlockToBase(Memory::IPC_MAPPING_VADDR, Memory::IPC_MAPPING_SIZE,
189  buffer, 0, static_cast<u32>(buffer->size()),
190  Kernel::MemoryState::Shared)
191  .Unwrap();
192  }
193 
194  cmd_buf[i++] = target_address + page_offset;
195  break;
196  }
197  default:
198  UNIMPLEMENTED_MSG("Unsupported handle translation: {:#010X}", descriptor);
199  }
200  }
201 
202  Memory::WriteBlock(*dst_process, dst_address, cmd_buf.data(), command_size * sizeof(u32));
203 
204  return RESULT_SUCCESS;
205 }
206 } // namespace Kernel
