151 void ERR_F::ThrowFatalError(Kernel::HLERequestContext& ctx) {
152  IPC::RequestParser rp{ctx, 1, 32, 0};
153 
154  LOG_CRITICAL(Service_ERR, "Fatal error");
155  const ErrInfo errinfo{rp.PopRaw<ErrInfo>()};
156  LOG_CRITICAL(Service_ERR, "Fatal error type: {}", GetErrType(errinfo.errinfo_common.specifier));
157  Core::System::GetInstance().SetStatus(Core::System::ResultStatus::FatalError);
158 
159  // Generic Info
160  LogGenericInfo(errinfo.errinfo_common);
161 
162  switch (static_cast<FatalErrType>(errinfo.errinfo_common.specifier)) {
163  case FatalErrType::Generic:
164  case FatalErrType::Corrupted:
165  case FatalErrType::CardRemoved:
166  case FatalErrType::Logged: {
167  LOG_CRITICAL(Service_ERR, "Datetime: {}", GetCurrentSystemTime());
168  break;
169  }
170  case FatalErrType::Exception: {
171  const auto& errtype{errinfo.exception};
172 
173  // Register Info
174  LOG_CRITICAL(Service_ERR, "ARM Registers:");
175  for (u32 index{}; index < errtype.exception_data.exception_context.arm_regs.size();
176  ++index) {
177  if (index < 13) {
178  LOG_DEBUG(Service_ERR, "r{}=0x{:08X}", index,
179  errtype.exception_data.exception_context.arm_regs.at(index));
180  } else if (index == 13) {
181  LOG_CRITICAL(Service_ERR, "SP=0x{:08X}",
182  errtype.exception_data.exception_context.arm_regs.at(index));
183  } else if (index == 14) {
184  LOG_CRITICAL(Service_ERR, "LR=0x{:08X}",
185  errtype.exception_data.exception_context.arm_regs.at(index));
186  } else if (index == 15) {
187  LOG_CRITICAL(Service_ERR, "PC=0x{:08X}",
188  errtype.exception_data.exception_context.arm_regs.at(index));
189  }
190  }
191  LOG_CRITICAL(Service_ERR, "CPSR=0x{:08X}", errtype.exception_data.exception_context.cpsr);
192 
193  // Exception Info
194  LOG_CRITICAL(
195  Service_ERR, "EXCEPTION TYPE: {}",
196  GetExceptionType(errtype.exception_data.exception_info.exception_type).c_str());
197  switch (static_cast<ExceptionType>(errtype.exception_data.exception_info.exception_type)) {
198  case ExceptionType::PrefetchAbort:
199  LOG_CRITICAL(Service_ERR, "IFSR: 0x{:08X}", errtype.exception_data.exception_info.sr);
200  LOG_CRITICAL(Service_ERR, "r15: 0x{:08X}", errtype.exception_data.exception_info.ar);
201  break;
202  case ExceptionType::DataAbort:
203  LOG_CRITICAL(Service_ERR, "DFSR: 0x{:08X}", errtype.exception_data.exception_info.sr);
204  LOG_CRITICAL(Service_ERR, "DFAR: 0x{:08X}", errtype.exception_data.exception_info.ar);
205  break;
206  case ExceptionType::VectorFP:
207  LOG_CRITICAL(Service_ERR, "FPEXC: 0x{:08X}",
208  errtype.exception_data.exception_info.fpinst);
209  LOG_CRITICAL(Service_ERR, "FINST: 0x{:08X}",
210  errtype.exception_data.exception_info.fpinst);
211  LOG_CRITICAL(Service_ERR, "FINST2: 0x{:08X}",
212  errtype.exception_data.exception_info.fpinst2);
213  break;
214  case ExceptionType::Undefined:
215  break; // Not logging exception_info for this case
216  }
217  LOG_CRITICAL(Service_ERR, "Datetime: {}", GetCurrentSystemTime());
218  break;
219  }
220 
221  case FatalErrType::ResultFailure: {
222  const auto& errtype{errinfo.result_failure};
223 
224  // Failure Message
225  LOG_CRITICAL(Service_ERR, "Failure Message: {}", errtype.message);
226  LOG_CRITICAL(Service_ERR, "Datetime: {}", GetCurrentSystemTime());
227  break;
228  }
229  }
230 
231  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
232  rb.Push(RESULT_SUCCESS);
233 }
