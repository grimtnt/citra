178 void JitShader::Compile_SwizzleSrc(Instruction instr, unsigned src_num, SourceRegister src_reg,
179  Xmm dest) {
180  Reg64 src_ptr;
181  std::size_t src_offset;
182 
183  if (src_reg.GetRegisterType() == RegisterType::FloatUniform) {
184  src_ptr = UNIFORMS;
185  src_offset = Uniforms::GetFloatUniformOffset(src_reg.GetIndex());
186  } else {
187  src_ptr = STATE;
188  src_offset = UnitState::InputOffset(src_reg);
189  }
190 
191  int src_offset_disp = (int)src_offset;
192  ASSERT_MSG(src_offset == src_offset_disp, "Source register offset too large for int type");
193 
194  unsigned operand_desc_id;
195 
196  const bool is_inverted =
197  (0 != (instr.opcode.Value().GetInfo().subtype & OpCode::Info::SrcInversed));
198 
199  unsigned address_register_index;
200  unsigned offset_src;
201 
202  if (instr.opcode.Value().EffectiveOpCode() == OpCode::Id::MAD ||
203  instr.opcode.Value().EffectiveOpCode() == OpCode::Id::MADI) {
204  operand_desc_id = instr.mad.operand_desc_id;
205  offset_src = is_inverted ? 3 : 2;
206  address_register_index = instr.mad.address_register_index;
207  } else {
208  operand_desc_id = instr.common.operand_desc_id;
209  offset_src = is_inverted ? 2 : 1;
210  address_register_index = instr.common.address_register_index;
211  }
212 
213  if (src_num == offset_src && address_register_index != 0) {
214  switch (address_register_index) {
215  case 1: // address offset 1
216  movaps(dest, xword[src_ptr + ADDROFFS_REG_0 + src_offset_disp]);
217  break;
218  case 2: // address offset 2
219  movaps(dest, xword[src_ptr + ADDROFFS_REG_1 + src_offset_disp]);
220  break;
221  case 3: // address offset 3
222  movaps(dest, xword[src_ptr + LOOPCOUNT_REG.cvt64() + src_offset_disp]);
223  break;
224  default:
225  UNREACHABLE();
226  break;
227  }
228  } else {
229  // Load the source
230  movaps(dest, xword[src_ptr + src_offset_disp]);
231  }
232 
233  SwizzlePattern swiz = {(*swizzle_data)[operand_desc_id]};
234 
235  // Generate instructions for source register swizzling as needed
236  u8 sel = swiz.GetRawSelector(src_num);
237  if (sel != NO_SRC_REG_SWIZZLE) {
238  // Selector component order needs to be reversed for the SHUFPS instruction
239  sel = ((sel & 0xc0) >> 6) | ((sel & 3) << 6) | ((sel & 0xc) << 2) | ((sel & 0x30) >> 2);
240 
241  // Shuffle inputs for swizzle
242  shufps(dest, dest, sel);
243  }
244 
245  // If the source register should be negated, flip the negative bit using XOR
246  const bool negate[] = {swiz.negate_src1, swiz.negate_src2, swiz.negate_src3};
247  if (negate[src_num - 1]) {
248  xorps(dest, NEGBIT);
249  }
250 }

252 void JitShader::Compile_DestEnable(Instruction instr, Xmm src) {
253  DestRegister dest;
254  unsigned operand_desc_id;
255  if (instr.opcode.Value().EffectiveOpCode() == OpCode::Id::MAD ||
256  instr.opcode.Value().EffectiveOpCode() == OpCode::Id::MADI) {
257  operand_desc_id = instr.mad.operand_desc_id;
258  dest = instr.mad.dest.Value();
259  } else {
260  operand_desc_id = instr.common.operand_desc_id;
261  dest = instr.common.dest.Value();
262  }
263 
264  SwizzlePattern swiz = {(*swizzle_data)[operand_desc_id]};
265 
266  std::size_t dest_offset_disp = UnitState::OutputOffset(dest);
267 
268  // If all components are enabled, write the result to the destination register
269  if (swiz.dest_mask == NO_DEST_REG_MASK) {
270  // Store dest back to memory
271  movaps(xword[STATE + dest_offset_disp], src);
272 
273  } else {
274  // Not all components are enabled, so mask the result when storing to the destination
275  // register...
276  movaps(SCRATCH, xword[STATE + dest_offset_disp]);
277 
278  u8 mask = ((swiz.dest_mask & 1) << 3) | ((swiz.dest_mask & 8) >> 3) |
279  ((swiz.dest_mask & 2) << 1) | ((swiz.dest_mask & 4) >> 1);
280  blendps(SCRATCH, src, mask);
281 
282  // Store dest back to memory
283  movaps(xword[STATE + dest_offset_disp], SCRATCH);
284  }
285 }

287 void JitShader::Compile_SanitizedMul(Xmm src1, Xmm src2, Xmm scratch) {
288  // 0 * inf and inf * 0 in the PICA should return 0 instead of NaN. This can be implemented by
289  // checking for NaNs before and after the multiplication. If the multiplication result is NaN
290  // where neither source was, this NaN was generated by a 0 * inf multiplication, and so the
291  // result should be transformed to 0 to match PICA fp rules.
292 
293  // Set scratch to mask of (src1 != NaN and src2 != NaN)
294  movaps(scratch, src1);
295  cmpordps(scratch, src2);
296 
297  mulps(src1, src2);
298 
299  // Set src2 to mask of (result == NaN)
300  movaps(src2, src1);
301  cmpunordps(src2, src2);
302 
303  // Clear components where scratch != src2 (i.e. if result is NaN where neither source was NaN)
304  xorps(scratch, src2);
305  andps(src1, scratch);
306 }

308 void JitShader::Compile_EvaluateCondition(Instruction instr) {
309  // Note: NXOR is used below to check for equality
310  switch (instr.flow_control.op) {
311  case Instruction::FlowControlType::Or:
312  mov(eax, COND0);
313  mov(ebx, COND1);
314  xor_(eax, (instr.flow_control.refx.Value() ^ 1));
315  xor_(ebx, (instr.flow_control.refy.Value() ^ 1));
316  or_(eax, ebx);
317  break;
318 
319  case Instruction::FlowControlType::And:
320  mov(eax, COND0);
321  mov(ebx, COND1);
322  xor_(eax, (instr.flow_control.refx.Value() ^ 1));
323  xor_(ebx, (instr.flow_control.refy.Value() ^ 1));
324  and_(eax, ebx);
325  break;
326 
327  case Instruction::FlowControlType::JustX:
328  mov(eax, COND0);
329  xor_(eax, (instr.flow_control.refx.Value() ^ 1));
330  break;
331 
332  case Instruction::FlowControlType::JustY:
333  mov(eax, COND1);
334  xor_(eax, (instr.flow_control.refy.Value() ^ 1));
335  break;
336  }
337 }

479 void JitShader::Compile_MOVA(Instruction instr) {
480  SwizzlePattern swiz = {(*swizzle_data)[instr.common.operand_desc_id]};
481 
482  if (!swiz.DestComponentEnabled(0) && !swiz.DestComponentEnabled(1)) {
483  return; // NoOp
484  }
485 
486  Compile_SwizzleSrc(instr, 1, instr.common.src1, SRC1);
487 
488  // Convert floats to integers using truncation (only care about X and Y components)
489  cvttps2dq(SRC1, SRC1);
490 
491  // Get result
492  movq(rax, SRC1);
493 
494  // Handle destination enable
495  if (swiz.DestComponentEnabled(0) && swiz.DestComponentEnabled(1)) {
496  // Move and sign-extend low 32 bits
497  movsxd(ADDROFFS_REG_0, eax);
498 
499  // Move and sign-extend high 32 bits
500  shr(rax, 32);
501  movsxd(ADDROFFS_REG_1, eax);
502 
503  // Multiply by 16 to be used as an offset later
504  shl(ADDROFFS_REG_0, 4);
505  shl(ADDROFFS_REG_1, 4);
506  } else {
507  if (swiz.DestComponentEnabled(0)) {
508  // Move and sign-extend low 32 bits
509  movsxd(ADDROFFS_REG_0, eax);
510 
511  // Multiply by 16 to be used as an offset later
512  shl(ADDROFFS_REG_0, 4);
513  } else if (swiz.DestComponentEnabled(1)) {
514  // Move and sign-extend high 32 bits
515  shr(rax, 32);
516  movsxd(ADDROFFS_REG_1, eax);
517 
518  // Multiply by 16 to be used as an offset later
519  shl(ADDROFFS_REG_1, 4);
520  }
521  }
522 }

553 void JitShader::Compile_END(Instruction instr) {
554  // Save conditional code
555  mov(byte[STATE + offsetof(UnitState, conditional_code[0])], COND0.cvt8());
556  mov(byte[STATE + offsetof(UnitState, conditional_code[1])], COND1.cvt8());
557 
558  // Save address/loop registers
559  sar(ADDROFFS_REG_0, 4);
560  sar(ADDROFFS_REG_1, 4);
561  sar(LOOPCOUNT_REG, 4);
562  mov(dword[STATE + offsetof(UnitState, address_registers[0])], ADDROFFS_REG_0.cvt32());
563  mov(dword[STATE + offsetof(UnitState, address_registers[1])], ADDROFFS_REG_1.cvt32());
564  mov(dword[STATE + offsetof(UnitState, address_registers[2])], LOOPCOUNT_REG);
565 
566  ABI_PopRegistersAndAdjustStack(*this, ABI_ALL_CALLEE_SAVED, 8, 16);
567  ret();
568 }

605 void JitShader::Compile_CMP(Instruction instr) {
606  using Op = Instruction::Common::CompareOpType::Op;
607  Op op_x = instr.common.compare_op.x;
608  Op op_y = instr.common.compare_op.y;
609 
610  Compile_SwizzleSrc(instr, 1, instr.common.src1, SRC1);
611  Compile_SwizzleSrc(instr, 2, instr.common.src2, SRC2);
612 
613  // SSE doesn't have greater-than (GT) or greater-equal (GE) comparison operators. You need to
614  // emulate them by swapping the lhs and rhs and using LT and LE. NLT and NLE can't be used here
615  // because they don't match when used with NaNs.
616  static const u8 cmp[] = {CMP_EQ, CMP_NEQ, CMP_LT, CMP_LE, CMP_LT, CMP_LE};
617 
618  bool invert_op_x = (op_x == Op::GreaterThan || op_x == Op::GreaterEqual);
619  Xmm lhs_x = invert_op_x ? SRC2 : SRC1;
620  Xmm rhs_x = invert_op_x ? SRC1 : SRC2;
621 
622  if (op_x == op_y) {
623  // Compare X-component and Y-component together
624  cmpps(lhs_x, rhs_x, cmp[op_x]);
625  movq(COND0, lhs_x);
626 
627  mov(COND1, COND0);
628  } else {
629  bool invert_op_y = (op_y == Op::GreaterThan || op_y == Op::GreaterEqual);
630  Xmm lhs_y = invert_op_y ? SRC2 : SRC1;
631  Xmm rhs_y = invert_op_y ? SRC1 : SRC2;
632 
633  // Compare X-component
634  movaps(SCRATCH, lhs_x);
635  cmpss(SCRATCH, rhs_x, cmp[op_x]);
636 
637  // Compare Y-component
638  cmpps(lhs_y, rhs_y, cmp[op_y]);
639 
640  movq(COND0, SCRATCH);
641  movq(COND1, lhs_y);
642  }
643 
644  shr(COND0.cvt32(), 31); // ignores upper 32 bits in source
645  shr(COND1, 63);
646 }

665 void JitShader::Compile_IF(Instruction instr) {
666  Compile_Assert(instr.flow_control.dest_offset >= program_counter,
667  "Backwards if-statements not supported");
668  Label l_else, l_endif;
669 
670  // Evaluate the "IF" condition
671  if (instr.opcode.Value() == OpCode::Id::IFU) {
672  Compile_UniformCondition(instr);
673  } else if (instr.opcode.Value() == OpCode::Id::IFC) {
674  Compile_EvaluateCondition(instr);
675  }
676  jz(l_else, T_NEAR);
677 
678  // Compile the code that corresponds to the condition evaluating as true
679  Compile_Block(instr.flow_control.dest_offset);
680 
681  // If there isn't an "ELSE" condition, we are done here
682  if (instr.flow_control.num_instructions == 0) {
683  L(l_else);
684  return;
685  }
686 
687  jmp(l_endif, T_NEAR);
688 
689  L(l_else);
690  // This code corresponds to the "ELSE" condition
691  // Comple the code that corresponds to the condition evaluating as false
692  Compile_Block(instr.flow_control.dest_offset + instr.flow_control.num_instructions);
693 
694  L(l_endif);
695 }

697 void JitShader::Compile_LOOP(Instruction instr) {
698  Compile_Assert(instr.flow_control.dest_offset >= program_counter,
699  "Backwards loops not supported");
700  Compile_Assert(!looping, "Nested loops not supported");
701 
702  looping = true;
703 
704  // This decodes the fields from the integer uniform at index instr.flow_control.int_uniform_id.
705  // The Y (LOOPCOUNT_REG) and Z (LOOPINC) component are kept multiplied by 16 (Left shifted by
706  // 4 bits) to be used as an offset into the 16-byte vector registers later
707  std::size_t offset = Uniforms::GetIntUniformOffset(instr.flow_control.int_uniform_id);
708  mov(LOOPCOUNT, dword[UNIFORMS + offset]);
709  mov(LOOPCOUNT_REG, LOOPCOUNT);
710  shr(LOOPCOUNT_REG, 4);
711  and_(LOOPCOUNT_REG, 0xFF0); // Y-component is the start
712  mov(LOOPINC, LOOPCOUNT);
713  shr(LOOPINC, 12);
714  and_(LOOPINC, 0xFF0); // Z-component is the incrementer
715  movzx(LOOPCOUNT, LOOPCOUNT.cvt8()); // X-component is iteration count
716  add(LOOPCOUNT, 1); // Iteration count is X-component + 1
717 
718  Label l_loop_start;
719  L(l_loop_start);
720 
721  loop_break_label = Xbyak::Label();
722  Compile_Block(instr.flow_control.dest_offset + 1);
723 
724  add(LOOPCOUNT_REG, LOOPINC); // Increment LOOPCOUNT_REG by Z-component
725  sub(LOOPCOUNT, 1); // Increment loop count by 1
726  jnz(l_loop_start); // Loop if not equal
727  L(*loop_break_label);
728  loop_break_label = boost::none;
729 
730  looping = false;
731 }

815 void JitShader::Compile_NextInstr() {
816  if (std::binary_search(return_offsets.begin(), return_offsets.end(), program_counter)) {
817  Compile_Return();
818  }
819 
820  L(instruction_labels[program_counter]);
821 
822  Instruction instr = {(*program_code)[program_counter++]};
823 
824  OpCode::Id opcode = instr.opcode.Value();
825  auto instr_func = instr_table[static_cast<unsigned>(opcode)];
826 
827  if (instr_func) {
828  // JIT the instruction!
829  ((*this).*instr_func)(instr);
830  } else {
831  // Unhandled instruction
832  LOG_CRITICAL(HW_GPU, "Unhandled instruction: 0x{:02x} (0x{:08x})",
833  static_cast<u32>(instr.opcode.Value().EffectiveOpCode()), instr.hex);
834  }

837 void JitShader::FindReturnOffsets() {
838  return_offsets.clear();
839 
840  for (std::size_t offset = 0; offset < program_code->size(); ++offset) {
841  Instruction instr = {(*program_code)[offset]};
842 
843  switch (instr.opcode.Value()) {
844  case OpCode::Id::CALL:
845  case OpCode::Id::CALLC:
846  case OpCode::Id::CALLU:
847  return_offsets.push_back(instr.flow_control.dest_offset +
848  instr.flow_control.num_instructions);
849  break;
850  default:
851  break;
852  }
853  }
854 
855  // Sort for efficient binary search later
856  std::sort(return_offsets.begin(), return_offsets.end());

859 void JitShader::Compile(const std::array<u32, MAX_PROGRAM_CODE_LENGTH>* program_code_,
860  const std::array<u32, MAX_SWIZZLE_DATA_LENGTH>* swizzle_data_) {
861  program_code = program_code_;
862  swizzle_data = swizzle_data_;
863 
864  // Reset flow control state
865  program = (CompiledShader*)getCurr();
866  program_counter = 0;
867  looping = false;
868  instruction_labels.fill(Xbyak::Label());
869 
870  // Find all `CALL` instructions and identify return locations
871  FindReturnOffsets();
872 
873  // The stack pointer is 8 modulo 16 at the entry of a procedure
874  // We reserve 16 bytes and assign a dummy value to the first 8 bytes, to catch any potential
875  // return checks (see Compile_Return) that happen in shader main routine.
876  ABI_PushRegistersAndAdjustStack(*this, ABI_ALL_CALLEE_SAVED, 8, 16);
877  mov(qword[rsp + 8], 0xFFFFFFFFFFFFFFFFULL);
878 
879  mov(UNIFORMS, ABI_PARAM1);
880  mov(STATE, ABI_PARAM2);
881 
882  // Load address/loop registers
883  movsxd(ADDROFFS_REG_0, dword[STATE + offsetof(UnitState, address_registers[0])]);
884  movsxd(ADDROFFS_REG_1, dword[STATE + offsetof(UnitState, address_registers[1])]);
885  mov(LOOPCOUNT_REG, dword[STATE + offsetof(UnitState, address_registers[2])]);
886  shl(ADDROFFS_REG_0, 4);
887  shl(ADDROFFS_REG_1, 4);
888  shl(LOOPCOUNT_REG, 4);
889 
890  // Load conditional code
891  mov(COND0, byte[STATE + offsetof(UnitState, conditional_code[0])]);
892  mov(COND1, byte[STATE + offsetof(UnitState, conditional_code[1])]);
893 
894  // Used to set a register to one
895  static const __m128 one = {1.f, 1.f, 1.f, 1.f};
896  mov(rax, reinterpret_cast<std::size_t>(&one));
897  movaps(ONE, xword[rax]);
898 
899  // Used to negate registers
900  static const __m128 neg = {-0.f, -0.f, -0.f, -0.f};
901  mov(rax, reinterpret_cast<std::size_t>(&neg));
902  movaps(NEGBIT, xword[rax]);
903 
904  // Jump to start of the shader program
905  jmp(ABI_PARAM3);
906 
907  // Compile entire program
908  Compile_Block(static_cast<unsigned>(program_code->size()));
909 
910  // Free memory that's no longer needed
911  program_code = nullptr;
912  swizzle_data = nullptr;
913  return_offsets.clear();
914  return_offsets.shrink_to_fit();
915 
916  ready();
917 
918  ASSERT_MSG(getSize() <= MAX_SHADER_SIZE, "Compiled a shader that exceeds the allocated size!");
919  LOG_DEBUG(HW_GPU, "Compiled shader size={}", getSize());
920 }

931 Xbyak::Label JitShader::CompilePrelude_Log2() {
932  Xbyak::Label subroutine;
933 
934  // SSE does not have a log instruction, thus we must approximate.
935  // We perform this approximation first performaing a range reduction into the range [1.0, 2.0).
936  // A minimax polynomial which was fit for the function log2(x) / (x - 1) is then evaluated.
937  // We multiply the result by (x - 1) then restore the result into the appropriate range.
938 
939  // Coefficients for the minimax polynomial.
940  // f(x) computes approximately log2(x) / (x - 1).
941  // f(x) = c4 + x * (c3 + x * (c2 + x * (c1 + x * c0)).
942  align(64);
943  const void* c0 = getCurr();
944  dd(0x3d74552f);
945  const void* c1 = getCurr();
946  dd(0xbeee7397);
947  const void* c2 = getCurr();
948  dd(0x3fbd96dd);
949  const void* c3 = getCurr();
950  dd(0xc02153f6);
951  const void* c4 = getCurr();
952  dd(0x4038d96c);
953 
954  align(16);
955  const void* negative_infinity_vector = getCurr();
956  dd(0xff800000);
957  dd(0xff800000);
958  dd(0xff800000);
959  dd(0xff800000);
960  const void* default_qnan_vector = getCurr();
961  dd(0x7fc00000);
962  dd(0x7fc00000);
963  dd(0x7fc00000);
964  dd(0x7fc00000);
965 
966  Xbyak::Label input_is_nan, input_is_zero, input_out_of_range;
967 
968  align(16);
969  L(input_out_of_range);
970  je(input_is_zero);
971  movaps(SRC1, xword[rip + default_qnan_vector]);
972  ret();
973  L(input_is_zero);
974  movaps(SRC1, xword[rip + negative_infinity_vector]);
975  ret();
976 
977  align(16);
978  L(subroutine);
979 
980  // Here we handle edge cases: input in {NaN, 0, -Inf, Negative}.
981  xorps(SCRATCH, SCRATCH);
982  ucomiss(SCRATCH, SRC1);
983  jp(input_is_nan);
984  jae(input_out_of_range);
985 
986  // Split input
987  movd(eax, SRC1);
988  mov(edx, eax);
989  and_(eax, 0x7f800000);
990  and_(edx, 0x007fffff);
991  movss(SCRATCH, xword[rip + c0]); // Preload c0.
992  or_(edx, 0x3f800000);
993  movd(SRC1, edx);
994  // SRC1 now contains the mantissa of the input.
995  mulss(SCRATCH, SRC1);
996  shr(eax, 23);
997  sub(eax, 0x7f);
998  cvtsi2ss(SCRATCH2, eax);
999  // SCRATCH2 now contains the exponent of the input.
1000 
1001  // Complete computation of polynomial
1002  addss(SCRATCH, xword[rip + c1]);
1003  mulss(SCRATCH, SRC1);
1004  addss(SCRATCH, xword[rip + c2]);
1005  mulss(SCRATCH, SRC1);
1006  addss(SCRATCH, xword[rip + c3]);
1007  mulss(SCRATCH, SRC1);
1008  subss(SRC1, ONE);
1009  addss(SCRATCH, xword[rip + c4]);
1010  mulss(SCRATCH, SRC1);
1011  addss(SCRATCH2, SCRATCH);
1012 
1013  // Duplicate result across vector
1014  xorps(SRC1, SRC1); // break dependency chain
1015  movss(SRC1, SCRATCH2);
1016  L(input_is_nan);
1017  shufps(SRC1, SRC1, _MM_SHUFFLE(0, 0, 0, 0));
1018 
1019  ret();
1020 
1021  return subroutine;
1022 }

1024 Xbyak::Label JitShader::CompilePrelude_Exp2() {
1025  Xbyak::Label subroutine;
1026 
1027  // SSE does not have a exp instruction, thus we must approximate.
1028  // We perform this approximation first performaing a range reduction into the range [-0.5, 0.5).
1029  // A minimax polynomial which was fit for the function exp2(x) is then evaluated.
1030  // We then restore the result into the appropriate range.
1031 
1032  align(64);
1033  const void* input_max = getCurr();
1034  dd(0x43010000);
1035  const void* input_min = getCurr();
1036  dd(0xc2fdffff);
1037  const void* c0 = getCurr();
1038  dd(0x3c5dbe69);
1039  const void* half = getCurr();
1040  dd(0x3f000000);
1041  const void* c1 = getCurr();
1042  dd(0x3d5509f9);
1043  const void* c2 = getCurr();
1044  dd(0x3e773cc5);
1045  const void* c3 = getCurr();
1046  dd(0x3f3168b3);
1047  const void* c4 = getCurr();
1048  dd(0x3f800016);
1049 
1050  Xbyak::Label ret_label;
1051 
1052  align(16);
1053  L(subroutine);
1054 
1055  // Handle edge cases
1056  ucomiss(SRC1, SRC1);
1057  jp(ret_label);
1058  // Clamp to maximum range since we shift the value directly into the exponent.
1059  minss(SRC1, xword[rip + input_max]);
1060  maxss(SRC1, xword[rip + input_min]);
1061 
1062  // Decompose input
1063  movss(SCRATCH, SRC1);
1064  movss(SCRATCH2, xword[rip + c0]); // Preload c0.
1065  subss(SCRATCH, xword[rip + half]);
1066  cvtss2si(eax, SCRATCH);
1067  cvtsi2ss(SCRATCH, eax);
1068  // SCRATCH now contains input rounded to the nearest integer.
1069  add(eax, 0x7f);
1070  subss(SRC1, SCRATCH);
1071  // SRC1 contains input - round(input), which is in [-0.5, 0.5).
1072  mulss(SCRATCH2, SRC1);
1073  shl(eax, 23);
1074  movd(SCRATCH, eax);
1075  // SCRATCH contains 2^(round(input)).
1076 
1077  // Complete computation of polynomial.
1078  addss(SCRATCH2, xword[rip + c1]);
1079  mulss(SCRATCH2, SRC1);
1080  addss(SCRATCH2, xword[rip + c2]);
1081  mulss(SCRATCH2, SRC1);
1082  addss(SCRATCH2, xword[rip + c3]);
1083  mulss(SRC1, SCRATCH2);
1084  addss(SRC1, xword[rip + c4]);
1085  mulss(SRC1, SCRATCH);
1086 
1087  // Duplicate result across vector
1088  L(ret_label);
1089  shufps(SRC1, SRC1, _MM_SHUFFLE(0, 0, 0, 0));
1090 
1091  ret();
1092 
1093  return subroutine;
1094 }
