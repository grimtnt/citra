21 Loader::ResultStatus CIAContainer::Load(const FileBackend& backend) {
22  std::vector<u8> header_data(sizeof(Header));
23 
24  // Load the CIA Header
25  ResultVal<std::size_t> read_result{backend.Read(0, sizeof(Header), header_data.data())};
26  if (read_result.Failed() || *read_result != sizeof(Header))
27  return Loader::ResultStatus::Error;
28 
29  Loader::ResultStatus result{LoadHeader(header_data)};
30  if (result != Loader::ResultStatus::Success)
31  return result;
32 
33  // Load Title Metadata
34  std::vector<u8> tmd_data(cia_header.tmd_size);
35  read_result = backend.Read(GetTitleMetadataOffset(), cia_header.tmd_size, tmd_data.data());
36  if (read_result.Failed() || *read_result != cia_header.tmd_size)
37  return Loader::ResultStatus::Error;
38 
39  result = LoadTitleMetadata(tmd_data);
40  if (result != Loader::ResultStatus::Success)
41  return result;
42 
43  // Load CIA Metadata
44  if (cia_header.meta_size) {
45  std::vector<u8> meta_data(sizeof(Metadata));
46  read_result = backend.Read(GetMetadataOffset(), sizeof(Metadata), meta_data.data());
47  if (read_result.Failed() || *read_result != sizeof(Metadata))
48  return Loader::ResultStatus::Error;
49 
50  result = LoadMetadata(meta_data);
51  if (result != Loader::ResultStatus::Success)
52  return result;
53  }
54 
55  return Loader::ResultStatus::Success;
56 }

58 Loader::ResultStatus CIAContainer::Load(const std::string& filepath) {
59  FileUtil::IOFile file{filepath, "rb"};
60  if (!file.IsOpen())
61  return Loader::ResultStatus::Error;
62 
63  // Load CIA Header
64  std::vector<u8> header_data(sizeof(Header));
65  if (file.ReadBytes(header_data.data(), sizeof(Header)) != sizeof(Header))
66  return Loader::ResultStatus::Error;
67 
68  Loader::ResultStatus result{LoadHeader(header_data)};
69  if (result != Loader::ResultStatus::Success)
70  return result;
71 
72  // Load Title Metadata
73  std::vector<u8> tmd_data(cia_header.tmd_size);
74  file.Seek(GetTitleMetadataOffset(), SEEK_SET);
75  if (file.ReadBytes(tmd_data.data(), cia_header.tmd_size) != cia_header.tmd_size)
76  return Loader::ResultStatus::Error;
77 
78  result = LoadTitleMetadata(tmd_data);
79  if (result != Loader::ResultStatus::Success)
80  return result;
81 
82  // Load CIA Metadata
83  if (cia_header.meta_size) {
84  std::vector<u8> meta_data(sizeof(Metadata));
85  file.Seek(GetMetadataOffset(), SEEK_SET);
86  if (file.ReadBytes(meta_data.data(), sizeof(Metadata)) != sizeof(Metadata))
87  return Loader::ResultStatus::Error;
88 
89  result = LoadMetadata(meta_data);
90  if (result != Loader::ResultStatus::Success)
91  return result;
92  }
93 
94  return Loader::ResultStatus::Success;
95 }
