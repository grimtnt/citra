14 inline int RegToIndex(const Xbyak::Reg& reg) {
15  using Kind = Xbyak::Reg::Kind;
16  ASSERT_MSG((reg.getKind() & (Kind::REG | Kind::XMM)) != 0,
17  "RegSet only support GPRs and XMM registers.");
18  ASSERT_MSG(reg.getIdx() < 16, "RegSet only supports XXM0-15.");
19  return reg.getIdx() + (reg.getKind() == Kind::REG ? 0 : 16);
20 }

154 inline void ABI_CalculateFrameSize(BitSet32 regs, std::size_t rsp_alignment, std::size_t needed_frame_size,
155  s32* out_subtraction, s32* out_xmm_offset) {
156  int count = (regs & ABI_ALL_GPRS).Count();
157  rsp_alignment -= count * 8;
158  std::size_t subtraction{};
159  int xmm_count = (regs & ABI_ALL_XMMS).Count();
160  if (xmm_count) {
161  // If we have any XMMs to save, we must align the stack here.
162  subtraction = rsp_alignment & 0xF;
163  }
164  subtraction += 0x10 * xmm_count;
165  std::size_t xmm_base_subtraction = subtraction;
166  subtraction += needed_frame_size;
167  subtraction += ABI_SHADOW_SPACE;
168  // Final alignment.
169  rsp_alignment -= subtraction;
170  subtraction += rsp_alignment & 0xF;
171 
172  *out_subtraction = (s32)subtraction;
173  *out_xmm_offset = (s32)(subtraction - xmm_base_subtraction);
174 }

176 inline std::size_t ABI_PushRegistersAndAdjustStack(Xbyak::CodeGenerator& code, BitSet32 regs,
177  std::size_t rsp_alignment, std::size_t needed_frame_size = 0) {
178  s32 subtraction, xmm_offset;
179  ABI_CalculateFrameSize(regs, rsp_alignment, needed_frame_size, &subtraction, &xmm_offset);
180 
181  for (int reg_index : (regs & ABI_ALL_GPRS)) {
182  code.push(IndexToReg64(reg_index));
183  }
184 
185  if (subtraction != 0) {
186  code.sub(code.rsp, subtraction);
187  }
188 
189  for (int reg_index : (regs & ABI_ALL_XMMS)) {
190  code.movaps(code.xword[code.rsp + xmm_offset], IndexToXmm(reg_index));
191  xmm_offset += 0x10;
192  }
193 
194  return ABI_SHADOW_SPACE;
195 }

197 inline void ABI_PopRegistersAndAdjustStack(Xbyak::CodeGenerator& code, BitSet32 regs,
198  std::size_t rsp_alignment, std::size_t needed_frame_size = 0) {
199  s32 subtraction, xmm_offset;
200  ABI_CalculateFrameSize(regs, rsp_alignment, needed_frame_size, &subtraction, &xmm_offset);
201 
202  for (int reg_index : (regs & ABI_ALL_XMMS)) {
203  code.movaps(IndexToXmm(reg_index), code.xword[code.rsp + xmm_offset]);
204  xmm_offset += 0x10;
205  }
206 
207  if (subtraction != 0) {
208  code.add(code.rsp, subtraction);
209  }
210 
211  // GPRs need to be popped in reverse order
212  for (int reg_index = 15; reg_index >= 0; reg_index--) {
213  if (regs[reg_index]) {
214  code.pop(IndexToReg64(reg_index));
215  }
216  }
217 }
