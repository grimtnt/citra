23 ResultCode ConversionConfiguration::SetInputLineWidth(u16 width) {
24  if (width == 0 || width > 1024 || width % 8 != 0) {
25  return ResultCode(ErrorDescription::OutOfRange, ErrorModule::CAM,
26  ErrorSummary::InvalidArgument, ErrorLevel::Usage); // 0xE0E053FD
27  }
28 
29  // Note: The hardware uses the register value 0 to represent a width of 1024, so for a width of
30  // 1024 the `camera` module would set the value 0 here, but we don't need to emulate this
31  // internal detail.
32  this->input_line_width = width;
33  return RESULT_SUCCESS;
34 }
35 
36 ResultCode ConversionConfiguration::SetInputLines(u16 lines) {
37  if (lines == 0 || lines > 1024) {
38  return ResultCode(ErrorDescription::OutOfRange, ErrorModule::CAM,
39  ErrorSummary::InvalidArgument, ErrorLevel::Usage); // 0xE0E053FD
40  }
41 
42  // Note: In what appears to be a bug, the `camera` module does not set the hardware register at
43  // all if `lines` is 1024, so the conversion uses the last value that was set. The intention
44  // was probably to set it to 0 like in SetInputLineWidth.
45  if (lines != 1024) {
46  this->input_lines = lines;
47  }
48  return RESULT_SUCCESS;
49 }

51 ResultCode ConversionConfiguration::SetStandardCoefficient(
52  StandardCoefficient standard_coefficient) {
53  std::size_t index = static_cast<std::size_t>(standard_coefficient);
54  if (index >= ARRAY_SIZE(standard_coefficients)) {
55  return ResultCode(ErrorDescription::InvalidEnumValue, ErrorModule::CAM,
56  ErrorSummary::InvalidArgument, ErrorLevel::Usage); // 0xE0E053ED
57  }
58 
59  std::memcpy(coefficients.data(), standard_coefficients[index].data(), sizeof(coefficients));
60  return RESULT_SUCCESS;
61 }

538 void Y2R_U::SetPackageParameter(Kernel::HLERequestContext& ctx) {
539  IPC::RequestParser rp{ctx, 0x29, 7, 0};
540  auto params{rp.PopRaw<ConversionParameters>()};
541 
542  conversion.input_format = params.input_format;
543  conversion.output_format = params.output_format;
544  conversion.rotation = params.rotation;
545  conversion.block_alignment = params.block_alignment;
546 
547  ResultCode result = conversion.SetInputLineWidth(params.input_line_width);
548 
549  if (result.IsError())
550  goto cleanup;
551 
552  result = conversion.SetInputLines(params.input_lines);
553 
554  if (result.IsError())
555  goto cleanup;
556 
557  result = conversion.SetStandardCoefficient(params.standard_coefficient);
558 
559  if (result.IsError())
560  goto cleanup;
561 
562  conversion.padding = params.padding;
563  conversion.alpha = params.alpha;
564 
565 cleanup:
566  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
567  rb.Push(result);
568
