36 System::ResultStatus System::RunLoop() {
37  status = ResultStatus::Success;
38  if (!cpu_core) {
39  return ResultStatus::ErrorNotInitialized;
40  }
41  if (!dsp_core->IsOutputAllowed()) {
42  // Draw black screens to the emulator window
43  VideoCore::g_renderer->SwapBuffers();
44  // Sleep for a frame or the PC would overheat
45  std::this_thread::sleep_for(std::chrono::milliseconds{16});
46  return ResultStatus::Success;
47  }
48  // If we don't have a currently active thread then don't execute instructions,
49  // instead advance to the next event and try to yield to the next thread
50  if (Kernel::GetCurrentThread() == nullptr) {
51  LOG_TRACE(Core_ARM11, "Idling");
52  CoreTiming::Idle();
53  CoreTiming::Advance();
54  PrepareReschedule();
55  } else {
56  CoreTiming::Advance();
57  cpu_core->Run();
58  }
59 
60  HW::Update();
61  Reschedule();
62 
63  if (jump_requested.exchange(false)) {
64  Jump();
65  } else if (shutdown_requested.exchange(false)) {
66  return ResultStatus::ShutdownRequested;
67  }
68  return status;
69 }

71 System::ResultStatus System::Load(EmuWindow& emu_window, const std::string& filepath) {
72  app_loader = Loader::GetLoader(filepath);
73 
74  if (!app_loader) {
75  LOG_CRITICAL(Core, "Failed to obtain loader for {}!", filepath);
76  return ResultStatus::ErrorGetLoader;
77  }
78  std::pair<boost::optional<u32>, Loader::ResultStatus> system_mode{
79  app_loader->LoadKernelSystemMode()};
80 
81  if (system_mode.second != Loader::ResultStatus::Success) {
82  LOG_CRITICAL(Core, "Failed to determine system mode (Error {})!",
83  static_cast<int>(system_mode.second));
84 
85  switch (system_mode.second) {
86  case Loader::ResultStatus::ErrorEncrypted:
87  return ResultStatus::ErrorLoader_ErrorEncrypted;
88  case Loader::ResultStatus::ErrorInvalidFormat:
89  return ResultStatus::ErrorLoader_ErrorInvalidFormat;
90  default:
91  return ResultStatus::ErrorSystemMode;
92  }
93  }
94 
95  ResultStatus init_result{Init(emu_window, system_mode.first.get())};
96  if (init_result != ResultStatus::Success) {
97  LOG_CRITICAL(Core, "Failed to initialize system (Error {})!",
98  static_cast<u32>(init_result));
99  System::Shutdown();
100  return init_result;
101  }
102 
103  const Loader::ResultStatus load_result{app_loader->Load(Kernel::g_current_process)};
104  if (Loader::ResultStatus::Success != load_result) {
105  LOG_CRITICAL(Core, "Failed to load ROM (Error {})!", static_cast<u32>(load_result));
106  System::Shutdown();
107 
108  switch (load_result) {
109  case Loader::ResultStatus::ErrorEncrypted:
110  return ResultStatus::ErrorLoader_ErrorEncrypted;
111  case Loader::ResultStatus::ErrorInvalidFormat:
112  return ResultStatus::ErrorLoader_ErrorInvalidFormat;
113  default:
114  return ResultStatus::ErrorLoader;
115  }
116  }
117  Memory::SetCurrentPageTable(&Kernel::g_current_process->vm_manager.page_table);
118  status = ResultStatus::Success;
119  m_emu_window = &emu_window;
120  m_filepath = filepath;
121  return status;
122 }

146 System::ResultStatus System::Init(EmuWindow& emu_window, u32 system_mode) {
147  LOG_DEBUG(HW_Memory, "initialized OK");
148 
149  CoreTiming::Init();
150 
151  cpu_core = std::make_unique<CPU>();
152 
153  qt_callbacks = std::make_unique<QtCallbacks>();
154 
155  dsp_core = std::make_unique<AudioCore::DspHle>();
156  dsp_core->SetSink(Settings::values.sink_id, Settings::values.audio_device_id);
157  dsp_core->EnableStretching(Settings::values.enable_audio_stretching);
158 
159  rpc_server = std::make_unique<RPC::RPCServer>();
160  service_manager = std::make_shared<Service::SM::ServiceManager>();
161  shared_page_handler = std::make_shared<SharedPage::Handler>();
162 
163  jump_requested = false;
164  shutdown_requested = false;
165  shell_open = true;
166 
167  // Initialize FS and CFG before kernel because kernel uses CFG (and Citra crashes if CFG is
168  // initialized before FS is initialized)
169  Service::FS::ArchiveInit();
170  Service::FS::InstallInterfaces(*service_manager);
171  Service::CFG::InstallInterfaces(*service_manager);
172  HW::Init();
173  Kernel::Init(system_mode);
174  Service::Init(service_manager);
175  CheatCore::Init();
176 
177  ResultStatus result{VideoCore::Init(emu_window)};
178  if (result != ResultStatus::Success) {
179  return result;
180  }
181 
182  LOG_DEBUG(Core, "Initialized OK");
183 
184  // Reset counters and set time origin to current frame
185  GetAndResetPerfStats();
186  perf_stats.BeginSystemFrame();
187 
188  return ResultStatus::Success;
189 }
