43 static bool VerifyBufferState(Kernel::Process& process, VAddr buffer_ptr, u32 size) {
44  auto vma = process.vm_manager.FindVMA(buffer_ptr);
45  return vma != process.vm_manager.vma_map.end() &&
46  vma->second.base + vma->second.size >= buffer_ptr + size &&
47  vma->second.permissions == Kernel::VMAPermission::ReadWrite &&
48  vma->second.meminfo_state == Kernel::MemoryState::Private;
49 }

51 void RO::Initialize(Kernel::HLERequestContext& ctx) {
52  IPC::RequestParser rp{ctx, 0x01, 3, 2};
53  VAddr crs_buffer_ptr{rp.Pop<u32>()};
54  u32 crs_size{rp.Pop<u32>()};
55  VAddr crs_address{rp.Pop<u32>()};
56  // TODO (wwylele): RO service checks the descriptor here and return error 0xD9001830 for
57  // incorrect descriptor. This error return should be probably built in IPC::RequestParser.
58  // All other service functions below have the same issue.
59  auto process{rp.PopObject<Kernel::Process>()};
60 
61  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
62 
63  ClientSlot* slot = GetSessionData(ctx.Session());
64  if (slot->loaded_crs != 0) {
65  LOG_ERROR(Service_LDR, "Already initialized");
66  rb.Push(ERROR_ALREADY_INITIALIZED);
67  return;
68  }
69 
70  if (crs_size < CRO_HEADER_SIZE) {
71  LOG_ERROR(Service_LDR, "CRS is too small");
72  rb.Push(ERROR_BUFFER_TOO_SMALL);
73  return;
74  }
75 
76  if (crs_buffer_ptr & Memory::PAGE_MASK) {
77  LOG_ERROR(Service_LDR, "CRS original address is not aligned");
78  rb.Push(ERROR_MISALIGNED_ADDRESS);
79  return;
80  }
81 
82  if (crs_address & Memory::PAGE_MASK) {
83  LOG_ERROR(Service_LDR, "CRS mapping address is not aligned");
84  rb.Push(ERROR_MISALIGNED_ADDRESS);
85  return;
86  }
87 
88  if (crs_size & Memory::PAGE_MASK) {
89  LOG_ERROR(Service_LDR, "CRS size is not aligned");
90  rb.Push(ERROR_MISALIGNED_SIZE);
91  return;
92  }
93 
94  if (!VerifyBufferState(*process, crs_buffer_ptr, crs_size)) {
95  LOG_ERROR(Service_LDR, "CRS original buffer is in invalid state");
96  rb.Push(ERROR_INVALID_MEMORY_STATE);
97  return;
98  }
99 
100  if (crs_address < Memory::PROCESS_IMAGE_VADDR ||
101  crs_address + crs_size > Memory::PROCESS_IMAGE_VADDR_END) {
102  LOG_ERROR(Service_LDR, "CRS mapping address is not in the process image region");
103  rb.Push(ERROR_ILLEGAL_ADDRESS);
104  return;
105  }
106 
107  ResultCode result = RESULT_SUCCESS;
108 
109  if (crs_buffer_ptr != crs_address) {
110  // TODO(wwylele): should be memory aliasing
111  std::shared_ptr<std::vector<u8>> crs_mem = std::make_shared<std::vector<u8>>(crs_size);
112  Memory::ReadBlock(crs_buffer_ptr, crs_mem->data(), crs_size);
113  result = process->vm_manager
114  .MapMemoryBlock(crs_address, crs_mem, 0, crs_size, Kernel::MemoryState::Code)
115  .Code();
116  if (result.IsError()) {
117  LOG_ERROR(Service_LDR, "Error mapping memory block {:08X}", result.raw);
118  rb.Push(result);
119  return;
120  }
121 
122  result =
123  process->vm_manager.ReprotectRange(crs_address, crs_size, Kernel::VMAPermission::Read);
124  if (result.IsError()) {
125  LOG_ERROR(Service_LDR, "Error reprotecting memory block {:08X}", result.raw);
126  rb.Push(result);
127  return;
128  }
129 
130  slot->memory_synchronizer.AddMemoryBlock(crs_address, crs_buffer_ptr, crs_size);
131  } else {
132  // Do nothing if buffer_ptr == address
133  // TODO(wwylele): verify this behaviour. This is only seen in the web browser app,
134  // and the actual behaviour is unclear. "Do nothing" is probably an incorrect implement.
135  // There is also a chance that another issue causes the app passing wrong arguments.
136  LOG_WARNING(Service_LDR, "crs_buffer_ptr == crs_address (0x{:08X})", crs_address);
137  }
138 
139  CROHelper crs(crs_address);
140  crs.InitCRS();
141 
142  result = crs.Rebase(0, crs_size, 0, 0, 0, 0, true);
143  if (result.IsError()) {
144  LOG_ERROR(Service_LDR, "Error rebasing CRS 0x{:08X}", result.raw);
145  rb.Push(result);
146  return;
147  }
148 
149  slot->memory_synchronizer.SynchronizeOriginalMemory(*process);
150 
151  slot->loaded_crs = crs_address;
152 
153  rb.Push(RESULT_SUCCESS);
154 
155  LOG_DEBUG(Service_LDR, "called, crs_buffer_ptr=0x{:08X}, crs_address=0x{:08X}, crs_size=0x{:X}",
156  crs_buffer_ptr, crs_address, crs_size);
157 }

183 void RO::LoadCRO(Kernel::HLERequestContext& ctx, bool link_on_load_bug_fix) {
184  IPC::RequestParser rp{ctx, static_cast<u16>(link_on_load_bug_fix ? 0x09 : 0x04), 11, 2};
185  VAddr cro_buffer_ptr{rp.Pop<u32>()};
186  VAddr cro_address{rp.Pop<u32>()};
187  u32 cro_size{rp.Pop<u32>()};
188  VAddr data_segment_address{rp.Pop<u32>()};
189  u32 zero{rp.Pop<u32>()};
190  u32 data_segment_size{rp.Pop<u32>()};
191  u32 bss_segment_address{rp.Pop<u32>()};
192  u32 bss_segment_size{rp.Pop<u32>()};
193  bool auto_link{rp.Pop<bool>()};
194  u32 fix_level{rp.Pop<u32>()};
195  VAddr crr_address{rp.Pop<u32>()};
196  auto process{rp.PopObject<Kernel::Process>()};
197 
198  LOG_DEBUG(Service_LDR,
199  "called ({}), cro_buffer_ptr=0x{:08X}, cro_address=0x{:08X}, cro_size={:X}, "
200  "data_segment_address=0x{:08X}, zero={}, data_segment_size=0x{:X}, "
201  "bss_segment_address=0x{:08X}, bss_segment_size=0x{:X}, auto_link={}, "
202  "fix_level={}, crr_address=0x{:08X}",
203  link_on_load_bug_fix ? "new" : "old", cro_buffer_ptr, cro_address, cro_size,
204  data_segment_address, zero, data_segment_size, bss_segment_address, bss_segment_size,
205  auto_link ? "true" : "false", fix_level, crr_address);
206 
207  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
208 
209  ClientSlot* slot = GetSessionData(ctx.Session());
210  if (slot->loaded_crs == 0) {
211  LOG_ERROR(Service_LDR, "Not initialized");
212  rb.Push(ERROR_NOT_INITIALIZED);
213  rb.Push<u32>(0);
214  return;
215  }
216 
217  if (cro_size < CRO_HEADER_SIZE) {
218  LOG_ERROR(Service_LDR, "CRO too small");
219  rb.Push(ERROR_BUFFER_TOO_SMALL);
220  rb.Push<u32>(0);
221  return;
222  }
223 
224  if (cro_buffer_ptr & Memory::PAGE_MASK) {
225  LOG_ERROR(Service_LDR, "CRO original address is not aligned");
226  rb.Push(ERROR_MISALIGNED_ADDRESS);
227  rb.Push<u32>(0);
228  return;
229  }
230 
231  if (cro_address & Memory::PAGE_MASK) {
232  LOG_ERROR(Service_LDR, "CRO mapping address is not aligned");
233  rb.Push(ERROR_MISALIGNED_ADDRESS);
234  rb.Push<u32>(0);
235  return;
236  }
237 
238  if (cro_size & Memory::PAGE_MASK) {
239  LOG_ERROR(Service_LDR, "CRO size is not aligned");
240  rb.Push(ERROR_MISALIGNED_SIZE);
241  rb.Push<u32>(0);
242  return;
243  }
244 
245  if (!VerifyBufferState(*process, cro_buffer_ptr, cro_size)) {
246  LOG_ERROR(Service_LDR, "CRO original buffer is in invalid state");
247  rb.Push(ERROR_INVALID_MEMORY_STATE);
248  rb.Push<u32>(0);
249  return;
250  }
251 
252  if (cro_address < Memory::PROCESS_IMAGE_VADDR ||
253  cro_address + cro_size > Memory::PROCESS_IMAGE_VADDR_END) {
254  LOG_ERROR(Service_LDR, "CRO mapping address is not in the process image region");
255  rb.Push(ERROR_ILLEGAL_ADDRESS);
256  rb.Push<u32>(0);
257  return;
258  }
259 
260  if (zero) {
261  LOG_ERROR(Service_LDR, "Zero is not zero {}", zero);
262  rb.Push(ResultCode(static_cast<ErrorDescription>(29), ErrorModule::RO,
263  ErrorSummary::Internal, ErrorLevel::Usage));
264  rb.Push<u32>(0);
265  return;
266  }
267 
268  ResultCode result = RESULT_SUCCESS;
269 
270  if (cro_buffer_ptr != cro_address) {
271  // TODO(wwylele): should be memory aliasing
272  std::shared_ptr<std::vector<u8>> cro_mem = std::make_shared<std::vector<u8>>(cro_size);
273  Memory::ReadBlock(cro_buffer_ptr, cro_mem->data(), cro_size);
274  result = process->vm_manager
275  .MapMemoryBlock(cro_address, cro_mem, 0, cro_size, Kernel::MemoryState::Code)
276  .Code();
277  if (result.IsError()) {
278  LOG_ERROR(Service_LDR, "Error mapping memory block {:08X}", result.raw);
279  rb.Push(result);
280  rb.Push<u32>(0);
281  return;
282  }
283 
284  result =
285  process->vm_manager.ReprotectRange(cro_address, cro_size, Kernel::VMAPermission::Read);
286  if (result.IsError()) {
287  LOG_ERROR(Service_LDR, "Error reprotecting memory block {:08X}", result.raw);
288  process->vm_manager.UnmapRange(cro_address, cro_size);
289  rb.Push(result);
290  rb.Push<u32>(0);
291  return;
292  }
293 
294  slot->memory_synchronizer.AddMemoryBlock(cro_address, cro_buffer_ptr, cro_size);
295  } else {
296  // Do nothing if buffer_ptr == address
297  // TODO(wwylele): verify this behaviour.
298  // This is derived from the case of LoadCRS with buffer_ptr==address,
299  // and is never seen in any game. "Do nothing" is probably an incorrect implement.
300  // There is also a chance that this case is just prohibited.
301  LOG_WARNING(Service_LDR, "cro_buffer_ptr == cro_address (0x{:08X})", cro_address);
302  }
303 
304  CROHelper cro(cro_address);
305 
306  result = cro.VerifyHash(cro_size, crr_address);
307  if (result.IsError()) {
308  LOG_ERROR(Service_LDR, "Error verifying CRO in CRR {:08X}", result.raw);
309  process->vm_manager.UnmapRange(cro_address, cro_size);
310  rb.Push(result);
311  rb.Push<u32>(0);
312  return;
313  }
314 
315  result = cro.Rebase(slot->loaded_crs, cro_size, data_segment_address, data_segment_size,
316  bss_segment_address, bss_segment_size, false);
317  if (result.IsError()) {
318  LOG_ERROR(Service_LDR, "Error rebasing CRO {:08X}", result.raw);
319  process->vm_manager.UnmapRange(cro_address, cro_size);
320  rb.Push(result);
321  rb.Push<u32>(0);
322  return;
323  }
324 
325  result = cro.Link(slot->loaded_crs, link_on_load_bug_fix);
326  if (result.IsError()) {
327  LOG_ERROR(Service_LDR, "Error linking CRO {:08X}", result.raw);
328  process->vm_manager.UnmapRange(cro_address, cro_size);
329  rb.Push(result);
330  rb.Push<u32>(0);
331  return;
332  }
333 
334  cro.Register(slot->loaded_crs, auto_link);
335 
336  u32 fix_size = cro.Fix(fix_level);
337 
338  slot->memory_synchronizer.SynchronizeOriginalMemory(*process);
339 
340  // TODO(wwylele): verify the behaviour when buffer_ptr == address
341  if (cro_buffer_ptr != cro_address) {
342  if (fix_size != cro_size) {
343  result = process->vm_manager.UnmapRange(cro_address + fix_size, cro_size - fix_size);
344  if (result.IsError()) {
345  LOG_ERROR(Service_LDR, "Error unmapping memory block {:08X}", result.raw);
346  process->vm_manager.UnmapRange(cro_address, cro_size);
347  rb.Push(result);
348  rb.Push<u32>(0);
349  return;
350  }
351  }
352 
353  // Changes the block size
354  slot->memory_synchronizer.ResizeMemoryBlock(cro_address, cro_buffer_ptr, fix_size);
355  }
356 
357  VAddr exe_begin;
358  u32 exe_size;
359  std::tie(exe_begin, exe_size) = cro.GetExecutablePages();
360  if (exe_begin) {
361  result = process->vm_manager.ReprotectRange(exe_begin, exe_size,
362  Kernel::VMAPermission::ReadExecute);
363  if (result.IsError()) {
364  LOG_ERROR(Service_LDR, "Error reprotecting memory block {:08X}", result.raw);
365  process->vm_manager.UnmapRange(cro_address, fix_size);
366  rb.Push(result);
367  rb.Push<u32>(0);
368  return;
369  }
370  }
371 
372  Core::GetCPU().InvalidateCacheRange(cro_address, cro_size);
373 
374  LOG_INFO(Service_LDR, "CRO {} loaded at 0x{:08X}, fixed_end=0x{:08X}", cro.ModuleName(),
375  cro_address, cro_address + fix_size);
376 
377  rb.Push(RESULT_SUCCESS, fix_size);
378 }

380 void RO::UnloadCRO(Kernel::HLERequestContext& ctx) {
381  IPC::RequestParser rp{ctx, 0x05, 3, 2};
382  VAddr cro_address{rp.Pop<u32>()};
383  u32 zero{rp.Pop<u32>()};
384  VAddr cro_buffer_ptr{rp.Pop<u32>()};
385  auto process{rp.PopObject<Kernel::Process>()};
386 
387  LOG_DEBUG(Service_LDR, "called, cro_address=0x{:08X}, zero={}, cro_buffer_ptr=0x{:08X}",
388  cro_address, zero, cro_buffer_ptr);
389 
390  CROHelper cro(cro_address);
391 
392  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
393 
394  ClientSlot* slot = GetSessionData(ctx.Session());
395  if (slot->loaded_crs == 0) {
396  LOG_ERROR(Service_LDR, "Not initialized");
397  rb.Push(ERROR_NOT_INITIALIZED);
398  return;
399  }
400 
401  if (cro_address & Memory::PAGE_MASK) {
402  LOG_ERROR(Service_LDR, "CRO address is not aligned");
403  rb.Push(ERROR_MISALIGNED_ADDRESS);
404  return;
405  }
406 
407  if (!cro.IsLoaded()) {
408  LOG_ERROR(Service_LDR, "Invalid or not loaded CRO");
409  rb.Push(ERROR_NOT_LOADED);
410  return;
411  }
412 
413  LOG_INFO(Service_LDR, "Unloading CRO {}", cro.ModuleName());
414 
415  u32 fixed_size = cro.GetFixedSize();
416 
417  cro.Unregister(slot->loaded_crs);
418 
419  ResultCode result = cro.Unlink(slot->loaded_crs);
420  if (result.IsError()) {
421  LOG_ERROR(Service_LDR, "Error unlinking CRO {:08X}", result.raw);
422  rb.Push(result);
423  return;
424  }
425 
426  // If the module is not fixed, clears all external/internal relocations
427  // to restore the state before loading, so that it can be loaded again(?)
428  if (!cro.IsFixed()) {
429  result = cro.ClearRelocations();
430  if (result.IsError()) {
431  LOG_ERROR(Service_LDR, "Error clearing relocations {:08X}", result.raw);
432  rb.Push(result);
433  return;
434  }
435  }
436 
437  cro.Unrebase(false);
438 
439  slot->memory_synchronizer.SynchronizeOriginalMemory(*process);
440 
441  // TODO(wwylele): verify the behaviour when buffer_ptr == address
442  if (cro_address != cro_buffer_ptr) {
443  result = process->vm_manager.UnmapRange(cro_address, fixed_size);
444  if (result.IsError()) {
445  LOG_ERROR(Service_LDR, "Error unmapping CRO {:08X}", result.raw);
446  }
447  slot->memory_synchronizer.RemoveMemoryBlock(cro_address, cro_buffer_ptr);
448  }
449 
450  Core::GetCPU().InvalidateCacheRange(cro_address, fixed_size);
451 
452  rb.Push(result);
453 }
