86 void SRV::GetServiceHandle(Kernel::HLERequestContext& ctx) {
87  IPC::RequestParser rp{ctx, 0x5, 4, 0};
88  auto name_buf{rp.PopRaw<std::array<char, 8>>()};
89  std::size_t name_len{rp.Pop<u32>()};
90  u32 flags{rp.Pop<u32>()};
91 
92  bool wait_until_available = (flags & 1) == 0;
93 
94  if (name_len > Service::kMaxPortSize) {
95  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
96  rb.Push(ERR_INVALID_NAME_SIZE);
97  LOG_ERROR(Service_SRV, "called name_len=0x{:X} -> ERR_INVALID_NAME_SIZE", name_len);
98  return;
99  }
100  std::string name(name_buf.data(), name_len);
101 
102  // TODO(yuriks): Permission checks go here
103 
104  auto get_handle = [name, this](Kernel::SharedPtr<Kernel::Thread> thread,
105  Kernel::HLERequestContext& ctx, ThreadWakeupReason reason) {
106  LOG_ERROR(Service_SRV, "called service={} wakeup", name);
107  auto client_port = service_manager->GetServicePort(name);
108 
109  auto session = client_port.Unwrap()->Connect();
110  if (session.Succeeded()) {
111  LOG_DEBUG(Service_SRV, "called service={} -> session={}", name,
112  (*session)->GetObjectId());
113  IPC::ResponseBuilder rb(ctx, 0x5, 1, 2);
114  rb.Push(session.Code());
115  rb.PushMoveObjects(std::move(session).Unwrap());
116  } else if (session.Code() == Kernel::ERR_MAX_CONNECTIONS_REACHED) {
117  LOG_WARNING(Service_SRV, "called service={} -> ERR_MAX_CONNECTIONS_REACHED", name);
118  std::mutex mutex;
119  std::unique_lock<std::mutex> lock{mutex};
120  std::condition_variable cv;
121  cv.wait(lock, [&]() -> bool {
122  session = client_port.Unwrap()->Connect();
123  return session.Code() == Kernel::ERR_MAX_CONNECTIONS_REACHED;
124  });
125  IPC::ResponseBuilder rb(ctx, 0x5, 1, 2);
126  rb.Push(session.Code());
127  rb.PushMoveObjects(std::move(session).Unwrap());
128  } else {
129  LOG_ERROR(Service_SRV, "called service={} -> error 0x{:08X}", name, session.Code().raw);
130  IPC::ResponseBuilder rb(ctx, 0x5, 1, 0);
131  rb.Push(session.Code());
132  }
133  };
134 
135  auto client_port = service_manager->GetServicePort(name);
136  if (client_port.Failed()) {
137  if (wait_until_available && client_port.Code() == ERR_SERVICE_NOT_REGISTERED) {
138  LOG_INFO(Service_SRV, "called service={} delayed", name);
139  Kernel::SharedPtr<Kernel::Event> get_service_handle_event =
140  ctx.SleepClientThread(Kernel::GetCurrentThread(), "GetServiceHandle",
141  std::chrono::nanoseconds(-1), get_handle);
142  get_service_handle_delayed_map[name] = std::move(get_service_handle_event);
143  return;
144  } else {
145  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
146  rb.Push(client_port.Code());
147  LOG_ERROR(Service_SRV, "called service={} -> error 0x{:08X}", name,
148  client_port.Code().raw);
149  return;
150  }
151  }
152 
153  auto session = client_port.Unwrap()->Connect();
154  if (session.Succeeded()) {
155  LOG_DEBUG(Service_SRV, "called service={} -> session={}", name, (*session)->GetObjectId());
156  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 2)};
157  rb.Push(session.Code());
158  rb.PushMoveObjects(std::move(session).Unwrap());
159  } else if (session.Code() == Kernel::ERR_MAX_CONNECTIONS_REACHED && wait_until_available) {
160  LOG_WARNING(Service_SRV, "called service={} -> ERR_MAX_CONNECTIONS_REACHED", name);
161  std::mutex mutex;
162  std::unique_lock<std::mutex> lock{mutex};
163  std::condition_variable cv;
164  cv.wait(lock, [&]() -> bool {
165  session = client_port.Unwrap()->Connect();
166  return session.Code() == Kernel::ERR_MAX_CONNECTIONS_REACHED;
167  });
168  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 2)};
169  rb.Push(session.Code());
170  rb.PushMoveObjects(std::move(session).Unwrap());
171  } else {
172  LOG_ERROR(Service_SRV, "called service={} -> error 0x{:08X}", name, session.Code().raw);
173  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
174  rb.Push(session.Code());
175  }
176 }

235 void SRV::RegisterService(Kernel::HLERequestContext& ctx) {
236  IPC::RequestParser rp{ctx, 0x3, 4, 0};
237 
238  auto name_buf{rp.PopRaw<std::array<char, 8>>()};
239  std::size_t name_len{rp.Pop<u32>()};
240  u32 max_sessions{rp.Pop<u32>()};
241 
242  std::string name(name_buf.data(), std::min(name_len, name_buf.size()));
243 
244  auto port = service_manager->RegisterService(name, max_sessions);
245 
246  if (port.Failed()) {
247  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 0)};
248  rb.Push(port.Code());
249  LOG_ERROR(Service_SRV, "called service={} -> error 0x{:08X}", name, port.Code().raw);
250  return;
251  }
252 
253  auto it = get_service_handle_delayed_map.find(name);
254  if (it != get_service_handle_delayed_map.end()) {
255  it->second->Signal();
256  get_service_handle_delayed_map.erase(it);
257  }
258 
259  IPC::ResponseBuilder rb{rp.MakeBuilder(1, 2)};
260  rb.Push(RESULT_SUCCESS);
261  rb.PushMoveObjects(port.Unwrap());
262 }
