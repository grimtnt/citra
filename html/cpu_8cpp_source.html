26 class UserCallbacks final : public Dynarmic::A32::UserCallbacks {
27 public:
28  explicit UserCallbacks(CPU& parent) : parent(parent) {
29  switch (Settings::values.ticks_mode) {
30  case Settings::TicksMode::Custom: {
31  custom_ticks = Settings::values.ticks;
32  use_custom_ticks = true;
33  break;
34  }
35  case Settings::TicksMode::Auto: {
36  u64 program_id{};
37  Core::System::GetInstance().GetAppLoader().ReadProgramId(program_id);
38  auto itr = custom_ticks_map.find(program_id);
39  if (itr != custom_ticks_map.end()) {
40  custom_ticks = itr->second;
41  use_custom_ticks = true;
42  } else {
43  custom_ticks = 0;
44  use_custom_ticks = false;
45  }
46  break;
47  }
48  case Settings::TicksMode::Accurate: {
49  custom_ticks = 0;
50  use_custom_ticks = false;
51  break;
52  }
53  }
54  }
55  ~UserCallbacks() = default;

217 void CPU::PageTableChanged() {
218  current_page_table = Memory::GetCurrentPageTable();
219 
220  auto iter = jits.find(current_page_table);
221  if (iter != jits.end()) {
222  jit = iter->second.get();
223  return;
224  }
225 
226  auto new_jit = MakeJit();
227  jit = new_jit.get();
228  jits.emplace(current_page_table, std::move(new_jit));
229 }
