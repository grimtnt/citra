73 // Remove any ending forward slashes from directory paths
74 // Modifies argument.
75 static void StripTailDirSlashes(std::string& fname) {
76  if (fname.length() > 1) {
77  std::size_t i{fname.length()};
78  while (i > 0 && fname[i - 1] == DIR_SEP_CHR)
79  --i;
80  fname.resize(i);
81  }
82  return;
83 }

85 // Returns true if file filename exists
86 bool Exists(const std::string& filename) {
87  struct stat file_info {};
88 
89  std::string copy{filename};
90  StripTailDirSlashes(copy);
91 
92 #ifdef _WIN32
93  // Windows needs a slash to identify a drive root
94  if (copy.size() != 0 && copy.back() == ':')
95  copy += DIR_SEP_CHR;
96 
97  int result = _wstat64(Common::UTF8ToUTF16W(copy).c_str(), &file_info);
98 #else
99  int result = stat(copy.c_str(), &file_info);
100 #endif
101 
102  return (result == 0);
103 }

155  LOG_ERROR(Common_Filesystem, "unlink failed on {}: {}", filename, GetLastErrorMsg());

163 // Returns true if successful, or path already exists.
164 bool CreateDir(const std::string& path) {
165  LOG_TRACE(Common_Filesystem, "directory {}", path);
166 #ifdef _WIN32
167  if (::CreateDirectoryW(Common::UTF8ToUTF16W(path).c_str(), nullptr))
168  return true;
169  DWORD error = GetLastError();
170  if (error == ERROR_ALREADY_EXISTS) {
171  LOG_DEBUG(Common_Filesystem, "CreateDirectory failed on {}: already exists", path);
172  return true;
173  }
174  LOG_ERROR(Common_Filesystem, "CreateDirectory failed on {}: {}", path, error);
175  return false;
176 #else
177  if (mkdir(path.c_str(), 0755) == 0)
178  return true;
179 
180  int err = errno;
181 
182  if (err == EEXIST) {
183  LOG_DEBUG(Common_Filesystem, "mkdir failed on {}: already exists", path);
184  return true;
185  }
186 
187  LOG_ERROR(Common_Filesystem, "mkdir failed on {}: {}", path, strerror(err));
188  return false;

192 // Creates the full path of fullPath returns true on success
193 bool CreateFullPath(const std::string& full_path) {
194  int panic_counter{100};
195  LOG_TRACE(Common_Filesystem, "path {}", full_path);
196 
197  if (FileUtil::Exists(full_path)) {
198  LOG_DEBUG(Common_Filesystem, "path exists {}", full_path);
199  return true;
200  }
201 
202  std::size_t position{};
203  while (true) {
204  // Find next sub path
205  position = full_path.find(DIR_SEP_CHR, position);
206 
207  // we're done, yay!
208  if (position == full_path.npos)
209  return true;
210 
211  // Include the '/' so the first call is CreateDir("/") rather than CreateDir("")
212  std::string const subPath(full_path.substr(0, position + 1));
213  if (!FileUtil::IsDirectory(subPath) && !FileUtil::CreateDir(subPath)) {
214  LOG_ERROR(Common, "CreateFullPath: directory creation failed");
215  return false;
216  }
217 
218  // A safety check
219  panic_counter--;
220  if (panic_counter <= 0) {
221  LOG_ERROR(Common, "CreateFullPath: directory structure is too deep");
222  return false;
223  }
224  position++;
225  }
226 }

266 // copies file src_filename to dest_filename, returns true on success
267 bool Copy(const std::string& src_filename, const std::string& dest_filename) {
268  LOG_TRACE(Common_Filesystem, "{} --> {}", src_filename, dest_filename);
269 #ifdef _WIN32
270  if (CopyFileW(Common::UTF8ToUTF16W(src_filename).c_str(),
271  Common::UTF8ToUTF16W(dest_filename).c_str(), FALSE))
272  return true;
273 
274  LOG_ERROR(Common_Filesystem, "failed {} --> {}: {}", src_filename, dest_filename,
275  GetLastErrorMsg());
276  return false;
277 #else
278  using CFilePointer = std::unique_ptr<FILE, decltype(&std::fclose)>;
279 
280  // Open input file
281  CFilePointer input{fopen(src_filename.c_str(), "rb"), std::fclose};
282  if (!input) {
283  LOG_ERROR(Common_Filesystem, "opening input failed {} --> {}: {}", src_filename,
284  dest_filename, GetLastErrorMsg());
285  return false;
286  }
287 
288  // open output file
289  CFilePointer output{fopen(dest_filename.c_str(), "wb"), std::fclose};
290  if (!output) {
291  LOG_ERROR(Common_Filesystem, "opening output failed {} --> {}: {}", src_filename,
292  dest_filename, GetLastErrorMsg());
293  return false;
294  }
295 
296  // copy loop
297  std::array<char, 1024> buffer{};
298  while (!feof(input.get())) {
299  // read input
300  std::size_t rnum{fread(buffer.data(), sizeof(char), buffer.size(), input.get())};
301  if (rnum != buffer.size()) {
302  if (ferror(input.get()) != 0) {
303  LOG_ERROR(Common_Filesystem, "failed reading from source, {} --> {}: {}",
304  src_filename, dest_filename, GetLastErrorMsg());
305  return false;
306  }
307  }
308 
309  // write output
310  std::size_t wnum{fwrite(buffer.data(), sizeof(char), rnum, output.get())};
311  if (wnum != rnum) {
312  LOG_ERROR(Common_Filesystem, "failed writing to output, {} --> {}: {}", src_filename,
313  dest_filename, GetLastErrorMsg());
314  return false;
315  }
316  }
317 
318  return true;
319 #endif
320 }

360 u64 GetSize(FILE* f) {
361  // can't use off_t here because it can be 32-bit
362  auto pos{ftello(f)};
363  if (fseeko(f, 0, SEEK_END) != 0) {
364  LOG_ERROR(Common_Filesystem, "GetSize: seek failed {}: {}", (void*)f, GetLastErrorMsg());

366  }
367  auto size{ftello(f)};
368  if ((size != pos) && (fseeko(f, pos, SEEK_SET) != 0)) {
369  LOG_ERROR(Common_Filesystem, "GetSize: seek failed {}: {}", (void*)f, GetLastErrorMsg());
370  return 0;
371  }
372  return size;

387 bool ForeachDirectoryEntry(u64* num_entries_out, const std::string& directory,
388  DirectoryEntryCallable callback) {
389  LOG_TRACE(Common_Filesystem, "directory {}", directory);
390 
391  // How many files + directories we found
392  u64 found_entries{};
393 
394  // Save the status of callback function
395  bool callback_error{};
396 
397 #ifdef _WIN32
398  // Find the first file in the directory.
399  WIN32_FIND_DATAW ffd;
400 
401  HANDLE handle_find = FindFirstFileW(Common::UTF8ToUTF16W(directory + "\\*").c_str(), &ffd);
402  if (handle_find == INVALID_HANDLE_VALUE) {
403  FindClose(handle_find);
404  return false;
405  }
406  // windows loop
407  do {
408  const std::string virtual_name(Common::UTF16ToUTF8(ffd.cFileName));
409 #else
410  DIR* dirp{opendir(directory.c_str())};
411  if (!dirp)
412  return false;
413 
414  // non windows loop
415  while (struct dirent* result = readdir(dirp)) {
416  const std::string virtual_name{result->d_name};
417 #endif
418 
419  if (virtual_name == "." || virtual_name == "..")
420  continue;
421 
422  u64 ret_entries{};
423  if (!callback(&ret_entries, directory, virtual_name)) {
424  callback_error = true;
425  break;
426  }
427  found_entries += ret_entries;
428 
429 #ifdef _WIN32
430  } while (FindNextFileW(handle_find, &ffd) != 0);
431  FindClose(handle_find);
432 #else
433  }
434  closedir(dirp);
435 #endif
436 
437  if (callback_error)
438  return false;
439 
440  // num_entries_out is allowed to be specified nullptr, in which case we shouldn't try to set it
441  if (num_entries_out != nullptr)
442  *num_entries_out = found_entries;
443  return true;
444 }

446 u64 ScanDirectoryTree(const std::string& directory, FSTEntry& parent_entry,
447  unsigned int recursion) {
448  const auto callback{[recursion, &parent_entry](u64* num_entries_out,
449  const std::string& directory,
450  const std::string& virtual_name) -> bool {
451  FSTEntry entry{};
452  entry.virtualName = virtual_name;
453  entry.physicalName = directory + DIR_SEP + virtual_name;
454 
455  if (IsDirectory(entry.physicalName)) {
456  entry.isDirectory = true;
457  // is a directory, lets go inside if we didn't recurse to often
458  if (recursion > 0) {
459  entry.size = ScanDirectoryTree(entry.physicalName, entry, recursion - 1);
460  *num_entries_out += entry.size;
461  } else {
462  entry.size = 0;
463  }
464  } else { // is a file
465  entry.isDirectory = false;
466  entry.size = GetSize(entry.physicalName);
467  }
468  (*num_entries_out)++;
469 
470  // Push into the tree
471  parent_entry.children.push_back(std::move(entry));
472  return true;
473  }};
474 
475  u64 num_entries{};
476  return ForeachDirectoryEntry(&num_entries, directory, callback) ? num_entries : 0;
477 }

479 bool DeleteDirRecursively(const std::string& directory, unsigned int recursion) {
480  const auto callback{[recursion](u64* num_entries_out, const std::string& directory,
481  const std::string& virtual_name) -> bool {
482  std::string new_path{directory + DIR_SEP_CHR + virtual_name};
483 
484  if (IsDirectory(new_path)) {
485  if (recursion == 0)
486  return false;
487  return DeleteDirRecursively(new_path, recursion - 1);
488  }
489  return Delete(new_path);
490  }};
491 
492  if (!ForeachDirectoryEntry(nullptr, directory, callback))
493  return false;
494 
495  // Delete the outermost directory
496  FileUtil::DeleteDir(directory);
497  return true;
498 }

500 // Create directory and copy contents (does not overwrite existing files)
501 void CopyDir(const std::string& source_path, const std::string& dest_path) {
502 #ifndef _WIN32
503  if (source_path == dest_path)
504  return;
505  if (!FileUtil::Exists(source_path))
506  return;
507  if (!FileUtil::Exists(dest_path))
508  FileUtil::CreateFullPath(dest_path);
509 
510  DIR* dirp{opendir(source_path.c_str())};
511  if (!dirp)
512  return;
513 
514  while (struct dirent* result = readdir(dirp)) {
515  const std::string virtualName(result->d_name);
516  // check for "." and ".."
517  if (((virtualName[0] == '.') && (virtualName[1] == '\0')) ||
518  ((virtualName[0] == '.') && (virtualName[1] == '.') && (virtualName[2] == '\0')))
519  continue;
520 
521  std::string source{}, dest{};
522  source = source_path + virtualName;
523  dest = dest_path + virtualName;
524  if (IsDirectory(source)) {
525  source += '/';
526  dest += '/';
527  if (!FileUtil::Exists(dest))
528  FileUtil::CreateFullPath(dest);
529  CopyDir(source, dest);
530  } else if (!FileUtil::Exists(dest))
531  FileUtil::Copy(source, dest);
532  }
533  closedir(dirp);
534 #endif
535 }

617 // Returns a string with a Citra data dir or file in the user's home
618 // directory. To be used in "multi-user" mode (that is, installed).
619 const std::string& GetUserPath(const unsigned int DirIDX, const std::string& newPath) {
620  static std::string paths[NUM_PATH_INDICES];
621 
622  // Set up all paths and files on the first run
623  if (paths[D_USER_IDX].empty()) {
624 #ifdef _WIN32
625  paths[D_USER_IDX] = GetExeDirectory() + DIR_SEP USERDATA_DIR DIR_SEP;
626  if (!FileUtil::IsDirectory(paths[D_USER_IDX])) {
627  paths[D_USER_IDX] = AppDataRoamingDirectory() + DIR_SEP EMU_DATA_DIR DIR_SEP;
628  } else {
629  LOG_INFO(Common_Filesystem, "Using the local user directory");
630  }
631 #else
632  if (FileUtil::Exists(ROOT_DIR DIR_SEP USERDATA_DIR)) {
633  paths[D_USER_IDX] = ROOT_DIR DIR_SEP USERDATA_DIR DIR_SEP;
634  } else {
635  std::string data_dir{GetDataDirectory()};
636  paths[D_USER_IDX] = data_dir + DIR_SEP EMU_DATA_DIR DIR_SEP;
637  }
638 #endif
639 
640  paths[D_CONFIG_IDX] = paths[D_USER_IDX] + CONFIG_DIR DIR_SEP;
641  paths[D_SDMC_IDX] = paths[D_USER_IDX] + SDMC_DIR DIR_SEP;
642  paths[D_NAND_IDX] = paths[D_USER_IDX] + NAND_DIR DIR_SEP;
643  paths[D_SYSDATA_IDX] = paths[D_USER_IDX] + SYSDATA_DIR DIR_SEP;
644  paths[D_LOGS_IDX] = paths[D_USER_IDX] + LOG_DIR DIR_SEP;
645  }
646 
647  if (!newPath.empty()) {
648  if (!FileUtil::IsDirectory(newPath)) {
649  LOG_ERROR(Common_Filesystem, "Invalid path specified {}", newPath);
650  return paths[DirIDX];
651  } else {
652  paths[DirIDX] = newPath;
653  }
654 
655  switch (DirIDX) {
656  case D_ROOT_IDX:
657  paths[D_USER_IDX] = paths[D_ROOT_IDX] + DIR_SEP;
658  break;
659 
660  case D_USER_IDX:
661  paths[D_USER_IDX] = paths[D_ROOT_IDX] + DIR_SEP;
662  paths[D_CONFIG_IDX] = paths[D_USER_IDX] + CONFIG_DIR DIR_SEP;
663  paths[D_SDMC_IDX] = paths[D_USER_IDX] + SDMC_DIR DIR_SEP;
664  paths[D_NAND_IDX] = paths[D_USER_IDX] + NAND_DIR DIR_SEP;
665  break;
666  }
667  }
668 
669  return paths[DirIDX];
670 }

693 void SplitFilename83(const std::string& filename, std::array<char, 9>& short_name,
694  std::array<char, 4>& extension) {
695  const std::string forbidden_characters{".\"/\\[]:;=, "};
696 
697  // On a FAT32 partition, 8.3 names are stored as a 11 bytes array, filled with spaces.
698  short_name = {{' ', ' ', ' ', ' ', ' ', ' ', ' ', ' ', '\0'}};
699  extension = {{' ', ' ', ' ', '\0'}};
700 
701  std::string::size_type point{filename.rfind('.')};
702  if (point == filename.size() - 1)
703  point = filename.rfind('.', point);
704 
705  // Get short name.
706  int j{};
707  for (char letter : filename.substr(0, point)) {
708  if (forbidden_characters.find(letter, 0) != std::string::npos)
709  continue;
710  if (j == 8) {
711  // TODO(Link Mauve): also do that for filenames containing a space.
712  // TODO(Link Mauve): handle multiple files having the same short name.
713  short_name[6] = '~';
714  short_name[7] = '1';
715  break;
716  }
717  short_name[j++] = toupper(letter);
718  }
719 
720  // Get extension.
721  if (point != std::string::npos) {
722  j = 0;
723  for (char letter : filename.substr(point + 1, 3))
724  extension[j++] = toupper(letter);
725  }
726 }
