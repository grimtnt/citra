27 ValidationError ValidateFilters(const SoftwareKeyboardConfig& config, const std::string& input) {
28  if ((config.filter_flags & SwkbdFilter_Digits) == SwkbdFilter_Digits) {
29  if (std::any_of(input.begin(), input.end(),
30  [](unsigned char c) { return std::isdigit(c); })) {
31  return ValidationError::DigitNotAllowed;
32  }
33  }
34  if ((config.filter_flags & SwkbdFilter_At) == SwkbdFilter_At) {
35  if (input.find('@') != std::string::npos) {
36  return ValidationError::AtSignNotAllowed;
37  }
38  }
39  if ((config.filter_flags & SwkbdFilter_Percent) == SwkbdFilter_Percent) {
40  if (input.find('%') != std::string::npos) {
41  return ValidationError::PercentNotAllowed;
42  }
43  }
44  if ((config.filter_flags & SwkbdFilter_Backslash) == SwkbdFilter_Backslash) {
45  if (input.find('\\') != std::string::npos) {
46  return ValidationError::BackslashNotAllowed;
47  }
48  }
49  if ((config.filter_flags & SwkbdFilter_Profanity) == SwkbdFilter_Profanity) {
50  // TODO: check the profanity filter
51  LOG_INFO(Applet_Swkbd, "App requested swkbd profanity filter, but its not implemented.");
52  }
53  if ((config.filter_flags & SwkbdFilter_Callback) == SwkbdFilter_Callback) {
54  // TODO: check the callback
55  LOG_INFO(Applet_Swkbd, "App requested a swkbd callback, but its not implemented.");
56  }
57  return ValidationError::None;
58 }

60 ValidationError ValidateInput(const SoftwareKeyboardConfig& config, const std::string& input) {
61  ValidationError error;
62  if ((error = ValidateFilters(config, input)) != ValidationError::None) {
63  return error;
64  }
65 
66  // TODO(jroweboy): Is max_text_length inclusive or exclusive?
67  if (input.size() > config.max_text_length) {
68  return ValidationError::MaxLengthExceeded;
69  }
70 
71  if (!config.multiline && (input.find('\n') != std::string::npos)) {
72  return ValidationError::NewLineNotAllowed;
73  }
74  auto is_blank = [&] {
75  return std::all_of(input.begin(), input.end(),
76  [](unsigned char c) { return std::isspace(c); });
77  };
78  auto is_empty = [&] { return input.empty(); };
79  switch (config.valid_input) {
80  case SwkbdValidInput::FixedLen:
81  if (input.size() != config.max_text_length) {
82  return ValidationError::FixedLengthRequired;
83  }
84  break;
85  case SwkbdValidInput::NotEmptyNotBlank:
86  if (is_blank()) {
87  return ValidationError::BlankInputNotAllowed;
88  }
89  if (is_empty()) {
90  return ValidationError::EmptyInputNotAllowed;
91  }
92  break;
93  case SwkbdValidInput::NotBlank:
94  if (is_blank()) {
95  return ValidationError::BlankInputNotAllowed;
96  }
97  break;
98  case SwkbdValidInput::NotEmpty:
99  if (is_empty()) {
100  return ValidationError::EmptyInputNotAllowed;
101  }
102  break;
103  case SwkbdValidInput::Anything:
104  break;
105  default:
106  // TODO(jroweboy): What does hardware do in this case?
107  LOG_CRITICAL(Frontend, "Application requested unknown validation method. Method: {}",
108  static_cast<u32>(config.valid_input));
109  UNREACHABLE();
110  }
111 
112  switch (config.type) {
113  case SwkbdType::QWERTY:
114  case SwkbdType::Western:
115  case SwkbdType::Normal:
116  return ValidationError::None;
117  case SwkbdType::Numpad:
118  return std::all_of(input.begin(), input.end(), [](const char c) { return std::isdigit(c); })
119  ? ValidationError::None
120  : ValidationError::InputNotNumber;
121  default:
122  return ValidationError::None;
123  }
124 }

126 ValidationError ValidateButton(const SoftwareKeyboardConfig& config, u8 button) {
127  switch (config.num_buttons_m1) {
128  case SwkbdButtonConfig::NoButton:
129  return ValidationError::None;
130  case SwkbdButtonConfig::SingleButton:
131  if (button != 0) {
132  return ValidationError::ButtonOutOfRange;
133  }
134  break;
135  case SwkbdButtonConfig::DualButton:
136  if (button > 1) {
137  return ValidationError::ButtonOutOfRange;
138  }
139  break;
140  case SwkbdButtonConfig::TripleButton:
141  if (button > 2) {
142  return ValidationError::ButtonOutOfRange;
143  }
144  break;
145  default:
146  UNREACHABLE();
147  }
148  return ValidationError::None;
149 }

151 ResultCode SoftwareKeyboard::ReceiveParameter(Service::APT::MessageParameter const& parameter) {
152  if (parameter.signal != Service::APT::SignalType::Request) {
153  LOG_ERROR(Applet_Swkbd, "unsupported signal {}", static_cast<u32>(parameter.signal));
154  UNIMPLEMENTED();
155  // TODO(Subv): Find the right error code
156  return ResultCode(-1);
157  }
158 
159  // The LibAppJustStarted message contains a buffer with the size of the framebuffer shared
160  // memory.
161  // Create the SharedMemory that will hold the framebuffer data
162  Service::APT::CaptureBufferInfo capture_info{};
163  ASSERT(sizeof(capture_info) == parameter.buffer.size());
164 
165  memcpy(&capture_info, parameter.buffer.data(), sizeof(capture_info));
166 
167  using Kernel::MemoryPermission;
168  // Allocate a heap block of the required size for this applet.
169  heap_memory = std::make_shared<std::vector<u8>>(capture_info.size);
170  // Create a SharedMemory that directly points to this heap block.
171  framebuffer_memory = Kernel::SharedMemory::CreateForApplet(
172  heap_memory, 0, static_cast<u32>(heap_memory->size()), MemoryPermission::ReadWrite,
173  MemoryPermission::ReadWrite, "SoftwareKeyboard Memory");
174 
175  // Send the response message with the newly created SharedMemory
176  Service::APT::MessageParameter result{};
177  result.signal = Service::APT::SignalType::Response;
178  result.buffer.clear();
179  result.destination_id = Service::APT::AppletId::Application;
180  result.sender_id = id;
181  result.object = framebuffer_memory;
182 
183  SendParameter(result);
184  return RESULT_SUCCESS;
185 }

202 void SoftwareKeyboard::Update() {
203  switch (Settings::values.keyboard_mode) {
204  case Settings::KeyboardMode::StdIn: {
205  std::string input{};
206  std::cout << "Software Keyboard" << std::endl;
207  // Display hint text
208  std::u16string hint{reinterpret_cast<char16_t*>(config.hint_text)};
209  if (!hint.empty()) {
210  std::cout << "Hint text: " << Common::UTF16ToUTF8(hint) << std::endl;
211  }
212  ValidationError error{ValidationError::ButtonOutOfRange};
213  auto ValidateInputString = [&]() -> bool {
214  ValidationError error{ValidateInput(config, input)};
215  if (error != ValidationError::None) {
216  switch (error) {
217  case ValidationError::AtSignNotAllowed:
218  std::cout << "Input must not contain the @ symbol" << std::endl;
219  break;
220  case ValidationError::BackslashNotAllowed:
221  std::cout << "Input must not contain the \\ symbol" << std::endl;
222  break;
223  case ValidationError::BlankInputNotAllowed:
224  std::cout << "Input must not be blank." << std::endl;
225  break;
226  case ValidationError::CallbackFailed:
227  std::cout << "Callbak failed." << std::endl;
228  break;
229  case ValidationError::DigitNotAllowed:
230  std::cout << "Input must not contain any digits" << std::endl;
231  break;
232  case ValidationError::EmptyInputNotAllowed:
233  std::cout << "Input must not be empty." << std::endl;
234  break;
235  case ValidationError::FixedLengthRequired:
236  std::cout << fmt::format("Input must be exactly {} characters.",
237  config.max_text_length)
238  << std::endl;
239  break;
240  case ValidationError::InputNotNumber:
241  std::cout << "All characters must be numbers." << std::endl;
242  break;
243  case ValidationError::MaxLengthExceeded:
244  std::cout << fmt::format("Input is longer than the maximum length. Max: {}",
245  config.max_text_length)
246  << std::endl;
247  break;
248  case ValidationError::PercentNotAllowed:
249  std::cout << "Input must not contain the % symbol" << std::endl;
250  break;
251  default:
252  UNREACHABLE();
253  }
254  }
255  return error == ValidationError::None;
256  };
257  do {
258  std::cout << "Enter the text you will send to the application:" << std::endl;
259  std::getline(std::cin, input);
260  } while (!ValidateInputString());
261 
262  std::string option_text{};
263  // convert all of the button texts into something we can output
264  // num_buttons is in the range of 0-2 so use <= instead of <
265  u32 num_buttons{static_cast<u32>(config.num_buttons_m1)};
266  for (u32 i{}; i <= num_buttons; ++i) {
267  std::string button_text{};
268  // apps are allowed to set custom text to display on the button
269  std::u16string custom_button_text{reinterpret_cast<char16_t*>(config.button_text[i])};
270  if (custom_button_text.empty()) {
271  // Use the system default text for that button
272  button_text = default_button_text[num_buttons][i];
273  } else {
274  button_text = Common::UTF16ToUTF8(custom_button_text);
275  }
276  option_text += "\t(" + std::to_string(i) + ") " + button_text + "\t";
277  }
278  std::string option{};
279  error = ValidationError::ButtonOutOfRange;
280  auto ValidateButtonString{[&]() -> bool {
281  bool valid{};
282  try {
283  u32 num{static_cast<u32>(std::stoul(option))};
284  valid = ValidateButton(config, static_cast<u8>(num)) == ValidationError::None;
285  if (!valid) {
286  std::cout << fmt::format("Please choose a number between 0 and {}",
287  static_cast<u32>(config.num_buttons_m1))
288  << std::endl;
289  }
290  } catch (const std::invalid_argument&) {
291  std::cout << "Unable to parse input as a number." << std::endl;
292  } catch (const std::out_of_range&) {
293  std::cout << "Input number is not valid." << std::endl;
294  }
295  return valid;
296  }};
297  do {
298  std::cout << "\nPlease type the number of the button you will press: \n"
299  << option_text << std::endl;
300  std::getline(std::cin, option);
301  } while (!ValidateButtonString());
302 
303  s32 button{static_cast<s32>(std::stol(option))};
304  switch (config.num_buttons_m1) {
305  case SwkbdButtonConfig::SingleButton:
306  config.return_code = SwkbdResult::D0Click;
307  break;
308  case SwkbdButtonConfig::DualButton:
309  if (button == 0) {
310  config.return_code = SwkbdResult::D1Click0;
311  } else {
312  config.return_code = SwkbdResult::D1Click1;
313  }
314  break;
315  case SwkbdButtonConfig::TripleButton:
316  if (button == 0) {
317  config.return_code = SwkbdResult::D2Click0;
318  } else if (button == 1) {
319  config.return_code = SwkbdResult::D2Click1;
320  } else {
321  config.return_code = SwkbdResult::D2Click2;
322  }
323  break;
324  default:
325  // TODO: what does the hardware do
326  LOG_WARNING(Applet_Swkbd, "Unknown option for num_buttons_m1: {}",
327  static_cast<u32>(config.num_buttons_m1));
328  config.return_code = SwkbdResult::None;
329  break;
330  }
331 
332  std::u16string utf16_input = Common::UTF8ToUTF16(input);
333  memcpy(text_memory->GetPointer(), utf16_input.c_str(),
334  utf16_input.length() * sizeof(char16_t));
335  config.text_length = static_cast<u16>(utf16_input.size());
336  config.text_offset = 0;
337  Finalize();
338  break;
339  }
340  case Settings::KeyboardMode::Qt: {
341  auto cb{Core::System::GetInstance().GetQtCallbacks().swkbd};
342  if (!cb)
343  UNREACHABLE_MSG("Qt keyboard callback is nullptr");
344  std::u16string text{};
345  cb(config, text);
346  memcpy(text_memory->GetPointer(), text.c_str(), text.length() * sizeof(char16_t));
347  Finalize();
348  break;
349  }
350  default:
351  UNREACHABLE_MSG("Unimplemented keyboard mode {}",
352  static_cast<int>(Settings::values.keyboard_mode));
353  break;
354  }
355 }
