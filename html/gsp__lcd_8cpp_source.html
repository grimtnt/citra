17 void GSP_LCD::SetBrightnessRaw(Kernel::HLERequestContext& ctx) {
18  IPC::RequestParser rp{ctx, 0xA, 2, 0};
19  u32 screen{rp.Pop<u32>()};
20  u32 brightness{std::clamp(rp.Pop<u32>(), MIN_BRIGHTNESS_RAW, MAX_BRIGHTNESS_RAW)};

30 #endif
