103  container.LoadTitleMetadata(data, container.GetTitleMetadataOffset());
104  FileSys::TitleMetadata tmd = container.GetTitleMetadata();
105  tmd.Print();

107  // If a TMD already exists for this app (ie 00000000.tmd), the incoming TMD
108  // will be the same plus one, (ie 00000001.tmd), both will be kept until
109  // the install is finalized and old contents can be discarded.
110  if (FileUtil::Exists(GetTitleMetadataPath(media_type, tmd.GetTitleID())))
111  is_update = true;
112 
113  std::string tmd_path = GetTitleMetadataPath(media_type, tmd.GetTitleID(), is_update);
114

116  std::string tmd_folder;
117  Common::SplitPath(tmd_path, &tmd_folder, nullptr, nullptr);
118  FileUtil::CreateFullPath(tmd_folder);
119 
120  // Save TMD so that we can start getting new .app paths
121  if (tmd.Save(tmd_path) != Loader::ResultStatus::Success)
122  return FileSys::ERROR_INSUFFICIENT_SPACE;
123 
124  // Create any other .app folders which may not exist yet
125  std::string app_folder;
126  Common::SplitPath(GetTitleContentPath(media_type, tmd.GetTitleID(),
127  FileSys::TMDContentIndex::Main, is_update),
128  &app_folder, nullptr, nullptr);

131  auto content_count{container.GetTitleMetadata().GetContentCount()};

134  if (title_key) {

148 ResultVal<std::size_t> CIAFile::WriteContentData(u64 offset, std::size_t length, const u8* buffer) {
149  // Data is not being buffered, so we have to keep track of how much of each <ID>.app
150  // has been written since we might get a written buffer which contains multiple .app
151  // contents or only part of a larger .app's contents.
152  u64 offset_max = offset + length;
153  for (int i{}; i < container.GetTitleMetadata().GetContentCount(); i++) {
154  if (content_written[i] < container.GetContentSize(i)) {
155  // The size, minimum unwritten offset, and maximum unwritten offset of this content
156  u64 size{container.GetContentSize(i)};
157  u64 range_min{container.GetContentOffset(i) + content_written[i]};
158  u64 range_max{container.GetContentOffset(i) + size};
159 
160  // The unwritten range for this content is beyond the buffered data we have
161  // or comes before the buffered data we have, so skip this content ID.
162  if (range_min > offset_max || range_max < offset)
163  continue;
164 
165  // Figure out how much of this content ID we have just recieved/can write out
166  u64 available_to_write{std::min(offset_max, range_max) - range_min};
167 
168  // Since the incoming TMD has already been written, we can use GetTitleContentPath
169  // to get the content paths to write to.
170  FileSys::TitleMetadata tmd{container.GetTitleMetadata()};
171  FileUtil::IOFile file{GetTitleContentPath(media_type, tmd.GetTitleID(), i, is_update),
172  content_written[i] ? "ab" : "wb"};
173 
174  if (!file.IsOpen())
175  return FileSys::ERROR_INSUFFICIENT_SPACE;
176 
177  std::vector<u8> temp(buffer + (range_min - offset),
178  buffer + (range_min - offset) + available_to_write);
179  if (tmd.GetContentTypeByIndex(static_cast<u16>(i)) &
180  FileSys::TMDContentTypeFlag::Encrypted) {
181  decryption_state->content[i].ProcessData(temp.data(), temp.data(), temp.size());
182  }
183  file.WriteBytes(temp.data(), temp.size());
184 
185  // Keep tabs on how much of this content ID has been written so new range_min
186  // values can be calculated.
187  content_written[i] += available_to_write;
188  LOG_DEBUG(Service_AM, "Wrote {:x} to content {}, total {:x}", available_to_write, i,
189  content_written[i]);
190  }
191  }
192 
193  return MakeResult<std::size_t>(length);
194 }

196 ResultVal<std::size_t> CIAFile::Write(u64 offset, std::size_t length, bool flush,
197  const u8* buffer) {
198  written += length;
199 
200  // TODO(shinyquagsire23): Can we assume that things will only be written in sequence?
201  // Does AM send an error if we write to things out of order?
202  // Or does it just ignore offsets and assume a set sequence of incoming data?
203 
204  // The data in CIAs is always stored CIA Header > Cert > Ticket > TMD > Content > Meta.
205  // The CIA Header describes Cert, Ticket, TMD, total content sizes, and TMD is needed for
206  // content sizes so it ends up becoming a problem of keeping track of how much has been
207  // written and what we have been able to pick up.
208  if (install_state == CIAInstallState::InstallStarted) {
209  std::size_t buf_copy_size{std::min(length, FileSys::CIA_HEADER_SIZE)};
210  std::size_t buf_max_size{
211  std::min(static_cast<std::size_t>(offset + length), FileSys::CIA_HEADER_SIZE)};
212  data.resize(buf_max_size);
213  memcpy(data.data() + offset, buffer, buf_copy_size);
214 
215  // We have enough data to load a CIA header and parse it.
216  if (written >= FileSys::CIA_HEADER_SIZE) {
217  container.LoadHeader(data);
218  container.Print();
219  install_state = CIAInstallState::HeaderLoaded;
220  }
221  }
222 
223  // If we don't have a header yet, we can't pull offsets of other sections
224  if (install_state == CIAInstallState::InstallStarted)
225  return MakeResult<std::size_t>(length);
226 
227  // If we have been given data before (or including) .app content, pull it into
228  // our buffer, but only pull *up to* the content offset, no further.
229  if (offset < container.GetContentOffset()) {
230  std::size_t buf_loaded{data.size()};
231  std::size_t copy_offset{std::max(static_cast<std::size_t>(offset), buf_loaded)};
232  std::size_t buf_offset{buf_loaded - offset};
233  std::size_t buf_copy_size{
234  std::min(length, static_cast<std::size_t>(container.GetContentOffset() - offset)) -
235  buf_loaded};
236  std::size_t buf_max_size{std::min(offset + length, container.GetContentOffset())};
237  data.resize(buf_max_size);
238  memcpy(data.data() + copy_offset, buffer + buf_offset, buf_copy_size);
239  }
240 
241  // TODO(shinyquagsire23): Write out .tik files to nand?
242 
243  // The end of our TMD is at the beginning of Content data, so ensure we have that much
244  // buffered before trying to parse.
245  if (written >= container.GetContentOffset() && install_state != CIAInstallState::TMDLoaded) {
246  auto result{WriteTicket()};
247  if (result.IsError())
248  return result;
249  result = WriteTitleMetadata();
250  if (result.IsError())
251  return result;
252  }
253 
254  // Content data sizes can only be retrieved from TMD data
255  if (install_state != CIAInstallState::TMDLoaded)
256  return MakeResult<std::size_t>(length);
257 
258  // From this point forward, data will no longer be buffered in data
259  auto result{WriteContentData(offset, length, buffer)};
260  if (result.Failed())
261  return result;
262 
263  return MakeResult<std::size_t>(length);
264 }

274 bool CIAFile::Close() const {
275  bool complete{true};
276  for (std::size_t i{}; i < container.GetTitleMetadata().GetContentCount(); i++) {
277  if (content_written[i] < container.GetContentSize(static_cast<u16>(i)))
278  complete = false;
279  }
280 
281  // Install aborted
282  if (!complete) {
283  LOG_ERROR(Service_AM, "CIAFile closed prematurely, aborting install...");
284  FileUtil::DeleteDir(GetTitlePath(media_type, container.GetTitleMetadata().GetTitleID()));
285  return true;
286  }
287 
288  // Clean up older content data if we installed newer content on top
289  std::string old_tmd_path{
290  GetTitleMetadataPath(media_type, container.GetTitleMetadata().GetTitleID(), false)};
291  std::string new_tmd_path{
292  GetTitleMetadataPath(media_type, container.GetTitleMetadata().GetTitleID(), true)};
293  if (FileUtil::Exists(new_tmd_path) && old_tmd_path != new_tmd_path) {
294  FileSys::TitleMetadata old_tmd{};
295  FileSys::TitleMetadata new_tmd{};
296 
297  old_tmd.Load(old_tmd_path);
298  new_tmd.Load(new_tmd_path);
299 
300  // For each content ID in the old TMD, check if there is a matching ID in the new
301  // TMD. If a CIA contains (and wrote to) an identical ID, it should be kept while
302  // IDs which only existed for the old TMD should be deleted.
303  for (u16 old_index = 0; old_index < old_tmd.GetContentCount(); old_index++) {
304  bool abort{};
305  for (u16 new_index = 0; new_index < new_tmd.GetContentCount(); new_index++) {
306  if (old_tmd.GetContentIDByIndex(old_index) ==
307  new_tmd.GetContentIDByIndex(new_index)) {
308  abort = true;
309  }
310  }
311  if (abort)
312  break;
313 
314  FileUtil::Delete(GetTitleContentPath(media_type, old_tmd.GetTitleID(), old_index));
315  }
316 
317  FileUtil::Delete(old_tmd_path);
318  }
319  return true;
320 }

324 InstallStatus InstallCIA(const std::string& path,
325  std::function<ProgressCallback>&& update_callback) {
326  LOG_INFO(Service_AM, "Installing {}...", path);
327 
328  if (!FileUtil::Exists(path)) {
329  LOG_ERROR(Service_AM, "File {} does not exist!", path);
330  return InstallStatus::ErrorFileNotFound;
331  }
332 
333  FileSys::CIAContainer container{};
334  if (container.Load(path) == Loader::ResultStatus::Success) {
335  Service::AM::CIAFile installFile{
336  Service::AM::GetTitleMediaType(container.GetTitleMetadata().GetTitleID())};
337 
338  bool title_key_available{container.GetTicket().GetTitleKey().is_initialized()};
339 
340  for (std::size_t i{}; i < container.GetTitleMetadata().GetContentCount(); i++) {
341  if ((container.GetTitleMetadata().GetContentTypeByIndex(static_cast<u16>(i)) &
342  FileSys::TMDContentTypeFlag::Encrypted) &&
343  !title_key_available) {
344  LOG_ERROR(Service_AM, "File {} is encrypted! Aborting...", path);
345  return InstallStatus::ErrorEncrypted;
346  }
347  }
348 
349  FileUtil::IOFile file{path, "rb"};
350  if (!file.IsOpen())
351  return InstallStatus::ErrorFailedToOpenFile;
352 
353  std::array<u8, 0x10000> buffer{};
354  std::size_t total_bytes_read{};
355  while (total_bytes_read != file.GetSize()) {
356  std::size_t bytes_read{file.ReadBytes(buffer.data(), buffer.size())};
357  auto result{installFile.Write(static_cast<u64>(total_bytes_read), bytes_read, true,
358  static_cast<u8*>(buffer.data()))};
359 
360  if (update_callback)
361  update_callback(total_bytes_read, file.GetSize());
362  if (result.Failed()) {
363  LOG_ERROR(Service_AM, "CIA file installation aborted with error code {:08x}",
364  result.Code().raw);
365  return InstallStatus::ErrorAborted;
366  }
367  total_bytes_read += bytes_read;
368  }
369  installFile.Close();
370 
371  LOG_INFO(Service_AM, "Installed {} successfully.", path);
372  return InstallStatus::Success;
373  }
374 
375  LOG_ERROR(Service_AM, "CIA file {} is invalid!", path);
376  return InstallStatus::ErrorInvalid;
377 }

393 std::string GetTitleMetadataPath(Service::FS::MediaType media_type, u64 tid, bool update) {
394  std::string content_path{GetTitlePath(media_type, tid) + "content/"};
395 
396  if (media_type == Service::FS::MediaType::GameCard) {
397  LOG_ERROR(Service_AM, "Invalid request for nonexistent gamecard title metadata!");
398  return "";
399  }
400 
401  // The TMD ID is usually held in the title databases, which we don't implement.
402  // For now, just scan for any .tmd files which exist, the smallest will be the
403  // base ID and the largest will be the (currently installing) update ID.
404  constexpr u32 MAX_TMD_ID{0xFFFFFFFF};
405  u32 base_id{MAX_TMD_ID};
406  u32 update_id{};
407  FileUtil::FSTEntry entries{};
408  FileUtil::ScanDirectoryTree(content_path, entries);
409  for (const FileUtil::FSTEntry& entry : entries.children) {
410  std::string filename_filename{}, filename_extension{};
411  Common::SplitPath(entry.virtualName, nullptr, &filename_filename, &filename_extension);
412 
413  if (filename_extension == ".tmd") {
414  u32 id{static_cast<u32>(std::stoul(filename_filename.c_str(), nullptr, 16))};
415  base_id = std::min(base_id, id);
416  update_id = std::max(update_id, id);
417  }
418  }
419 
420  // If we didn't find anything, default to 00000000.tmd for it to be created.
421  if (base_id == MAX_TMD_ID)
422  base_id = 0;
423 
424  // Update ID should be one more than the last, if it hasn't been created yet.
425  if (base_id == update_id)
426  update_id++;
427 
428  return content_path + fmt::format("{:08x}.tmd", (update ? update_id : base_id));
429 }
