162 inline DescriptorType GetDescriptorType(u32 descriptor) {
163  // Note: Those checks must be done in this order
164  if (IsHandleDescriptor(descriptor))
165  return (DescriptorType)(descriptor & 0x30);
166 
167  // handle the fact that the following descriptors can have rights
168  if (descriptor & MappedBuffer)
169  return MappedBuffer;
170 
171  if (descriptor & PXIBuffer)
172  return PXIBuffer;
173 
174  return StaticBuffer;
175 }
