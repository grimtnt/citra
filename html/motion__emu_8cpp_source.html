36  void BeginTilt(int x, int y) {
37  mouse_origin = Math::MakeVec(x, y);
38  is_tilting = true;
39  }

41  void Tilt(int x, int y) {
42  auto mouse_move{Math::MakeVec(x, y) - mouse_origin};
43  if (is_tilting) {
44  std::lock_guard<std::mutex> guard(tilt_mutex);
45  if (mouse_move.x == 0 && mouse_move.y == 0) {
46  tilt_angle = 0;
47  } else {
48  tilt_direction = mouse_move.Cast<float>();
49  tilt_angle = std::clamp(tilt_direction.Normalize() * sensitivity, 0.0f,
50  MathUtil::PI * this->tilt_clamp / 180.0f);
51  }
52  }
53  }
54 
55  void EndTilt() {
56  std::lock_guard<std::mutex> guard(tilt_mutex);
57  tilt_angle = 0;
58  is_tilting = false;
59  }

91  void MotionEmuThread() {
92  auto update_time{std::chrono::steady_clock::now()};
93  Math::Quaternion<float> q{MakeQuaternion(Math::Vec3<float>(), 0)};
94  Math::Quaternion<float> old_q{};
95 
96  while (!shutdown_event.WaitUntil(update_time)) {
97  update_time += update_duration;
98  old_q = q;
99 
100  {
101  std::lock_guard<std::mutex> guard(tilt_mutex);
102 
103  // Find the quaternion describing current 3DS tilting
104  q = MakeQuaternion(Math::MakeVec(-tilt_direction.y, 0.0f, tilt_direction.x),
105  tilt_angle);
106  }
107 
108  auto inv_q{q.Inverse()};
109 
110  // Set the gravity vector in world space
111  auto gravity{Math::MakeVec(0.0f, -1.0f, 0.0f)};
112 
113  // Find the angular rate vector in world space
114  auto angular_rate{((q - old_q) * inv_q).xyz * 2};
115  angular_rate *= 1000 / update_millisecond / MathUtil::PI * 180;
116 
117  // Transform the two vectors from world space to 3DS space
118  gravity = QuaternionRotate(inv_q, gravity);
119  angular_rate = QuaternionRotate(inv_q, angular_rate);
120 
121  // Update the sensor state
122  {
123  std::lock_guard<std::mutex> guard(status_mutex);
124  status = std::make_tuple(gravity, angular_rate);
125  }
126  }
127  }
128 };
