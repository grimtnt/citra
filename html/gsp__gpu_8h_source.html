159 struct CommandBuffer {
160  union {
161  u32 hex;
162 
163  // Current command index. This index is updated by GSP module after loading the command
164  // data, right before the command is processed. When this index is updated by GSP module,
165  // the total commands field is decreased by one as well.
166  BitField<0, 8, u32> index;
167 
168  // Total commands to process, must not be value 0 when GSP module handles commands. This
169  // must be <=15 when writing a command to shared memory. This is incremented by the
170  // application when writing a command to shared memory, after increasing this value
171  // TriggerCmdReqQueue is only used if this field is value 1.
172  BitField<8, 8, u32> number_commands;
173  };
174 
175  u32 unk[7];
176 
177  Command commands[0xF];
178 };
