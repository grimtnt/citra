172  template <typename T>
173  std::size_t ReadArray(T* data, std::size_t length) {
174  static_assert(std::is_standard_layout<T>(),
175  "Given array does not consist of standard layout objects");
176 #if (__GNUC__ >= 5) || defined(__clang__) || defined(_MSC_VER)
177  static_assert(std::is_trivially_copyable<T>(),
178  "Given array does not consist of trivially copyable objects");
179 #endif
180 
181  if (!IsOpen()) {
182  m_good = false;
183  return std::numeric_limits<std::size_t>::max();
184  }
185 
186  std::size_t items_read = std::fread(data, sizeof(T), length, m_file);
187  if (items_read != length)
188  m_good = false;
189 
190  return items_read;
191  }
192 
193  template <typename T>
194  std::size_t WriteArray(const T* data, std::size_t length) {
195  static_assert(std::is_standard_layout<T>(),
196  "Given array does not consist of standard layout objects");
197 #if (__GNUC__ >= 5) || defined(__clang__) || defined(_MSC_VER)
198  static_assert(std::is_trivially_copyable<T>(),
199  "Given array does not consist of trivially copyable objects");
200 #endif
201 
202  if (!IsOpen()) {
203  m_good = false;
204  return std::numeric_limits<std::size_t>::max();
205  }
206 
207  std::size_t items_written = std::fwrite(data, sizeof(T), length, m_file);
208  if (items_written != length)
209  m_good = false;
210 
211  return items_written;
212  }
