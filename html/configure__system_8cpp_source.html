241 void ConfigureSystem::setConfiguration() {
242  enabled = !Core::System::GetInstance().IsPoweredOn();
243 
244  ui->combo_init_clock->setCurrentIndex(static_cast<u8>(Settings::values.init_clock));
245  QDateTime date_time{};
246  date_time.setTime_t(Settings::values.init_time);
247  ui->edit_init_time->setDateTime(date_time);
248 
249  if (!enabled) {
250  cfg = Service::CFG::GetCurrentModule();
251  ReadSystemSettings();
252  ui->group_system_settings->setEnabled(false);
253  } else {
254  // This tab is enabled only when game is not running (i.e. all service are not initialized).
255  // Temporarily register archive types and load the config savegame file to memory.
256  Service::FS::RegisterArchiveTypes();
257  cfg = std::make_shared<Service::CFG::Module>();
258  Service::FS::UnregisterArchiveTypes();
259 
260  ReadSystemSettings();
261  ui->label_disable_info->hide();
262  }

304 void ConfigureSystem::applyConfiguration() {
305  if (!enabled)
306  return;
307 
308  bool modified{};
309 
310  // apply username
311  std::u16string new_username{ui->edit_username->text().toStdU16String()};
312  if (new_username != username) {
313  cfg->SetUsername(new_username);
314  modified = true;
315  }
316 
317  // apply birthday
318  int new_birthmonth{ui->combo_birthmonth->currentIndex() + 1};
319  int new_birthday{ui->combo_birthday->currentIndex() + 1};
320  if (birthmonth != new_birthmonth || birthday != new_birthday) {
321  cfg->SetBirthday(new_birthmonth, new_birthday);
322  modified = true;
323  }
324 
325  // apply language
326  int new_language{ui->combo_language->currentIndex()};
327  if (language_index != new_language) {
328  cfg->SetSystemLanguage(static_cast<Service::CFG::SystemLanguage>(new_language));
329  modified = true;
330  }
331 
332  // apply sound
333  int new_sound{ui->combo_sound->currentIndex()};
334  if (sound_index != new_sound) {
335  cfg->SetSoundOutputMode(static_cast<Service::CFG::SoundOutputMode>(new_sound));
336  modified = true;
337  }
338 
339  // apply model
340  int new_model{ui->combo_model->currentIndex()};
341  if (model_index != new_model) {
342  cfg->SetSystemModel(static_cast<Service::CFG::SystemModel>(new_model));
343  modified = true;
344  }
345 
346  // apply country
347  u8 new_country{static_cast<u8>(ui->combo_country->currentData().toInt())};
348  if (country_code != new_country) {
349  cfg->SetCountryCode(new_country);
350  modified = true;
351  }
352 
353  // update the config savegame if any item except region is modified.
354  if (modified)
355  cfg->UpdateConfigNANDSavegame();
356 
357  Settings::values.init_clock =
358  static_cast<Settings::InitClock>(ui->combo_init_clock->currentIndex());
359  Settings::values.init_time = ui->edit_init_time->dateTime().toTime_t();
360  Settings::values.region_value = ui->region_combobox->currentIndex() - 1;
361  Settings::Apply();
362 }

364 void ConfigureSystem::updateBirthdayComboBox(int birthmonth_index) {
365  if (birthmonth_index < 0 || birthmonth_index >= 12)
366  return;
367 
368  // store current day selection
369  int birthday_index{ui->combo_birthday->currentIndex()};
370 
371  // get number of days in the new selected month
372  int days{days_in_month[birthmonth_index]};
373 
374  // if the selected day is out of range,
375  // reset it to 1st
376  if (birthday_index < 0 || birthday_index >= days)
377  birthday_index = 0;
378 
379  // update the day combo box
380  ui->combo_birthday->clear();
381  for (int i = 1; i <= days; ++i) {
382  ui->combo_birthday->addItem(QString::number(i));
383  }
384 
385  // restore the day selection
386  ui->combo_birthday->setCurrentIndex(birthday_index);
387 }
