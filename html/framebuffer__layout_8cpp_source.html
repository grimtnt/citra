23 // Finds the largest size subrectangle contained in window area that is confined to the aspect ratio
24 template <class T>
25 static MathUtil::Rectangle<T> maxRectangle(MathUtil::Rectangle<T> window_area,
26  float screen_aspect_ratio) {
27  float scale = std::min(static_cast<float>(window_area.GetWidth()),
28  window_area.GetHeight() / screen_aspect_ratio);
29  return MathUtil::Rectangle<T>{0, 0, static_cast<T>(std::round(scale)),
30  static_cast<T>(std::round(scale * screen_aspect_ratio))};
31 }

33 FramebufferLayout DefaultFrameLayout(unsigned width, unsigned height, bool swapped) {
34  ASSERT(width > 0);
35  ASSERT(height > 0);
36 
37  FramebufferLayout res{width, height, true, true, {}, {}};
38  // Default layout gives equal screen sizes to the top and bottom screen
39  MathUtil::Rectangle<unsigned> screen_window_area{0, 0, width, height / 2};
40  MathUtil::Rectangle<unsigned> top_screen{
41  maxRectangle(screen_window_area, TOP_SCREEN_ASPECT_RATIO)};
42  MathUtil::Rectangle<unsigned> bot_screen{
43  maxRectangle(screen_window_area, BOT_SCREEN_ASPECT_RATIO)};
44 
45  float window_aspect_ratio{static_cast<float>(height) / width};
46  // both screens height are taken into account by multiplying by 2
47  float emulation_aspect_ratio{TOP_SCREEN_ASPECT_RATIO * 2};
48 
49  if (window_aspect_ratio < emulation_aspect_ratio) {
50  // Apply borders to the left and right sides of the window.
51  top_screen =
52  top_screen.TranslateX((screen_window_area.GetWidth() - top_screen.GetWidth()) / 2);
53  bot_screen =
54  bot_screen.TranslateX((screen_window_area.GetWidth() - bot_screen.GetWidth()) / 2);
55  } else {
56  // Window is narrower than the emulation content => apply borders to the top and bottom
57  // Recalculate the bottom screen to account for the width difference between top and bottom
58  screen_window_area = {0, 0, width, top_screen.GetHeight()};
59  bot_screen = maxRectangle(screen_window_area, BOT_SCREEN_ASPECT_RATIO);
60  bot_screen = bot_screen.TranslateX((top_screen.GetWidth() - bot_screen.GetWidth()) / 2);
61  if (swapped) {
62  bot_screen = bot_screen.TranslateY(height / 2 - bot_screen.GetHeight());
63  } else {
64  top_screen = top_screen.TranslateY(height / 2 - top_screen.GetHeight());
65  }
66  }
67  // Move the top screen to the bottom if we are swapped.
68  res.top_screen = swapped ? top_screen.TranslateY(height / 2) : top_screen;
69  res.bottom_screen = swapped ? bot_screen : bot_screen.TranslateY(height / 2);
70  return res;
71 }

73 FramebufferLayout SingleFrameLayout(unsigned width, unsigned height, bool swapped) {
74  ASSERT(width > 0);
75  ASSERT(height > 0);
76  // The drawing code needs at least somewhat valid values for both screens
77  // so just calculate them both even if the other isn't showing.
78  FramebufferLayout res{width, height, !swapped, swapped, {}, {}};
79 
80  MathUtil::Rectangle<unsigned> screen_window_area{0, 0, width, height};
81  MathUtil::Rectangle<unsigned> top_screen{
82  maxRectangle(screen_window_area, TOP_SCREEN_ASPECT_RATIO)};
83  MathUtil::Rectangle<unsigned> bot_screen{
84  maxRectangle(screen_window_area, BOT_SCREEN_ASPECT_RATIO)};
85 
86  float window_aspect_ratio{static_cast<float>(height) / width};
87  float emulation_aspect_ratio{(swapped) ? BOT_SCREEN_ASPECT_RATIO : TOP_SCREEN_ASPECT_RATIO};
88 
89  if (window_aspect_ratio < emulation_aspect_ratio) {
90  top_screen =
91  top_screen.TranslateX((screen_window_area.GetWidth() - top_screen.GetWidth()) / 2);
92  bot_screen =
93  bot_screen.TranslateX((screen_window_area.GetWidth() - bot_screen.GetWidth()) / 2);
94  } else {
95  top_screen = top_screen.TranslateY((height - top_screen.GetHeight()) / 2);
96  bot_screen = bot_screen.TranslateY((height - bot_screen.GetHeight()) / 2);
97  }
98  res.top_screen = top_screen;
99  res.bottom_screen = bot_screen;

103 FramebufferLayout MediumFrameLayout(unsigned width, unsigned height, bool swapped) {
104  ASSERT(width > 0);
105  ASSERT(height > 0);
106 
107  FramebufferLayout res{width, height, true, true, {}, {}};
108  float window_aspect_ratio{static_cast<float>(height) / width};
109  float emulation_aspect_ratio{
110  swapped ? Core::kScreenBottomHeight * 4 /
111  (Core::kScreenBottomWidth * 6.0f + Core::kScreenTopWidth * 0.61f)
112  : Core::kScreenTopHeight * 4 /
113  (Core::kScreenTopWidth * 5.5f + Core::kScreenBottomWidth * 0.33f)};
114  float large_screen_aspect_ratio{swapped ? BOT_SCREEN_ASPECT_RATIO : TOP_SCREEN_ASPECT_RATIO};
115  float small_screen_aspect_ratio{swapped ? TOP_SCREEN_ASPECT_RATIO : BOT_SCREEN_ASPECT_RATIO};
116 
117  MathUtil::Rectangle<unsigned> screen_window_area{0, 0, width, height};
118  MathUtil::Rectangle<unsigned> total_rect{
119  maxRectangle(screen_window_area, emulation_aspect_ratio)};
120  MathUtil::Rectangle<unsigned> large_screen{maxRectangle(total_rect, large_screen_aspect_ratio)};
121  MathUtil::Rectangle<unsigned> fourth_size_rect{total_rect.Scale(.55f)};
122  MathUtil::Rectangle<unsigned> small_screen{
123  maxRectangle(fourth_size_rect, small_screen_aspect_ratio)};
124 
125  if (window_aspect_ratio < emulation_aspect_ratio) {
126  large_screen =
127  large_screen.TranslateX((screen_window_area.GetWidth() - total_rect.GetWidth()) / 2);
128  } else {
129  large_screen = large_screen.TranslateY((height - total_rect.GetHeight()) / 2);
130  }
131  // Shift the small screen to the bottom right corner
132  small_screen =
133  small_screen.TranslateX(large_screen.right)
134  .TranslateY(large_screen.GetHeight() + large_screen.top - small_screen.GetHeight());
135  res.top_screen = swapped ? small_screen : large_screen;
136  res.bottom_screen = swapped ? large_screen : small_screen;
137  return res;
138 }

141  ASSERT(width > 0);
142  ASSERT(height > 0);
143 
144  FramebufferLayout res{width, height, true, true, {}, {}};
145  // Split the window into two parts. Give 4x width to the main screen and 1x width to the small
146  // To do that, find the total emulation box and maximize that based on window size
147  float window_aspect_ratio{static_cast<float>(height) / width};
148  float emulation_aspect_ratio{
149  swapped ? Core::kScreenBottomHeight * 4 /
150  (Core::kScreenBottomWidth * 4.0f + Core::kScreenTopWidth)
151  : Core::kScreenTopHeight * 4 /
152  (Core::kScreenTopWidth * 4.0f + Core::kScreenBottomWidth)};
153  float large_screen_aspect_ratio{swapped ? BOT_SCREEN_ASPECT_RATIO : TOP_SCREEN_ASPECT_RATIO};
154  float small_screen_aspect_ratio{swapped ? TOP_SCREEN_ASPECT_RATIO : BOT_SCREEN_ASPECT_RATIO};
155 
156  MathUtil::Rectangle<unsigned> screen_window_area{0, 0, width, height};
157  MathUtil::Rectangle<unsigned> total_rect{
158  maxRectangle(screen_window_area, emulation_aspect_ratio)};
159  MathUtil::Rectangle<unsigned> large_screen{maxRectangle(total_rect, large_screen_aspect_ratio)};

179 FramebufferLayout SideFrameLayout(unsigned width, unsigned height, bool swapped) {
180  ASSERT(width > 0);
181  ASSERT(height > 0);
182 
183  FramebufferLayout res{width, height, true, true, {}, {}};
184  // Aspect ratio of both screens side by side
185  const float emulation_aspect_ratio{static_cast<float>(Core::kScreenTopHeight) /
186  (Core::kScreenTopWidth + Core::kScreenBottomWidth)};
187  float window_aspect_ratio{static_cast<float>(height) / width};
188  MathUtil::Rectangle<unsigned> screen_window_area{0, 0, width, height};
189  // Find largest Rectangle that can fit in the window size with the given aspect ratio
190  MathUtil::Rectangle<unsigned> screen_rect{
191  maxRectangle(screen_window_area, emulation_aspect_ratio)};
192  // Find sizes of top and bottom screen
193  MathUtil::Rectangle<unsigned> top_screen{maxRectangle(screen_rect, TOP_SCREEN_ASPECT_RATIO)};
194  MathUtil::Rectangle<unsigned> bot_screen{maxRectangle(screen_rect, BOT_SCREEN_ASPECT_RATIO)};
195 
196  if (window_aspect_ratio < emulation_aspect_ratio) {
197  // Apply borders to the left and right sides of the window.
198  u32 shift_horizontal{(screen_window_area.GetWidth() - screen_rect.GetWidth()) / 2};
199  top_screen = top_screen.TranslateX(shift_horizontal);
200  bot_screen = bot_screen.TranslateX(shift_horizontal);
201  } else {
202  // Window is narrower than the emulation content => apply borders to the top and bottom
203  u32 shift_vertical{(screen_window_area.GetHeight() - screen_rect.GetHeight()) / 2};
204  top_screen = top_screen.TranslateY(shift_vertical);
205  bot_screen = bot_screen.TranslateY(shift_vertical);
206  }
207  // Move the top screen to the right if we are swapped.
208  res.top_screen = swapped ? top_screen.TranslateX(bot_screen.GetWidth()) : top_screen;
209  res.bottom_screen = swapped ? bot_screen : bot_screen.TranslateX(top_screen.GetWidth());
210  return res;
211 }

231 FramebufferLayout FrameLayoutFromResolutionScale(u16 res_scale) {
232  FramebufferLayout layout{};
233  if (Settings::values.custom_layout == true) {
234  layout = CustomFrameLayout(
235  std::max(Settings::values.custom_top_right, Settings::values.custom_bottom_right),
236  std::max(Settings::values.custom_top_bottom, Settings::values.custom_bottom_bottom));
237  } else {
238  int width{}, height{};
239  // TODO: add support for medium layout
240  switch (Settings::values.layout_option) {
241  case Settings::LayoutOption::SingleScreen:
242  if (Settings::values.swap_screen) {
243  width = Core::kScreenBottomWidth * res_scale;
244  height = Core::kScreenBottomHeight * res_scale;
245  } else {
246  width = Core::kScreenTopWidth * res_scale;
247  height = Core::kScreenTopHeight * res_scale;
248  }
249  layout = SingleFrameLayout(width, height, Settings::values.swap_screen);
250  break;
251  case Settings::LayoutOption::LargeScreen:
252  if (Settings::values.swap_screen) {
253  width = (Core::kScreenBottomWidth + Core::kScreenTopWidth / 4) * res_scale;
254  height = Core::kScreenBottomHeight * res_scale;
255  } else {
256  width = (Core::kScreenTopWidth + Core::kScreenBottomWidth / 4) * res_scale;
257  height = Core::kScreenTopHeight * res_scale;
258  }
259  layout = LargeFrameLayout(width, height, Settings::values.swap_screen);
260  break;
261  case Settings::LayoutOption::SideScreen:
262  width = (Core::kScreenTopWidth + Core::kScreenBottomWidth) * res_scale;
263  height = Core::kScreenTopHeight * res_scale;
264  layout = SideFrameLayout(width, height, Settings::values.swap_screen);
265  break;
266  case Settings::LayoutOption::Default:
267  default:
268  width = Core::kScreenTopWidth * res_scale;
269  height = (Core::kScreenTopHeight + Core::kScreenBottomHeight) * res_scale;
270  layout = DefaultFrameLayout(width, height, Settings::values.swap_screen);
271  break;
272  }
273  }
274  return layout;
275 }
