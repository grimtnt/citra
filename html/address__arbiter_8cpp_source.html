24 void AddressArbiter::ResumeAllThreads(VAddr address) {
25  // Determine which threads are waiting on this address, those should be woken up.
26  auto itr = std::stable_partition(waiting_threads.begin(), waiting_threads.end(),
27  [address](const auto& thread) {
28  ASSERT_MSG(thread->status == THREADSTATUS_WAIT_ARB,
29  "Inconsistent AddressArbiter state");
30  return thread->wait_address != address;
31  });
32 
33  // Wake up all the found threads
34  std::for_each(itr, waiting_threads.end(), [](auto& thread) { thread->ResumeFromWait(); });
35 
36  // Remove the woken up threads from the wait list.
37  waiting_threads.erase(itr, waiting_threads.end());
38 }

40 SharedPtr<Thread> AddressArbiter::ResumeHighestPriorityThread(VAddr address) {
41  // Determine which threads are waiting on this address, those should be considered for wakeup.
42  auto matches_start = std::stable_partition(
43  waiting_threads.begin(), waiting_threads.end(), [address](const auto& thread) {
44  ASSERT_MSG(thread->status == THREADSTATUS_WAIT_ARB,
45  "Inconsistent AddressArbiter state");
46  return thread->wait_address != address;
47  });
48 
49  // Iterate through threads, find highest priority thread that is waiting to be arbitrated.
50  // Note: The real kernel will pick the first thread in the list if more than one have the
51  // same highest priority value. Lower priority values mean higher priority.
52  auto itr = std::min_element(matches_start, waiting_threads.end(),
53  [](const auto& lhs, const auto& rhs) {
54  return lhs->current_priority < rhs->current_priority;
55  });
56 
57  if (itr == waiting_threads.end())
58  return nullptr;
59 
60  auto thread = *itr;
61  thread->ResumeFromWait();
62 
63  waiting_threads.erase(itr);
64  return thread;
65 }

78 ResultCode AddressArbiter::ArbitrateAddress(SharedPtr<Thread> thread, ArbitrationType type,
79  VAddr address, s32 value, u64 nanoseconds) {
80 
81  auto timeout_callback = [this](ThreadWakeupReason reason, SharedPtr<Thread> thread,
82  SharedPtr<WaitObject> object) {
83  ASSERT(reason == ThreadWakeupReason::Timeout);
84  // Remove the newly-awakened thread from the Arbiter's waiting list.
85  waiting_threads.erase(std::remove(waiting_threads.begin(), waiting_threads.end(), thread),
86  waiting_threads.end());
87  };
88 
89  switch (type) {
90 
91  // Signal thread(s) waiting for arbitrate address...
92  case ArbitrationType::Signal:
93  // Negative value means resume all threads
94  if (value < 0) {
95  ResumeAllThreads(address);
96  } else {
97  // Resume first N threads
98  for (int i{}; i < value; i++)
99  ResumeHighestPriorityThread(address);
100  }
101  break;
102 
103  // Wait current thread (acquire the arbiter)...
104  case ArbitrationType::WaitIfLessThan:
105  if ((s32)Memory::Read32(address) < value) {
106  WaitThread(std::move(thread), address);
107  }
108  break;
109  case ArbitrationType::WaitIfLessThanWithTimeout:
110  if ((s32)Memory::Read32(address) < value) {
111  thread->wakeup_callback = timeout_callback;
112  thread->WakeAfterDelay(nanoseconds);
113  WaitThread(std::move(thread), address);
114  }
115  break;
116  case ArbitrationType::DecrementAndWaitIfLessThan: {
117  s32 memory_value = Memory::Read32(address);
118  if (memory_value < value) {
119  // Only change the memory value if the thread should wait
120  Memory::Write32(address, (s32)memory_value - 1);
121  WaitThread(std::move(thread), address);
122  }
123  break;
124  }
125  case ArbitrationType::DecrementAndWaitIfLessThanWithTimeout: {
126  s32 memory_value = Memory::Read32(address);
127  if (memory_value < value) {
128  // Only change the memory value if the thread should wait
129  Memory::Write32(address, (s32)memory_value - 1);
130  thread->wakeup_callback = timeout_callback;
131  thread->WakeAfterDelay(nanoseconds);
132  WaitThread(std::move(thread), address);
133  }
134  break;
135  }
136 
137  default:
138  LOG_ERROR(Kernel, "unknown type {}", static_cast<u32>(type));
139  return ERR_INVALID_ENUM_VALUE_FND;
140  }
141 
142  // The calls that use a timeout seem to always return a Timeout error even if they did not put
143  // the thread to sleep
144  if (type == ArbitrationType::WaitIfLessThanWithTimeout ||
145  type == ArbitrationType::DecrementAndWaitIfLessThanWithTimeout) {
146 
147  return RESULT_TIMEOUT;
148  }
149  return RESULT_SUCCESS;
150 }
