22  return std::make_tuple(0.0f, 0.0f);
23  constexpr float SQRT_HALF{0.707106781f};
24  int x{}, y{};
25 
26  if (right->GetStatus())

31  ++y;
32  if (down->GetStatus())
33  --y;
34 
35  float coef = modifier->GetStatus() ? modifier_scale : 1.0f;
36  return std::make_tuple(x * coef * (y == 0 ? 1.0f : SQRT_HALF),
37  y * coef * (x == 0 ? 1.0f : SQRT_HALF));
38  }
