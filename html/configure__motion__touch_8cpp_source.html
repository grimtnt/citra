16 CalibrationConfigurationDialog::CalibrationConfigurationDialog(QWidget* parent,
17  const std::string& host, u16 port,
18  u8 pad_index, u16 client_id)
19  : QDialog{parent} {
20  layout = new QVBoxLayout;
21  status_label = new QLabel("Communicating with the server...");
22  cancel_button = new QPushButton("Cancel");
23  connect(cancel_button, &QPushButton::clicked, this, [this] {
24  if (!completed)
25  job->Stop();
26  accept();
27  });
28  layout->addWidget(status_label);
29  layout->addWidget(cancel_button);
30  setLayout(layout);
31 
32  using namespace InputCommon::CemuhookUDP;
33  job = std::move(std::make_unique<CalibrationConfigurationJob>(
34  host, port, pad_index, client_id,
35  [this](CalibrationConfigurationJob::Status status) {
36  QString text{};
37  switch (status) {
38  case CalibrationConfigurationJob::Status::Ready:
39  text = "Touch the top left corner <br>of your touchpad.";
40  break;
41  case CalibrationConfigurationJob::Status::Stage1Completed:
42  text = "Now touch the bottom right corner <br>of your touchpad.";
43  break;
44  case CalibrationConfigurationJob::Status::Completed:
45  text = "Configuration completed!";
46  break;
47  }
48  QMetaObject::invokeMethod(this, "UpdateLabelText", Q_ARG(QString, text));
49  if (status == CalibrationConfigurationJob::Status::Completed) {
50  QMetaObject::invokeMethod(this, "UpdateButtonText", Q_ARG(QString, "OK"));
51  }
52  },
53  [this](u16 min_x_, u16 min_y_, u16 max_x_, u16 max_y_) {
54  completed = true;
55  min_x = min_x_;
56  min_y = min_y_;
57  max_x = max_x_;
58  max_y = max_y_;
59  }));
60 }

246 void ConfigureMotionTouch::applyConfiguration() {
247  if (!CanCloseDialog())
248  return;
249 
250  std::string motion_engine{ui->motion_provider->currentData().toString().toStdString()};
251  std::string touch_engine{ui->touch_provider->currentData().toString().toStdString()};
252 
253  Common::ParamPackage motion_param{}, touch_param{};
254  motion_param.Set("engine", motion_engine);
255  touch_param.Set("engine", touch_engine);
256 
257  if (motion_engine == "motion_emu") {
258  motion_param.Set("sensitivity", static_cast<float>(ui->motion_sensitivity->value()));
259  }
260 
261  if (touch_engine == "cemuhookudp") {
262  touch_param.Set("min_x", min_x);
263  touch_param.Set("min_y", min_y);
264  touch_param.Set("max_x", max_x);
265  touch_param.Set("max_y", max_y);
266  }
267 
268  Settings::values.motion_device = motion_param.Serialize();
269  Settings::values.touch_device = touch_param.Serialize();
270  Settings::values.udp_input_address = ui->udp_server->text().toStdString();
271  Settings::values.udp_input_port = static_cast<u16>(ui->udp_port->text().toInt());
272  Settings::values.udp_pad_index = static_cast<u8>(ui->udp_pad_index->currentIndex());
273  InputCommon::ReloadInputDevices();
274 
275  accept();
276 }
