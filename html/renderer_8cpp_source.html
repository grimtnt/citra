24 constexpr char vertex_shader[]{R"(
25 #version 150 core
26 
27 in vec2 vert_position;
28 in vec2 vert_tex_coord;
29 out vec2 frag_tex_coord;
30 
31 // This is a truncated 3x3 matrix for 2D transformations:
32 // The upper-left 2x2 submatrix performs scaling/rotation/mirroring.
33 // The third column performs translation.
34 // The third row could be used for projection, which we don't need in 2D. It hence is assumed to
35 // implicitly be [0, 0, 1]
36 uniform mat3x2 modelview_matrix;
37 
38 void main() {
39  // Multiply input position by the rotscale part of the matrix and then manually translate by
40  // the last column. This is equivalent to using a full 3x3 matrix and expanding the vector
41  // to `vec3(vert_position.xy, 1.0)`
42  gl_Position = vec4(mat2(modelview_matrix) * vert_position + modelview_matrix[2], 0.0, 1.0);
43  frag_tex_coord = vert_tex_coord;
44 }
45 )"};

82 static std::array<GLfloat, 6> MakeOrthographicMatrix(const float width, const float height) {
83  std::array<GLfloat, 6> matrix{}; // Laid out in column-major order
84 
85  // clang-format off
86  matrix[0] = 2.f / width; matrix[2] = 0.f; matrix[4] = -1.f;
87  matrix[1] = 0.f; matrix[3] = -2.f / height; matrix[5] = 1.f;
88  // Last matrix row is implicitly assumed to be [0, 0, 1].
89  // clang-format on
90 
91  return matrix;
92 }

100 void Renderer::SwapBuffers() {
101  // Maintain the rasterizer's OpenGLState as a priority
102  OpenGLState prev_state{OpenGLState::GetCurState()};
103  state.Apply();
104 
105  for (int i : {0, 1, 2}) {
106  int fb_id{i == 2 ? 1 : 0};
107  const auto& framebuffer{GPU::g_regs.framebuffer_config[fb_id]};
108 
109  // Main LCD (0): 0x1ED02204, Sub LCD (1): 0x1ED02A04
110  u32 lcd_color_addr{static_cast<u32>((fb_id == 0) ? LCD_REG_INDEX(color_fill_top)
111  : LCD_REG_INDEX(color_fill_bottom))};
112  lcd_color_addr = HW::VADDR_LCD + 4 * lcd_color_addr;
113  LCD::Regs::ColorFill color_fill = {0};
114  LCD::Read(color_fill.raw, lcd_color_addr);
115 
116  if (color_fill.is_enabled) {
117  LoadColorToActiveGLTexture(color_fill.color_r, color_fill.color_g, color_fill.color_b,
118  screen_infos[i].texture);
119 
120  // Resize the texture in case the framebuffer size has changed
121  screen_infos[i].texture.width = 1;
122  screen_infos[i].texture.height = 1;
123  } else {
124  if (screen_infos[i].texture.width != (GLsizei)framebuffer.width ||
125  screen_infos[i].texture.height != (GLsizei)framebuffer.height ||
126  screen_infos[i].texture.format != framebuffer.color_format) {
127  // Reallocate texture if the framebuffer size has changed.
128  // This is expected to not happen very often and hence should not be a
129  // performance problem.
130  ConfigureFramebufferTexture(screen_infos[i].texture, framebuffer);
131  }
132  LoadFBToScreenInfo(framebuffer, screen_infos[i], i == 1);
133 
134  // Resize the texture in case the framebuffer size has changed
135  screen_infos[i].texture.width = framebuffer.width;
136  screen_infos[i].texture.height = framebuffer.height;
137  }
138  }
139 
140  if (VideoCore::g_renderer_screenshot_requested) {
141  // Draw this frame to the screenshot framebuffer
142  screenshot_framebuffer.Create();
143  GLuint old_read_fb = state.draw.read_framebuffer;
144  GLuint old_draw_fb = state.draw.draw_framebuffer;
145  state.draw.read_framebuffer = state.draw.draw_framebuffer = screenshot_framebuffer.handle;
146  state.Apply();
147 
148  Layout::FramebufferLayout layout{VideoCore::g_screenshot_framebuffer_layout};
149 
150  GLuint renderbuffer;
151  glGenRenderbuffers(1, &renderbuffer);
152  glBindRenderbuffer(GL_RENDERBUFFER, renderbuffer);
153  glRenderbufferStorage(GL_RENDERBUFFER, GL_RGB8, layout.width, layout.height);
154  glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER,
155  renderbuffer);
156 
157  DrawScreens(layout);
158 
159  glReadPixels(0, 0, layout.width, layout.height, GL_BGRA, GL_UNSIGNED_INT_8_8_8_8_REV,
160  VideoCore::g_screenshot_bits);
161 
162  screenshot_framebuffer.Release();
163  state.draw.read_framebuffer = old_read_fb;
164  state.draw.draw_framebuffer = old_draw_fb;
165  state.Apply();
166  glDeleteRenderbuffers(1, &renderbuffer);
167 
168  VideoCore::g_screenshot_complete_callback();
169  VideoCore::g_renderer_screenshot_requested = false;
170  }
171 
172  DrawScreens(render_window.GetFramebufferLayout());
173 
174  Core::System::GetInstance().perf_stats.EndSystemFrame();
175 
176  // Swap buffers
177  render_window.PollEvents();
178  render_window.SwapBuffers();
179 
180  Core::System::GetInstance().frame_limiter.DoFrameLimiting(CoreTiming::GetGlobalTimeUs());
181  Core::System::GetInstance().perf_stats.BeginSystemFrame();
182 
183  prev_state.Apply();
184 }

189 void Renderer::LoadFBToScreenInfo(const GPU::Regs::FramebufferConfig& framebuffer,
190  ScreenInfo& screen_info, bool right_eye) {
191 
192  if (framebuffer.address_right1 == 0 || framebuffer.address_right2 == 0)
193  right_eye = false;
194 
195  const PAddr framebuffer_addr{
196  framebuffer.active_fb == 0
197  ? (!right_eye ? framebuffer.address_left1 : framebuffer.address_right1)
198  : (!right_eye ? framebuffer.address_left2 : framebuffer.address_right2)};
199 
200  LOG_TRACE(Render, "0x{:08x} bytes from 0x{:08x}({:x}{}), fmt {}",
201  framebuffer.stride * framebuffer.height, framebuffer_addr, (int)framebuffer.width,
202  (int)framebuffer.height, (int)framebuffer.format);
203 
204  int bpp{GPU::Regs::BytesPerPixel(framebuffer.color_format)};
205  std::size_t pixel_stride{framebuffer.stride / bpp};
206 
207  // OpenGL only supports specifying a stride in units of pixels, not bytes, unfortunately
208  ASSERT(pixel_stride * bpp == framebuffer.stride);
209 
210  // Ensure no bad interactions with GL_UNPACK_ALIGNMENT, which by default
211  // only allows rows to have a memory alignement of 4.
212  ASSERT(pixel_stride % 4 == 0);
213 
214  if (!rasterizer->AccelerateDisplay(framebuffer, framebuffer_addr,
215  static_cast<u32>(pixel_stride), screen_info)) {
216  // Reset the screen info's display texture to its own permanent texture
217  screen_info.display_texture = screen_info.texture.resource.handle;
218  screen_info.display_texcoords = MathUtil::Rectangle<float>(0.f, 0.f, 1.f, 1.f);
219 
220  Memory::RasterizerFlushRegion(framebuffer_addr, framebuffer.stride * framebuffer.height);
221 
222  const u8* framebuffer_data{Memory::GetPhysicalPointer(framebuffer_addr)};
223 
224  state.texture_units[0].texture_2d = screen_info.texture.resource.handle;
225  state.Apply();
226 
227  glActiveTexture(GL_TEXTURE0);
228  glPixelStorei(GL_UNPACK_ROW_LENGTH, (GLint)pixel_stride);
229 
230  // Update existing texture
231  // TODO: Test what happens on hardware when you change the framebuffer dimensions so that
232  // they differ from the LCD resolution.
233  // TODO: Applications could theoretically crash Citra here by specifying too large
234  // framebuffer sizes. We should make sure that this cannot happen.
235  glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, framebuffer.width, framebuffer.height,
236  screen_info.texture.gl_format, screen_info.texture.gl_type,
237  framebuffer_data);
238 
239  glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
240 
241  state.texture_units[0].texture_2d = 0;
242  state.Apply();
243  }

268 void Renderer::InitOpenGLObjects() {
269  // Link shaders and get variable locations
270  shader.Create(vertex_shader, fragment_shader);
271  state.draw.shader_program = shader.handle;
272  state.Apply();
273  uniform_modelview_matrix = glGetUniformLocation(shader.handle, "modelview_matrix");
274  uniform_color_texture = glGetUniformLocation(shader.handle, "color_texture");
275  attrib_position = glGetAttribLocation(shader.handle, "vert_position");
276  attrib_tex_coord = glGetAttribLocation(shader.handle, "vert_tex_coord");
277 
278  // Generate VBO handle for drawing
279  vertex_buffer.Create();
280 
281  // Generate VAO
282  vertex_array.Create();
283 
284  state.draw.vertex_array = vertex_array.handle;
285  state.draw.vertex_buffer = vertex_buffer.handle;
286  state.draw.uniform_buffer = 0;
287  state.Apply();
288 
289  // Attach vertex data to VAO
290  glBufferData(GL_ARRAY_BUFFER, sizeof(ScreenRectVertex) * 4, nullptr, GL_STREAM_DRAW);
291  glVertexAttribPointer(attrib_position, 2, GL_FLOAT, GL_FALSE, sizeof(ScreenRectVertex),
292  (GLvoid*)offsetof(ScreenRectVertex, position));
293  glVertexAttribPointer(attrib_tex_coord, 2, GL_FLOAT, GL_FALSE, sizeof(ScreenRectVertex),
294  (GLvoid*)offsetof(ScreenRectVertex, tex_coord));
295  glEnableVertexAttribArray(attrib_position);
296  glEnableVertexAttribArray(attrib_tex_coord);
297 
298  // Allocate textures for each screen
299  for (auto& screen_info : screen_infos) {
300  screen_info.texture.resource.Create();
301 
302  // Allocation of storage is deferred until the first frame, when we
303  // know the framebuffer size.
304 
305  state.texture_units[0].texture_2d = screen_info.texture.resource.handle;
306  state.Apply();
307 
308  glActiveTexture(GL_TEXTURE0);
309  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAX_LEVEL, 0);
310  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
311  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
312  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
313  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
314 
315  screen_info.display_texture = screen_info.texture.resource.handle;
316  }
317 
318  state.texture_units[0].texture_2d = 0;
319  state.Apply();
320 }

322 void Renderer::ConfigureFramebufferTexture(TextureInfo& texture,
323  const GPU::Regs::FramebufferConfig& framebuffer) {
324  GPU::Regs::PixelFormat format{framebuffer.color_format};
325  GLint internal_format{};
326 
327  texture.format = format;
328  texture.width = framebuffer.width;
329  texture.height = framebuffer.height;
330 
331  switch (format) {
332  case GPU::Regs::PixelFormat::RGBA8:
333  internal_format = GL_RGBA;
334  texture.gl_format = GL_RGBA;
335  texture.gl_type = GL_UNSIGNED_INT_8_8_8_8;
336  break;
337 
338  case GPU::Regs::PixelFormat::RGB8:
339  // This pixel format uses BGR since GL_UNSIGNED_BYTE specifies byte-order, unlike every
340  // specific OpenGL type used in this function using native-endian (that is, little-endian
341  // mostly everywhere) for words or half-words.
342  // TODO: check how those behave on big-endian processors.
343  internal_format = GL_RGB;
344  texture.gl_format = GL_BGR;
345  texture.gl_type = GL_UNSIGNED_BYTE;
346  break;
347 
348  case GPU::Regs::PixelFormat::RGB565:
349  internal_format = GL_RGB;
350  texture.gl_format = GL_RGB;
351  texture.gl_type = GL_UNSIGNED_SHORT_5_6_5;
352  break;
353 
354  case GPU::Regs::PixelFormat::RGB5A1:
355  internal_format = GL_RGBA;
356  texture.gl_format = GL_RGBA;
357  texture.gl_type = GL_UNSIGNED_SHORT_5_5_5_1;
358  break;
359 
360  case GPU::Regs::PixelFormat::RGBA4:
361  internal_format = GL_RGBA;
362  texture.gl_format = GL_RGBA;
363  texture.gl_type = GL_UNSIGNED_SHORT_4_4_4_4;
364  break;
365 
366  default:
367  UNIMPLEMENTED();
368  }
369 
370  state.texture_units[0].texture_2d = texture.resource.handle;
371  state.Apply();
372 
373  glActiveTexture(GL_TEXTURE0);
374  glTexImage2D(GL_TEXTURE_2D, 0, internal_format, texture.width, texture.height, 0,
375  texture.gl_format, texture.gl_type, nullptr);
376 
377  state.texture_units[0].texture_2d = 0;
378  state.Apply();
379 }

409 void Renderer::DrawScreens(const Layout::FramebufferLayout& layout) {
410  if (VideoCore::g_renderer_bg_color_update_requested.exchange(false)) {
411  // Update background color before drawing
412  glClearColor(Settings::values.bg_red, Settings::values.bg_green, Settings::values.bg_blue,
413  0.0f);
414  }
415 
416  const auto& top_screen{layout.top_screen};
417  const auto& bottom_screen{layout.bottom_screen};
418 
419  glViewport(0, 0, layout.width, layout.height);
420  glClear(GL_COLOR_BUFFER_BIT);
421 
422  // Set projection matrix
423  std::array<GLfloat, 6> ortho_matrix{
424  MakeOrthographicMatrix((float)layout.width, (float)layout.height)};
425  glUniformMatrix3x2fv(uniform_modelview_matrix, 1, GL_FALSE, ortho_matrix.data());
426 
427  // Bind texture in Texture Unit 0
428  glActiveTexture(GL_TEXTURE0);
429  glUniform1i(uniform_color_texture, 0);
430 
431  if (!Core::System::GetInstance().IsShellOpen())
432  return;
433  if (layout.top_screen_enabled) {
434  if (Settings::values.factor_3d == 0) {
435  DrawSingleScreenRotated(screen_infos[0], (float)top_screen.left, (float)top_screen.top,
436  (float)top_screen.GetWidth(), (float)top_screen.GetHeight());
437  } else {
438  DrawSingleScreenRotated(screen_infos[0], (float)top_screen.left / 2,
439  (float)top_screen.top, (float)top_screen.GetWidth() / 2,
440  (float)top_screen.GetHeight());
441  DrawSingleScreenRotated(screen_infos[1],
442  ((float)top_screen.left / 2) + ((float)layout.width / 2),
443  (float)top_screen.top, (float)top_screen.GetWidth() / 2,
444  (float)top_screen.GetHeight());
445  }
446  }
447  if (layout.bottom_screen_enabled) {
448  if (Settings::values.factor_3d == 0) {
449  DrawSingleScreenRotated(screen_infos[2], (float)bottom_screen.left,
450  (float)bottom_screen.top, (float)bottom_screen.GetWidth(),
451  (float)bottom_screen.GetHeight());
452  } else {
453  DrawSingleScreenRotated(screen_infos[2], (float)bottom_screen.left / 2,
454  (float)bottom_screen.top, (float)bottom_screen.GetWidth() / 2,
455  (float)bottom_screen.GetHeight());
