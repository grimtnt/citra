26 ParamPackage::ParamPackage(const std::string& serialized) {
27  if (serialized == EMPTY_PLACEHOLDER) {
28  return;
29  }
30 
31  std::vector<std::string> pairs;
32  Common::SplitString(serialized, PARAM_SEPARATOR, pairs);
33 
34  for (const std::string& pair : pairs) {
35  std::vector<std::string> key_value;
36  Common::SplitString(pair, KEY_VALUE_SEPARATOR, key_value);
37  if (key_value.size() != 2) {
38  LOG_ERROR(Common, "invalid key pair {}", pair);
39  continue;
40  }
41 
42  for (std::string& part : key_value) {
43  part = Common::ReplaceAll(part, KEY_VALUE_SEPARATOR_ESCAPE, {KEY_VALUE_SEPARATOR});
44  part = Common::ReplaceAll(part, PARAM_SEPARATOR_ESCAPE, {PARAM_SEPARATOR});
45  part = Common::ReplaceAll(part, ESCAPE_CHARACTER_ESCAPE, {ESCAPE_CHARACTER});
46  }
47 
48  Set(key_value[0], std::move(key_value[1]));
49  }
50 }

54 std::string ParamPackage::Serialize() const {
55  if (data.empty())
56  return EMPTY_PLACEHOLDER;
57 
58  std::string result{};
59 
60  for (const auto& pair : data) {
61  std::array<std::string, 2> key_value{{pair.first, pair.second}};
62  for (std::string& part : key_value) {
63  part = Common::ReplaceAll(part, {ESCAPE_CHARACTER}, ESCAPE_CHARACTER_ESCAPE);
64  part = Common::ReplaceAll(part, {PARAM_SEPARATOR}, PARAM_SEPARATOR_ESCAPE);
65  part = Common::ReplaceAll(part, {KEY_VALUE_SEPARATOR}, KEY_VALUE_SEPARATOR_ESCAPE);
66  }
67  result += key_value[0] + KEY_VALUE_SEPARATOR + key_value[1] + PARAM_SEPARATOR;
68  }
69 
70  result.pop_back(); // discard the trailing PARAM_SEPARATOR
71  return result;
72 }
