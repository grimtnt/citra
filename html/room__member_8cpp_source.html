109 void RoomMember::RoomMemberImpl::MemberLoop() {
110  // Receive packets while the connection is open
111  while (IsConnected()) {
112  std::lock_guard<std::mutex> lock{network_mutex};
113  ENetEvent event;
114  if (enet_host_service(client, &event, 100) > 0) {
115  switch (event.type) {
116  case ENET_EVENT_TYPE_RECEIVE:
117  switch (event.packet->data[0]) {
118  case IdWifiPacket:
119  HandleWifiPacket(&event);
120  break;
121  case IdJoinSuccess:
122  HandleJoinPacket(&event); // Get the MAC Address for the client
123  break;
124  case IdMacCollision:
125  SetState(State::MacCollision);
126  break;
127  case IdCloseRoom:
128  SetState(State::LostConnection);
129  break;
130  }
131  enet_packet_destroy(event.packet);
132  break;
133  case ENET_EVENT_TYPE_DISCONNECT:
134  SetState(State::LostConnection);
135  break;
136  }
137  }
138  {
139  std::lock_guard<std::mutex> lock{send_list_mutex};
140  for (const auto& packet : send_list) {
141  ENetPacket* enetPacket = enet_packet_create(packet.GetData(), packet.GetDataSize(),
142  ENET_PACKET_FLAG_RELIABLE);
143  enet_peer_send(server, 0, enetPacket);
144  }
145  enet_host_flush(client);
146  send_list.clear();
147  }
148  }
149  Disconnect();
150 };

180 void RoomMember::RoomMemberImpl::HandleWifiPacket(const ENetEvent* event) {
181  WifiPacket wifi_packet{};
182  Packet packet{};
183  packet.Append(event->packet->data, event->packet->dataLength);
184 
185  // Ignore the first byte, which is the message id.
186  packet.IgnoreBytes(sizeof(u8)); // Ignore the message type
187 
188  // Parse the WifiPacket from the packet
189  u8 frame_type;
190  packet >> frame_type;
191  WifiPacket::PacketType type = static_cast<WifiPacket::PacketType>(frame_type);
192 
193  wifi_packet.type = type;
194  packet >> wifi_packet.channel;
195  packet >> wifi_packet.transmitter_address;
196  packet >> wifi_packet.destination_address;
197  packet >> wifi_packet.data;
198 
199  Invoke<WifiPacket>(wifi_packet);
200 }

202 void RoomMember::RoomMemberImpl::Disconnect() {
203  if (!server)
204  return;
205  enet_peer_disconnect(server, 0);
206 
207  ENetEvent event;
208  while (enet_host_service(client, &event, ConnectionTimeoutMs) > 0) {
209  switch (event.type) {
210  case ENET_EVENT_TYPE_RECEIVE:
211  enet_packet_destroy(event.packet); // Ignore all incoming data
212  break;
213  case ENET_EVENT_TYPE_DISCONNECT:
214  server = nullptr;
215  return;
216  case ENET_EVENT_TYPE_NONE:
217  case ENET_EVENT_TYPE_CONNECT:
218  break;
219  }
220  }
221  // didn't disconnect gracefully force disconnect
222  enet_peer_reset(server);
223  server = nullptr;
224 }

274 void RoomMember::Join(const char* server_addr, u16 server_port, const MacAddress& preferred_mac) {
275  // If the member is connected, kill the connection first
276  if (room_member_impl->loop_thread && room_member_impl->loop_thread->joinable()) {
277  Leave();
278  }
279  // If the thread isn't running but the ptr still exists, reset it
280  else if (room_member_impl->loop_thread) {
281  room_member_impl->loop_thread.reset();
282  }
283 
284  if (!room_member_impl->client) {
285  room_member_impl->client = enet_host_create(nullptr, 1, NumChannels, 0, 0);
286  ASSERT_MSG(room_member_impl->client != nullptr, "Could not create client");
287  }
288 
289  room_member_impl->SetState(State::Joining);
290 
291  ENetAddress address{};
292  enet_address_set_host(&address, server_addr);
293  address.port = server_port;
294  room_member_impl->server =
295  enet_host_connect(room_member_impl->client, &address, NumChannels, 0);
296 
297  if (!room_member_impl->server) {
298  room_member_impl->SetState(State::Error);
299  return;
300  }
301 
302  ENetEvent event{};
303  int net = enet_host_service(room_member_impl->client, &event, ConnectionTimeoutMs);
304  if (net > 0 && event.type == ENET_EVENT_TYPE_CONNECT) {
305  room_member_impl->StartLoop();
306  room_member_impl->SendJoinRequest(preferred_mac);
307  } else {
308  enet_peer_disconnect(room_member_impl->server, 0);
309  room_member_impl->SetState(State::CouldNotConnect);
310  }
311 }
