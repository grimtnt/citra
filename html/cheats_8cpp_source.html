324 void CheatDialog::OnScan(bool is_next_scan) {
325  int value_type{};
326  int search_type{};
327  QString search_value{};
328  bool convert_hex{};
329 
330  try {
331  value_type = ui->cbValueType->currentIndex();
332  search_type = ui->cbScanType->currentIndex();
333  search_value = ui->txtSearch->text();
334  convert_hex = ui->chkHex->isChecked();
335  } catch (const std::exception& e) {
336  LOG_CRITICAL(Frontend, "Exception when scanning: {}", e.what());
337  ui->txtSearch->clear();
338  return;
339  }
340 
341  std::function<bool(int, int, int)> comparer = [&](int a, int b, int c) {
342  return Equals(a, b, c);
343  };
344 
345  switch (search_type) {
346  case 0: { // Equals
347  comparer = [&](int a, int b, int c) { return Equals(a, b, c); };
348  break;
349  }
350  case 1: { // Greater Than
351  comparer = [&](int a, int b, int c) { return GreaterThan(a, b, c); };
352  break;
353  }
354  case 2: { // Less Than
355  comparer = [&](int a, int b, int c) { return LessThan(a, b, c); };
356  break;
357  }
358  case 3: { // Between
359  comparer = [&](int a, int b, int c) { return Between(a, b, c); };
360  break;
361  }
362  }
363 
364  int base = (ui->chkHex->isChecked()) ? 16 : 10;
365 
366  switch (value_type) {
367  case 0: { // u32
368  u32 value = search_value.toUInt(nullptr, base);
369  if (!is_next_scan)
370  previous_found = FirstSearch<u32>(value, comparer);
371  else
372  previous_found = NextSearch<u32>(value, comparer);
373  break;
374  }
375 
376  case 1: { // u16
377  u16 value = search_value.toUShort(nullptr, base);
378  if (!is_next_scan)
379  previous_found = FirstSearch<u16>(value, comparer);
380  else
381  previous_found = NextSearch<u16>(value, comparer);
382  break;
383  }
384  case 2: { // u8
385  u8 value = static_cast<u8>(search_value.toUShort(nullptr, base));
386  if (!is_next_scan)
387  previous_found = FirstSearch<u8>(value, comparer);
388  else
389  previous_found = NextSearch<u8>(value, comparer);
390  break;
391  }
392  }
393 
394  ui->tableFound->setRowCount(0);
395 
396  if (previous_found.size() > 50000) {
397  ui->lblCount->setText("Count: 50000+");
398  } else {
399  LoadTable(previous_found);
400  ui->lblCount->setText(QString("Count: %1").arg(previous_found.size()));
401  }

476 std::vector<FoundItem> CheatDialog::FirstSearch(const T value,
477  std::function<bool(int, int, int)> comparer) {
478  u32 start_address{0x00000000};
479  u32 end_address{0x08000000 + 0x08000000};
480  std::vector<int> address_in_use{};
481  std::vector<FoundItem> results{};
482  int base{ui->chkHex->isChecked() ? 16 : 10};
483  T search_to_value{static_cast<T>(ui->txtSearchTo->text().toInt(nullptr, base))};
484 
485  for (u32 i = start_address; i < end_address; i += 4096) {
486  if (Memory::IsValidVirtualAddress(i)) {
487  address_in_use.push_back(i);
488  }
489  }
490 
491  for (auto& range : address_in_use) {
492  for (int i = range; i < (range + 4096); i++) {
493  T result{Read<T>(i)};
494  if (comparer(result, value, search_to_value)) {
495  FoundItem item;
496  item.address = IntToHex(i);
497  item.value = QString::number(result);
498  results.push_back(item);
499  }
500  }
501  }
502  return results;
503 }
