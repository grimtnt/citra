12 static void SetShaderUniformBlockBinding(GLuint shader, const char* name, UniformBindings binding,
13  std::size_t expected_size) {
14  GLuint ub_index{glGetUniformBlockIndex(shader, name)};
15  if (ub_index == GL_INVALID_INDEX) {
16  return;
17  }
18  GLint ub_size{};
19  glGetActiveUniformBlockiv(shader, ub_index, GL_UNIFORM_BLOCK_DATA_SIZE, &ub_size);
20  glGetActiveUniformBlockiv(shader, ub_index, GL_UNIFORM_BLOCK_DATA_SIZE, &ub_size);
21  ASSERT_MSG(ub_size == expected_size, "Uniform block size did not match! Got {}, expected {}",
22  static_cast<int>(ub_size), expected_size);
23  glUniformBlockBinding(shader, ub_index, static_cast<GLuint>(binding));
24 }

94 class ShaderStage {
95 public:
96  explicit ShaderStage(bool separable) {
97  if (separable) {
98  shader_or_program = Program();
99  } else {
100  shader_or_program = Shader();
101  }
102  }
103 
104  void Create(const char* source, GLenum type) {
105  if (shader_or_program.which() == 0) {
106  boost::get<Shader>(shader_or_program).Create(source, type);
107  } else {
108  Shader shader;
109  shader.Create(source, type);
110  Program& program = boost::get<Program>(shader_or_program);
111  program.Create(true, {shader.handle});
112  SetShaderUniformBlockBindings(program.handle);
113  SetShaderSamplerBindings(program.handle);
114  }
115  }
116 
117  GLuint GetHandle() const {
118  if (shader_or_program.which() == 0) {
119  return boost::get<Shader>(shader_or_program).handle;
120  } else {
121  return boost::get<Program>(shader_or_program).handle;
122  }
123  }
124 
125 private:
126  boost::variant<Shader, Program> shader_or_program;
127 };

161 // This is a cache designed for shaders translated from PICA shaders. The first cache matches the
162 // config structure like a normal cache does. On cache miss, the second cache matches the generated
163 // GLSL code. The configuration is like this because there might be leftover code in the PICA shader
164 // program buffer from the previous shader, which is hashed into the config, resulting several
165 // different config values from the same shader program.
166 template <typename KeyConfigType,
167  boost::optional<std::string> (*CodeGenerator)(const Pica::Shader::ShaderSetup&,
168  const KeyConfigType&, bool),
169  GLenum ShaderType>
170 class ShaderDoubleCache {
171 public:
172  explicit ShaderDoubleCache(bool separable) : separable(separable) {}
173  GLuint Get(const KeyConfigType& key, const Pica::Shader::ShaderSetup& setup) {
174  auto map_it = shader_map.find(key);
175  if (map_it == shader_map.end()) {
176  auto program_opt = CodeGenerator(setup, key, separable);
177  if (!program_opt) {
178  shader_map[key] = nullptr;
179  return 0;
180  }
181 
182  std::string& program = program_opt.get();
183  auto [iter, new_shader] = shader_cache.emplace(program, ShaderStage{separable});
184  ShaderStage& cached_shader = iter->second;
185  if (new_shader) {
186  cached_shader.Create(program.c_str(), ShaderType);
187  }
188  shader_map[key] = &cached_shader;
189  return cached_shader.GetHandle();
190  }
191 
192  if (map_it->second == nullptr) {
193  return 0;
194  }
195 
196  return map_it->second->GetHandle();
197  }
198 
199 private:
200  bool separable;
201  std::unordered_map<KeyConfigType, ShaderStage*> shader_map;
202  std::unordered_map<std::string, ShaderStage> shader_cache;
203 };

309 void ShaderProgramManager::ApplyTo(OpenGLState& state) {
310  if (impl->separable) {
311  if (impl->is_amd) {
312  // Without this reseting, AMD sometimes freezes when one stage is changed but not for
313  // the others.
314  // On the other hand, including this reset seems to introduce memory leak in Intel
315  // Graphics.
316  glUseProgramStages(
317  impl->pipeline.handle,
318  GL_VERTEX_SHADER_BIT | GL_GEOMETRY_SHADER_BIT | GL_FRAGMENT_SHADER_BIT, 0);
319  }
320 
321  glUseProgramStages(impl->pipeline.handle, GL_VERTEX_SHADER_BIT, impl->current.vs);
322  glUseProgramStages(impl->pipeline.handle, GL_GEOMETRY_SHADER_BIT, impl->current.gs);
323  glUseProgramStages(impl->pipeline.handle, GL_FRAGMENT_SHADER_BIT, impl->current.fs);
324  state.draw.shader_program = 0;
325  state.draw.program_pipeline = impl->pipeline.handle;
326  } else {
327  Program& cached_program = impl->program_cache[impl->current];
328  if (cached_program.handle == 0) {
329  cached_program.Create(false, {impl->current.vs, impl->current.gs, impl->current.fs});
330  SetShaderUniformBlockBindings(cached_program.handle);
331  SetShaderSamplerBindings(cached_program.handle);
332  }
333  state.draw.shader_program = cached_program.handle;
334  }
