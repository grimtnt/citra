84  color[3] = v.color.w.ToFloat32();
85  tex_coord0[0] = v.tc0.x.ToFloat32();
86  tex_coord0[1] = v.tc0.y.ToFloat32();
87  tex_coord1[0] = v.tc1.x.ToFloat32();
88  tex_coord1[1] = v.tc1.y.ToFloat32();
89  tex_coord2[0] = v.tc2.x.ToFloat32();
90  tex_coord2[1] = v.tc2.y.ToFloat32();
91  tex_coord0_w = v.tc0_w.ToFloat32();
92  normquat[0] = v.quat.x.ToFloat32();
93  normquat[1] = v.quat.y.ToFloat32();
94  normquat[2] = v.quat.z.ToFloat32();
95  normquat[3] = v.quat.w.ToFloat32();
96  view[0] = v.view.x.ToFloat32();
97  view[1] = v.view.y.ToFloat32();
98  view[2] = v.view.z.ToFloat32();
99 
100  if (flip_quaternion) {
101  for (float& x : normquat) {
102  x = -x;
103  }
104  }
105  }
