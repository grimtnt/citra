17 SharedPtr<SharedMemory> SharedMemory::Create(SharedPtr<Process> owner_process, u32 size,
18  MemoryPermission permissions,
19  MemoryPermission other_permissions, VAddr address,
20  MemoryRegion region, std::string name) {
21  SharedPtr<SharedMemory> shared_memory(new SharedMemory);
22 
23  shared_memory->owner_process = owner_process;
24  shared_memory->name = std::move(name);
25  shared_memory->size = size;
26  shared_memory->permissions = permissions;
27  shared_memory->other_permissions = other_permissions;
28 
29  if (address == 0) {
30  // We need to allocate a block from the Linear Heap ourselves.
31  // We'll manually allocate some memory from the linear heap in the specified region.
32  MemoryRegionInfo* memory_region = GetMemoryRegion(region);
33  auto& linheap_memory = memory_region->linear_heap_memory;
34 
35  ASSERT_MSG(linheap_memory->size() + size <= memory_region->size,
36  "Not enough space in region to allocate shared memory!");
37 
38  shared_memory->backing_block = linheap_memory;
39  shared_memory->backing_block_offset = linheap_memory->size();
40  // Allocate some memory from the end of the linear heap for this region.
41  linheap_memory->insert(linheap_memory->end(), size, 0);
42  memory_region->used += size;
43 
44  shared_memory->linear_heap_phys_address =
45  Memory::FCRAM_PADDR + memory_region->base +
46  static_cast<PAddr>(shared_memory->backing_block_offset);
47 
48  // Increase the amount of used linear heap memory for the owner process.
49  if (shared_memory->owner_process != nullptr) {
50  shared_memory->owner_process->linear_heap_used += size;
51  }
52 
53  // Refresh the address mappings for the current process.
54  if (Kernel::g_current_process != nullptr) {
55  Kernel::g_current_process->vm_manager.RefreshMemoryBlockMappings(linheap_memory.get());
56  }
57  } else {
58  auto& vm_manager = shared_memory->owner_process->vm_manager;
59  // The memory is already available and mapped in the owner process.
60  auto vma = vm_manager.FindVMA(address);
61  ASSERT_MSG(vma != vm_manager.vma_map.end(), "Invalid memory address");
62  ASSERT_MSG(vma->second.backing_block, "Backing block doesn't exist for address");
63 
64  // The returned VMA might be a bigger one encompassing the desired address.
65  auto vma_offset = address - vma->first;
66  ASSERT_MSG(vma_offset + size <= vma->second.size,
67  "Shared memory exceeds bounds of mapped block");
68 
69  shared_memory->backing_block = vma->second.backing_block;
70  shared_memory->backing_block_offset = vma->second.offset + vma_offset;
71  }
72 
73  shared_memory->base_address = address;
74  return shared_memory;
75 }

96 ResultCode SharedMemory::Map(Process* target_process, VAddr address, MemoryPermission permissions,
97  MemoryPermission other_permissions) {
98 
99  MemoryPermission own_other_permissions =
100  target_process == owner_process ? this->permissions : this->other_permissions;
101 
102  // Automatically allocated memory blocks can only be mapped with other_permissions = DontCare
103  if (base_address == 0 && other_permissions != MemoryPermission::DontCare) {
104  return ERR_INVALID_COMBINATION;
105  }
106 
107  // Error out if the requested permissions don't match what the creator process allows.
108  if (static_cast<u32>(permissions) & ~static_cast<u32>(own_other_permissions)) {
109  LOG_ERROR(Kernel, "cannot map id={}, address=0x{:08X}, name={}, permissions don't match",
110  GetObjectId(), address, name);
111  return ERR_INVALID_COMBINATION;
112  }
113 
114  // Heap-backed memory blocks can not be mapped with other_permissions = DontCare
115  if (base_address != 0 && other_permissions == MemoryPermission::DontCare) {
116  LOG_ERROR(Kernel, "cannot map id={}, address=0x{:08X}, name={}, permissions don't match",
117  GetObjectId(), address, name.c_str());
118  return ERR_INVALID_COMBINATION;
119  }
120 
121  // Error out if the provided permissions are not compatible with what the creator process needs.
122  if (other_permissions != MemoryPermission::DontCare &&
123  static_cast<u32>(this->permissions) & ~static_cast<u32>(other_permissions)) {
124  LOG_ERROR(Kernel, "cannot map id={}, address=0x{:08X}, name={}, permissions don't match",
125  GetObjectId(), address, name);
126  return ERR_WRONG_PERMISSION;
127  }
128 
129  // TODO(Subv): Check for the Shared Device Mem flag in the creator process.
130  /*if (was_created_with_shared_device_mem && address != 0) {
131  return ResultCode(ErrorDescription::InvalidCombination, ErrorModule::OS,
132  ErrorSummary::InvalidArgument, ErrorLevel::Usage);
133  }*/
134 
135  // TODO(Subv): The same process that created a SharedMemory object
136  // can not map it in its own address space unless it was created with addr=0, result 0xD900182C.
137 
138  if (address != 0) {
139  if (address < Memory::HEAP_VADDR || address + size >= Memory::SHARED_MEMORY_VADDR_END) {
140  LOG_ERROR(Kernel, "cannot map id={}, address=0x{:08X}, name={}, invalid address",
141  GetObjectId(), address, name);
142  return ERR_INVALID_ADDRESS;
143  }
144  }
145 
146  VAddr target_address = address;
147 
148  if (base_address == 0 && target_address == 0) {
149  // Calculate the address at which to map the memory block.
150  target_address = Memory::PhysicalToVirtualAddress(linear_heap_phys_address).value();
151  }
152 
153  // Map the memory block into the target process
154  auto result = target_process->vm_manager.MapMemoryBlock(
155  target_address, backing_block, backing_block_offset, size, MemoryState::Shared);
156  if (result.Failed()) {
157  LOG_ERROR(
158  Kernel,
159  "cannot map id={}, target_address=0x{:08X} name={}, error mapping to virtual memory",
160  GetObjectId(), target_address, name);
161  return result.Code();
162  }
163 
164  return target_process->vm_manager.ReprotectRange(target_address, size,
165  ConvertPermissions(permissions));
166 }
