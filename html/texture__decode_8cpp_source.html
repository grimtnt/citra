23 size_t CalculateTileSize(TextureFormat format) {
24  switch (format) {
25  case TextureFormat::RGBA8:
26  return 4 * TILE_SIZE;
27 
28  case TextureFormat::RGB8:
29  return 3 * TILE_SIZE;
30 
31  case TextureFormat::RGB5A1:
32  case TextureFormat::RGB565:
33  case TextureFormat::RGBA4:
34  case TextureFormat::IA8:
35  case TextureFormat::RG8:
36  return 2 * TILE_SIZE;
37 
38  case TextureFormat::I8:
39  case TextureFormat::A8:
40  case TextureFormat::IA4:
41  return 1 * TILE_SIZE;
42 
43  case TextureFormat::I4:
44  case TextureFormat::A4:
45  return TILE_SIZE / 2;
46 
47  case TextureFormat::ETC1:
48  return ETC1_SUBTILES * 8;
49 
50  case TextureFormat::ETC1A4:
51  return ETC1_SUBTILES * 16;
52 
53  default: // placeholder for yet unknown formats
54  UNIMPLEMENTED();
55  return 0;
56  }
57 }

59 Math::Vec4<u8> LookupTexture(const u8* source, unsigned int x, unsigned int y,
60  const TextureInfo& info) {
61  // Coordinate in tiles
62  const unsigned int coarse_x{x / 8};
63  const unsigned int coarse_y{y / 8};
64 
65  // Coordinate inside the tile
66  const unsigned int fine_x{x % 8};
67  const unsigned int fine_y{y % 8};
68 
69  const u8* line{source + coarse_y * info.stride};
70  const u8* tile{line + coarse_x * CalculateTileSize(info.format)};
71  return LookupTexelInTile(tile, fine_x, fine_y, info);
72 }

74 Math::Vec4<u8> LookupTexelInTile(const u8* source, unsigned int x, unsigned int y,
75  const TextureInfo& info) {
76  DEBUG_ASSERT(x < 8);
77  DEBUG_ASSERT(y < 8);
78 
79  using VideoCore::MortonInterleave;
80 
81  switch (info.format) {
82  case TextureFormat::RGBA8: {
83  auto res{Color::DecodeRGBA8(source + MortonInterleave(x, y) * 4)};
84  return {res.r(), res.g(), res.b(), res.a()};
85  }
86 
87  case TextureFormat::RGB8: {
88  auto res{Color::DecodeRGB8(source + MortonInterleave(x, y) * 3)};
89  return {res.r(), res.g(), res.b(), 255};
90  }
91 
92  case TextureFormat::RGB5A1: {
93  auto res{Color::DecodeRGB5A1(source + MortonInterleave(x, y) * 2)};
94  return {res.r(), res.g(), res.b(), res.a()};
95  }
96 
97  case TextureFormat::RGB565: {
98  auto res{Color::DecodeRGB565(source + MortonInterleave(x, y) * 2)};
99  return {res.r(), res.g(), res.b(), 255};
100  }
101 
102  case TextureFormat::RGBA4: {
103  auto res{Color::DecodeRGBA4(source + MortonInterleave(x, y) * 2)};
104  return {res.r(), res.g(), res.b(), res.a()};
105  }
106 
107  case TextureFormat::IA8: {
108  const u8* source_ptr{source + MortonInterleave(x, y) * 2};
109 
110  return {source_ptr[1], source_ptr[1], source_ptr[1], source_ptr[0]};
111  }
112 
113  case TextureFormat::RG8: {
114  auto res{Color::DecodeRG8(source + MortonInterleave(x, y) * 2)};
115  return {res.r(), res.g(), 0, 255};
116  }
117 
118  case TextureFormat::I8: {
119  const u8* source_ptr{source + MortonInterleave(x, y)};
120  return {*source_ptr, *source_ptr, *source_ptr, 255};
121  }
122 
123  case TextureFormat::A8: {
124  const u8* source_ptr{source + MortonInterleave(x, y)};
125 
126  return {0, 0, 0, *source_ptr};
127  }
128 
129  case TextureFormat::IA4: {
130  const u8* source_ptr{source + MortonInterleave(x, y)};
131 
132  u8 i{Color::Convert4To8(((*source_ptr) & 0xF0) >> 4)};
133  u8 a{Color::Convert4To8((*source_ptr) & 0xF)};
134 
135  return {i, i, i, a};
136  }
137 
138  case TextureFormat::I4: {
139  u32 morton_offset{MortonInterleave(x, y)};
140  const u8* source_ptr{source + morton_offset / 2};
141 
142  u8 i{static_cast<u8>((morton_offset % 2) ? ((*source_ptr & 0xF0) >> 4)
143  : (*source_ptr & 0xF))};
144  i = Color::Convert4To8(i);
145 
146  return {i, i, i, 255};
147  }
148 
149  case TextureFormat::A4: {
150  u32 morton_offset{MortonInterleave(x, y)};
151  const u8* source_ptr{source + morton_offset / 2};
152 
153  u8 a{static_cast<u8>((morton_offset % 2) ? ((*source_ptr & 0xF0) >> 4)
154  : (*source_ptr & 0xF))};
155  a = Color::Convert4To8(a);
156 
157  return {0, 0, 0, a};
158  }
159 
160  case TextureFormat::ETC1:
161  case TextureFormat::ETC1A4: {
162  bool has_alpha{info.format == TextureFormat::ETC1A4};
163  int subtile_size{has_alpha ? 16 : 8};
164 
165  // ETC1 further subdivides each 8x8 tile into four 4x4 subtiles
166  constexpr unsigned int subtile_width{4};
167  constexpr unsigned int subtile_height{4};
168 
169  unsigned int subtile_index{(x / subtile_width) + 2 * (y / subtile_height)};
170  x %= subtile_width;
171  y %= subtile_height;
172 
173  const u8* subtile_ptr{source + subtile_index * subtile_size};
174 
175  u8 alpha{255};
176  if (has_alpha) {
177  u64_le packed_alpha;
178  memcpy(&packed_alpha, subtile_ptr, sizeof(u64));
179  subtile_ptr += sizeof(u64);
180 
181  alpha = Color::Convert4To8((packed_alpha >> (4 * (x * subtile_width + y))) & 0xF);
182  }
183 
184  u64_le subtile_data;
185  memcpy(&subtile_data, subtile_ptr, sizeof(u64));
186 
187  return Math::MakeVec(SampleETC1Subtile(subtile_data, x, y), alpha);
188  }
189 
190  default:
191  LOG_ERROR(HW_GPU, "Unknown texture format: {:x}", (u32)info.format);
192  DEBUG_ASSERT(false);
193  return {};
194  }
195 }
