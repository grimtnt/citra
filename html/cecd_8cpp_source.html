30 void Module::Interface::Open(Kernel::HLERequestContext& ctx) {
31  IPC::RequestParser rp{ctx, 0x01, 3, 2};
32  const u32 ncch_program_id{rp.Pop<u32>()};
33  const CecDataPathType path_type{rp.PopEnum<CecDataPathType>()};
34  CecOpenMode open_mode{};
35  open_mode.raw = rp.Pop<u32>();
36  rp.PopPID();
37 
38  FileSys::Path path{cecd->GetCecDataPathTypeAsString(path_type, ncch_program_id).data()};
39  FileSys::Mode mode;
40  mode.read_flag.Assign(1);
41  mode.write_flag.Assign(1);
42  mode.create_flag.Assign(1);
43 
44  SessionData* session_data{GetSessionData(ctx.Session())};
45  session_data->ncch_program_id = ncch_program_id;
46  session_data->open_mode.raw = open_mode.raw;
47  session_data->data_path_type = path_type;
48  session_data->path = path;
49 
50  IPC::ResponseBuilder rb{rp.MakeBuilder(2, 0)};
51  switch (path_type) {
52  case CecDataPathType::RootDir:
53  case CecDataPathType::MboxDir:
54  case CecDataPathType::InboxDir:
55  case CecDataPathType::OutboxDir: {
56  auto dir_result =
57  Service::FS::OpenDirectoryFromArchive(cecd->cecd_system_save_data_archive, path);
58  if (dir_result.Failed()) {
59  if (open_mode.create) {
60  Service::FS::CreateDirectoryFromArchive(cecd->cecd_system_save_data_archive, path);
61  rb.Push(RESULT_SUCCESS);
62  } else {
63  LOG_DEBUG(Service_CECD, "Failed to open directory: {}", path.AsString());
64  rb.Push(ResultCode(ErrorDescription::NoData, ErrorModule::CEC,
65  ErrorSummary::NotFound, ErrorLevel::Status));
66  }
67  rb.Push<u32>(0); // Zero entries
68  } else {
69  constexpr u32 max_entries{32}; // reasonable value, just over max boxes 24
70  auto directory{dir_result.Unwrap()};
71 
72  // Actual reading into vector seems to be required for entry count
73  std::vector<FileSys::Entry> entries(max_entries);
74  const u32 entry_count{directory->backend->Read(max_entries, entries.data())};
75 
76  LOG_DEBUG(Service_CECD, "Number of entries found: {}", entry_count);
77 
78  rb.Push(RESULT_SUCCESS);
79  rb.Push<u32>(entry_count); // Entry count
80  directory->backend->Close();
81  }
82  break;
83  }
84  default: { // If not directory, then it is a file
85  auto file_result{
86  Service::FS::OpenFileFromArchive(cecd->cecd_system_save_data_archive, path, mode)};
87  if (file_result.Failed()) {
88  LOG_DEBUG(Service_CECD, "Failed to open file: {}", path.AsString());
89  rb.Push(ResultCode(ErrorDescription::NoData, ErrorModule::CEC, ErrorSummary::NotFound,
90  ErrorLevel::Status));
91  rb.Push<u32>(0); // No file size
92  } else {
93  session_data->file = std::move(file_result.Unwrap());
94  rb.Push(RESULT_SUCCESS);
95  rb.Push<u32>(session_data->file->backend->GetSize()); // Return file size
96  }
97 
98  if (path_type == CecDataPathType::MboxProgramId) {
99  std::vector<u8> program_id(8);
100  u64_le le_program_id{Kernel::g_current_process->codeset->program_id};
101  std::memcpy(program_id.data(), &le_program_id, sizeof(u64));
102  session_data->file->backend->Write(0, sizeof(u64), true, program_id.data());
103  session_data->file->backend->Close();
104  }
105  }
106  }
107 
108  LOG_DEBUG(Service_CECD,
109  "called, ncch_program_id={:#010x}, path_type={:#04x}, path={}, "
110  "open_mode: raw={:#x}, unknown={}, read={}, write={}, create={}, check={}",
111  ncch_program_id, static_cast<u32>(path_type), path.AsString(), open_mode.raw,
112  open_mode.unknown, open_mode.read, open_mode.write, open_mode.create,
113  open_mode.check);
114 }

285  buffer.data() + msg_header.header_size + msg_header.body_size, 0x20);

294

300  if (verify_hmac)

302  else
303  LOG_DEBUG(Service_CECD, "Verification failed");
304 
305  rb.Push(RESULT_SUCCESS);
306  rb.Push<u32>(bytes_read);
307  } else {
308  rb.Push(ResultCode(ErrorDescription::NoData, ErrorModule::CEC, ErrorSummary::NotFound,
309  ErrorLevel::Status));
310  rb.Push<u32>(0); // zero bytes read
311  }
312 
313  rb.PushMappedBuffer(message_id_buffer);
314  rb.PushMappedBuffer(hmac_key_buffer);
315  rb.PushMappedBuffer(write_buffer);
316 
317  LOG_DEBUG(
318  Service_CECD,
319  "called, ncch_program_id={:#010x}, is_outbox={}, message_id_size={:#x}, buffer_size={:#x}",
320  ncch_program_id, is_outbox, message_id_size, buffer_size);

932 void Module::CheckAndUpdateFile(const CecDataPathType path_type, const u32 ncch_program_id,
933  std::vector<u8>& file_buffer) {
934  constexpr u32 max_num_boxes{24};
935  constexpr u32 name_size{16}; // fixed size 16 characters long
936  constexpr u32 valid_name_size{8}; // 8 characters are valid, the rest are null
937  const std::size_t file_size{file_buffer.size()};
938 
939  switch (path_type) {
940  case CecDataPathType::MboxList: {
941  CecMBoxListHeader mbox_list_header = {};
942  std::memcpy(&mbox_list_header, file_buffer.data(), sizeof(CecMBoxListHeader));
943 
944  LOG_DEBUG(Service_CECD, "CecMBoxList: magic={:#06x}, version={:#06x}, num_boxes={:#06x}",
945  mbox_list_header.magic, mbox_list_header.version, mbox_list_header.num_boxes);
946 
947  if (file_size != sizeof(CecMBoxListHeader)) { // 0x18C
948  LOG_DEBUG(Service_CECD, "CecMBoxListHeader size is incorrect: {}", file_size);
949  }
950 
951  if (mbox_list_header.magic != 0x6868) { // 'hh'
952  if (mbox_list_header.magic == 0 || mbox_list_header.magic == 0xFFFF) {
953  LOG_DEBUG(Service_CECD, "CecMBoxListHeader magic number is not set");
954  } else {
955  LOG_DEBUG(Service_CECD, "CecMBoxListHeader magic number is incorrect: {}",
956  mbox_list_header.magic);
957  }
958  std::memset(&mbox_list_header, 0, sizeof(CecMBoxListHeader));
959  mbox_list_header.magic = 0x6868;
960  }
961 
962  if (mbox_list_header.version != 0x01) { // Not quite sure if it is a version
963  if (mbox_list_header.version == 0)
964  LOG_DEBUG(Service_CECD, "CecMBoxListHeader version is not set");
965  else
966  LOG_DEBUG(Service_CECD, "CecMBoxListHeader version is incorrect: {}",
967  mbox_list_header.version);
968  mbox_list_header.version = 0x01;
969  }
970 
971  if (mbox_list_header.num_boxes > 24) {
972  LOG_DEBUG(Service_CECD, "CecMBoxListHeader number of boxes is too large: {}",
973  mbox_list_header.num_boxes);
974  } else {
975  std::vector<u8> name_buffer(name_size);
976  std::memset(name_buffer.data(), 0, name_size);
977 
978  if (ncch_program_id != 0) {
979  std::string name{fmt::format("{:08x}", ncch_program_id)};
980  std::memcpy(name_buffer.data(), name.data(), name.size());
981 
982  bool already_activated{};
983  for (u16_le i{}; i < mbox_list_header.num_boxes; i++) {
984  // Box names start at offset 0xC, are 16 char long, first 8 id, last 8 null
985  if (std::memcmp(name_buffer.data(), &mbox_list_header.box_names[i],
986  valid_name_size) == 0) {
987  LOG_DEBUG(Service_CECD, "Title already activated");
988  already_activated = true;
989  }
990  };
991 
992  if (!already_activated) {
993  if (mbox_list_header.num_boxes < max_num_boxes) { // max boxes
994  LOG_DEBUG(Service_CECD, "Adding title to mboxlist____: {}", name);
995  std::memcpy(&mbox_list_header.box_names[mbox_list_header.num_boxes],
996  name_buffer.data(), name_size);
997  mbox_list_header.num_boxes++;
998  }
999  }
1000  } else { // ncch_program_id == 0, remove/update activated boxes
1004  FileSys::Path root_path{
1005  GetCecDataPathTypeAsString(CecDataPathType::RootDir, 0).data()};
1006 
1007  auto dir_result{Service::FS::OpenDirectoryFromArchive(cecd_system_save_data_archive,
1008  root_path)};
1009 
1010  auto root_dir{dir_result.Unwrap()};
1011  std::vector<FileSys::Entry> entries(max_num_boxes + 1); // + 1 mboxlist
1012  const u32 entry_count{root_dir->backend->Read(max_num_boxes + 1, entries.data())};
1013  root_dir->backend->Close();
1014 
1015  LOG_DEBUG(Service_CECD, "Number of entries found in /CEC: {}", entry_count);
1016 
1017  std::string mbox_list_name{"MBoxList____"};
1018  std::string file_name{};
1019  std::u16string u16_filename{};
1020 
1021  // Loop through entries but don't add mboxlist____ to itself.
1022  for (u32 i{}; i < entry_count; i++) {
1023  u16_filename = std::u16string(entries[i].filename);
1024  file_name = Common::UTF16ToUTF8(u16_filename);
1025 
1026  if (mbox_list_name.compare(file_name) != 0) {
1027  LOG_DEBUG(Service_CECD, "Adding title to mboxlist____: {}", file_name);
1028  std::memcpy(&mbox_list_header.box_names[mbox_list_header.num_boxes++],
1029  file_name.data(), valid_name_size);
1030  }
1031  }
1032  }
1033  }
1034  std::memcpy(file_buffer.data(), &mbox_list_header, sizeof(CecMBoxListHeader));
1035  break;
1036  }
1037  case CecDataPathType::MboxInfo: {
1038  CecMBoxInfoHeader mbox_info_header = {};
1039  std::memcpy(&mbox_info_header, file_buffer.data(), sizeof(CecMBoxInfoHeader));
1040 
1041  LOG_DEBUG(Service_CECD,
1042  "CecMBoxInfoHeader: magic={:#06x}, program_id={:#010x}, "
1043  "private_id={:#010x}, flag={:#04x}, flag2={:#04x}",
1044  mbox_info_header.magic, mbox_info_header.program_id, mbox_info_header.private_id,
1045  mbox_info_header.flag, mbox_info_header.flag2);
