72  const Math::Vec3<u8> GetRGB(unsigned int x, unsigned int y) const {
73  unsigned int texel{4 * x + y};
74 
75  if (flip)
76  std::swap(x, y);
77 
78  // Lookup base value
79  Math::Vec3<int> ret{};
80  if (differential_mode) {
81  ret.r() = static_cast<int>(differential.r);
82  ret.g() = static_cast<int>(differential.g);
83  ret.b() = static_cast<int>(differential.b);
84  if (x >= 2) {
85  ret.r() += static_cast<int>(differential.dr);
86  ret.g() += static_cast<int>(differential.dg);
87  ret.b() += static_cast<int>(differential.db);
88  }
89  ret.r() = Color::Convert5To8(ret.r());
90  ret.g() = Color::Convert5To8(ret.g());
91  ret.b() = Color::Convert5To8(ret.b());
92  } else {
93  if (x < 2) {
94  ret.r() = Color::Convert4To8(static_cast<u8>(separate.r1));
95  ret.g() = Color::Convert4To8(static_cast<u8>(separate.g1));
96  ret.b() = Color::Convert4To8(static_cast<u8>(separate.b1));
97  } else {
98  ret.r() = Color::Convert4To8(static_cast<u8>(separate.r2));
99  ret.g() = Color::Convert4To8(static_cast<u8>(separate.g2));
100  ret.b() = Color::Convert4To8(static_cast<u8>(separate.b2));
101  }
102  }
103 
104  // Add modifier
105  int table_index{static_cast<int>((x < 2) ? table_index_1.Value() : table_index_2.Value())};
106 
107  int modifier = etc1_modifier_table[table_index][GetTableSubIndex(texel)];
108  if (GetNegationFlag(texel))
109  modifier *= -1;
110 
111  ret.r() = std::clamp(ret.r() + modifier, 0, 255);
112  ret.g() = std::clamp(ret.g() + modifier, 0, 255);
113  ret.b() = std::clamp(ret.b() + modifier, 0, 255);
114 
115  return ret.Cast<u8>();
116  }
117 };
