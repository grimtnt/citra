101  static bool IsLightingSamplerSupported(LightingConfig config, LightingSampler sampler) {
102  switch (sampler) {
103  case LightingSampler::Distribution0:
104  return (config != LightingConfig::Config1);
105 
106  case LightingSampler::Distribution1:
107  return (config != LightingConfig::Config0) && (config != LightingConfig::Config1) &&
108  (config != LightingConfig::Config5);
109 
110  case LightingSampler::SpotlightAttenuation:
111  return (config != LightingConfig::Config2) && (config != LightingConfig::Config3);
112 
113  case LightingSampler::Fresnel:
114  return (config != LightingConfig::Config0) && (config != LightingConfig::Config2) &&
115  (config != LightingConfig::Config4);
116 
117  case LightingSampler::ReflectRed:
118  return (config != LightingConfig::Config3);
119 
120  case LightingSampler::ReflectGreen:
121  case LightingSampler::ReflectBlue:
122  return (config == LightingConfig::Config4) || (config == LightingConfig::Config5) ||
123  (config == LightingConfig::Config7);
124  default:
125  UNREACHABLE_MSG("Regs::IsLightingSamplerSupported: Reached unreachable section, "
126  "sampler should be one of Distribution0, Distribution1, "
127  "SpotlightAttenuation, Fresnel, ReflectRed, ReflectGreen or "
128  "ReflectBlue, instead got %i",
129  static_cast<int>(config));
130  }
131  }
