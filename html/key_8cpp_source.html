59 AESKey HexToKey(const std::string& hex) {

72 void LoadPresetKeys() {
73  const std::string filepath = FileUtil::GetUserPath(D_SYSDATA_IDX) + AES_KEYS;
74  FileUtil::CreateFullPath(filepath); // Create path if not already created
75  std::ifstream file;
76  OpenFStream(file, filepath, std::ios_base::in);
77  if (!file) {
78  return;
79  }
80 
81  while (!file.eof()) {
82  std::string line;
83  std::getline(file, line);
84  std::vector<std::string> parts;
85  Common::SplitString(line, '=', parts);
86  if (parts.size() != 2) {
87  LOG_ERROR(HW_AES, "Failed to parse {}", line);
88  continue;
89  }
90 
91  const std::string& name = parts[0];
92  AESKey key;
93  try {
94  key = HexToKey(parts[1]);
95  } catch (const std::logic_error& e) {
96  LOG_ERROR(HW_AES, "Invalid key {}: {}", parts[1], e.what());
97  continue;
98  }
99 
100  if (name == "generator") {
101  generator_constant = key;
102  continue;
103  }
104 
105  std::size_t common_key_index;
106  if (std::sscanf(name.c_str(), "common%zd", &common_key_index) == 1) {
107  if (common_key_index >= common_key_y_slots.size()) {
108  LOG_ERROR(HW_AES, "Invalid common key index {}", common_key_index);
109  } else {
110  common_key_y_slots[common_key_index] = key;
111  }
112  continue;
113  }
114 
115  std::size_t slot_id;
116  char key_type;
117  if (std::sscanf(name.c_str(), "slot0x%zXKey%c", &slot_id, &key_type) != 2) {
118  LOG_ERROR(HW_AES, "Invalid key name {}", name);
119  continue;
120  }
121 
122  if (slot_id >= MaxKeySlotID) {
123  LOG_ERROR(HW_AES, "Out of range slot ID {:#X}", slot_id);
124  continue;
125  }
126 
127  switch (key_type) {
128  case 'X':
129  key_slots.at(slot_id).SetKeyX(key);
130  break;
131  case 'Y':
132  key_slots.at(slot_id).SetKeyY(key);
133  break;
134  case 'N':
135  key_slots.at(slot_id).SetNormalKey(key);
136  break;
137  default:
138  LOG_ERROR(HW_AES, "Invalid key type {}", key_type);
139  break;
140  }
141  }
142 }
