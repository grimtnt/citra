16 static std::chrono::seconds GetInitTime() {
17  switch (Settings::values.init_clock) {
18  case Settings::InitClock::SystemTime: {
19  auto now{std::chrono::system_clock::now()};
20  // If the system time is in daylight saving, we give an additional hour to console time
21  std::time_t now_time_t{std::chrono::system_clock::to_time_t(now)};
22  std::tm* now_tm{std::localtime(&now_time_t)};
23  if (now_tm && now_tm->tm_isdst > 0)
24  now = now + std::chrono::hours(1);
25  return std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch());
26  }
27  case Settings::InitClock::FixedTime:
28  return std::chrono::seconds(Settings::values.init_time);
29  default:
30  UNREACHABLE();
31  }
32 }

63 u64 Handler::GetSystemTime() const {
64  std::chrono::milliseconds now{init_time +
65  std::chrono::milliseconds(CoreTiming::GetGlobalTimeUs() / 1000)};
66 
67  // 3DS system does't allow user to set a time before Jan 1 2000,
68  // so we use it as an auxiliary epoch to calculate the console time.
69  std::tm epoch_tm{};
70  epoch_tm.tm_sec = 0;
71  epoch_tm.tm_min = 0;
72  epoch_tm.tm_hour = 0;
73  epoch_tm.tm_mday = 1;
74  epoch_tm.tm_mon = 0;
75  epoch_tm.tm_year = 100;
76  epoch_tm.tm_isdst = 0;
77  s64 epoch{std::mktime(&epoch_tm) * 1000};
78 
79  // 3DS console time uses Jan 1 1900 as internal epoch,
80  // so we use the milliseconds between 1900 and 2000 as base console time
81  u64 console_time{3155673600000ULL};
82 
83  // Only when system time is after 2000, we set it as 3DS system time
84  if (now.count() > epoch) {
85  console_time += (now.count() - epoch);
86  }
87 
88  return console_time;
89 }
