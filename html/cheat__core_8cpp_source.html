60 std::vector<std::shared_ptr<CheatBase>> CheatEngine::ReadFileContents() {
61  std::string file_path{GetFilePath()};
62  std::string contents{};
63  FileUtil::ReadFileToString(true, file_path.c_str(), contents);
64  std::vector<std::string> lines{};
65  Common::SplitString(contents, '\n', lines);
66 
67  std::string code_type{
68  "Gateway"}; // If more cheat types added, need to specify which type in parsing.
69  std::vector<CheatLine> cheat_lines{};
70  std::vector<std::shared_ptr<CheatBase>> cheats{};
71  std::string name{};
72  bool enabled{};
73  for (std::size_t i{}; i < lines.size(); i++) {
74  std::string current_line{lines[i]};
75  current_line = Common::Trim(current_line);
76  if (!current_line.empty()) {
77  if (current_line.compare(0, 2, "+[") == 0) { // Enabled code
78  if (!cheat_lines.empty()) {
79  if (code_type == "Gateway")
80  cheats.push_back(
81  std::make_shared<GatewayCheat>(cheat_lines, enabled, name));
82  }
83  name = current_line.substr(2, current_line.length() - 3);
84  cheat_lines.clear();
85  enabled = true;
86  continue;
87  } else if (current_line.front() == '[') { // Disabled code
88  if (!cheat_lines.empty()) {
89  if (code_type == "Gateway")
90  cheats.push_back(
91  std::make_shared<GatewayCheat>(cheat_lines, enabled, name));
92  }
93  name = current_line.substr(1, current_line.length() - 2);
94  cheat_lines.clear();
95  enabled = false;
96  continue;
97  } else if (current_line.front() != '*') {
98  cheat_lines.emplace_back(std::move(current_line));
99  }
100  }
101  if (i == lines.size() - 1) { // End of file
102  if (!cheat_lines.empty()) {
103  if (code_type == "Gateway")
104  cheats.push_back(std::make_shared<GatewayCheat>(cheat_lines, enabled, name));
105  }
106  }
107  }
108  return cheats;
109 }

133 void GatewayCheat::Execute() {
134  if (!enabled)
135  return;
136  u32 addr{};
137  u32 reg{};
138  u32 offset{};
139  u32 val{};
140  int if_flag{};
141  u32 loop_count{};
142  std::size_t loopbackline{};
143  bool loop_flag{};
144  for (std::size_t i{}; i < cheat_lines.size(); i++) {
145  CheatLine line{cheat_lines[i]};
146  if (line.type == CheatType::Null)
147  continue;
148  addr = line.address;
149  val = line.value;
150  if (if_flag > 0) {
151  if (line.type == CheatType::Patch)
152  i += (line.value + 7) / 8;
153  if (line.type == CheatType::Terminator)
154  if_flag--; // ENDIF
155  if (line.type == CheatType::FullTerminator) {
156  // NEXT & Flush
157  if (loop_flag)
158  i = loopbackline - 1;
159  else {
160  offset = 0;
161  reg = 0;
162  loop_count = 0;
163  if_flag = 0;
164  loop_flag = false;
165  }
166  }
167  continue;
168  }
169 
170  switch (line.type) {
171  case CheatType::Write32: { // 0XXXXXXX YYYYYYYY word[XXXXXXX+offset] = YYYYYYYY
172  addr = line.address + offset;
173  Memory::Write32(addr, val);
174  Core::GetCPU().InvalidateCacheRange(addr, sizeof(u32));
175  break;
176  }
177  case CheatType::Write16: { // 1XXXXXXX 0000YYYY half[XXXXXXX+offset] = YYYY
178  addr = line.address + offset;
179  Memory::Write16(addr, static_cast<u16>(val));
180  Core::GetCPU().InvalidateCacheRange(addr, sizeof(u16));
181  break;
182  }
183  case CheatType::Write8: { // 2XXXXXXX 000000YY byte[XXXXXXX+offset] = YY
184  addr = line.address + offset;
185  Memory::Write8(addr, static_cast<u8>(val));
186  Core::GetCPU().InvalidateCacheRange(addr, sizeof(u8));
187  break;
188  }
189  case CheatType::GreaterThan32: { // 3XXXXXXX YYYYYYYY IF YYYYYYYY > word[XXXXXXX]
190  // ;unsigned
191  if (line.address == 0)
192  line.address = offset;
193  val = Memory::Read32(line.address);
194  if (line.value > val) {
195  if (if_flag > 0)
196  if_flag--;
197  } else {
198  if_flag++;
199  }
200  break;
201  }
202  case CheatType::LessThan32: { // 4XXXXXXX YYYYYYYY IF YYYYYYYY < word[XXXXXXX] ;unsigned
203  if (line.address == 0)
204  line.address = offset;
205  val = Memory::Read32(line.address);
206  if (line.value < val) {
207  if (if_flag > 0)
208  if_flag--;
209  } else {
210  if_flag++;
211  }
212  break;
213  }
214  case CheatType::EqualTo32: { // 5XXXXXXX YYYYYYYY IF YYYYYYYY = word[XXXXXXX]
215  if (line.address == 0)
216  line.address = offset;
217  val = Memory::Read32(line.address);
218  if (line.value == val) {
219  if (if_flag > 0)
220  if_flag--;
221  } else {
222  if_flag++;
223  }
224  break;
225  }
226  case CheatType::NotEqualTo32: { // 6XXXXXXX YYYYYYYY IF YYYYYYYY <> word[XXXXXXX]
227  if (line.address == 0)
228  line.address = offset;
229  val = Memory::Read32(line.address);
230  if (line.value != val) {
231  if (if_flag > 0)
232  if_flag--;
233  } else {
234  if_flag++;
235  }
236  break;
237  }
238  case CheatType::GreaterThan16: { // 7XXXXXXX ZZZZYYYY IF YYYY > ((not ZZZZ) AND
239  // half[XXXXXXX])
240  if (line.address == 0)
241  line.address = offset;
242  u32 x{line.address & 0x0FFFFFFF};
243  u32 y{line.value & 0xFFFF};
244  u32 z{line.value >> 16};
245  val = Memory::Read16(x);
246  if (y > (u16)((~z) & val)) {
247  if (if_flag > 0)
248  if_flag--;
249  } else {
250  if_flag++;
251  }
252  break;
253  }
254  case CheatType::LessThan16: { // 8XXXXXXX ZZZZYYYY IF YYYY < ((not ZZZZ) AND
255  // half[XXXXXXX])
256  if (line.address == 0)
257  line.address = offset;
258  u32 x{line.address & 0x0FFFFFFF};
259  u32 y{line.value & 0xFFFF};
260  u32 z{line.value >> 16};
261  val = Memory::Read16(x);
262  if (y < (u16)((~z) & val)) {
263  if (if_flag > 0)
264  if_flag--;
265  } else {
266  if_flag++;
267  }
268  break;
269  }
270  case CheatType::EqualTo16: { // 9XXXXXXX ZZZZYYYY IF YYYY = ((not ZZZZ) AND half[XXXXXXX])
271  if (line.address == 0)
272  line.address = offset;
273  u32 x{line.address & 0x0FFFFFFF};
274  u32 y{line.value & 0xFFFF};
275  u32 z{line.value >> 16};
276  val = Memory::Read16(x);
277  if (y == (u16)((~z) & val)) {
278  if (if_flag > 0)
279  if_flag--;
280  } else {
281  if_flag++;
282  }
283  break;
284  }
285  case CheatType::NotEqualTo16: { // AXXXXXXX ZZZZYYYY IF YYYY <> ((not ZZZZ) AND
286  // half[XXXXXXX])
287  if (line.address == 0)
288  line.address = offset;
289  u32 x{line.address & 0x0FFFFFFF};
290  u32 y{line.value & 0xFFFF};
291  u32 z{line.value >> 16};
292  val = Memory::Read16(x);
293  if (y != (u16)((~z) & val)) {
294  if (if_flag > 0)
295  if_flag--;
296  } else {
297  if_flag++;
298  }
299  break;
300  }
301  case CheatType::LoadOffset: { // BXXXXXXX 00000000 offset = word[XXXXXXX+offset]
302  addr = line.address + offset;
303  offset = Memory::Read32(addr);
304  break;
305  }
306  case CheatType::Loop: {
307  if (loop_count < (line.value + 1))
308  loop_flag = true;
309  else
310  loop_flag = false;
311  loop_count++;
312  loopbackline = i;
313  break;
314  }
315  case CheatType::Terminator: {
316  break;
317  }
318  case CheatType::LoopExecuteVariant: {
319  if (loop_flag)
320  i = loopbackline - 1;
321  break;
322  }
323  case CheatType::FullTerminator: {
324  if (loop_flag)
325  i = loopbackline - 1;
326  else {
327  offset = 0;
328  reg = 0;
329  loop_count = 0;
330  if_flag = 0;
331  loop_flag = false;
332  }
333  break;
334  }
335  case CheatType::SetOffset: {
336  offset = line.value;
337  break;
338  }
339  case CheatType::AddValue: {
340  reg += line.value;
341  break;
342  }
343  case CheatType::SetValue: {
344  reg = line.value;
345  break;
346  }
347  case CheatType::IncrementiveWrite32: {
348  addr = line.value + offset;
349  Memory::Write32(addr, reg);
350  offset += 4;
351  break;
352  }
353  case CheatType::IncrementiveWrite16: {
354  addr = line.value + offset;
355  Memory::Write16(addr, static_cast<u16>(reg));
356  offset += 2;
357  break;
358  }
359  case CheatType::IncrementiveWrite8: {
360  addr = line.value + offset;
361  Memory::Write8(addr, static_cast<u8>(reg));
362  offset += 1;
363  break;
364  }
365  case CheatType::Load32: {
366  addr = line.value + offset;
367  reg = Memory::Read32(addr);
368  break;
369  }
370  case CheatType::Load16: {
371  addr = line.value + offset;
372  reg = Memory::Read16(addr);
373  break;
374  }
375  case CheatType::Load8: {
376  addr = line.value + offset;
377  reg = Memory::Read8(addr);
378  break;
379  }
380  case CheatType::AddOffset: {
381  offset += line.value;
382  break;
383  }
384  case CheatType::Joker: {
385  auto state{Service::HID::GetInputsThisFrame()};
386  bool result{(state.hex & line.value) == line.value};
387  if (result) {
388  if (if_flag > 0)
389  if_flag--;
390  } else {
391  if_flag++;
392  }
393  break;
394  }
395  case CheatType::Patch: {
396  // Patch Code (Miscellaneous Memory Manipulation Codes)
397  // EXXXXXXX YYYYYYYY
398  // Copies YYYYYYYY bytes from (current code location + 8) to [XXXXXXXX + offset].
399  u32 x{line.address & 0x0FFFFFFF};
400  u32 y{line.value};
401  addr = x + offset;
402  {
403  u32 j{}, t{}, b{};
404  if (y > 0)
405  i++; // skip over the current code
406  while (y >= 4) {
407  u32 tmp{(t == 0) ? cheat_lines[i].address : cheat_lines[i].value};
408  if (t == 1)
409  i++;
410  t ^= 1;
411  Memory::Write32(addr, tmp);
412  addr += 4;
413  y -= 4;
414  }
415  while (y > 0) {
416  u32 tmp{((t == 0) ? cheat_lines[i].address : cheat_lines[i].value) >> b};
417  Memory::Write8(addr, tmp);
418  addr += 1;
419  y -= 1;
420  b += 4;
421  }
422  }
423  break;
424  }
425  }
426  }
