351 // swizzlers - create a subvector of specific components
352 // e.g. Vec2 uv() { return Vec2(x,y); }
353 // _DEFINE_SWIZZLER2 defines a single such function, DEFINE_SWIZZLER2 defines all of them for all
354 // component names (x<->r) and permutations (xy<->yx)
355 #define _DEFINE_SWIZZLER2(a, b, name) \
356  constexpr Vec2<T> name() const { \
357  return Vec2<T>(a, b); \
358  }
359 #define DEFINE_SWIZZLER2(a, b, a2, b2, a3, b3, a4, b4) \
360  _DEFINE_SWIZZLER2(a, b, a##b); \
361  _DEFINE_SWIZZLER2(a, b, a2##b2); \
362  _DEFINE_SWIZZLER2(a, b, a3##b3); \
363  _DEFINE_SWIZZLER2(a, b, a4##b4); \
364  _DEFINE_SWIZZLER2(b, a, b##a); \
365  _DEFINE_SWIZZLER2(b, a, b2##a2); \
366  _DEFINE_SWIZZLER2(b, a, b3##a3); \
367  _DEFINE_SWIZZLER2(b, a, b4##a4)
368 
369  DEFINE_SWIZZLER2(x, y, r, g, u, v, s, t);
370  DEFINE_SWIZZLER2(x, z, r, b, u, w, s, q);
371  DEFINE_SWIZZLER2(y, z, g, b, v, w, t, q);
372 #undef DEFINE_SWIZZLER2
373 #undef _DEFINE_SWIZZLER2
374 };
