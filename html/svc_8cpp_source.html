63 static ResultCode ControlMemory(u32* out_addr, u32 operation, u32 addr0, u32 addr1, u32 size,
64  u32 permissions) {
65  LOG_DEBUG(Kernel_SVC,
66  "called operation=0x{:08X}, addr0=0x{:08X}, addr1=0x{:08X}, size=0x{:X}, "
67  "permissions=0x{:08X}",
68  operation, addr0, addr1, size, permissions);
69 
70  if ((addr0 & Memory::PAGE_MASK) != 0 || (addr1 & Memory::PAGE_MASK) != 0) {
71  return ERR_MISALIGNED_ADDRESS;
72  }
73  if ((size & Memory::PAGE_MASK) != 0) {
74  return ERR_MISALIGNED_SIZE;
75  }
76 
77  u32 region{operation & MEMOP_REGION_MASK};
78  operation &= ~MEMOP_REGION_MASK;
79 
80  if (region != 0) {
81  LOG_WARNING(Kernel_SVC, "ControlMemory with specified region not supported, region={:X}",
82  region);
83  }
84 
85  if ((permissions & (u32)MemoryPermission::ReadWrite) != permissions) {
86  return ERR_INVALID_COMBINATION;
87  }
88  VMAPermission vma_permissions{(VMAPermission)permissions};
89 
90  auto& process{*g_current_process};
91 
92  switch (operation & MEMOP_OPERATION_MASK) {
93  case MEMOP_FREE: {
94  // TODO(Subv): What happens if an application tries to FREE a block of memory that has a
95  // SharedMemory pointing to it?
96  if (addr0 >= Memory::HEAP_VADDR && addr0 < Memory::HEAP_VADDR_END) {
97  ResultCode result{process.HeapFree(addr0, size)};
98  if (result.IsError())
99  return result;
100  } else if (addr0 >= process.GetLinearHeapBase() && addr0 < process.GetLinearHeapLimit()) {
101  ResultCode result{process.LinearFree(addr0, size)};
102  if (result.IsError())
103  return result;
104  } else {
105  return ERR_INVALID_ADDRESS;
106  }
107  *out_addr = addr0;
108  break;
109  }
110 
111  case MEMOP_COMMIT: {
112  if (operation & MEMOP_LINEAR) {
113  CASCADE_RESULT(*out_addr, process.LinearAllocate(addr0, size, vma_permissions));
114  } else {
115  CASCADE_RESULT(*out_addr, process.HeapAllocate(addr0, size, vma_permissions));
116  }
117  break;
118  }
119 
120  case MEMOP_MAP: {
121  // TODO: This is just a hack to avoid regressions until memory aliasing is implemented
122  CASCADE_RESULT(*out_addr, process.HeapAllocate(addr0, size, vma_permissions));
123  break;
124  }
125 
126  case MEMOP_UNMAP: {
127  // TODO: This is just a hack to avoid regressions until memory aliasing is implemented
128  ResultCode result{process.HeapFree(addr0, size)};
129  if (result.IsError())
130  return result;
131  break;
132  }
133 
134  case MEMOP_PROTECT: {
135  ResultCode result{process.vm_manager.ReprotectRange(addr0, size, vma_permissions)};
136  if (result.IsError())
137  return result;
138  break;
139  }
140 
141  default:
142  LOG_ERROR(Kernel_SVC, "unknown operation=0x{:08X}", operation);
143  return ERR_INVALID_COMBINATION;
144  }
145 
146  process.vm_manager.LogLayout(Log::Level::Trace);
147 
148  return RESULT_SUCCESS;
149 }

233  continue;
234

238  // TODO(Subv): When are the other running/ready threads terminated?

366 static ResultCode WaitSynchronization1(Handle handle, s64 nano_seconds) {
367  auto object{g_handle_table.Get<WaitObject>(handle)};
368  Thread* thread{GetCurrentThread()};
369 
370  if (object == nullptr)
371  return ERR_INVALID_HANDLE;
372 
373  LOG_TRACE(Kernel_SVC, "called handle=0x{:08X}({}:{}), nanoseconds={}", handle,
374  object->GetTypeName(), object->GetName(), nano_seconds);
375 
376  if (object->ShouldWait(thread)) {
377  if (nano_seconds == 0)
378  return RESULT_TIMEOUT;
379 
380  thread->wait_objects = {object};
381  object->AddWaitingThread(thread);
382  thread->status = THREADSTATUS_WAIT_SYNCH_ANY;
383 
384  // Create an event to wake the thread up after the specified nanosecond delay has passed
385  thread->WakeAfterDelay(nano_seconds);
386 
387  thread->wakeup_callback = [](ThreadWakeupReason reason, SharedPtr<Thread> thread,
388  SharedPtr<WaitObject> object) {
389  ASSERT(thread->status == THREADSTATUS_WAIT_SYNCH_ANY);
390 
391  if (reason == ThreadWakeupReason::Timeout) {
392  thread->SetWaitSynchronizationResult(RESULT_TIMEOUT);
393  return;
394  }
395 
396  ASSERT(reason == ThreadWakeupReason::Signal);
397  thread->SetWaitSynchronizationResult(RESULT_SUCCESS);
398 
399  // WaitSynchronization1 doesn't have an output index like WaitSynchronizationN, so we
400  // don't have to do anything else here.
401  };
402 
403  Core::System::GetInstance().PrepareReschedule();
404 
405  // Note: The output of this SVC will be set to RESULT_SUCCESS if the thread
406  // resumes due to a signal in its wait objects.
407  // Otherwise we retain the default value of timeout.
408  return RESULT_TIMEOUT;
409  }
410 
411  object->Acquire(thread);
412 
413  return RESULT_SUCCESS;
414 }

417 static ResultCode WaitSynchronizationN(s32* out, VAddr handles_address, s32 handle_count,
418  bool wait_all, s64 nano_seconds) {
419  Thread* thread{GetCurrentThread()};
420 
421  if (!Memory::IsValidVirtualAddress(handles_address))
422  return ERR_INVALID_POINTER;
423 
424  // NOTE: on real hardware, there is no nullptr check for 'out' (tested with firmware 4.4). If
425  // this happens, the running application will crash.
426  ASSERT_MSG(out != nullptr, "invalid output pointer specified!");
427 
428  // Check if 'handle_count' is invalid
429  if (handle_count < 0)
430  return ERR_OUT_OF_RANGE;
431 
432  using ObjectPtr = SharedPtr<WaitObject>;
433  std::vector<ObjectPtr> objects(handle_count);
434 
435  for (int i{}; i < handle_count; ++i) {
436  Handle handle{Memory::Read32(handles_address + i * sizeof(Handle))};
437  auto object{g_handle_table.Get<WaitObject>(handle)};
438  if (object == nullptr)
439  return ERR_INVALID_HANDLE;
440  objects[i] = object;
441  }
442 
443  if (wait_all) {
444  bool all_available{
445  std::all_of(objects.begin(), objects.end(),
446  [thread](const ObjectPtr& object) { return !object->ShouldWait(thread); })};
447  if (all_available) {
448  // We can acquire all objects right now, do so.
449  for (auto& object : objects)
450  object->Acquire(thread);
451  // Note: In this case, the `out` parameter is not set,
452  // and retains whatever value it had before.
453  return RESULT_SUCCESS;
454  }
455 
456  // Not all objects were available right now, prepare to suspend the thread.
457 
458  // If a timeout value of 0 was provided, just return the Timeout error code instead of
459  // suspending the thread.
460  if (nano_seconds == 0)
461  return RESULT_TIMEOUT;
462 
463  // Put the thread to sleep
464  thread->status = THREADSTATUS_WAIT_SYNCH_ALL;
465 
466  // Add the thread to each of the objects' waiting threads.
467  for (auto& object : objects) {
468  object->AddWaitingThread(thread);
469  }
470 
471  thread->wait_objects = std::move(objects);
472 
473  // Create an event to wake the thread up after the specified nanosecond delay has passed
474  thread->WakeAfterDelay(nano_seconds);
475 
476  thread->wakeup_callback = [](ThreadWakeupReason reason, SharedPtr<Thread> thread,
477  SharedPtr<WaitObject> object) {
478  ASSERT(thread->status == THREADSTATUS_WAIT_SYNCH_ALL);
479 
480  if (reason == ThreadWakeupReason::Timeout) {
481  thread->SetWaitSynchronizationResult(RESULT_TIMEOUT);
482  return;
483  }
484 
485  ASSERT(reason == ThreadWakeupReason::Signal);
486 
487  thread->SetWaitSynchronizationResult(RESULT_SUCCESS);
488  // The wait_all case does not update the output index.
489  };
490 
491  Core::System::GetInstance().PrepareReschedule();
492 
493  // This value gets set to -1 by default in this case, it is not modified after this.
494  *out = -1;
495  // Note: The output of this SVC will be set to RESULT_SUCCESS if the thread resumes due to
496  // a signal in one of its wait objects.
497  return RESULT_TIMEOUT;
498  } else {
499  // Find the first object that is acquirable in the provided list of objects
500  auto itr{std::find_if(objects.begin(), objects.end(), [thread](const ObjectPtr& object) {
501  return !object->ShouldWait(thread);
502  })};
503 
504  if (itr != objects.end()) {
505  // We found a ready object, acquire it and set the result value
506  WaitObject* object{itr->get()};
507  object->Acquire(thread);
508  *out = static_cast<s32>(std::distance(objects.begin(), itr));
509  return RESULT_SUCCESS;
510  }
511 
512  // No objects were ready to be acquired, prepare to suspend the thread.
513 
514  // If a timeout value of 0 was provided, just return the Timeout error code instead of
515  // suspending the thread.
516  if (nano_seconds == 0)
517  return RESULT_TIMEOUT;
518 
519  // Put the thread to sleep
520  thread->status = THREADSTATUS_WAIT_SYNCH_ANY;
521 
522  // Add the thread to each of the objects' waiting threads.
523  for (std::size_t i{}; i < objects.size(); ++i) {
524  WaitObject* object{objects[i].get()};
525  object->AddWaitingThread(thread);
526  }
527 
528  thread->wait_objects = std::move(objects);
529 
530  // Note: If no handles and no timeout were given, then the thread will deadlock, this is
531  // consistent with hardware behavior.
532 
533  // Create an event to wake the thread up after the specified nanosecond delay has passed
534  thread->WakeAfterDelay(nano_seconds);
535 
536  thread->wakeup_callback = [](ThreadWakeupReason reason, SharedPtr<Thread> thread,
537  SharedPtr<WaitObject> object) {
538  ASSERT(thread->status == THREADSTATUS_WAIT_SYNCH_ANY);
539 
540  if (reason == ThreadWakeupReason::Timeout) {
541  thread->SetWaitSynchronizationResult(RESULT_TIMEOUT);
542  return;
543  }
544 
545  ASSERT(reason == ThreadWakeupReason::Signal);
546 
547  thread->SetWaitSynchronizationResult(RESULT_SUCCESS);
548  thread->SetWaitSynchronizationOutput(thread->GetWaitObjectIndex(object.get()));
549  };
550 
551  Core::System::GetInstance().PrepareReschedule();
552 
553  // Note: The output of this SVC will be set to RESULT_SUCCESS if the thread resumes due to a
554  // signal in one of its wait objects.
555  // Otherwise we retain the default value of timeout, and -1 in the out parameter
556  *out = -1;
557  return RESULT_TIMEOUT;
558  }
559 }

561 static ResultCode ReceiveIPCRequest(SharedPtr<ServerSession> server_session,
562  SharedPtr<Thread> thread) {
563  if (server_session->parent->client == nullptr) {
564  return ERR_SESSION_CLOSED_BY_REMOTE;
565  }
566 
567  VAddr target_address{thread->GetCommandBufferAddress()};
568  VAddr source_address{server_session->currently_handling->GetCommandBufferAddress()};
569 
570  ResultCode translation_result{TranslateCommandBuffer(server_session->currently_handling, thread,
571  source_address, target_address, false)};
572 
573  // If a translation error occurred, immediately resume the client thread.
574  if (translation_result.IsError()) {
575  // Set the output of SendSyncRequest in the client thread to the translation output.
576  server_session->currently_handling->SetWaitSynchronizationResult(translation_result);
577 
578  server_session->currently_handling->ResumeFromWait();
579  server_session->currently_handling = nullptr;
580 
581  // TODO(Subv): This path should try to wait again on the same objects.
582  ASSERT_MSG(false, "ReplyAndReceive translation error behavior unimplemented");
583  }
584 
585  return translation_result;
586 }

589 static ResultCode ReplyAndReceive(s32* index, VAddr handles_address, s32 handle_count,
590  Handle reply_target) {
591  if (!Memory::IsValidVirtualAddress(handles_address))
592  return ERR_INVALID_POINTER;
593 
594  // Check if 'handle_count' is invalid
595  if (handle_count < 0)
596  return ERR_OUT_OF_RANGE;
597 
598  using ObjectPtr = SharedPtr<WaitObject>;
599  std::vector<ObjectPtr> objects(handle_count);
600 
601  for (int i{}; i < handle_count; ++i) {
602  Handle handle{Memory::Read32(handles_address + i * sizeof(Handle))};
603  auto object{g_handle_table.Get<WaitObject>(handle)};
604  if (object == nullptr)
605  return ERR_INVALID_HANDLE;
606  objects[i] = object;
607  }
608 
609  // We are also sending a command reply.
610  // Do not send a reply if the command id in the command buffer is 0xFFFF.
611  u32* cmd_buff{GetCommandBuffer()};
612  IPC::Header header{cmd_buff[0]};
613  if (reply_target != 0 && header.command_id != 0xFFFF) {
614  auto session{g_handle_table.Get<ServerSession>(reply_target)};
615  if (session == nullptr)
616  return ERR_INVALID_HANDLE;
617 
618  auto request_thread{std::move(session->currently_handling)};
619 
620  // Mark the request as "handled".
621  session->currently_handling = nullptr;
622 
623  // Error out if there's no request thread or the session was closed.
624  // TODO(Subv): Is the same error code (ClosedByRemote) returned for both of these cases?
625  if (request_thread == nullptr || session->parent->client == nullptr) {
626  *index = -1;
627  return ERR_SESSION_CLOSED_BY_REMOTE;
628  }
629 
630  VAddr source_address{GetCurrentThread()->GetCommandBufferAddress()};
631  VAddr target_address{request_thread->GetCommandBufferAddress()};
632 
633  ResultCode translation_result{TranslateCommandBuffer(
634  Kernel::GetCurrentThread(), request_thread, source_address, target_address, true)};
635 
636  // Note: The real kernel seems to always panic if the Server->Client buffer translation
637  // fails for whatever reason.
638  ASSERT(translation_result.IsSuccess());
639 
640  // Note: The scheduler is not invoked here.
641  request_thread->ResumeFromWait();
642  }
643 
644  if (handle_count == 0) {
645  *index = 0;
646  // The kernel uses this value as a placeholder for the real error, and returns it when we
647  // pass no handles and do not perform any reply.
648  if (reply_target == 0 || header.command_id == 0xFFFF)
649  return ResultCode(0xE7E3FFFF);
650 
651  return RESULT_SUCCESS;
652  }
653 
654  auto thread{GetCurrentThread()};
655 
656  // Find the first object that is acquirable in the provided list of objects
657  auto itr{std::find_if(objects.begin(), objects.end(), [thread](const ObjectPtr& object) {
658  return !object->ShouldWait(thread);
659  })};
660 
661  if (itr != objects.end()) {
662  // We found a ready object, acquire it and set the result value
663  WaitObject* object{itr->get()};
664  object->Acquire(thread);
665  *index = static_cast<s32>(std::distance(objects.begin(), itr));
666 
667  if (object->GetHandleType() != HandleType::ServerSession)
668  return RESULT_SUCCESS;
669 
670  auto server_session{static_cast<ServerSession*>(object)};
671  return ReceiveIPCRequest(server_session, GetCurrentThread());
672  }
673 
674  // No objects were ready to be acquired, prepare to suspend the thread.
675 
676  // Put the thread to sleep
677  thread->status = THREADSTATUS_WAIT_SYNCH_ANY;
678 
679  // Add the thread to each of the objects' waiting threads.
680  for (std::size_t i{}; i < objects.size(); ++i) {
681  WaitObject* object{objects[i].get()};
682  object->AddWaitingThread(thread);
683  }
684 
685  thread->wait_objects = std::move(objects);
686 
687  thread->wakeup_callback = [](ThreadWakeupReason reason, SharedPtr<Thread> thread,
688  SharedPtr<WaitObject> object) {
689  ASSERT(thread->status == THREADSTATUS_WAIT_SYNCH_ANY);
690  ASSERT(reason == ThreadWakeupReason::Signal);
691 
692  ResultCode result{RESULT_SUCCESS};
693 
694  if (object->GetHandleType() == HandleType::ServerSession) {
695  auto server_session{DynamicObjectCast<ServerSession>(object)};
696  result = ReceiveIPCRequest(server_session, thread);
697  }
698 
699  thread->SetWaitSynchronizationResult(result);
700  thread->SetWaitSynchronizationOutput(thread->GetWaitObjectIndex(object.get()));
701  };
702 
703  Core::System::GetInstance().PrepareReschedule();
704 
705  // Note: The output of this SVC will be set to RESULT_SUCCESS if the thread resumes due to a
706  // signal in one of its wait objects, or to 0xC8A01836 if there was a translation error.
707  // By default the index is set to -1.
708  *index = -1;
709  return RESULT_SUCCESS;
710 }

1159 static ResultCode CreateMemoryBlock(Handle* out_handle, u32 addr, u32 size, u32 my_permission,
1160  u32 other_permission) {

1162  return ERR_MISALIGNED_SIZE;

1165

1167  // SharedMemory blocks can not be created with Execute permissions
1168  switch (permission) {
1169  case MemoryPermission::None:
1170  case MemoryPermission::Read:
1171  case MemoryPermission::Write:
1172  case MemoryPermission::ReadWrite:
1173  case MemoryPermission::DontCare:
1174  return true;
1175  default:
1176  return false;
1177  }
1178  }};

1198  region = g_current_process->flags.memory_region;

1201  SharedMemory::Create(g_current_process, size, static_cast<MemoryPermission>(my_permission),
