45 static bool LZSS_Decompress(const u8* compressed, u32 compressed_size, u8* decompressed,
46  u32 decompressed_size) {
47  const u8* footer{compressed + compressed_size - 8};
48 
49  u32 buffer_top_and_bottom{};
50  std::memcpy(&buffer_top_and_bottom, footer, sizeof(u32));
51 
52  u32 out{decompressed_size};
53  u32 index{compressed_size - ((buffer_top_and_bottom >> 24) & 0xFF)};
54  u32 stop_index{compressed_size - (buffer_top_and_bottom & 0xFFFFFF)};
55 
56  memset(decompressed, 0, decompressed_size);
57  memcpy(decompressed, compressed, compressed_size);
58 
59  while (index > stop_index) {
60  u8 control = compressed[--index];
61 
62  for (unsigned i{}; i < 8; i++) {
63  if (index <= stop_index)
64  break;
65  if (index <= 0)
66  break;
67  if (out <= 0)
68  break;
69 
70  if (control & 0x80) {
71  // Check if compression is out of bounds
72  if (index < 2)
73  return false;
74  index -= 2;
75 
76  u32 segment_offset = compressed[index] | (compressed[index + 1] << 8);
77  u32 segment_size = ((segment_offset >> 12) & 15) + 3;
78  segment_offset &= 0x0FFF;
79  segment_offset += 2;
80 
81  // Check if compression is out of bounds
82  if (out < segment_size)
83  return false;
84 
85  for (unsigned j{}; j < segment_size; j++) {
86  // Check if compression is out of bounds
87  if (out + segment_offset >= decompressed_size)
88  return false;
89 
90  u8 data = decompressed[out + segment_offset];
91  decompressed[--out] = data;
92  }
93  } else {
94  // Check if compression is out of bounds
95  if (out < 1)
96  return false;
97  decompressed[--out] = compressed[--index];
98  }
99  control <<= 1;
100  }
101  }
102  return true;
103 }

124 Loader::ResultStatus NCCHContainer::Load() {
125  if (is_loaded)
126  return Loader::ResultStatus::Success;
127 
128  if (file.IsOpen()) {
129  // Reset read pointer in case this file has been read before.
130  file.Seek(ncch_offset, SEEK_SET);
131 
132  if (file.ReadBytes(&ncch_header, sizeof(NCCH_Header)) != sizeof(NCCH_Header))
133  return Loader::ResultStatus::Error;
134 
135  // Skip NCSD header and load first NCCH (NCSD is just a container of NCCH files)...
136  if (Loader::MakeMagic('N', 'C', 'S', 'D') == ncch_header.magic) {
137  LOG_DEBUG(Service_FS, "Only loading the first (bootable) NCCH within the NCSD file!");
138  ncch_offset += 0x4000;
139  file.Seek(ncch_offset, SEEK_SET);
140  file.ReadBytes(&ncch_header, sizeof(NCCH_Header));
141  }
142 
143  // Verify we are loading the correct file type...
144  if (Loader::MakeMagic('N', 'C', 'C', 'H') != ncch_header.magic)
145  return Loader::ResultStatus::ErrorInvalidFormat;
146 
147  has_header = true;
148  bool failed_to_decrypt{};
149  if (!ncch_header.no_crypto) {
150  is_encrypted = true;
151 
152  // Find primary and secondary keys
153  if (ncch_header.fixed_key) {
154  LOG_DEBUG(Service_FS, "Fixed-key crypto");
155  primary_key.fill(0);
156  secondary_key.fill(0);
157  } else {
158  using namespace HW::AES;
159  InitKeys();
160  std::array<u8, 16> key_y_primary{}, key_y_secondary{};
161 
162  std::copy(ncch_header.signature, ncch_header.signature + key_y_primary.size(),
163  key_y_primary.begin());
164 
165  if (!ncch_header.seed_crypto) {
166  key_y_secondary = key_y_primary;
167  } else {
168  // Seed crypto is unimplemented.
169  LOG_ERROR(Service_FS, "Unsupported seed crypto");
170  failed_to_decrypt = true;
171  }
172 
173  SetKeyY(KeySlotID::NCCHSecure1, key_y_primary);
174  if (!IsNormalKeyAvailable(KeySlotID::NCCHSecure1)) {
175  LOG_ERROR(Service_FS, "Secure1 KeyX missing");
176  failed_to_decrypt = true;
177  }
178  primary_key = GetNormalKey(KeySlotID::NCCHSecure1);
179 
180  switch (ncch_header.secondary_key_slot) {
181  case 0:
182  LOG_DEBUG(Service_FS, "Secure1 crypto");
183  secondary_key = primary_key;
184  break;
185  case 1:
186  LOG_DEBUG(Service_FS, "Secure2 crypto");
187  SetKeyY(KeySlotID::NCCHSecure2, key_y_secondary);
188  if (!IsNormalKeyAvailable(KeySlotID::NCCHSecure2)) {
189  LOG_ERROR(Service_FS, "Secure2 KeyX missing");
190  failed_to_decrypt = true;
191  }
192  secondary_key = GetNormalKey(KeySlotID::NCCHSecure2);
193  break;
194  case 10:
195  LOG_DEBUG(Service_FS, "Secure3 crypto");
196  SetKeyY(KeySlotID::NCCHSecure3, key_y_secondary);
197  if (!IsNormalKeyAvailable(KeySlotID::NCCHSecure3)) {
198  LOG_ERROR(Service_FS, "Secure3 KeyX missing");
199  failed_to_decrypt = true;
200  }
201  secondary_key = GetNormalKey(KeySlotID::NCCHSecure3);
202  break;
203  case 11:
204  LOG_DEBUG(Service_FS, "Secure4 crypto");
205  SetKeyY(KeySlotID::NCCHSecure4, key_y_secondary);
206  if (!IsNormalKeyAvailable(KeySlotID::NCCHSecure4)) {
207  LOG_ERROR(Service_FS, "Secure4 KeyX missing");
208  failed_to_decrypt = true;
209  }
210  secondary_key = GetNormalKey(KeySlotID::NCCHSecure4);
211  break;
212  }
213  }
214 
215  // Find CTR for each section
216  // Written with reference to
217  // https://github.com/d0k3/GodMode9/blob/99af6a73be48fa7872649aaa7456136da0df7938/arm9/source/game/ncch.c#L34-L52
218  if (ncch_header.version == 0 || ncch_header.version == 2) {
219  LOG_DEBUG(Loader, "NCCH version 0/2");
220  // In this version, CTR for each section is a magic number prefixed by partition ID
221  // (reverse order)
222  std::reverse_copy(ncch_header.partition_id, ncch_header.partition_id + 8,
223  exheader_ctr.begin());
224  exefs_ctr = romfs_ctr = exheader_ctr;
225  exheader_ctr[8] = 1;
226  exefs_ctr[8] = 2;
227  romfs_ctr[8] = 3;
228  } else if (ncch_header.version == 1) {
229  LOG_DEBUG(Loader, "NCCH version 1");
230  // In this version, CTR for each section is the section offset prefixed by partition
231  // ID, as if the entire NCCH image is encrypted using a single CTR stream.
232  std::copy(ncch_header.partition_id, ncch_header.partition_id + 8,
233  exheader_ctr.begin());
234  exefs_ctr = romfs_ctr = exheader_ctr;
235  auto u32ToBEArray = [](u32 value) -> std::array<u8, 4> {
236  return std::array<u8, 4>{
237  static_cast<u8>(value >> 24),
238  static_cast<u8>((value >> 16) & 0xFF),
239  static_cast<u8>((value >> 8) & 0xFF),
240  static_cast<u8>(value & 0xFF),
241  };
242  };
243  auto offset_exheader{u32ToBEArray(0x200)}; // exheader offset
244  auto offset_exefs{u32ToBEArray(ncch_header.exefs_offset * kBlockSize)};
245  auto offset_romfs{u32ToBEArray(ncch_header.romfs_offset * kBlockSize)};
246  std::copy(offset_exheader.begin(), offset_exheader.end(),
247  exheader_ctr.begin() + 12);
248  std::copy(offset_exefs.begin(), offset_exefs.end(), exefs_ctr.begin() + 12);
249  std::copy(offset_romfs.begin(), offset_romfs.end(), romfs_ctr.begin() + 12);
250  } else {
251  LOG_ERROR(Service_FS, "Unknown NCCH version {}", ncch_header.version);
252  failed_to_decrypt = true;
253  }
254  } else {
255  LOG_DEBUG(Service_FS, "No crypto");
256  is_encrypted = false;
257  }
258 
259  // System archives and DLC don't have an extended header but have RomFS
260  if (ncch_header.extended_header_size) {
261  if (file.ReadBytes(&exheader_header, sizeof(ExHeader_Header)) !=
262  sizeof(ExHeader_Header))
263  return Loader::ResultStatus::Error;
264 
265  if (is_encrypted) {
266  // This ID check is masked to low 32-bit as a toleration to ill-formed ROM created
267  // by merging games and its updates.
268  if ((exheader_header.system_info.jump_id & 0xFFFFFFFF) ==
269  (ncch_header.program_id & 0xFFFFFFFF)) {
270  LOG_WARNING(Service_FS, "NCCH is marked as encrypted but with decrypted "
271  "exheader. Force no crypto scheme.");
272  is_encrypted = false;
273  } else {
274  if (failed_to_decrypt) {
275  LOG_ERROR(Service_FS, "Failed to decrypt");
276  return Loader::ResultStatus::ErrorEncrypted;
277  }
278  CryptoPP::byte* data{reinterpret_cast<CryptoPP::byte*>(&exheader_header)};
279  CryptoPP::CTR_Mode<CryptoPP::AES>::Decryption(
280  primary_key.data(), primary_key.size(), exheader_ctr.data())
281  .ProcessData(data, data, sizeof(exheader_header));
282  }
283  }
284 
285  is_compressed = (exheader_header.codeset_info.flags.flag & 1) == 1;
286  u32 entry_point{exheader_header.codeset_info.text.address};
287  u32 code_size{exheader_header.codeset_info.text.code_size};
288  u32 stack_size{exheader_header.codeset_info.stack_size};
289  u32 bss_size{exheader_header.codeset_info.bss_size};
290  u32 core_version{exheader_header.arm11_system_local_caps.core_version};
291  u8 priority{exheader_header.arm11_system_local_caps.priority};
292  u8 resource_limit_category{
293  exheader_header.arm11_system_local_caps.resource_limit_category};
294 
295  LOG_DEBUG(Service_FS, "Name: {}",
296  exheader_header.codeset_info.name);
297  LOG_DEBUG(Service_FS, "Program ID: {:016X}", ncch_header.program_id);
298  LOG_DEBUG(Service_FS, "Code compressed: {}", is_compressed ? "yes" : "no");
299  LOG_DEBUG(Service_FS, "Entry point: 0x{:08X}", entry_point);
300  LOG_DEBUG(Service_FS, "Code size: 0x{:08X}", code_size);
301  LOG_DEBUG(Service_FS, "Stack size: 0x{:08X}", stack_size);
302  LOG_DEBUG(Service_FS, "Bss size: 0x{:08X}", bss_size);
303  LOG_DEBUG(Service_FS, "Core version: {}", core_version);
304  LOG_DEBUG(Service_FS, "Thread priority: 0x{:X}", priority);
305  LOG_DEBUG(Service_FS, "Resource limit category: {}", resource_limit_category);
306  LOG_DEBUG(Service_FS, "System Mode: {}",
307  static_cast<int>(exheader_header.arm11_system_local_caps.system_mode));
308 
309  has_exheader = true;
310  }
311 
312  // DLC can have an ExeFS and a RomFS but no extended header
313  if (ncch_header.exefs_size) {
314  exefs_offset = ncch_header.exefs_offset * kBlockSize;
315  u32 exefs_size{ncch_header.exefs_size * kBlockSize};
316 
317  LOG_DEBUG(Service_FS, "ExeFS offset: 0x{:08X}", exefs_offset);
318  LOG_DEBUG(Service_FS, "ExeFS size: 0x{:08X}", exefs_size);
319 
320  file.Seek(exefs_offset + ncch_offset, SEEK_SET);
321  if (file.ReadBytes(&exefs_header, sizeof(ExeFs_Header)) != sizeof(ExeFs_Header))
322  return Loader::ResultStatus::Error;
323 
324  if (is_encrypted) {
325  CryptoPP::byte* data{reinterpret_cast<CryptoPP::byte*>(&exefs_header)};
326  CryptoPP::CTR_Mode<CryptoPP::AES>::Decryption(primary_key.data(),
327  primary_key.size(), exefs_ctr.data())
328  .ProcessData(data, data, sizeof(exefs_header));
329  }
330 
331  exefs_file = FileUtil::IOFile{filepath, "rb"};
332  has_exefs = true;
333  }
334 
335  if (ncch_header.romfs_offset != 0 && ncch_header.romfs_size != 0)
336  has_romfs = true;
337  }
338 
339  LoadOverrides();
340 
341  // We need at least one of these or overrides, practically
342  if (!(has_exefs || has_romfs || is_tainted))
343  return Loader::ResultStatus::Error;
344 
345  is_loaded = true;
346  return Loader::ResultStatus::Success;
347 }

349 Loader::ResultStatus NCCHContainer::LoadOverrides() {
350  // Check for split-off files, mark the archive as tainted if we will use them
351  std::string romfs_override{filepath + ".romfs"};
352  if (FileUtil::Exists(romfs_override)) {
353  is_tainted = true;
354  }
355 
356  // If we have a split-off exefs file/folder, it takes priority
357  std::string exefs_override{filepath + ".exefs"};
358  std::string exefsdir_override{filepath + ".exefsdir/"};
359  if (FileUtil::Exists(exefs_override)) {
360  exefs_file = FileUtil::IOFile{exefs_override, "rb"};
361 
362  if (exefs_file.ReadBytes(&exefs_header, sizeof(ExeFs_Header)) == sizeof(ExeFs_Header)) {
363  LOG_DEBUG(Service_FS, "Loading ExeFS section from {}", exefs_override);
364  exefs_offset = 0;
365  is_tainted = true;
366  has_exefs = true;
367  } else {
368  exefs_file = FileUtil::IOFile(filepath, "rb");
369  }
370  } else if (FileUtil::Exists(exefsdir_override) && FileUtil::IsDirectory(exefsdir_override)) {
371  is_tainted = true;
372  }
373 
374  if (is_tainted)
375  LOG_WARNING(Service_FS,
376  "Loaded NCCH {} is tainted, application behavior may not be as expected!",
377  filepath);
378 
379  return Loader::ResultStatus::Success;
380 }

382 Loader::ResultStatus NCCHContainer::LoadSectionExeFS(const char* name, std::vector<u8>& buffer) {
383  Loader::ResultStatus result{Load()};
384  if (result != Loader::ResultStatus::Success)
385  return result;
386 
387  // Check if we have files that can drop-in and replace
388  result = LoadOverrideExeFSSection(name, buffer);
389  if (result == Loader::ResultStatus::Success || !has_exefs)
390  return result;
391 
392  // As of firmware 5.0.0-11 the logo is stored between the access descriptor and the plain region
393  // instead of the ExeFS.
394  if (std::strcmp(name, "logo") == 0) {
395  if (ncch_header.logo_region_offset && ncch_header.logo_region_size) {
396  std::size_t logo_offset{ncch_header.logo_region_offset * kBlockSize};
397  std::size_t logo_size{ncch_header.logo_region_size * kBlockSize};
398 
399  buffer.resize(logo_size);
400  file.Seek(ncch_offset + logo_offset, SEEK_SET);
401 
402  if (file.ReadBytes(buffer.data(), logo_size) != logo_size) {
403  LOG_ERROR(Service_FS, "Could not read NCCH logo");
404  return Loader::ResultStatus::Error;
405  }
406  return Loader::ResultStatus::Success;
407  } else {
408  LOG_INFO(Service_FS, "Attempting to load logo from the ExeFS");
409  }
410  }
411 
412  // If we don't have any separate files, we'll need a full ExeFS
413  if (!exefs_file.IsOpen())
414  return Loader::ResultStatus::Error;
415 
416  LOG_DEBUG(Service_FS, "{} sections:", kMaxSections);
417  // Iterate through the ExeFs archive until we find a section with the specified name...
418  for (unsigned section_number{}; section_number < kMaxSections; section_number++) {
419  const auto& section{exefs_header.section[section_number]};
420 
421  // Load the specified section...
422  if (strcmp(section.name, name) == 0) {
423  LOG_DEBUG(Service_FS, "{} - offset: 0x{:08X}, size: 0x{:08X}, name: {}", section_number,
424  section.offset, section.size, section.name);
425 
426  s64 section_offset{static_cast<s64>(
427  (section.offset + exefs_offset + sizeof(ExeFs_Header) + ncch_offset))};
428  exefs_file.Seek(section_offset, SEEK_SET);
429 
430  std::array<u8, 16> key{};
431  if (strcmp(section.name, "icon") == 0 || strcmp(section.name, "banner") == 0) {
432  key = primary_key;
433  } else {
434  key = secondary_key;
435  }
436 
437  CryptoPP::CTR_Mode<CryptoPP::AES>::Decryption dec(key.data(), key.size(),
438  exefs_ctr.data());
439  dec.Seek(section.offset + sizeof(ExeFs_Header));
440 
441  if (strcmp(section.name, ".code") == 0 && is_compressed) {
442  // Section is compressed, read compressed .code section...
443  std::unique_ptr<u8[]> temp_buffer{};
444  try {
445  temp_buffer.reset(new u8[section.size]);
446  } catch (std::bad_alloc&) {
447  return Loader::ResultStatus::ErrorMemoryAllocationFailed;
448  }
449 
450  if (exefs_file.ReadBytes(&temp_buffer[0], section.size) != section.size)
451  return Loader::ResultStatus::Error;
452 
453  if (is_encrypted) {
454  dec.ProcessData(&temp_buffer[0], &temp_buffer[0], section.size);
455  }
456 
457  // Decompress .code section...
458  u32 decompressed_size{LZSS_GetDecompressedSize(&temp_buffer[0], section.size)};
459  buffer.resize(decompressed_size);
460  if (!LZSS_Decompress(&temp_buffer[0], section.size, &buffer[0], decompressed_size))
461  return Loader::ResultStatus::ErrorInvalidFormat;
462  } else {
463  // Section is uncompressed...
464  buffer.resize(section.size);
465  if (exefs_file.ReadBytes(&buffer[0], section.size) != section.size)
466  return Loader::ResultStatus::Error;
467  if (is_encrypted) {
468  dec.ProcessData(&buffer[0], &buffer[0], section.size);
469  }
470  }
471  return Loader::ResultStatus::Success;
472  }
473  }
474  return Loader::ResultStatus::ErrorNotUsed;
475 }

477 Loader::ResultStatus NCCHContainer::LoadOverrideExeFSSection(const char* name,
478  std::vector<u8>& buffer) {
479  std::string override_name{};
480 
481  // Map our section name to the extracted equivalent
482  if (!strcmp(name, ".code"))
483  override_name = "code.bin";
484  else if (!strcmp(name, "icon"))
485  override_name = "code.bin";
486  else if (!strcmp(name, "banner"))
487  override_name = "banner.bnr";
488  else if (!strcmp(name, "logo"))
489  override_name = "logo.bcma.lz";
490  else
491  return Loader::ResultStatus::Error;

493  std::string section_override{filepath + ".exefsdir/" + override_name};
494  FileUtil::IOFile section_file{section_override, "rb"};

496  if (section_file.IsOpen()) {

498  buffer.resize(section_size);
499 
500  section_file.Seek(0, SEEK_SET);
501  if (section_file.ReadBytes(&buffer[0], section_size) == section_size) {
502  LOG_WARNING(Service_FS, "File {} overriding built-in ExeFS file", section_override);
503  return Loader::ResultStatus::Success;
504  }
505  }

509 Loader::ResultStatus NCCHContainer::ReadRomFS(std::shared_ptr<RomFSReader>& romfs_file) {
510  Loader::ResultStatus result{Load()};
511  if (result != Loader::ResultStatus::Success)
512  return result;
513 
514  if (ReadOverrideRomFS(romfs_file) == Loader::ResultStatus::Success)
515  return Loader::ResultStatus::Success;
516 
517  if (!has_romfs) {
518  LOG_DEBUG(Service_FS, "RomFS requested from NCCH which has no RomFS");
519  return Loader::ResultStatus::ErrorNotUsed;
520  }
521 
522  if (!file.IsOpen())
523  return Loader::ResultStatus::Error;
524 
525  u32 romfs_offset{ncch_offset + (ncch_header.romfs_offset * kBlockSize) + 0x1000};
526  u32 romfs_size{(ncch_header.romfs_size * kBlockSize) - 0x1000};
527 
528  LOG_DEBUG(Service_FS, "RomFS offset: 0x{:08X}", romfs_offset);
529  LOG_DEBUG(Service_FS, "RomFS size: 0x{:08X}", romfs_size);
530 
531  if (file.GetSize() < romfs_offset + romfs_size)
532  return Loader::ResultStatus::Error;
533 
534  // We reopen the file, to allow its position to be independent from file's
535  FileUtil::IOFile romfs_file_inner{filepath, "rb"};
536  if (!romfs_file_inner.IsOpen())
537  return Loader::ResultStatus::Error;
538 
539  if (is_encrypted) {
540  romfs_file = std::make_shared<RomFSReader>(std::move(romfs_file_inner), romfs_offset,
541  romfs_size, secondary_key, romfs_ctr, 0x1000);
542  } else {
543  romfs_file =
544  std::make_shared<RomFSReader>(std::move(romfs_file_inner), romfs_offset, romfs_size);
545  }
546 
547  return Loader::ResultStatus::Success;
548 }
